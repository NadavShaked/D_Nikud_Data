אֵיךְ קוֹרְאִים לְךָ, אֵיפֹה אַתָּה גָּר, הַפָּרָמֵדִיק עֲדַיִין צוֹעֵק, דָּבַק בַּמְּשִׂימָה שֶׁלּוֹ, אֲבָל הַנַּעַר לֹא מְגַלֶּה אֵיךְ קוֹרְאִים לוֹ וְגַם לֹא אֵיפֹה הוּא גָּר, "אֲנִי בִּכְלָל לֹא מִכָּאן," הוּא אוֹמֵר.
תִּיכָּנֵס, נִתְקַשֵּׁר לַהוֹרִים שֶׁלְּךָ.
מִזְּמַן שַׂמְנוּ עָלֶיךָ עֵין, יא בֶּן כֶּלֶב, עָלֵינוּ לֹא הִצְלַחְתָּ לַעֲבוֹד!
שָׁב תֹּאכַל אִיתָּנוּ, עַד שֶׁהָאֲווֹז שֶׁלְּךָ מִתְבַּשֵּׁל.
אָז תִּיכָּנֵס לְכָאן, נִיקַּח אוֹתְךָ, אוֹמֵר הַנַּהָג בְּנִימָה יְדִידוּתִית.

------------------
יֵשׁ לְךָ אִישָּׁה, יֵשׁ לְךָ יְלָדִים, יֵשׁ לָךְ עֲבוֹדָה, יֵשׁ לְךָ מְכוֹנִית, בַּיִת בְּפַרְבָּר הָעִיר, כְּפִי הַנִּרְאֶה תָּמוּת מְאוּשָּׁר וִילָדֶיךָ יִבְכּוּ בַּלְּווֹיָה שֶׁלְּךָ, הַשְּׁכֵנִים יַבִּיעוּ צַעַר עַל לֶכְתֵּךְ.
אַתָּה תִּישָּׁאֵר עִם מָרְתָה, יִהְיֶה לְךָ טוֹב אֶצְלָהּ.
אַתָּה חוֹשֵׁב שֶׁאֲנַחְנוּ רוֹצִים לִשְׁמוֹעַ שׁוּב פַּעַם אֵיךְ הִיא נֶעֶלְמָה פִּתְאוֹם וְאֵיךְ הַשּׁוֹטְרִים בָּאוּ וּבָעֲטוּ לְךָ בַּבֶּטֶן כִּי הֵם רָאוּ אֶתְכֶם?

------------------
אֲנִי סוֹלַחַת לָךְ אֶת מוֹתִי, בִּתְנַאי שֶׁתִּתְחַתֵּן עִם בַּת-חָסוּתִי, לְיַזְבָּטָה אִיווֹנוֹבְנָה.
לְמַעֲשֶׂה, זֶה מְשַׁעֲמֵם אוֹתָהּ, אֲבָל כְּדֵי לֹא לְצַעֵר אֶת אָבִיהָ הִיא לוֹחֶשֶׁת בְּהַכְנָעָה: "אֲנִי מְאוֹד-מְאוֹד מוֹדֶה לְךָ, אַבָּא יָקָר.
סְלַח לִי בְּבַקָּשָׁה, אוֹמֵר אָבִיהָ שֶׁל נַדְיָה, "תַּסְכִּים אוּלַי לְשַׁחְרֵר אֶת הַפִּיל שֶׁלָּךְ אֵלַי הַבַּיְתָה לִזְמַן-מָה?
יֵשׁ לְךָ הַכְשָׁרָה שֶׁל מוּמְחֶה לְנַפֵּץ.
‒ אַתָּה רוֹצֶה לְהִשְׁתַּמֵּט מֵחוֹבָתְךָ?

------------------
הַדָּבָר הֲכִי גָּרוּעַ שֶׁיֵּשׁ לְךָ זֶה רֶגֶל שְׁבוּרָה, וְזֶה בְּדֶרֶךְ כְּלָל לֹא מַצָּב סוֹפָנִי, אֶלָּא אִם כֵּן אַתָּה סוּס, אָמְרָה.
זֶה שֶׁאַתָּה שָׁם קוֹנְדּוֹם לֹא אוֹמֵר שֶׁלֹּא מְזַיְּינִים אוֹתְךָ.
לָמָּה אַתָּה לֹא קוֹרֵא לְעַצְמְךָ עַל שָׁם בֶּנְגָ'מִין אוֹ בְּנֵי?

------------------
אֲנִי יָכוֹל לְהָרִיחַ אֶת הַפַּחַד שֶׁלְּךָ, בֶּן זוֹנָה.
קַח אַחְרָיוּת עַל הַמַּעֲשִׂים שֶׁלְּךָ!
לֹא יָכוֹלְתִּי לָדַעַת שֶׁזֶּה הַבַּיִת שֶׁלְּךָ, אָחִי!
יֵשׁ לְךָ יֶלֶד, אַתָּה אוֹמֵר?
אִם תַּגִּיד אַתְּ זֶה עוֹד פַּעַם אַחַת אֲנִי תּוֹקֵעַ לָךְ כַּדּוּר בַּבֶּרֶךְ.
אֲנִי אֲשַׁלֵּם לָךְ כַּמָּה שֶׁתִּרְצֶה.
פָּלִיסִיטָה הָיְתָה מְסַלֶּקֶת אוֹתוֹ בְּנִימוּס מִן הַבַּיִת: "דַּי לְךָ, אָדוֹן דֶּה גֶּרְמַנְוִויל!
לְשֵׁירוּתְךָ, אֲדוֹנִי!

------------------
אוּלַי הָיִיתָ מַשְׁאִיר לָהּ כַּמָּה זִיכְרוֹנוֹת רָעִים.
טוֹב, הִיא אָמְרָה, "אֲנִי מַנִּיחָה שֶׁהָיִיתָ צָרִיךְ לְהַעֲסִיק אֶת עַצְמְךָ."
בֶּטַח גַּם אַתָּה עָבַדְתָּ אִיתָּם, אָמְרָה סְטֶיְיסִי לְאַנְטוֹן, וְהוּא הִנְהֵן בְּהֶיסַּח דַּעַת.
סְלַח לִי, אָמְרָה וְהִתְאַמְּצָה לִפְשׁוֹט אֶת זְרוֹעָהּ, "חָשַׁבְתִּי שֶׁאַתָּה הוּא שֶׁהָרַגְתָּ אוֹתוֹ!"
הָיִיתִי כֹּה בּוֹטַחַת וְכֹה טְהוֹרָה, וְאַתָּה בָּאתָ כְּדֵי לֶאֱהוֹב אוֹתִי, וַאֲנִי אָהַבְתִּי אוֹתְךָ!
אַתָּה קָנִיתָ שַׁרְשֶׁרֶת יַהֲלוֹמִים?

------------------
מָה אַתָּה יוֹצֵא כְּשֶׁלֹּא קָרְאוּ לֵךְ!
כִּי חָמֵשׁ אָחוּז מִשָּׁלוֹשׁ מִילְיַארְד, אַתָּה יָכוֹל לְחַשְׁבֵּן אֶת זֶה בְּעַצְמְךָ, זֶה יוֹצֵא מֵאָה חֲמִישִּׁים מִילְיוֹן!
וְאָז, הוֹפִיעַ הַסִּיכּוּן הָאֲמִיתִּי הָרִאשׁוֹן : "לַחֲצוֹת אֶת שַׁעַר בְּרַנְדֶּנְבּוּרְג: וְזֶה הָיָה צַר, אֲנִי אוֹמֵר לְךָ, בֶּנְאָדָם, כְּמוֹ אֵצֶל בְּתוּלָה: 'אַתָּה תִּבְדּוֹק מִשְּׂמֹאל; אֲנִי מִיָּמִין.

------------------
חִיפַּשְׂתִּי בְּעֵינָיו סִימָן לְכָךְ שֶׁהוּא מְזַהֶה אוֹתִי, זוֹכֵר מַשֶּׁהוּ.
וּבִילִּיתִי אֶת הַיָּמִים הַבָּאִים בְּמִשְׂחֲקֵי שֵׁשׁ בֵּשׁ אֵינְסוֹפִיִּים, אִיתָּהּ וְעִם הָבְנֵי דּוֹדִים שֶׁלָּהּ, מְנַסָּה נוֹאָשׁוֹת לְהַרְשִׁים אוֹתָהּ.
אֲנִי מְכַבֶּה אֶת זֶה וּפוֹתֵחַ אֶת מְעַבֵּד הַתַּמְלִילִים.
אִמִּי וַאֲנִי מַעֲרִיכִים מְאוֹד אֶת גְּמִישׁוּתְךָ וּשְׂמֵחִים שֶׁנִּזְכֶּה לִפְקוֹד אֶת חֲדַר הַבְּרִיחָה שֶׁלָּכֶם אֲשֶׁר אַתְּ חִידוֹתָיו אַתָּה מַרְבֶּה לְשֶׁבַח.

------------------
אַחֵר כָּךְ, כַּאֲשֶׁר הוּא מוֹתִירָה שְׁבוּרָה וּמוּשְׁפֶּלֶת, לְבַדָּהּ עִם חֲרָטוֹתֶיהָ, עִם מַחְשְׁבוֹתֶיהָ עַל הֶעָבָר, עִם אַכְזָבוֹתֶיהָ מִן הָאַהֲבָה, הוּא זוֹנֵחַ אוֹתָהּ, כִּמְעַט אֵינוֹ מְזַהֶה אוֹתָהּ, נוֹטֵשׁ אוֹתָהּ לְגוֹרָלָהּ הַמַּר; הוּא אַף שׂוֹנֵא אוֹתָהּ מִדֵּי פַּעַם.
לַאֲמִיתּוֹ שֶׁל דָּבָר, הָיִינוּ טִיפְּשִׁים, כִּי עַכְשָׁיו, כַּעֲבוֹר כָּל כָּךְ הַרְבֵּה שָׁנִים, קָשֶׁה לִי לְהַאֲמִין שֶׁלֹּא עָלָה עַל דַּעְתֵּנוּ לְהִתְנַגֵּד לְכָל מָה שֶׁהַמּוֹרֶה הוֹרָה לָנוּ לַעֲשׂוֹת.

------------------
אִם אַתָּה רוֹאֶה פִּתְאוֹם בָּאִי בַּרְנָשׁ יָחֵף וּמוּזְנָח מִתְגַּנֵּב בֵּין הַשִּׂיחִים קָרוֹב לַיָּם, וּבוֹ בַּזְּמַן אַתָּה מַבְחִין בְּמֶרְחָק שֶׁל פָּחוֹת מִקִּילוֹמֶטֶר וָחֵצִי מֵהַחוֹף בְּמִפְרָשִׂית מְלֵאָה כּוּשִׁים מִתְרַחֶקֶת בִּמְהִירוּת, לֹא הָיִיתָ חוֹשֵׁב שֶׁהָאִישׁ הַזֶּה נָפַל מֵהַשָּׁמַיִם, נָכוֹן?
הָיִיתָ צָרִיךְ לִרְאוֹת אוֹתָם קָמִים לִתְחִיָּיה וְנִרְתָּמִים שׁוּב לָעֲבוֹדָה שֶׁלָּהֶם!
אַתָּה מֵעֵז לְהַכְחִישׁ שֶׁמִּשָּׁם בָּאתָ?

------------------
הַאִם נָבִיא הִנְּךָ שֶׁאֵין לְהַזִּיק לוֹ, אוֹ בְּנוֹ שֶׁל נָבִיא שֶׁלֹּא אוּכַל לִפְגּוֹעַ בּוֹ?
אָנָּא, עֲזוֹב אֶת עִירִי הַלַּיְלָה, כִּי כָּל עוֹד הִנְּךָ בָּעִיר, לֹא אוּכַל לִהְיוֹת אָדוֹן בָּהּ.
תֵּן לִי אֶת מַחֲצִית אוֹצָרְךָ, וְאֵלֵךְ לִי.
עַל כֵּן, בּוֹא וְקָחֶנָּה, וְכָל אוֹצְרוֹת הָעוֹלָם יִהְיוּ שֶׁלְּךָ.
בּוֹא עִמִּי וְאַרְאֶה אוֹתָן לְךָ.
אַל תַּטְרִיד אֶת עַצְמְךָ, אֶלָּא בּוֹא אִתִּי אֶל עִיר אַחֶרֶת.
הַנַּח לִי עַתָּה לְסַפֵּר לְךָ עַל כְּאֵב הָעוֹלָם, אוּלַי תִּשְׁמַע לִי.

------------------
