וְעַרְפִלֵּי הַטֹּהַר יִמּוֹגוּ, יִמַּסּוּ בִּמְרוֹם הָרָקִיעַ,

יָעֵמּוּ אֶת-זְהַב תַּלְתַּלֵּיהֶן, וְרוּחַ חֲרִישִׁית תָּנִיעַ
בַּלָּט אֶת-הַקָּמָה הַנָּמָה, הַחֹלְמָה בֵּין אוֹרוֹת וּצְלָלִים.
פִּתְאֹם נֵעוֹר רוּחַ וַיַּךְ בַּשִּׁבֳּלִים,

נִפלֶה דַרְכְּךָ מֵאֱנוֹשׁ וּתְשַׁנֶּה לְךָ מֵהֶם טָעַם,

הִנֵּה תַּפּוּחַ.

וְשִׁפְעַת אוֹר וְשִׁפְעַת צְלָלִים.
בַּפַּרְדֵּס יֵשׁ מַחֲבֹאֵי שֶׁקֶט,
עֵץ פְּרִי וּנְטִיעִים מְגֻדָּלִים.
וַיְהִי הַיּוֹם בַּעֲלוֹת הַקַּיִץ,

וַאֲנִי עִם-פְּנִנָּה שַׁאֲרָתִי
לָשׂוּחַ אֶל-הַגַּן יָצָאתִי.
הִיא עַלְמָה רַכָּה וַאֲנִי אִישׁ צָעִיר;
וַנְּצַחֵק יַחְדָּו לְרוּחַ קָיִץ,

אַשְׁרֶיךָ, צָעִיר רֹדֵם, אֲשֶׁר שָׂפָם גַּאֲוָתֶךָ. 
הוֹד וְהָדָר תִּלְבַּשׁ, נַפְתָּ מוֹר בִּגְדוֹתֶיךָ; 
מִצְּנִיף רֹאשְׁךָ הַנָּבוּב עַד נַעֲלֵי הָרַגְלָיִם – 
כֻּלְּךָ אוֹמֵר בָּחוּר תַּאֲוָה לָעֵינָיִם! 
זֶה קָדְקֹד הַשֵּׂעָר וִיפִי זֶה הַבְּלוֹרִית,

וּנְתַנְיָה רָאָה מִמַּאֲרָבוֹ אֶת-בַּעַל הַכְּנָפַיִם וַיָּפָג לִבּוֹ.
כִּמְעַט בָּקְעָה צְעָקָה גְדוֹלָה מִפִּיו, צַעֲקַת תִּמָּהוֹן וָפַחַד.
הֲלֹא זֶה הָעַיִט הנוֹרָא, בֶּן הַשּׁחַץ, אֲשֶר מָשָׁהוּ מִן הַמַּיִם וַיְבִיאֵהוּ הֲלֹם.
הַכֵּר הִכִּירוֹ בְּלֹבֶן נוֹצָתוֹ וּבְכָל-מַרְאֵהוּ הָאָיֹם.
וּבְכָל-זֹאת הִתְאַפֵּק נְתַנְיָה וַיֵּשֶׁב בְּמַאֲרָבוֹ דוּמָם וְעֵינָיו כָּלוֹת לְאַחֲרִית דָּבָר. 

------------------
“לְכִי שׁוּבִי, בִּתִּי, לְבֵית אָבִיךְ, וְהָיִית לְבֶן-מֶלֶךְ כְּעֶרְכֵּךְ, כִּי לָמָּה תֹאבְדִי בִגְלָלִי בַּמִּדְבָּר וְאַתְּ עֲנֻגָּה וְרַכָּה וְלֹא יָדַעַתְּ מַחְסוֹר מֵעוֹדֵךְ. לְכִי שׁוּבִי, בִּתִּי, וַאֲנִי כַּאֲשֶׁראָבַדְתִּי אָבַדְתִּי.”
וַתֹּאמֶר נַעֲמָה:
“לֹא, כִּי אַחֲרֶיךָ אֵלֵךְ וְעִמְּךָ אֶתְעַנֶּה, כִּי אִם אֵין אַתָּה עִמִּי לָמָּה לִי חַיִּים.”
וַתִּשָׂא קוֹלָהּ וַתֵּבְךְ.

------------------

וַיַּחֲרִישׁוּ הַבַּחוּרִים וַיַּעַצְרוּ רוּחָם, כִּי-נָפַל פַּחַד אֱלֹהִים עֲלֵיהֶם וּלְבָבָם הָגָה אֵימָה סְתוּמָה.
וַיַּדְבִּיקוּ אֶת-אָזְנָם אֶל-הַדְּמָמָה, כְּהַדְבֵּק הַצָּמֵא בַּמִּדְבָּר אֶת-אָזְנוֹ אֶל-סֶלַע אִלֵּם, מְדַמֶּה לְהַקְשִׁיב הֶמְיַת מַעְיָן חָתוּם מִלֵּב הָאֶבֶן… 

------------------
אָז יִפְתַּח הַמֶּלֶךְ בְּדֶרֶךְ נַעֲלָמָה אֶת-כַּפֹּרֶת הַתֵּבָה – וְעֵינֵי כָל-הַקָּהָל הֻכּוּ בַסַּנְוֵרִים.
בְּתוֹךְ הַתֵּבָה עַל-פְּנֵי מַצַּע רַךְ, רֶפֶד מֶשִׁי יָרֹק, הִזְהִירָה בְאֶלֶף נְגֹהוֹת עַזִים וַתְּפַרְפֵּר בְּזַהֲרוּרֶיהָ כְּנֶפֶשׁ חַיָּה – בָּרֶקֶת גְדוֹלָה ונִפְלָאָה, עַיִן בְּעַיִן כַּבָּרֶקֶת אֲשֶר עַל הַשּׁלְחָן בָּרֶקֶת תְּאוֹמָה. 
וְכָל-הַקָּהָל נִדְהַם בָרִאשׁוֹנָה לַמַּרְאֶה וַיְהִי כְּאִלֵּם.

------------------
הֲשָׁמַעְתָּ בְּמִי נָתְנוּ עֵינָם?
מַרְבִּית עַם הָאָרֶץ
יֵט לִבָּם אַחֲרֵי הַגְּרַף נְגִיד לֻקְסֶמְבֻּרְגְּ. 
אַשְׁרֵינוּ
כִּי עָמַדְנוּ בֶאֱמוּנָתֵנוּ לַקֵּיסָר. עַתָּה
יֵשׁ לַצֶּדֶק תִּקְוָה.
הֵן יְבַקֵּשׁ לוֹ בֵּינֵינוּ
הַמּוֹשֵׁל הֶחָדָשׁ בַּעֲלֵי בְרִית אַמִּיצֵי רוּחַ
וּבְיַד אוֹסְטְרִיָּה אוֹיַבְתֵּנוּ לֹא יַסְגִּירֵנוּ.
רָאשֵׁי הָעָם וְנִכְבַּדָּיו הִנֵּה הֵם בָּזֶה לְפָנֶיךָ!
מָה הַדָּבָר, פֶּטֶרְמַן?
צִיר בָּא בִּמְגִלַּת סֵפֶר.
קוּם פְּתַח וּקְרָא.
"לִשְׁלוּמֵי אֱמוּנֵי עָם, שׁוֹכְנֵי אֶרֶץ

------------------
וְהַגַּן מָלֵא כֻלוֹ מִצְהֲלוֹת צִפֳּרֵי בֹקֶר וּתְרוּעַת כַּנְפֵי רְנָנִים, רַחַף פַּרְפְּרֵי רִקְמָה וַהֲמִית זְבוּבֵי תְכֵלֶת וָכָתֶם.
רִבֲבוֹת פִּיפִיוֹת יִתְּנוּ קוֹל מִבֵּין עֳפָאִים וּבְרַד צְלִילֵי עֹז עִם גַּרְגְּרֵי זְהַב שֶׁמֶשׁ יַחְדַּו יִתְּכוּ אָרְצָה.

------------------
וְתַת לַדְּבָרִים פִּתְרוֹנִים לֹא הֵמָּה.
הֲיָבוֹא קֵיסָר אֶל הַקַּפִּיטוֹלִיּוֹן מָחָר?
קַסְקָא: בּוֹא יָבוֹא. גַּם שָׁאַל מֵעִם אַנְטוֹנְיּוּס
לְהוֹדִיעֲךָ כִּי מָחָר יִהְיֶה שָׁמָּה.
צִיצֵרוֹן: וּבְכֵן הֱיֵה שָׁלוֹם. לֵיל סַגְרִיר אֲשֶׁר כָּזֶה –
לֹא עֵת שׁוֹטֵט וָשׂוּחַ הוּא.
קַסְקָא לֵךְ לְשָׁלוֹם, צִיצֵרוֹן.
קַסִּיּוּס: מִי בָזֶה?
קַסְקָא אִישׁ רוֹמֵי.
קַסִּיּוּס: הֲקוֹלְךָ הוּא, קַסְקָא?
קַסְקָא: אֹזֶן בּוֹחֶנֶת לָךְ. מָה-נּוֹרָא הַלַּיְלָה, קַסִּיּוּס.

------------------
חֲלוֹם בְּתוֹךְ חֲלוֹם
מַשְׁמִים כַּמָּוֶת בַּלַּיְלָה יָשַׁבְתִּי
לְבַדִּי בְּחַדְרִי לְאוֹר פְּתִילָה כֵהָה,
חָשַׁבְתִּי הָגִיתִי, הָגִיתִי חָשַׁבְתִּי,
עַד רָפָה בִּי מֹחִי וְנַפְשִׁי יָגֵעָה.
וָאָנוּם בְּהָקִיץ וּבְהָקִיץ חָלַמְתִּי
חֲלוֹמוֹת בְּטֵלִים, הִרְהוּרֵי מַחֲשָׁבוֹת,
וּלְנֶגְדִּי בֶּחָזוֹן שִׁוִּיתִי, דּוֹמַמְתִּי
רֶפֶשׁ, גַּל אַשְׁפָּה וּבְלוֹי סְחָבוֹת.
מִבְּלוֹי הַסְּחָבוֹת מְקַבְּלִים תָּפְרוּ,
תַּרְמִילִים עֲמֻקִּים, שׁוּלֵיהֶם נָשָׁקְתִּי,
וָאֶלְעַג לַחֲלוֹמוֹת שֶׁשָּׁוְא יְדַבֵּרוּ

------------------
עַתָּה יָדַע לִבּוֹ כִּי אָכֵן צָפוּי הוּא עוֹד אֶל יוֹם פְּלָאוֹת בָּאִי הַזֶּה, וְאוּלָם טֶרֶם יָבִין עוֹד אֶל נָכוֹן מַה-מִשְׁפַּט הַפְּלָאוֹת הָהֵן וּלְמָתַי תָּבֹאנָה.
אַחַת הִגִּיד לוֹ לִבּוֹ, כִּי הַבָּרֶקֶת הִיא הִיא אֲשֶׁר תַּנְחֵהוּ לִקְרָאתָן וְהִיא הִיא אֲשֶׁר תִּתְמֹךְ גּוֹרָלוֹ.
אֵין זֹאת כִּי אִם נֶאֶמְנוּ דִּבְרֵי אָבִיו עַל אֶבֶן הַסְּגֻלָּה הַזֹּאת וְעַל כֹּחָהּ הַנֶּעְלָם.

------------------
וְאוּלָם הַחֲמוֹר הַמְּגֻדָּף לֹא נָתַן בֶּעָפָר פִּיו גַּם הוּא, וּבַהֲרִימוֹ רֹאשׁוֹ מִלַּחֵךְ מָחָה תְחִלָה בִלְשׁוֹנוֹ אֶת-שְׂפָתָיו אַחַת הֵנָּה וְאַחַת הֵנָּה וְאַחַר נָהַק פִּתְאֹם נְהָקָה גְדוֹלָה וּמָרָה, נִהֲקַת חָמָס וְעֶלְבּוֹן כָּבֵד, וַיְהִי כְּמֵשִׁיב עַל גְּדוּפָה לֵאמֹר: 
“יִסָּכֵר פִּי דוֹבְרֵי שֶׁקֶר וָכָסֶל! לֹא אֲנִי, כִּי אִם אַתָּה הָאֱוִיל! פֶּתִי יַאֲמִין לְכָל-דָּבָר! 
------------------
וְרָחַף כָּל-לֵב מִמָּגוֹר סָתוּם וְלָבַשׁ חֲרָדוֹת –
הַאֵין נִשְׁמַת הַמֵּת הִיא הַבָּאָה בַלָּט בְּעַד הַחַלּוֹן
אֶל הַצּוּלָה הַקְּטַנָּה בְּמִקְוֵה הַזְּכוּכִית לִטְבִילָתָהּ,
וּבַעֲשׂוֹתָהּ רֶגַע קָטֹן, כְּדֵי טְבִילָה אַחַת, בַּמַּיִם,
וְעָלְתָה מִן הָרַחֲצָה וְהִסְתַּפְּגָה בָאֲלוּנְטִית הַנְּקִיָּה,
וּמְטֹהֶרֶת וְזַכָּה תָּשׁוּב בַּלָּאט לִמְנוּחָתָהּ כְּשֶׁבָּאָה.
אִי נְשָׁמָה טְהוֹרָה! אִם אֱמֶת נָכוֹן הַדָּבָר
כִּי אַתְּ הִיא הַבָּאָה הֵנָּה לִפְקֹד אֶת-נָוֵךְ,

------------------
דַּם יוֹנֵק וָשָׂב עַל-כֻּתָּנְתְּךָ –
וְלֹא יִמַּח לָנֶצַח, לָנֶצַח.
וְאִם יֶשׁ-צֶדֶק – יוֹפַע מִיָּד!
אַךְ אִם-אַחֲרֵי הִשָּׁמְדִי מִתַּחַת רָקִיעַ
הַצֶּדֶק יוֹפִיעַ –
יְמֻגַּר-נָא כִסְאוֹ לָעַד!
וּבְרֶשַׁע עוֹלָמִים שָׁמַיִם יִמָּקּוּ;
אַף-אַתֶּם לְכוּ, זֵדִים, בַּחֲמַסְכֶם זֶה
וּבְדִמְכֶם חֲיוּ וְהִנָּקוּ.
וְאָרוּר הָאוֹמֵר: נְקֹם!
נְקָמָה כָזֹאת, נִקְמַת דַּם יֶלֶד קָטָן
עוֹד לֹא-בָרָא הַשָּׂטָן –
וְיִקֹּב הַדָּם אֶת-הַתְּהוֹם!
יִקֹּב הַדָּם עַד תְּהֹמוֹת מַחֲשַׁכִּים,
וְאָכַל בַּחֹשֶׁךְ וְחָתַר שָׁם

------------------
עַל-חַלְמִישׁ כַּנּוֹ אֶ חָ ד נִשְׁאַר, עֵד הֶהָרֶס, 
הַמִּגְדָּל בְּרֹאשׁ הַסֶּלַע; 
עֲרִירִי נִצָּב שָׁם, וּכְבוֹא הַחַרְסָה 
יַד-סֵתֶר תַּדְלִיק בְּרֹאשׁוֹ נֵר הַפֶּלֶא, 
וּכְמוֹ שֶׁהֵאִיר – יָאִיר. 
לְמִי הוּא מֵאִיר שָׁם וְלָמָּה? – 
וּסְבִיבָיו מֵת הַיָּם מְלֹא מֵאָה פַרְסָה, 
וּמֵאָה פַרְסָה סְבִיבָיו שְׁמָמָה, שְׁמָמָה 
הַסִּירוֹת לֹא תָבֹאנָה עוֹד בִּגְבוּלוֹ: 
הָאַחַת נִתְקְעָה בַּיָּם הַקָּרוּשׁ, 
בְּחֶשְׁכַת לַיְלָה אַחַת אָבַד דַּרְכָּהּ, 
וְאַחַת מָצְאָה קֶבֶר לָהּ עַל קַרְקַע 

------------------
וּמוֹדַע הָאַלְמָנָה טָמַן אֶת-כַּדֵּי הַדְּבַשׁ בְּאוֹצָרוֹ בְּאַחַת הַפִּנּוֹת וַיִּשְׁכָּחֵן. וַיְהִי הַיּוֹם וַיַּעַשׂ הָאִישׁ מִשְׁתֶּה בְּבֵיתוֹ, כִּי הֵבִיא אִשָּׁה לִבְנוֹ, וַיָּשֶׂם דְּבַשׁ לִפְנֵי הַקְּרוּאִים, וְלֹא מָצָא לְכֻלָּם. וַיִּזְכֹּר אֶת הַפִּקָּדוֹן, וַיָבֹא לְאוֹצָרוֹ לִשְׁאָל-לוֹ מְּעַט דְּבַשׁ מִן הַכַּדִּים, וַיְהִי בַּהֲרִיקוֹ אֶת-הַכַּד הָאַחַת –וְהִנֵּה זָהָב וָכֶסֶף נוֹצְצִים מִקִּרְבָּהּ. 
------------------
אוֹרִים חַמִּים וּגְדוֹלִים עַל-פָּנַי מַבְהִיקִים.
מַה-נָּמוֹג הַלֵּב! עֵינַי אֶפְקַח, אֶסְגֹּרָה –
אֱלֹהִים, שְׁטָפַתְנִי הָאוֹרָה!
הוֹ, אֵלַי בֹּאוּ, זַכִּים! צַפְרִירֵי הַתֹּם!
אֶל-מִתַּחַת לִסְדִינִי הַצָּחֹר, הַצָּח!
שָׁם נִתְפַּלַּשׁ, נִתַּפַּל עַד-יִכּוֹן הַיּוֹם,
וּפִזַּזְתֶּם עַל-עוֹרִי וּבְשָׂרִי הָרָךְ.
הוֹ, אֵלַי! עַל-תַּלְתַּלַּי, עַפְעַפַּי, שִׂפְתוֹתָי,
אֱלֵי גוּמוֹת לְחָיַי, מַעֲמַקֵּי בָבוֹתָי;
הֲזִכּוּנִי, שִׁטְפוּנִי, אֶל-לִבִּי חֲדֹרוּ,

אָבִיב אָבִיב וּשְׁלוֹמוֹ, קַיִץ קַיִץ וּבִרְכָתוֹ – 
וְלוּא אֶלֶף פְּעָמִים מִשֶּׁהוּ חֶלְקִי בַּחַיִּים יִצְעַר,

עוֹד עֵינָיו מַיְשִׁירוֹת שָׁמָּה – וְהִנֵּה דְמוּת אִשּׁה עוֹלָה מן הַמְּצוּלָה, עַל רֹאשָׁהּ יָצִיץ נֵזֶר, וּבָרֶקֶת גְּדוֹלָה דוֹלֶקֶת עַל מִצְחָהּ, וּמַרְאֵה הַדְּמוּת מַרְאֵה אִמּוֹ, וְעֵין הַבָּרֶקֶת כְּעֵין אֶבֶן הַחֵן אֲשֶׁר אִתּוֹ בַחֲגוֹרָתוֹ, גַּל רַחֲמִים וַחֲנִינָה הֵצִיף אֶת-לֵב הָעֶלֶם וַיֹּאמֶר לִצְעֹק: “אִמִּי, רְאִי בְנֵךְ!” 
------------------
וְתֻכִּיֵּי שֵׁן בְּרָאשֵׁיהֶם, פּוֹרְשִׂים כְּנָפַיִם לְעֻמַּת נִשְׁרֵי זָהָב. וְקַלְעֵי שֵׁשׁ לָבָן, כְּמִפְרְשֵׂי עָבִים דַּקִּים, פְּרוּשִׂים עַל הָעֵצִים, וּמְנַעַנְעֵי בְּדֹלַח וּזְכוּכִית יְקָרָה תְלוּיִים בָּהֶם, וּבְהִנּוֹעַ הַקְּלָעִים יָזוּעוּ הַמְּנַעַנְעִים וְצָלְלוּ חֶרֶשׁ, וּנְגֹהוֹת הַקֶּשֶׁת לָהֶם, וּצְלִילָם דַּק וְעָרֵב, כְּהֶמְיַת כִּנּוֹר דָּוִד בִּנְשֹׁב-בּוֹ רוּחַ שָחַר.

וַיַּרְא הַטַּבָּח כִּי יֵשׁ עַתָּה מָקוֹם לְהִתְגַּדֵּר, וַתִּצְלַח עָלָיו הָרוּחַ, וּבְשַׁסְעוֹ אֶת-הָאֲסַפְסוּף בְּאֶפֶס יָד, כְּשַׁסְּעוֹ בַקַּרְדֹּם אֶת-הַגְּדִי הַטָּבוּחַ, גָּעַר כִּמְהַתֵּל:

“בְּסֵפֶר ‘וַיִּקְרָא’ כְּמִשְׁפָּט כָּל-הַיְּלָדִים.”

------------------
כֻּלִי בוֹעֵר בְּאֵשׁ אַהֲבָה וּמִתְהַפֵּךְ בְּצִירֵי מַאֲוַיִּים נַעֲלָמִים, וְנַפְשִׁי תָחִיל, תְּפַרְפֵּר, כִּטְלֵה חָלָב בֵּין שִׁנֵּי כְּפִיר רָעֵב, וְעֵינִי בוֹכִיָּה וְשִׁנַּי נוֹשְׁכוֹת וּשְׂפָתַי לוֹחֲשׁוֹת בְּרַעַד חֵשֶׁק תְּפִלַּת חַטָּאת לְאֵל חַיַּי אָנִי… וּבָאָה רִנַּת הַזָּקֵן בְּקִרְבִּי כְּמַעְיָן טָהוֹר מְפַכֶּה אֶל-יָם נִגְרָשׁ… וָאֶקַל מְאֹד בְּעֵינַי וָאֶתְעַנֶּה מְאֹד וָאֲבַטֵּא נְדָרִים בַּעֲלֶטֶת נַפְשִׁי… וְאֶת-הַזָּקֵן יָרֵאתִי, פֶּן-יִמְצָא אֶת-נַפְשִׁי חֲלָלָה וּשְׂרָפָהּ

“שְׁמִי מַחְשְׁמָה!”

מִי לֹא שָׁמַע אֶת-שֵׁמַע מִשְׁתֶּה שְׁלֹמֹה? וְאוּלָם כַּשֶּׁפַע הַזֶּה לֹא רָאֲתָה עוֹד עַיִן גַּם בְּיָמָיו.

וְהָאִשּׁה רָאֲתָה כִּי כָבְדָה שְׁנָתוֹ, וַתֶּאֱזֹר חַיִל, וַתִּשְׁלֹף אֶת-הַחֶרֶב וַתַּךְ בְּכֹח עַל צַוָּארוֹ, – וַיִּיקַץ הָאִישׁ וַיִּנָּעֵר, וַיַּרְא אֶת-אִשְׁתּוֹ נִצֶּבֶת עָלָיו וְהַחֶרֶב הַשּׁלוּפָה בְיָדָהּ, וַיִּתְחַלְחַל מְאֹד, וַיִּתְפֹּשׂ בְּיָדָהּ, וַיִּתֵּן עָלֵיהָ בְקוֹלוֹ:

------------------
עוֹדֶנּוּ מַבִּיט כֹּה וָכֹה – וְהִנֵּה הַדּוּכִיפַת בָּאָה מִצַּד תֵּימָן. וַיַּעַט אֵלֶיהָ וַיֹּאמֶר לְתָפְשָׂהּ, וַתִּתְחַנַּן אֵלָיו הַדּוּכִיפַת וַתֹּאמַר: הִשְׁבַּעְתִּיךָ בֵּאלֹהִים אֲשֶר אִמֵּץ כֹּחֲךָ וַיְחַזֶּקְךָ מִמֶּנִּי, אִם תִּפְגַּע בִּי וְאִם תַּעֲשֶׂה לִי רָעָה." וַיִּרֶף הַשָּׁלָךְ מִמֶּנָּה וַיֹּאמַר: 
“אוֹי לָךְ, שִׁכְּלָתֶךְ אִמֵּךְ! קָצַף הַמֶּלֶךְ עָלַיִךְ וְגַם נִּשְׁבַּע לְיַסְּרֵךְ אוֹ לְשַׁחֲטֵךְ.”

הָאוֹפֶה וְהַקַּטָּף

וַיֶּחֱרַד אֲבִישַׁי וַיֹאמַר: אֵין זֹאת כִּי-אִם רָעָה גְדוֹלָה נֶגֶד פְּנֵי הַמֶּלֶךְ.
וַיִּלְבַּש חֲלִיצָתוֹ וַיְמַהֵר אֶל בֵּית הַמֶּלֶךְ, וְהִנֵּה לֹא שָׁב עוֹד הַמֶּלֶךְ מִצֵּידוֹ, וּכְבָר רַד הַיּוֹם.

וְצַיָּד אֶחָד מֵאַנְשֵׁי הַמָּקוֹם הִתְבּוֹנֵן אֶל דַּרְכֵי הַדּוּכִיפַתִּים, וּבְטָמְנוֹ לָהֶם פַּחִים, וַיִּתֵּן שֶׁבֶר רְאִי בְּתוֹכָם, וַתֵּרֵד דּוּכִיפַת לִרְאוֹת צַלְמָהּ בָּרְאִי וּלְהִתְהַדֵּר, כְּמִשְׁפָּטָהּ, וּבַעֲשׂוֹתָהּ כֹּה וָכֹה וַתֵּאָחֵז בַּפַּח וַתִּלָּכֵּד. וַיְהִי כִּרְאוֹת הַצַּיָּד אֶת הָעֲטָרָה הַנּוֹצֶצֶת בְּרֹאשׁ הַצִּפּוֹר, וַיְמַהֵר וַיִּמְלֹק רֹאשָׁה וַיִּשָׂא מֵעָלָיו אֶת- הָעֲטָרָה, וַיְבִיאֶנָּה לִפְנֵי צוֹרֵף נְחֹשֶׁת. לָדַעַת מִפִּיו מָה הִיא. 
------------------
וְאַתָּה גַם-אַתָּה, בֶּן-אָדָם, סְגֹר בַּעַדְךָ הַשַּׁעַר,
וְנִסְגַּרְתָּ פֹּה בָּאֲפֵלָה וּבַקַּרְקַע תִּכְבֹּשׁ עֵינֶיךָ
וְנִצַּבְתָּ כֹּה עַד-בּוֹשׁ וְהִתְיַחַדְתָּ עִם-הַצַּעַר
וּמִלֵּאתָ בּוֹ אֶת-לְבָבְךָ לְכֹל יְמֵי חַיֶּיךָ,
וּבְיוֹם תְּרֻשַּׁשׁ נַפְשְּךָ וּבַאֲבֹד כָּל חֵילָהּ –
וְהָיָה הוּא לְךָ לִפְלֵיטָה וּלְמַעְיַן תַּרְעֵלָה,

וּלְפָתְךָ וְהֵעִיק עָלֶיךָ כְּהָעֵק חֲלוֹם זְוָעָה;
וּבְחֵיקְךָ תִשָּׂאֶנּוּ אֶל-אַרְבַּע רוּחוֹת הַשָּׁמַיִם,

“עֻזִּיאֵל! רְצוֹן הַמֶּלֶךְ כְּטַל עֲלֵי עֵשֶׂב, וַחֲמָתוֹ – מַלְאֲכֵי מָוֶת.
אִם הַמֶּלֶךְ צִוָּה, מִי זֶה יָזִיד לַמְרוֹת פִּיו.
לֵךְ מֵעָלַי וַעֲשֵׂה כַּאֲשֶׁר צִוִּיתִיךְ, כִּי בְנַפְשְׁךָ הִיא.”
וַיִּתֵּן הַמֶּלֶךְ חֶרֶב בְּיָדוֹ וַיְשַׁלְּחֵהוּ.
וַיָּבֹא עֻזִּיאֵל אֶל בֵּיתוֹ וַיֵּשֶׁב אֶל הַשּׁלְחָן וְהוּא עָצֵב וְנֶאֱנָח וְהַחֶרֶב מִתַּחַת לְמַדָּיו.
וַתִּשְׁאָלֵהוּ אִשְׁתּוֹ:
“מֶה עָכַר אֶת רוֹחֲךָ-אִישִׁי? וּמַדּוּעַ פָּנֶיךָ זוֹעֲפִים?”

------------------
“הִנֵּה שָמַעְתִּי עָלֵיךָ כִּי אִישׁ חַיִל וּנְבוֹן דָבָר אַתָּה, וַתִּיטַב בְּעֵינַי לְשִׂימְךָ עֶלְיוֹן עַל כָּל-שָׂרֵי הַחַיִל אֲשֶׁר לִי, אַךְ כִּי תִשְׁמַע בְּקוֹלִי לְכָל-אֲשֶׁר אֲצַוֶּךָ.”
וַיַּעַן עֻזִּיאֵל:
“עַבְדְּךָ אָנִי, וְכָל-אֲשֶׁר תֹּאמַר אֵלַי אֶעֱשֶׂה, וְהַיִיתִי-נָא כְּאֶחָד מְעַבְדֵי אֲדוֹנִי הַמֶּלֶךְ הַקְּטַנִּים.”
וַיֹּאמֶר לוֹ הַמֶּלֶךְ:

------------------
אַחַת יִנָּבֵא לִי לִבִּי, וְנַפְשִׁי יוֹדַעַת מְאֹד,
כִּי-פָקֹד יִפְקְדֵנִי הֶחָזוֹן וִיקַר מַאֲוַיַּי יַשִּׂיגֵנִי.
וּבְבֹא הָרֶגַע הַנֶּאְדָּר, וְהָיָה כְּנִיד עַפְעַף הַנֵּצַח
בַּהֲצִיצוֹ עָלַי פִּתְאֹם בְּעַד צֹהַר נִקְרַע בַּמָּרוֹם –
וְנִפְקְחוּ עֵינַי וָאֹרוּ, וְזַכּוּ כְּעֵינֵי הַיָּלֶד,
וּמַרְאוֹת יַלְדוּתִי כֻּלָּם עַל צְבָא חֲלִיפוֹתָם יַחַד
עִם כָּל-יְקוּמָם סָבִיב יִנְהֲרוּ דוּמָם דֶּרֶךְ נַפְשִׁי,
יָסֹלּוּ עָלַי אָרְחוֹת זָהֳרָם כַּחֲלֹמוֹת בֹּקֶר בְּהִירִים,

------------------
וּבַצַּר לִלְבָבוֹ מְאֹד, וּמְצָאוּהוּ חֲלֹמוֹתָיו הַגְּדוֹלִים וּמַדְוָיו הַנֶּאֱמָנִים, וְהִשִּׂיגוּהוּ כְּגַלֵּי הַיָּם – וְיָצָא הָעֶלֶם בַּשַּׁחַר אֶל-מִחוּץ לָעִיר וְנִשְׁעַן שָׁם תַּחַת עַרְעָר עַל-שְׂפַת נַחַל נִרְדָּם, וְנָשָׂא עֵינָיו אֶל-אַיֶּלֶת הַשַּׁחַר, וּבִקֵּשׁ צַלְמָהּ בְּמֵימֵי הַנַּחַל, וְעָצַם אֶת-עֵינָיו, וְהִבִּיט גַּם-אֶל תְּהוֹם נַפְשׁוֹ, וְעָמַד עַד-בּוֹשׁ, וְהֶחֱרִישׁ עִם-כָּל-הָעוֹלָם בִּיגוֹנוֹ הַגָּדוֹל, יְגוֹן הַיָּחִיד.


------------------
מִבְּלִי שֶׁנֶּאֱהָבָה, מִבְּלִי שֶׁאָהֵבָה.
וּשְׁתֵּי הָעֵינַיִם שֶׁלִּרְאוֹת נִבְרָאוּ,
שֶׁלִּרְאוֹת יָכְלוּ הָאָרֶץ וּמְלֹאָהּ,
תִּדְעַכְנָה, תֶּחְשַׁכְנָה וּמְאוּמָה לֹא רָאוּ,
וּבְלִי חֲמֹד מְאוּמָה תִּכְלֶינָה בַשֹּׁאָה;
וּבְחַיִּים כְּבָר בָּטְלוּ כַּצֶּלֶם תָּנוּעַ,
וּתְתַפֵּשׂ כִּשְׂמָמִית בִּמְסִלּוֹת חָדֵלוּ –
וְלָמָּה זֶה נוֹצְרוּ לַשָּׁוְא, וּמַדּוּעַ
תֹּאבַדְנָה עַל חִנָּם הַנְּשָׁמוֹת הָאֵלֶּוּ?
וְיֵשׁ אֲשֶׁר-שָׁמְמָה גַּם רוּחַ הַנַּעַר
בַּעֲמָלָהּ הַקְּלֹקֵל וּכְמוֹ הִתְמַרְמָרָה;

------------------
– / וְהַפַּעֲמוֹנִים עוֹדָם הוֹמִים הָמֹה וְהָלֹם, / וּמֵרָאשֵׁי הַמִּגְדָּלִים מְנַפְנְפִים הַדְּגָלִים, / וּגְבִירוֹת מֵרָאשֵׁי טִירוֹת מַמְטִירוֹת בְּצָלִים, / וּבְרֹאשָׁן כְּלִילֵי-פְאֵר: מִקְלְעוֹת בְּצַלְצַלִים, – / וְהַתֻּפִּים מַרְעִימִים וְהַצֶּלְצְלִים מְצַלְצְלִים.
יב
