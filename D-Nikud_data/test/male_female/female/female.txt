הַחֲתִימָה עוֹמֶדֶת בְּנִיגּוּד אַכְזָרִי לַנִּימָה שֶׁל הַמִּכְתָּב עַצְמוֹ, וְגַם כָּךְ, הִיא לֹא רוֹאָה בָּהּ שֶׁמֶץ שֶׁל אִירוֹנְיָה.
כְּשֶׁהֵם יַחַד בַּקָּרָוָואן הַמַּחְסוֹר בַּמֶּרְחָב מִחְיָיה בִּלְתִּי נִסְבָּל, וְהִיא רַק רוֹצָה שֶׁהוּא יִתְפּוֹגֵג.
הִיא רוֹצָה לִבְרוֹחַ מִשָּׁם, אֲבָל מַשֶּׁהוּ בַּצּוּרוֹת הַמְּעוּיָּנוֹת שֶׁמְּקַשְּׁטוֹת אֶת הַקַּשְׂקַשִּׂים שֶׁלּוֹ מְהַפְּנֵט וּמוֹנֵעַ מִמֶּנָּה לְהַתִּיק אֶת מַבָּטָהּ.

------------------
כְּדֵי לְהָסִיר מֵעָלֵינוּ אֶת הַחֲטִיפָה, הִיא אוֹמֶרֶת וְחוֹשֶׁבֶת עַל הַמִּישּׁוּשִׁים שֶׁנֶּאֶלְצָה לִסְבּוֹל, מִישּׁוּשִׁים קַלִּים, לַאֲמִיתּוֹ שֶׁל דָּבָר, וְעַל כָּךְ הִיא מוֹדָה.
מִישֶׁהִי חֲצוּפָה, מִישֶׁהִי שֶׁעֲדַיִין מַאֲמִינָה בַּסִּיפּוּרִים מֵהַסּוּג הַזֶּה, שֶׁאוֹסֶפֶת אוֹתָם וּמְנַסָּה לְסַפֵּר אוֹתָם כְּמוֹ שֶׁצָּרִיךְ.

------------------
הַגַּן הַכָּחוֹל מְאוֹתֵת לָהּ "בּוֹאִי, תַּעַבְרִי דַּרְכִּי, תְּצַלְצְלִי בָּאִינְטֶרְקוֹם, רָאִיסָה תְּקַבֵּל אוֹתְךָ יָפֶה".
אַל תַּחֲזִירִי אִתְּךָ שׁוּם דָּבָר מִלְּבַד הַכּוֹבַע הָאָדָם!
תַּחֲזִירִי אוֹתִי אִתְּךָ!
אֲנִי שָׁמַעְתִּי בְּמוֹ-אָזְנִי אֵיךְ אַתְּ מֵאֵימַת עַל אָחִיךָ הַקָּטָן, וַאֲנִי מְאוֹד מֻפְתַּעַת מִמְּךָ.
אֶת הַחְזָרַת אוֹתִי אִתְּךָ!
אַתְּ מְצִיקָה לְאַחֶיךָ, גְּבֶרֶת צְעִירָה?
אַתְּ רוֹאֶה כַּמָּה אָחִיךָ הַקָּטָן סומך עָלֶיךָ, גְּבֶרֶת צְעִירָה?

------------------
פֶדְּרִיגוֹ עָנָה לָהּ בְּהַכְנָעָה: "גְּבִירְתִּי, אֵינִי זוֹכֵר שֶׁרָעָה כָּלְשֶׁהִי בָּאָה עָלַי בְּעֶטְיֵךְ מִיְּמֵי, אֶלָּא רוֹב טוֹב בִּלְבַד, עַד כִּי אִם הָיָה בִּי עֵרֶךְ כָּלְשֶׁהוּ, בָּא לִי בִּזְכוּת עֶרְכֵּךְ וּבִזְכוּת הָאַהֲבָה שֶׁרָחַשְׁתִּי לְךָ.
"גְּבִירְתִּי, לְאַחַר שֶׁרָצָה הָאֵל כִּי בְּךָ יִדְבַּק לִבִּי, חָשַׁבְתִּי שֶׁיַּד הַגּוֹרָל פָּגְעָה בִּי רַבּוֹת וְעַל כך הִתְלוֹנַנְתִּי."

------------------
הָיִינוּ נִלְהָבִים כָּל כָּךְ עַד שֶׁלֹּא הִבְחַנּוּ בִּמְנַקָּה שֶׁנִּיצְּבָה מוּלֵנוּ, קְשִׁישָׁה נְמוּכַת קוֹמָה וּמְגוּבֶּנֶת, בַּעֲלַת עֵינַיִים כְּחוּלּוֹת דְּהוּיוֹת שֶׁתָּמִיד דָּמְעוּ.
הִיא פְּרוֹפֵסוֹר, אַתְּ יוֹדַעַת, מַרְצָה לְאֶסְתֵּטִיקָה שֶׁל הָעֵת הָעַתִּיקָה — הִיא עֲלוּלָה לְהִיבֶּהָלָה.
מוֹרָה טוּרְקִיָּה שֶׁהָפְכָה לִמְלוֹנָאִית מָסְרָה לִי דּוּחַ מוּדְפָּס שֶׁחִיבְּרָה וּבוֹ חָשְׂפָה אֶת שֶׁקֶר הַשּׁוֹאָה הָאַרְמֵנִית, כְּדֵי שֶׁאֶמְסוֹר אוֹתוֹ לִידֵי הַמִּמְשָׁל הָאָמֵרִיקָאִי.

------------------
פָּגַשְׁתִּי פַּעַם עוֹלָה מֵאַרְגֶּנְטִינָה שֶׁהָיְתָה חֲבֵרָה בַּהִסְתַּדְּרוּת.
זֶה לֹא מָה שֶׁהָרוֹעָה חָשְׁבָה.
קִיבַּלְנוּ לֶחֶם טָרִי מֵהָאוֹפָה הַחֲמוּדָה.
כּוֹרָה מִטַּנְזַנְיָה נֶהֶפְכָה בְּרֶגַע אֶחָד לַמִּילְיוֹנֵרִית לְאַחַר שֶׁמָּכְרָה שְׁתֵּי אַבְנֵי טֶנְזֵנִיט.
הַמָּלְוָוה הִצִּיעָה עֲשֶׂרֶת אֲלָפִים שְׁקָלִים.
הַמְּמוּנָּה רָתְחָה מִזַּעַם.
יֵשׁ לִי מַנְחָה, קוֹרְאִים לָהּ שִׁיר.
לִפְנֵי שֶׁעָבְדָה כִּמְנַקָּה הִיא הָיְתָה דּוֹקְטוֹר לְפִיזִיקָה.

------------------
אַתְּ יְכוֹלָה לִבְנוֹת עַל כָּל הַשֶּׁטַח, פְּרָט לְשִׁבְעַת הָאַקֶרִים בְּרֹאשׁ הַגִּבְעָה וְלַחֲמִישָּׁה-עָשָׂר אַקֶרִים נוֹסָפִים מִסְּבִיבָם, וְעָלַיִךְ לִבְנוֹת לִי חוֹמָה גְּבוֹהָה סְבִיב שִׁבְעַת הָאַקֶרִים.
תְּסַפְּרִי לָנוּ אֵיךְ לִמְצוֹא אוֹתָן, וּלְאַחַר הַחֲתוּנּוֹת שֶׁלָּנוּ נָשִׁיב אֶת רַגְלַיִךְ לָאוֹר הַיּוֹם.
אָמְרוּ הַנְּסִיכִים בְּנִימָה מְאַיֶּמֶת; "אִם לֹא תְּסַפְּרִי לָנוּ, לְעוֹלָם לֹא תּוּכְלִי לַעֲמוֹד עַל רַגְלַיִךְ."

------------------
קוֹנְצָ'ה, קוֹדֶם הִבְטַחְתִּי לְךָ שֶׁאֲנִי אֶעֱזוֹב אֶת זֶה כִּי לֹא הִתְחַשֵּׁק לִי לְהִתְוַוכֵּחַ, אָמַרְתִּי לָהּ.
הוּא הִפְנָה מַבָּט כָּעוּס אֶל מוֹרַשְׁקִינָה וְחָשׁ כֵּיצַד חוֹבֵט הַטֵּנוֹר הַגַּבְרִי שֶׁלָּהּ עַל עוֹר הַתּוֹף שֶׁלּוֹ, לֹא הֵבִין דָּבָר וְחָשַׁב: "הַשֵּׁד הֵבִיא אוֹתְךָ אֵלַי.
אָנָה סְרַגְיַיבְנָה, אִם אַתְּ עוֹשֶׂה אֶת זֶה, אַתְּ הוֹרֶגֶת אֶת אַבָּא שֶׁלָּךְ.
– שָׁאַל יוֹנָה אֶת סוּסָתוֹ, כְּשֶׁהוּא מַבְחִין בִּבְרַק עֵינֶיהָ – לַעֲסִי לְךָ בְּתֵיאָבוֹן.

------------------
כָּל פַּעַם שֶׁהָיִיתִי בַּדִּירָה הָיִיתִי מַבִּיטָה לְעֵבֶר הַוִּוילוֹן וּמְדַמֶּה שֶׁיֵּשׁ לִי עֵינֵי רֶנְטְגֶּן שֶׁחוֹשְׂפוֹת אֶת הַמִּסְתּוֹרִין.
הִיא הָיְתָה בַּעֲלַת גַּאֲוָוה, חָשָׁה הֵיטֵב אֶת מַצָּבָהּ, וְהִתְבּוֹנְנָה עַל סְבִיבָה בְּקוֹצֶר-רוּחַ, מְצַפֶּה לְגוֹאֵל.
הַשָּׁלוֹשׁ נִיצֵּחַ, אָמְרָה מָרִיָּה, מַרְאָה אֶת הַקְּלָף שֶׁלָּהּ.
הִיא עָמְדָה סָמוּךְ לַכְּנִיסָה, מְכַסָּה אֶת פָּנֶיהָ בְּצַוְּוארוֹן מִפַּרְוַות-בּוֹנֶה: עֵינֶיהָ הַשְּׁחוֹרוֹת הִבְרִיקוּ מִתַּחַת לַכּוֹבַע.

------------------
