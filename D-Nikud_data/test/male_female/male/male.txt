כָּל יוֹם אַתָּה נוֹסֵעַ בַּכְּבִישׁ הַזֶּה, וְאֵינְךָ מַכִּיר אִישׁ, זֶה אֵזוֹר שִׁיכּוּנִים אָפוֹר וּמוּזְנָח, אִישׁ אֵינוֹ מִסְתַּכֵּל, כָּל הָאוֹרוֹת כְּבוּיִים.
אַתָּה מִסְתּוֹבֵב סְבִיבְךָ וּבוֹדֵק, אוּלַי מִישֶׁהוּ בָּא, אוּלַי מִישֶׁהוּ מִסְתַּתֵּר מֵאֲחוֹרֵי פַּחֵי הָאַשְׁפָּה וּמִסְתַּכֵּל, אֲבָל אֵין אַף אֶחָד בְּשׁוּם מָקוֹם.
אֵין לָךְ בִּכְלָל סוֹלְלָה בַּטֵּלֵפוֹן, וְאַתָּה תּוֹפֵס שֶׁאַף אֶחָד לֹא הָיָה עוֹנָה, גַּם אִילּוּ הָיְתָה לְךָ.

------------------
כְּדֵי לְהָסִיר מֵעָלֵינוּ אֶת הַחֲטִיפָה, הוּא אוֹמֵר וְחוֹשֵׁב עַל הַמִּישּׁוּשִׁים שֶׁנֶּאֱלַץ לִסְבּוֹל, מִישּׁוּשִׁים קַלִּים, לַאֲמִיתּוֹ שֶׁל דָּבָר, וְעַל כָּךְ הוּא מוֹדֶה.
מִישֶׁהוּ חָצוּף, מִישֶׁהוּ שֶׁעֲדַיִין מַאֲמִין בַּסִּיפּוּרִים מֵהַסּוּג הַזֶּה, שֶׁאוֹסֵף אוֹתָם וּמְנַסֶּה לְסַפֵּר אוֹתָם כְּמוֹ שֶׁצָּרִיךְ.

------------------
בְּשֵׁם הַיְּדִידוּת אֱמוֹר לִי, מָה גָּרַם לְךָ לְעִיצָּבוֹן הַלֵּב הַזֶּה וּלְחוּלְשַׁת הַגּוּף?
אַחֵר כך הִיא לָקְחָה אֶת יָדִי בְּיָדָהּ וְהוֹלִיכָה אוֹתִי אֶל הַחַלּוֹן וְאָמְרָה: "בּוֹא וְאַרְאֶה לְךָ אֶת סוֹדוֹתֵיהֶם הַחֲשׂוּפִים שֶׁל אוֹתָם אֲנָשִׁים אֲשֶׁר סֵירַבְתִּי לָלֶכֶת בְּדַרְכָּם.
בּוֹא וְאֶקַּח אוֹתְךָ עַתָּה אֶל הַבַּיִת הַנֶּהְדָּר שֶׁסְּבִיבוֹ גִּינּוֹת יְפֵהפִיּוֹת.
כַּאֲשֶׁר הַמָּוֶות מַכֶּה קָרוֹב אֵלֶיךָ אַתָּה סוֹבֵל.

------------------
לִהְיוֹת עוֹלֶה מֵאֶרֶץ רְחוֹקָה פֵּירוּשׁוֹ לִהְיוֹת בּוֹדֵד.
הָיִיתִי רוֹעֶה, הָיִיתִי זַמָּר, הָיִיתִי כָּל מִינֵי דְּבָרִים.
שָׁאַלְנוּ אֶת הַכּוֹרֶה מֵאֵיפֹה הוּא בָּא.
לֹא חָשַׁבְתִּי שֶׁאֶזְדַּקֵּק לְטוֹבוֹת מִמַּלְוֶוה בְּרִיבִּית מֵחוּרְבָּן.
אוֹרֶן כַּעַס עַל הַמְּמוּנֶּה הַחַשְׁדָן.
אֲנִי לֹא חוֹשֵׁב שֶׁאֶהְיֶה מִנְחָה.
לֹא כְּדַאי שֶׁתִּהְיֶה מְנַקֶּה.
הַמַּרְצֶה עָשִׁיר.
הַיֶּלֶד הַזֶּה, עוֹלֶה מֵאַרְהָ"ב, יוֹדֵעַ מָה הוּא עוֹשֶׂה.
אָסַפְתִּי אוֹתָם כְּמוֹ הָרוֹעֶה הֶחָבִיב שֶׁאֲנִי.
------------------
כָּל כָּךְ אֲנִי רוֹצֶה לְהַגִּיד לוֹ אֶת זֶה אֲבָל בְּסוֹף רַק מְהַנְהֵן, כְּדֵי לְצַמְצֵם כַּמָּה שֶׁיּוֹתֵר אֶת הַזְּמַן שֶׁל הַדִּיבּוּרִים וּלְהַרְוִויחַ עוֹד כַּמָּה דַּקּוֹת שֶׁל שֶׁקֶט.
עַכְשָׁיו בְּיַחַד הוּא רוֹצֶה שֶׁנִּבְדּוֹק שֶׁהַדֶּלֶת נְעוּלָה, כְּמוֹ שֶׁלִּימַּדְתִּי אֶת עַצְמִי וְאוֹתוֹ.
הָאִישָּׁה שֶׁלּוֹ אוֹמֶרֶת שֶׁצְּפַרְדֵּעַ, וְהַבַּעַל שֶׁלָּהּ עוֹנֶה שֶׁלֹּא יָכוֹל לִהְיוֹת וְשֶׁמָּה הִיא מְבִינָה מֵהַחַיִּים שֶׁלָּהּ.

------------------
אַחֲרֵי שָׁנָה הִיא אָמְרָה לְמֶלֶךְ – "יַקִּירִי, אַתָּה חַיָּב לַעֲשׂוֹת מַשֶּׁהוּ כְּדֵי לְהַסִּיחַ אֶת דַּעְתְּךָ."
הִיא אָמְרָה – "הַשָּׁב אֶת הַשְּׁטָחִים הַיְּרוּקִּים, הוֹד מַלְכוּתְךָ."
אַתָּה דַּיְּקָן כְּתָמִיד, בְּדִיּוּק כְּפִי שֶׁנִּכְתַּב בְּהַמְלָצוֹתֶיךָ.
אַתָּה יָכוֹל לִבְנוֹת עַל כָּל הַשֶּׁטַח, פְּרָט לְשִׁבְעַת הָאַקֶרִים בְּרֹאשׁ הַגִּבְעָה וּלְחַמְשָׁה-עָשָׂר אַקֶרִים נוֹסָפִים מִסְּבִיבָם, וְעָלֶיךָ לִבְנוֹת לִי חוֹמָה גְּבוֹהָה סְבִיב שִׁבְעַת הָאַקֶרִים.

------------------
בַּהֲלִיכָה זְהִירָה הוּא מְנַסֶּה לִשְׁמוֹר שֶׁהַקָּפֶה לֹא יִישָּׁפֵךְ בַּדֶּרֶךְ אֶל הַשּׁוּלְחָן, וּמַנִּיחַ אֶת הַכּוֹס בְּמָקוֹם שֶׁבּוֹ הָעַכְבָּר לֹא יָכוֹל לְהַגִּיעַ אֵלֶיהָ.
אָה, גָ'מִיל, אַתָּה הֲרֵי לֹא מְצַפֶּה שֶׁאֶסְתַּפֵּק בַּטִּיפּ-טִיפָּה שֶׁאַתָּה מַצִּיעַ לִי?

------------------
אִילּוּ הָיִיתִי בֶּן שְׁבַע-עֶשְׂרֵה, אָמַר אֶלְיָקוּם בְּלִבּוֹ, הָיִיתִי יָכוֹל, כְּמוֹ כְּלוּם, לִקְפּוֹץ עָלֶיךָ, יָשָׁר עַל הָרִצְפָּה, בַּמָּקוֹם שֶׁאַתָּה יוֹשֵׁב, וְהָיִיתִי יָכוֹל לְחַבֵּק אוֹתְךָ בְּכוֹחַ כָּזֶה, שֶׁהָיִיתָ מִתְפָּרֵק לַחֲתִיכוֹת.
הָיִיתָ נִבְהַל לִרְאוֹת כַּמָּה אֲנִי חָזָק.
אַתָּה, אַתָּה הָיִיתָ הַיֶּלֶד הַקָּטָן שֶׁכָּל הַזְּמַן הִבִּיט אֵלַי, כָּל הַזְּמַן, כְּשֶׁיָּשַׁבְתִּי שָׁם וְקָרָאתִי, וְכוּלָּם נִרְדְּמוּ?

------------------
אֵינִי יָכוֹל לִשְׁפּוֹט אוֹתָם, כִּי הָיִיתִי אֶחָד מֵהֶם, אֲבָל אֲנִי מִזְדַּהֶה אִיתָּם בְּכָל לִבִּי.
ג'וֹרְג' פְּסֶלְמֶנְצֶר הָיָה מִתְחַזֶּה שֶׁטָּעַן כִּי מוֹצָאוֹ בַּמִּזְרָח הָרָחוֹק.
רוֹרִי עָקַב אַחֲרָיו בְּהַעֲרָצָה מֵעַל לְכוֹס הַבּוֹרְבּוֹן שֶׁלּוֹ, מִתְקַשֶּׁה לְהַאֲמִין שֶׁהוּא יוֹשֵׁב עִם וְזוֹבֵי אַחֲרֵי פְּעָמִים רַבּוֹת כָּל כָּךְ שֶׁרָאָה אוֹתוֹ מִסְתַּלֵּק לְדַרְכּוֹ בְּמוֹנִיּוֹת, וְהִרְגִּישׁ אָז שֶׁהַדָּבָר הֲכִי חָשׁוּב בָּעוֹלָם הוּא לְהֵיעָלֵם אִיתּוֹ.

------------------
אִם יֵשׁ דָּבָר אֶחָד שֶׁאֲנִי לֹא מְפַקְפֵּק בּוֹ לְרֶגַע הוּא שֶׁבְּכָל מַצָּב שֶׁאֲנִי לֹא אֶהְיֶה בּוֹ, תָּמִיד אוֹכֶל לִסְמוֹךְ עָלֶיךָ שֶׁתַּעֲמוֹד לְצִדִּי וְשֶׁתַּעֲזוֹב הַכּוֹל כְּדֵי לַעֲזוֹר לִי.
וְתָשִׂים לֵב שֶׁלֹּא יַעְקְבוּ אַחֲרֶיךָ כְּמוֹ שֶׁאַתָּה הָיִיתָ עוֹקֵב אַחֲרַיי," צָחַק אָבִי בְּקוֹל רַע.
הַבְּעָיָה הָיְיתָה שֶׁהַתַּרְגּוּם שֶׁל בּוּרְלָא מְנוּקָּד וְאַתָּה חָשַׁבְתָּ שֶׁאַתָּה גָּדוֹל וְשֶׁסְּפָרִים מְנוּקָּדִים הֵם תִּינוֹקִיִּים בִּשְׁבִילְךָ.

------------------
טִי כְּבָר הָיָה עַל הַקַּרְקַע, חוֹפֵר בְּתוֹךְ הַהֲרִיסוֹת, מַפְנֶה מָקוֹם לְאוֹרֶךְ הַקִּיר הַחִיצוֹנִי.
הוּא קוּרְץ לִי, מְנַסֶּה לְהָמֵיס אֶת לִבִּי.
אֲנִי מְכַבֶּה אֶת הַטֵּלֵוִויזְיָה וְעוֹצֵם אֶת עֵינַי וְאָז הַנַּיָּיד שֶׁלִּי מְצַלְצֵל.

------------------
