"בְּשִּׂמְחָה נָבוֹא לַבָּר מִצְוָה", אָמְרָה נֹעָה בִּמְּבוּכָה, "וְהַקִּשְׁקוּשִׁים בֶּאֱמֶת יְכוֹלִים לְחַכּוֹת".
שָׂמַחְתִּי מְאוֹד! וְאִמָּא בֵּינְתַיִם דִּבְּרָה עִם דּוֹדָה מִרְים , וְאָמְנָם דּוֹדָה מִרְים הִסְכִּימָה שֶׁהִיא וְהַבַּת שֶׁלָּהּ נֹעָה בְּעֶזְרַת ה' תַּגַּעְנָה לַבַּר מִצְוָה! שָׁמְעוּ זֹאת הַבְּנוֹת דּוֹדוֹת שֶׁל אַבָּא שֶׁיָּשְׁבוּ לְיָדֵינוּ, חָשְׁבוּ מַחְשָׁבָה שְׁנִיָּה, וְאָמְרוּ שֶׁאוּלַי בְּכָל זֹאת הֵן תַּגַּעְנָה וְשֶׁנִּהְיָה בְּקֶשֶׁר.

------------------
הִיא אָחוֹת שֶׁל אַבָּא, וְגָרָה בְּחֵיפָה.
כְּבָר הַרְבֵּה שָׁנִים שֶׁאֵין לָהֶם יְלָדִים, וְלִפְנֵי כְּשָׁנָה לָקַח אַבָּא אֶת בַּעלָהּ, דּוֹד אַבְנֵר, לָרַבִּי שֶׁלָּנוּ, רַבִּי אַבְרָהָם, וְהוּא בֵּרֵךְ אוֹתָם בַּחֹם, וְהִנֵּה – בְּשָּׁעָה טוֹבָה נוֹלַד לָהֶם תִּינוֹק חָמוּד!
"וְנִסַּע לְבַקֵּר אוֹתָהּ?" שָׁאַלְתִּי.
"אַבָּא יִסַּע לְבֵית הַחוֹלִים בְּעֶזְרַת ה' מָחָר בַּצָּהֳרָיִים", אָמְרָה אִמָּא.
מְאוֹד הִתְרַגַּשְׁתִּי וְאַחַר כָּךְ הָיָה קָשֶׁה לִי לְהֵרָדֵם.

------------------
"אַבָּא שֶׁבַּשָּׁמַיִם", קָרָאתִי בִּתְּחִנָּה מִתּוֹךְ הַלֵּב, "בְּבַקָּשָׁה תַּדְרִיךְ אוֹתָנוּ, תְּכַוֵּן אוֹתָנוּ, תִּהְיֶה אִתָּנוּ תָּמִיד! קִרְיַת רִמּוֹנִים כֻּלָּהּ מְחַכָּה לִהְיוֹת שֶׁלְּךָ, מְחַכָּה לִשְׁמֹעַ אֶת קוֹלְךָ, מְחַכָּה לָלֶכֶת בְּדַרְכְּךָ.
בְּבַקָּשָׁה תַּשְׁרֶה עָלֵינוּ אֶת שְׁכִינָתְךָ, תִּשְׁלַח לָנוּ אֶת בִּרְכַתְךָ, תַּנְחִיל לָנוּ אֶת שַׁבְּתוֹתֶיךָ, וְתוֹשִׁיעֵנוּ בָּרַחֲמִים!" גַּם אִמָּא הִתְפַּלְּלָה מִכָּל הַלֵּב, עָלֵינוּ, וְעַל כָּל עַם יִשְׂרָאֵל.

------------------
הָיָה זֶה אוֹתוֹ יְהוּדִי, שֶׁהִרְגִּישׁ לְפֶתַע פִּתְאוֹם בָּאוֹרָה הֶעָצוּם שֶׁל הַשַּׁבָּת.
אִישׁ לֹא יִחֵס חֲשִׁיבוּת לִבְגָדָיו שֶׁשִּׁדְּרוּ חֻלִּין וְנִכּוּר.
הַכֹּל הִתְרַכְּזוּ אַךְ וְרַק בְּפָנָיו, שֶׁקַּרְנוֹ בָּאוֹר שֶׁל שַׁבָּת קֹדֶשׁ.
הוּא בָּא לְהִסְתּוֹפֵף בַּחֲצֵרוֹ שֶׁל הַחוֹזֶה וְהָפַךְ לִהְיוֹת אֶחָד מֵחֲסִידָיו.
בָּחוּר צָעִיר אֶחָד, נִפְצַע בְּמִלְחֶמֶת יוֹם הַכִּפּוּרִים, וּלְאַחַר סִידְרַת נִתּוּחִים נִשְׁאַר בְּלִי אֶפְשָׁרוּת לְהָזִיז אֶת אַחַת מְרַגְּלָיו.

------------------
דַּוְקָא בִּגְלַל שֶׁלֹּא מְצַיְּרִים וְלֹא נוֹסְעִים לְשׁוּם מָקוֹם, דַּוְקָא כָּךְ בֶּאֱמֶת יְכוֹלִים לֵהָנוֹת".
"וְגַם יֵשׁ אֹכֶל טָעִים וּמְיֻחָד" אָמַר שְׁלוֹמִי.
"וְגַם מַמְתַּקִּים" אָמַרְתִּי וְצָחַקְתִּי.
וּמֵאָז הִגִּיעָה אֵלֵינוּ הִלָּה כָּל שָׁבוּעַ לִתְפִלַּת "קַבָּלַת הַשַּׁבָּת", וְגַם אַבָּא שֶׁלָּהּ הִתְחִיל לְהַגִּיעַ בִּקְבִיעוּת לְבֵית הַכְּנֶסֶת .
הַיּוֹם כְּבָר עָבְרוּ כַּמָּה חֳדָשִׁים מֵאוֹתָהּ שַׁבָּת מְיֻחֶדֶת.

------------------
הָרַב גּוּטְנִיק נִכְנַס אֶל בֵּית הַמִּשְׁפַּחַה, וּבְסַבְלָנוּת רָבָה עָנָה לִבְנֵי הַמִּשְׁפָּחָה עַל כָּל שְׁאֵלוֹתֵיהֶם, וְהִדְרִיךְ אוֹתָם בִּשְׁמִירַת תּוֹרָה וּמִצְווֹת כְּהִלְכָתָן.
בְּנֵי הַמִּשְׁפָּחָה הִקְשִׁיבוּ קֶשֶׁב רָב לִדְבָרָיו וְאַף רָשְׁמוּ אוֹתָם.
לְבַסּוֹף, נִפְרְדוּ מִמֶּנּוּ בְּסִּפּוּק רָב וּמֵרֹב הִתְרַגְּשׁוּת שָׁכְחוּ לִשְׁאֹל לִשְׁמוֹ, וְכָךְ נֶעֱלַם "הַמַּלְאָךְ" שה' שָׁלַח לָהֶם.

------------------
גַּם אֵלּוּ שֶׁלֹּא הָיוּ בַּדְּרָשָׁה, שָׁמְעוּ עָלֶיהָ מְחַבְרֵיהֶם, וְסִפְּרוּ לְחַבְרֵיהֶם, וְכֻלָּם כֻּלָּם בָּאוּ וְרָצוּ חֹמֶר, וְרָצוּ הֶסְבֵּרִים, וְרָצוּ לִשְׁמֹעַ עוֹד וְעוֹד.
יָדַעְנוּ שֶׂה' עָשָׂה אֶת כָּל זֶה, וְהוֹדֵינוּ לוֹ כָּל הַזְּמַן.
הֵבֵאנוּ עוֹד וְעוֹד חוֹבְרוֹת הַדְרָכָה, בְּעִבְרִית, בְּרוּסִית וְגַם בְּצָרְפָתִית וּבְאַנְגְּלִית.

------------------
שְׂאוֹר הוּא חֲתִיכַת בָּצֵק שֶׁשָּׁהֲתָה זְמַן רַב בּוֹ תָּפְחָה וְהֶחֱמִיצָה, וְיֵשׁ לָהּ תְּכוּנוֹת 'מְדַבְּקוֹת'.
כַּאֲשֶׁר אֲנִי מְצָרֵף אוֹתָהּ לָעִסָּה הַחֲדָשָׁה הִיא מְאִיצָה בַּבָּצֵק אֶת תַּהֲלִיךְ הַהַחְמָצָה.
לָמָּה חָשׁוּב לִי לְהַסְבִּיר לָכֶם עַל הַשְּׂאוֹר? כִּי חֲזַ"ל מְכַנִּים אֶת הַיֵּצֶר הָרַע שֶׁבְּתוֹכֵנוּ בְּשֵׁם "שְׂאוֹר שֶׁבָּעִסָּה", מִי שֶׁגּוֹרֵם לְהַתְסִיס אוֹתָנוּ לִדְבָרִים שֶׁאֵינָם רְצוּיִים, וּמִכָּאן אָנוּ מַגִּיעִים לַנּוֹשֵׂא הָעִקָּרִי שֶׁלָּנוּ – הֶחָמֵץ שֶׁבַּלֵּב.

------------------
אָמְנָם מֻתָּר לַיּוֹלֶדֶת לִנְסֹעַ בְּשַּׁבָּת לַבַּיִת הַחוֹלִים! אֲבָל אִם יֵשׁ אֶפְשָׁרוּת לְהִתְאָרֵחַ אֵצֶל מִשְׁפָּחָה קְרוֹבָה וְלַחְסֹךְ אֶת הַנְּסִיעָה הַזּוֹ, זֶה בְּוַדַּאי יוֹתֵר נָעִים.
וְכָךְ בְּיוֹם שִׁשִּׁי הִתְאַרְגַּנּוּ, אָרַזְנוּ אֶת הַחֲפָצִים, אִמָּא אָפְתָה עוּגָה וּפַשְׁטִידָה וַאֲנִי הֵכַנְתִּי מָנָה אַחֲרוֹנָה, אַבָּא הוֹרִיד אֶת הַמִּזְוָדוֹת לָאוֹטוֹ, שְׁלוֹמִי הֵכִין דְּבָר תּוֹרָה נֶחְמָד – וְכָךְ נָסַעְנוּ כֻּלָּנוּ לַבַּיִת שֶׁל דּוֹדָה רִבְקָה.


------------------
וּבְכָל יוֹם שִׁשִּׁי הָיְתָה הָאִשָּׁה הַזּוֹ בָּאָה וּמְבִיאָה אִתָּהּ מַגָּשׁ נָאֶה וּמָלֵא בָּעוּגוֹת וְעוּגִיּוֹת, לַחֲלֻקָּה לְעוֹבְרִים וְשָׁבִים, כְּדֵי שֶׁיְּבָרְכוּ וְיִטְעֲמוּ מֵהֶן בְּשַּׁבָּת קוֹדֶשׁ, וְכָךְ יְעַנְּגוּ אֶת הַשַּׁבָּת.
הַזְּמַן חָלַף לוֹ בִּמְהִירוּת, וּבְכָל שָׁבוּעַ וְשָׁבוּעַ הִצְטָרְפוּ אֵלֵינוּ בָּרוּךְ ה' עוֹד וְעוֹד פְּעִילִים וְעוֹד וְעוֹד מִשְׁפָּחוֹת הִצְטָרְפוּ לִפְּרוֹיֶקְט הָאֵרוּחַ.

------------------
הָאִסּוּר הַגּוֹרֵף שֶׁל אֲכִילַת חָמֵץ בְּפֶסַח, וַאֲפִלּוּ שֶׁלֹּא לְהַשְׁהוֹתוֹ בִּרְשׁוּתֵנוּ, הוּא מֵהַמִּצְווֹת הָרִאשׁוֹנוֹת שֶׁקִּבַּלְנוּ, וַאֲפִלּוּ עוֹד לִפְנֵי שֶׁיָּצָאנוּ מִמִּצְרַיִם וְנַעֲשֵׂינוּ לְעַם.
לְאִסּוּרִים אֵלּוּ נוֹסֶפֶת, כַּמּוּבָן, מִצְוַת אֲכִילַת מַצָּה בְּלֵיל הַסֵּדֶר.
מָה בֵּין חָמֵץ לְמַצָּה?
קַו דַּק שֶׁמַּבְדִּיל

------------------
אַחֲרֵי הַפַּארְק אֲנַחְנוּ חוֹזְרִים, כָּל מִשְׁפָּחָה לַבַּיִת שֶׁלָּהּ, וְרַק בּוֹדְקִים שתּוּקְתּוּק נִשְׁאַר בְּתוֹךְ הַכְּלוּב שֶׁלּוֹ וְלֹא הֶחְלִיט שׁוּב לָצֵאת לְטִיּוּל חָדָשׁ.
אַבָּא שֶׁלָּנוּ אוֹמֵר שֶׁבִּזְכוּת תּוּקְתּוּק הוּא גִּלָּה מִשְׁפָּחָה חֲדָשָׁה מַקְסִימָה.
אִמָּא שֶׁלָּנוּ אוֹמֶרֶת שֶׁבִּזְכוּתוֹ מָצְאָה חֲבֵרָה מְרַתֶּקֶת וּבִזְכוּתוֹ לָמַדְנוּ הַרְבֵּה עַל בְּרִיאוֹת וְזֶה מְאוֹד מְאוֹד חָשׁוּב.

------------------
הַבָּחוּר הִגִּיעַ לְשָׁם בְּרִיצָה, וּבְהִתְרַגְּשׁוּת לֹא רְגִילָה דִּבֵּר עִם מִשְׁפַּחְתּוֹ, וְסִפּוּר לָהֶם בִּפְרוֹטְרוֹט אֶת אֲשֶׁר קָרָה בְּבֵית הָרַב.
בַּחוּרֵי הַיְּשִׁיבָה שֶׁעָמְדוּ בַּמָּקוֹם שָׁמְעוּ בְּהִשְׁתָּאוּת אֶת סִפּוּרוֹ, נִתְפְּסוּ לַאֲוִירַת הִתְעַלּוּת נִפְלָאָה, שִׁלְּבוּ יָד בַּיָּד וְיָצְאוּ בְּרִקּוּד שֶׁל שִׂמְחָה בְּיַחַד עִם הַבָּחוּר, לְהוֹדוֹת לְהַקָּדוֹשׁ בָּרוּךְ הוּא עַל הַיְּשׁוּעָה הַגְּדוֹלָה.

------------------
הַתְּחוּשָׁה שֶׁל מְסִירוֹת הַנֶּפֶשׁ לִשְׁמִירַת הַשַּׁבָּת הַקְּדוֹשָׁה גָּרְמָה לִי לְהִתְעַלּוֹת רוּחָנִית שֶׁלֹּא זָכִיתִי לִכְמוֹתָהּ שׁוּב".
פַּעַם אַחַת בְּלֵּיל שַׁבָּת, הִגִּיעָה עֵת הַמִּשְׁמֶרֶת שֶׁל הַסְּטַיְפְּלֶר לִהְיוֹת הַשּׁוֹמֵר בַּמַּחַנֶה.
הַכְּפוֹר הָיָה עָז.
הַסֵּדֶר בַּמַּחַנֶה הָיָה שֶׁהַחַיָּל הַיּוֹשֵׁב בַּמִּשְׁמֶרֶת לוֹבֵשׁ עַל עָצְמוֹ מְעִיל פַּרְוָה עָבֶה כְּדֵי לְהָגֵן עָלָיו מִפְּנֵי הַקֹּר הֶחָזָק, וְכָל חַיָּל מַעֲבִיר אֶת הַמְּעִיל לַחַיָּל בַּמִּשְׁמֶרֶת אַחֲרָיו.

------------------
הִרְעִיפָה הָרַבָּנִית בְּרֵכוֹת עַל רֹאשׁ הַנּוֹסֵעַ הַנָּדִיב, שֶׁכְּבָר הָיָה רָחוֹק מִשָּׁם.
"יְהִי רָצוֹן", מִלְמְלָה הָרַבָּנִית, "שֶׁבִּזְכוּת הַנֵּרוֹת הַלָּלוּ יָאִיר לְךָ אוֹר הַשַּׁבָּת.
שֶׁתִּזְכֶּה לְאוֹר שֶׁל שַׁבָּת קֹדֶשׁ!!!!"

------------------
פָּשׁוּט לְהוֹדוֹת לַה' שֶׁעָזַר לִי בְּבִעוּר הֶחָמֵץ!
שֶׁנִּזְכֶּה לְבַעֵר מֵהַלֵּב אֶת כָּל הֶחָמֵץ הָרַע!
שַׁבַּת שָׁלוֹם וּמְבֹרָךְ!
------------------
לְמָחֳרָת נָסַע אַבָּא לְבַקֵּר אֶת צִלָּה בְּבֵית הַחוֹלִים רָמְבָּ"ם בְּחֵיפָה, וְחָזַר עִם תְּמוּנוֹת שֶׁל צִלָה עִם הַתִּינוֹק שֶׁלָּהּ, שֶׁהוּא בֶּאֱמֶת מְאוֹד מְאוֹד חָמוּד, עִם פָּנִים עֲגוּלוֹת וְעֵינַיִם פְּתוּחוֹת וְיָפוֹת! כָּל כָּךְ שָׂמַחְתִּי שֶׁבִּקַּשְׁתִּי מֵאַבָּא לְהַדְפִּיס לִי תְּמוּנָה אַחַת שֶׁאוּכָל לְהַרְאוֹת לַחֲבֵרוֹת בַּכִּתָּה שֶׁלִּי.
סִפַּרְתִּי לַחֲבֵרוֹת שֶׁנּוֹלַד לִי בֵּן דּוֹד שֶׁהוּא בֵּן דּוֹד מְאוֹד חָמוּד וְרוֹאִים עָלָיו שֶׁהוּא מְאוֹד מְאוֹד מְיֻחָד!

------------------
אַחֲרֵי שְׁבוּעַיִם נָסַעְנוּ, קְבוּצָה שֶׁל מִשְׁפָּחוֹת מִבֵּית הַכְּנֶסֶת שֶׁלָּנוּ בְּקִרְיַת רִמּוֹנִים, לְחֻפְשָׁה לְמּוֹשָׁב בַּצָּפוֹן.
כְּבָר סִפַּרְתִּי לָכֶם עַל בֵּית הַכְּנֶסֶת הַשְּׁכוּנָתִי שֶׁלָּנוּ, בְּקִרְיַת רִמּוֹנִים.
בְּבֵּית הַכְּנֶסֶת יֵשׁ כְּמָאתַיִם מִתְפַּלְּלִים כֵּן יִרְבּוּ, חֶלְקָם מְבֻגָּרִים, חֶלְקָם בַּגִּילָאִים שֶׁל אַבָּא שֶׁלִּי, וְחֶלְקָם יְלָדִים בַּגִּיל שֶׁל יָאִיר וַאֲפִלּוּ קְטַנִּים יוֹתֵר.

------------------
אַחַר כָּךְ הִגִּיעַ גַּם הַמּוֹהֵל, וְאַבָּא הִתְיַשֵּׁב עַל כִּסֵּא הַסַּנְדָּק, וְכָל הָאוֹרְחִים נֶעֶמְדוּ מִסָּבִיב וְאָמְרוּ אֶת נֻסָּח הַתְּפִלָּה עִם הַמּוֹהֵל בְּיַחַד.
הִתְרַגַּשְׁתִּי מְאוֹד.
יָדַעְתִּי שֶׁאֵלִיָּהוּ הַנָּבִיא נִמְצָא כָּאן.
וְלַתִּינוֹק הֶחָדָשׁ קָרְאוּ נְתַנְאֵל, אוֹ בַּקִּצּוּר: נָתִי.
