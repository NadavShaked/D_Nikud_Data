הוּדְחַק אֶל אֲחוֹרֵי מוֹחִי כְּמוֹ כָּל שְׁאָר הַדְּבָרִים שֶׁלֹּא יָכוֹלְתִּי לְהִתְמוֹדֵד אִיתָּם. וְרֹאשׁ וְרִאשׁוֹנָה לְכוּלָּם, הַשְּׁאֵלָה: הַאִם זוֹ הָיְיתָה אַשְׁמָתִי? הַאִם פִסְפַסְתִּי מַשֶּׁהוּ? פְּרָט מֵידָע שֶׁהֵיעָדְרוֹ בְּרֶגַע הָאֱמֶת הוֹבִיל לְמוֹת כּוּלָּם? מָה בְּדִיּוּק קָרָה שָׁם, בָּרְגָעִים הָאַחֲרוֹנִים?


------------------
אֲנַחְנוּ לֹא בְּטוּחִים אֵיךְ, אוֹ לָמָּה. אֲנַחְנוּ מַמְשִׁיכִים לְשַׁדֵּר בְּתִקְוָוה שֶׁבְּשָׁלָב מְסוּיָּם הַחֲסִימָה תּוּסַר, שֶׁהַמְּסָרִים הָאֵלֶּה יַגִּיעוּ אֲלֵיכֶם בְּדֶרֶךְ זוֹ אוֹ אַחֶרֶת. אִם אַתֶּם רוֹאִים אֶת זֶה, בּוֹאוּ לְחַלֵּץ אוֹתָנוּ. תָּבִיאוּ בַּמְבָּלִיק. מצ"ב מְסָרִים פְּרָטִיִּים שֶׁל חַבְרֵי הַמִּשְׁלַחַת לְמִשְׁפָּחוֹת.


------------------
אֲבָל מִתַּחַת לִפְנֵי הַקַּרְקַע יֶשְׁנָם שׁוֹרָשִׁים, רַבִּים לְפָחוֹת כְּמִסְפַּר הָעֲנָפִים שֶׁיֶּשְׁנָם מֵעָלֶיהָ. לְמַעֲשֶׂה, הַמּוֹחַ הָיָה אָמוּר לְהוֹדִיעַ לָנוּ שֶׁיֵּשׁ כָּאן מַשֶּׁהוּ דְּמוּי מִשְׁקוֹלֶת-יָד, אַךְ הוּא אֵינוֹ עוֹשֶׂה זֹאת. 
------------------
רָאשִׁי מְסוּחְרָר? הַאִם הַהַעֲרָצָה הָעִיוֶּורֶת שֶׁלִּי לִמְפַקֵּד הַמִּשְׁלַחַת הַבּוֹגֵר, הַנָּאֶה, עִיוְּורָה אוֹתִי לִפְרָט קְרִיטִי כָּלְשֶׁהוּ שֶׁהוֹבִיל אֶת כּוּלְּכֶם לְמוֹתְכֶם? לֹא נוֹתַר לִי אֶת מִי לְהַאֲשִׁים מִלְּבַד אֶת עַצְמִי, אָז... זוֹ חַיֶּיבֶת לִהְיוֹת אַשְׁמָתִי, לֹא?

אֲנִי מְאַרְגֶּנֶת אֶת הַתִּיעוּד בְּצוּרָה נוֹחָה יוֹתֵר לְנִיתּוּחַ וּמַתְחִילָה לַעֲבוֹר עַל הַחוֹמָרִים – שֶׁאֲנִי רוֹאֶה לָרִאשׁוֹנָה. אֲבָל... 
------------------
הָיָה לִי בָּרוּר שֶׁהֵם לֹא יִטְרְחוּ לְשַׁחְרֵר קְבָצִים מוּתְאָמִים לְהַיּוֹם, הֵם חַיָּיבִים לַחֲשׂוֹף אוֹתָם, אֲבָל הֵם לֹא חַיָּיבִים לְהָקֵל עַל אֲנָשִׁים לְנַתֵּחַ אוֹתָם.


------------------
אֲנִי מְקַוֶּוה שֶׁהַסֵּפֶר הַזֶּה יֶשְׁנָהּ אֶת הַתְּפִיסָה הַזֹּאת, וְשֶׁהוּא יִגְרוֹם לְשִׁינּוּי תּוֹךְ שֶׁיַּעֲשֶׂה שִׁימּוּשׁ מוּצְלָח בַּיְּכוֹלֶת הַנִּפְלָאָה שֶׁיֵּשׁ לַסְּפָרִים לְהַרְאוֹת לָנוּ מֵעֵבֶר לְמָה שֶׁבִּיכוֹלְתֵּנוּ לִרְאוֹת בָּעוֹלָם סְבִיבֵנוּ: עֵצִים אֵינָם כַּפּוֹת! וּתְחוּשַׁת בֶּטֶן הִיא תְּחוּשָׁה מְבוֹרֶכֶת!
