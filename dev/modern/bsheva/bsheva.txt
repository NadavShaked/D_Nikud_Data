עו"ד מִיכָאֵל דְּבוֹרִין, הַמְּיַיצֵּג אֶת מְכִינַת עַצְמוֹנָה וְאֶת הָרַב אָסָף נֶאַמְבּוּרְג, מָסַר: "מְדוּבָּר בְּאֶחָד הַמִּקְרִים הַחֲמוּרִים בְּיוֹתֵר שֶׁל הוֹצָאַת לָשׁוֹן הָרַע הַזְּכוּרִים לִי בַּשָּׁנִים הָאַחֲרוֹנוֹת. 
------------------
אָז אֵיךְ בְּדִיּוּק מַצְלִיחַ הַגָּמָל לִשְׁתּוֹת כָּל כָּךְ הַרְבֵּה מַיִם בְּבַת אַחַת? בְּהַשְׁוָואָה לִבְנֵי אָדָם, פְּרוֹפ' יָגִיל מְצַיֵּין כִּי עַל מְנַת לְפַצּוֹת בִּשְׁתִיָּיה עַל אִיבּוּד 6 לִיטֶר מַיִם בֶּן אָדָם זָקוּק לִכְ-24 שָׁעוֹת. הַסִּיבָּה הִיא שֶׁאִם הַקֶּצֶב הָיָה גָּדוֹל יוֹתֵר, הָיָה יָכוֹל לְהִיגָּרֵם נֶזֶק לְכַדּוּרִיּוֹת הַדָּם שֶׁעֲלוּלוֹת לְהִתְפּוֹצֵץ מִלַּחַץ אוֹסְמוֹטִי גָּבוֹהַּ. אָז אֵיךְ הַגָּמָל עוֹשֶׂה זֹאת?


------------------
מִיָּד הָרַב תִּיקֵּן אוֹתוֹ וְאָמַר שֶׁיֵּשׁ לוֹמַר "רַב אַלּוּף גּוֹרֶן"! הַשִּׂיחָה הַקְּצָרָה הַזֹּאת הִתְנַהֲלָה דַּי בִּרְצִינוּת וְלֹא חַשְׁתִּי שֶׁיֵּשׁ כָּאן בְּדִיחָה כָּלְשֶׁהִי. הָרַמַטְכָּ"ל שׁוּב חוֹזֵר וּמְתַקֵּן: "אַתָּה. אַלּוּף הָרַב גּוֹרֶן, אֲנִי, רַב אַלּוּף, מַדּוּעַ אַתָּה טוֹעֵן שֶׁאַתָּה רַב אַלּוּף גּוֹרֶן"?! מֵשִׁיב לוֹ הָרַב: "אַתָּה הָרַמַטְכָּ"ל וַאֲנִי רַב אַלּוּף!" הָרַמַטְכָּ"ל אֵינוֹ מְוַותֵּר וּבְמַבָּט שֶׁל חִיּוּךְ עַל פָּנָיו מְבַקֵּשׁ הֶסְבֵּר לְדִבְרֵי הָרַב. 
------------------
בֵּית הַמִּדְרָשׁ שֶׁלָּנוּ יַמְשִׁיךְ לְהַצְמִיחַ עוֹד אַלְפֵי אֲנָשִׁים שֶׁמַּאֲמִינִים בִּמְדִינַת יִשְׂרָאֵל, בִּצְבָא הַהֲגָנָה לְיִשְׂרָאֵל וּבְעַם יִשְׂרָאֵל וּמְשַׁלְּבִים בְּהִצְטַיְּינוּת בֵּין סָפְרָא לְסַיָּיפָא. הִשְׁתַּלְּבוּת זוֹ בַּחֶבְרָה הַיִּשְׂרְאֵלִית הַיּוֹנֶקֶת מֵחִיבּוּר עָמוֹק לְבֵית הַמִּדְרָשׁ לֹא נִיתָּן לַעֲצִירָה זוֹ. מִי שֶׁמְּנַסֶּה לְפַרְסֵם מֵידָע שִׁקְרִי וְשָׁגוּי עָלָיו לָשֵׂאת בַּתּוֹצָאוֹת".


------------------
פְּסַק הַדִּין הַנִּפְרָשׁ עַל פְּנֵי 51 עַמּוּדִים אוֹתוֹ כָּתַב הַשּׁוֹפֵט לַמְבֵּרְט בְּסוֹף הַשָּׁבוּעַ הָאַחֲרוֹן, קוֹבֵעַ כִּי רוּסְיָה וּמַחֲזִיקֵי נְכָסֶיהָ בְּאַרְהָ"ב אֵינָם יְכוֹלִים לְעַרְעֵר עַל פְּסִיקָתוֹ בֶּעָבָר וְכִי עֲבוּר כָּל יוֹם שֶׁבּוֹ הַמַּעֲצָמָה הָרוּסִית לֹא תָּשִׁיב אֶת ''אוֹסֶף שֶׁנֵּיאוֹרְסאהן'' הַמֵּכִיל אֶת סִפְרֵי אַדְמוֹרֵי חַבָּ''ד לִתְנוּעַת חַבָּ''ד, תְּחוּיַּב רוּסְיָה בִּסְכוּם שֶׁל 50 אֶלֶף דּוֹלָר בְּכָל יוֹם.


------------------
לֹא שֶׁאֵין אֲחֵרִים, אֲבָל אִישׁ אֵינוֹ כֹּה רָהוּט, מְתוּמְצָת וּבָהִיר, מְמוּקָּד וְנָהִיר, שֶׁדְּבָרָיו הַמְּתוּקִים קוֹלְעִים לְלִיבּוֹ שֶׁל כָּל יְהוּדִי, אִם לַמְדָן וְאִם עַם הָאָרֶץ.

6. 
------------------
הַמְּכִינָה הַקְּדַם צְבָאִית ''עֹצֶם'' שֶׁבַּיִּישּׁוּב נָוֶוה הִגִּישָׁה הַיּוֹם (חֲמִישִׁי) לְבֵית הַמִּשְׁפָּט הַשָּׁלוֹם בְּתֵל אָבִיב תְּבִיעַת דִּיבָּה בְּסַךְ מִילְיוֹן שְׁקָלִים נֶגֶד חֲדָשׁוֹת 13 וְהַכַּתָּב עָמְרִי מֵנִיב.

הַתְּבִיעָה מַגִּיעָה, כַּזָּכוּר, בְּעִקְבוֹת פִּרְסוּם מְגַמָּתִי וְשִׁקְרִי נֶגֶד הַמְּכִינָה שֶׁשּׁוּדַּר בַּמַּהֲדוּרָה הַמֶּרְכָּזִית שֶׁל 'חֲדָשׁוֹת 13' וְעָסְקָה בְּהִתְבַּטְּאוּיוֹתָיו לִכְאוֹרָה שֶׁל הָרַב אָסָף נָאוּמְבּוּרְג נֶגֶד צַהַ"ל.
