הוּא פִיזִית נִמְצָא בַּשֶּׁבִי שֶׁלָּהּ אַךְ לִיבּוֹ וּמַחְשְׁבוֹתָיו אִיתִּי וְשֶׁלִּי.
כָּל הַיּוֹם וְכָל הַלַּיְלָה אֲנַחְנוּ מְדַבְּרִים וּמַרְגִּישִׁים אַהֲבָה שֶׁלֹּא נִיתֶּנֶת לְמִימּוּשׁ.
גַּם אִם נִהְיֶה עִם אֲחֵרִים הַלֵּב שֶׁלָּנוּ שַׁיָּיךְ אֶחָד לַשְּׁנִיָּה לְעוֹלָמִים.
אַמְשִׁיךְ מְאוּחָר יוֹתֵר לִכְתּוֹב נֶאֱלֶצֶת לְסַיֵּים אֲהוּבִי הִתְקַשֵּׁר :)
------------------
זֶה כָּכָה הָיָה מֵהַתְחָלָה, אֲנִי גָּדוֹל מֵהַבַּת שֶׁלָּהּ בְּאַרְבַּע שָׁנִים(אנִי 25 הִיא 21) וּכְבָר אֵיךְ שֶׁנִּכְנַסְתִּי לַבַּיִת הִיא 'נִצְמְדָה אֵלַיי' יוֹתֵר מִדַּי.
חִילְּקָה לִי מַחְמָאוֹת, בִּיקְּשָׁה שֶׁאֲנִי אֵלֵךְ אִיתָּהּ כָּל פַּעַם לַסּוֹפֵר, וּבִכְלָל לֹא מַמָּשׁ הָיְיתָה נוֹתֶנֶת לִי מְנוּחָה, תָּמִיד אָהֲבָה לְדַבֵּר הַרְבֵּה וְהִרְגַּשְׁתִּי כְּאִילּוּ הִיא מְנַסָּה לְהוֹצִיא מִמֶּנִּי דְּבָרִים

------------------
הִצַּעְתִּי לוֹ שֶׁנַּעֲשָׂה יְשִׁיבָה שֶׁהוּא יָבִיא חָבֵר וַאֲנִי יָבִיא חֶבְרָה (סִיפֵּר שֶׁגָּר לְבַד) .
הוּא הִתְנַגֵּד וּבַהֶמְשֵׁךְ הֵבַנְתִּי שֶׁבְּעֶצֶם הוּא מְעוּנְיָין לִשְׁכַּב אִיתִּי.
הוּא טוֹעֵן שֶׁאֵין שׁוּם דָּבָר פָּסוּל אִם הָאִישָּׁה זוֹרֶמֶת וְשֶׁאֲנִי אַפְסִיק עִם הַהַגְדָּרוֹת שֶׁל "זוֹנָה".
סִיפֵּר שֶׁהָיוּ לוֹ קְשָׁרִים שֶׁהִתְחִילוּ מִסְּטוּץ וְנִמְשְׁכוּ לְשָׁנָה .

------------------
אִם אָכֵן דַּעַת רוֹב הַגּוֹלְשִׁים שֶׁלֹּא מְדוּבָּר בְּיִיעוּל אֶלָּא בְּמִטְרָד, אָנְסָה לְהִתְלוֹנֵן "בַּדְּרָגִים הַגְּבוֹהִים" לְמַעַן שִׁינּוּי הַשֵּׁירוּת.
------------------
בַּתְּחִילָּה אָמְרוּ שֶׁבְּעָיָה כִּי הַמָּסָךְ שָׁבוּר וְהֵם לֹא מַמָּשׁ רוֹאִים מָה הֵם עוֹשִׂים, אֲבָל מִשֶּׁהֵבִינוּ כַּמָּה חָשׁוּב לִי הָעִנְיָין שָׁלְפוּ לִכְבוֹדִי אֶת הַתּוֹתָח שֶׁל הַמֶּרְכָּז, הַהוּא שֶׁפַּעַם כְּשֶׁהָיוּ מְתַקְּנִים בַּמָּקוֹם, הָיָה טֶכְנַאי מַעְבָּדָה וְיוֹדֵעַ לְתַפְעֵל מַכְשִׁירִים בַּעֲצִימַת עֵינַיִים, וְהוּא הֶעֱבִיר לִי אֶת הַזִּכְרוֹנוֹת.

------------------
תִּיכְנַנְתִּי לָשֶׁבֶת בְּחוֹסֶר מַעַשׂ, לִגְמוֹעַ כּוֹסוֹת קָפֶה מִמְּכוֹנָה וּלְהַמְתִּין לַמַּכְשִׁיר הַמְּתוּקָּן,
אֲבָל כָּאן חִיכְּתָה לִי הַפְתָּעָה! הַטֵּלֵפוֹן יִישָּׁלַח לְתִיקּוּן בְּמַעְבָּדָה מֶרְכָּזִית וְיַחֲזוֹר לִנְקוּדַּת הַשֵּׁירוּת בְּעוֹד 48 שָׁעוֹת,
בֵּנְתַיִים אַתָּה מְקַבֵּל מַכְשִׁיר חִילּוּפִי.
לָמָּה? שָׁאַלְתִּי,
זֶה בִּשְׁבִיל לְיַיעֵל אֶת מַתַּן הַשֵּׁירוּת! נֶאֱמַר לִי.
