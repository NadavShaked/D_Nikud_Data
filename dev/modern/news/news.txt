נִגְבּוּ עֵדוּיוֹת מִנָּשִׁים רַבּוֹת שֶׁתֵּיאֲרוּ אֶת מַהוּת וְטִיב הַקֶּשֶׁר בֵּינָן לְבֵין הֶחָשׁוּד.
חֵלֶק מֵהָעֵדוּיוֹת הֶעֱלוּ חֲשָׁד לְבִיצּוּעַ עֲבֵירוֹת מִין, אוּלָם אֵלֶּה הִתְיַישְּׁנוּ".

עוֹד נִמְסַר בְּהוֹדָעַת מִשְׂרַד הַמִּשְׁפָּטִים כִּי "תְּלוּנוֹת אֲחֵרוֹת, שֶׁלֹּא הִתְיַישְּׁנוּ, לֹא חָצוּ אֶת הֶרֶף הַפְּלִילִי הַנִּדְרָשׁ לַעֲבֵירָה שֶׁל הַטְרָדָה מִינִית.

------------------
הַחִיבּוּר נַעֲשֶׂה בְּעֶזְרַת הַתְּחוּשׁוֹת שֶׁאֲנַחְנוּ מַרְגִּישִׁים בְּעֵת שֶׁאֲנַחְנוּ שׁוֹמְעִים אֶת אוֹתוֹ שִׁיר בַּפַּעַם הָרִאשׁוֹנָה.
כַּאֲשֶׁר הַשִּׁיר הַזֶּה מְקוּשָּׁר לִרְגָשׁוֹת מְסוּיָּמִים, חֲזָקִים בִּמְיוּחָד, גַּם אִם נִשְׁמָע אוֹתוֹ שָׁנִים רַבּוֹת לְאַחַר אוֹתוֹ פַּעַם רִאשׁוֹנָה, הָרְגָשׁוֹת הַלָּלוּ יַעֲלוּ שׁוּב וְנִיזָּכֵר בְּכָל מָה שֶׁהָיָה סְבִיב הַשִּׁיר.


------------------
גַּם לֹא לְהַרְגִּישׁ שֶׁאַתָּה מַשְׁאִיר מִישֶׁהוּ בַּצַּד, אוֹ לֹא מִתְחַשֵּׁב בְּדֵעוֹת שֶׁלּוֹ, פָּשׁוּט לַשֶּׁקֶט שֶׁלְּךָ עִם עַצְמְךָ".

"שְׁאֵלָה קָשָׁה.
אֲנִי רְגִילָה לָטוּשׁ בַּשָּׁנִים הָאַחֲרוֹנוֹת, וּבַיְּלָדוֹת לָאַמְבַּטְיָה.
אֲנִי אֵלֵךְ עַל מָה שֶׁיּוֹתֵר נוֹחַ לִי פְּרַקְטִית וְזֶה טוּשׁ".

טֵלֵוִויזְיָה בַּסָּלוֹן אוֹ בַּחֲדַר הַשֵּׁינָה?

"בַּסָּלוֹן.
זֶה מֶרְחָב יוֹתֵר לָשֶׁבֶת עִם הָאֲנָשִׁים, יֵשׁ לַטֵּלֵוִויזְיָה אַקְט יוֹתֵר חֶבְרָתִי.

------------------
הַתַּקְרִית אֵירְעָה בְּמֶרְחָב הַמּוּכָּר לָאוֹיֵב, מֵאַחַר וּבֶעָבָר פָּעַל בּוֹ בֵּית חוֹלִים שָׂדֶה בִּתְקוּפַת "שְׁכֵנוֹת טוֹבָה".
מִדּוֹבֵר צַהַ"ל נִמְסַר כִּי "כּוֹחַ מְיוּחָד שֶׁבִּיצֵּעַ מַאֲרַב בְּאֵזוֹר דְּרוֹם רָמַת הַגּוֹלָן, סָמוּךְ לְמוּצָּב צְבָאִי, זִיהָה חוּלְיָה שֶׁל מִסְפַּר מְחַבְּלִים מְנִיחִים מִטְעָנִים בְּקִרְבַת גֶּדֶר הַגְּבוּל. הַכּוֹחַ וּכְלֵי טַיִס שֶׁל צַהַ"ל בִּיצְּעוּ יְרִי בְּמַקְבִּיל עַל הַחוּלְיָה שֶׁמָּנְתָה אַרְבָּעָה מְחַבְּלִים וּפָגְעוּ בָּהֶם". 

------------------
כְּשֶׁנִּכְנַסְנוּ לֹא הָיָה שָׁם כִּמְעַט רִיהוּט, וְנוֹרָא רָצִיתִי שָׁטִיחַ לַסָּלוֹן וְכָרִיּוֹת צִבְעוֹנִיּוֹת וְגַם לְהָבִיא אֲבִיזָרִים דֵּקוֹרָטִיבִיִּים מֵהַבַּיִת שֶׁלִּי, כְּדֵי לְהַרְגִּישׁ יוֹתֵר בְּנוֹחַ וְשֶׁתִּהְיֶה אֲוִוירָה יוֹתֵר חַמָּה".

דָּנִית מְסַפֶּרֶת שֶׁ"אָמִיר שׁוּתָּף מְעוּלֶּה לַדִּירָה", וְהֵם מִתְחַלְּקִים בַּמַּטָּלוֹת.

------------------
זוֹ מַכָּה קָשָׁה לְאִירָאן וְהַהֶישֵּׂג הֶחָשׁוּב בְּיוֹתֵר לְמִמְשָׁל טְרַאמְפּ בַּמַּאֲבָק בַּטֵּרוֹר מֵאָז חִיסּוּל מַנְהִיג דָּאעֶשׁ אַבּוּ בַּכְּר אֶל-בְּגַדְּדִי.
מִיהוּ קָאסֶם סוּלִימָאנִי? מָה מַשְׁמָעוּת חִיסּוּלוֹ וְכֵיצַד תָּגִיב אִירָאן?

יַד הַבַּרְזֶל שֶׁל טְרַאמְפּ
חִיסּוּל סוּלִימָאנִי הוּא צַעַד יוֹצֵא דּוֹפֶן בַּמַּאֲבָק הָאָמֵרִיקָנִי נֶגֶד אִירָאן.
זוֹ הַפַּעַם הָרִאשׁוֹנָה שֶׁבְּאוֹפֶן רִשְׁמִי מְחַסֶּלֶת אַרְהָ"ב בָּכִיר אִירָאנִי מֵאָז הַמַּהְפֵּכָה הָאִיסְלָאמִית בְּ-1979.

------------------
כְּכָל שֶׁיִּתְוַוסְּפוּ עוֹד מְכוֹנוֹת הַנְּשָׁמָה זֶה יַגְדִּיל אֶת מִרְוָוח הַטָּעוּת וְזֶה יִילָּקַח בְּחֶשְׁבּוֹן בְּשִׁיקּוּל הַדַּעַת בְּשִׁחְרוּר הַמֶּשֶׁק.
"אָנוּ לֹא רוֹצִים לְהַגִּיעַ לְסִיטוּאַצְיָה שֶׁצְּרִיכִים לִבְחוֹר מִי מַשְׁאִירִים בַּחַיִּים וְאֶת מִי מְנַתְּקִים מֵהַמְּכוֹנָה", הוֹסִיף הַשַּׂר.
עִיקָּרוֹן מַנְחֶה נוֹסָף בְּגִיבּוּשׁ הַתָּכְנִית הָאַסְטְרָטֵגִית הוּא הִימָּצְאוּתָם שֶׁל בְּדִיקוֹת סְרוֹלוֹגְיוֹת אֲמִינוּת.

------------------
לִיבֶּרְמָן הָפַךְ אֶת הַנּוֹשֵׂא לְגוּלַּת הַכּוֹתֶרֶת שֶׁל תִּפְקוּדוֹ בְּמִשְׂרַד הַבִּיטָּחוֹן, וְעַל כָּךְ מַגִּיעִים לוֹ שְׁבָחִים, אֲבָל בֵּינְתַיִים נִשְׁאֲרָה הַחְלָטַת הַמֶּמְשָׁלָה עַל הַנְּיָיר וְלֹא הוּחַל בְּבִיצּוּעַ.
הַשְּׁנָתַיִים שֶׁחָלְפוּ מֵאָז עֲלוּלוֹת לְהִתְבָּרֵר כְּעִיכּוּב שֶׁיִּגְרוֹם בְּכִייָּה לְדוֹרוֹת, תַּרְתֵּי מַשְׁמָע.


------------------
עִם זֹאת חָשׁוּב לְצַיֵּין שֶׁאֵין עֲדַיִין כּוֹנָנִים כָּאֵלֶּה בַּשּׁוּק כָּךְ שֶׁלֹּא יָדוּעַ כַּמָּה הֵם יַעֲלוּ בַּהַתְחָלָה, אַךְ כַּנִּרְאֶה שֶׁלֹּא פָּחוֹת מֵאֵלֶּה שֶׁל מַיְקְרוֹסוֹפְט, בַּהֶמְשֵׁךְ עִם הַתַּחֲרוּת הַמְּחִירִים כַּנִּרְאֶה יֵרְדוּ יוֹתֵר מַהֵר בַּצַּד שֶׁל סוֹנִי.

חָשׁוּב לְצַיֵּין שֶׁבְּסוֹפוֹ שֶׁל דָּבָר לֹא הַנְּתוּנִים הַטֶּכְנִיִּים יַכְרִיעוּ אֶת הַנִּיצָּחוֹן שֶׁל צַד זֶה אוֹ אַחֵר.

------------------
בַּפּוֹסְט שֶׁפִּירְסֵם בְּפֶיְיסְבּוּק זַעַם קָנִיאֵל: "ד"ר וַויְנְשְׁטֵיְין, בּוֹאִי אֲנִי אֲסַפֵּר לְךָ מָה זֶה אַנְשֵׁי הַיְיטֵק עֲשִׁירִים, כְּשֶׁהֵקַמְנוּ אֶת הַחֶבְרָה לִפְנֵי 9 שָׁנִים, עָבַדְנוּ לְלֹא מַשְׂכּוֹרֶת 3 שָׁנִים, וּלְאַחַר שֶׁהִשַּׂגְנוּ מִימּוּן מִינִימָלִי הִרְוַוחְנוּ שְׂכַר מִינִימוּם.
לָמַדְנוּ לִחְיוֹת כִּמְעַט בְּלִי כֶּסֶף, חִיסַּלְנוּ אֶת כָּל הַחֶסְכוֹנוֹת וְכָל קוּפּוֹת הַפִּיצּוּיִים שֶׁהָיָה אֶפְשָׁר לִפְתּוֹחַ.

------------------
הַחֲשִׁיבוּת הַמֶּרְכָּזִית שֶׁל מֶחְקָר זֶה הִיא בְּכָךְ שֶׁהוּא מַדְגִּים אֵיךְ שִׁינּוּיֵי אוֹרְחוֹת חַיִּים יְכוֹלִים לְהַשְׁפִּיעַ מַשְׁמָעוּתִית עַל הַפְּעִילוּת שֶׁל גַּנִּים שׁוֹנִים בַּגּוּף, וְכָךְ לְהַשְׁפִּיעַ עַל תַּהֲלִיכֵי מַחֲלָה", מוֹסִיפָה בַּר-גִּיל.
מֶחְקָר נוֹסָף שֶׁהִתְפַּרְסֵם לָאַחֲרוֹנָה, עֵקֶב אַחֲרֵי כְּ-1,000 חוֹלֵי סַרְטָן הַמְּעִי הַגַּס שֶׁסִּייְּמוּ אֶת הַטִּיפּוּלִים הַכִּימוֹתֵרַפִּיִּים.

------------------
הִיא נָחֲתָה נְחִיתָה קָשָׁה עַל הַיָּרֵחַ בְּ-11 בְּאַפְּרִיל הַשָּׁנָה, וְלַמְרוֹת שֶׁהַמְּשִׂימָה לֹא הוּשְׁלְמָה כַּמְּתוּכְנָן, מְדוּבָּר בְּהֶישֵּׂג יוֹצֵא דּוֹפֶן - מְדִינָה שְׁלֵמָה נִסְחֲפָה אַחֲרֵי הַחֲלָלִית וְנוֹצַר "אֵפֶקְט בְּרֵאשִׁית", בְּדוֹמֶה לְאֵפֶקְט אַפּוֹלוֹ בִּשְׁנוֹת הַ-70.
 
אֶת הָאֵירוּעַ פָּתַח הַפִילַנְתְּרוֹפּ מוֹרִיס קַאהַן- נָשִׂיא עֲמוּתַּת SpaceIL, וּמִי שֶׁתָּרַם אֶת הַחֵלֶק הָאֲרִי לַפְּרוֹיֶקְט שֶׁעֲלוּתוֹ הָיְתָה 100 מִילְיוֹן דּוֹלָרִים.

------------------
וְאוּלָם, יֶשְׁנָם שְׁנֵי גּוֹרְמִים מְעַכְּבִים שֶׁצָּרִיךְ לְהִתְחַשֵּׁב בָּהֶם.

הָרִאשׁוֹן הוּא שֶׁלְּפָחוֹת בַּיָּמִים הָרִאשׁוֹנִים, חֵיל הָאֲוִויר וְצַהַ"ל יִצְטָרְכוּ לִפְעוֹל כְּשֶׁכָּל שְׂדוֹת הַתְּעוּפָה, רוֹב מַחְסְנֵי הַחֵירוּם, מְתַקְּנֵי הַבַּקָּרָה וְאִיסּוּף הַמּוֹדִיעִין נִמְצָאִים תַּחַת מַטָּחִים כְּבֵדִים שֶׁל טִילִים, רָקֵטוֹת, מָלָ"טִים וְרַחֲפָנִים.

------------------
לֹא אוֹהֶבֶת לִשְׁהוֹת יוֹתֵר מִדַּי זְמַן בְּמַיִם כִּי אָז נִהְיוֹת לִי אֶצְבְּעוֹת סָבְתָא, וְזֶה לֹא כֵּיף אָז מַעֲדִיפָה לַעֲשׂוֹת אֶת זֶה מַהֵר".

טֵלֵוִויזְיָה בַּסָּלוֹן אוֹ בַּחֲדַר הַשֵּׁינָה?

"טֵלֵוִויזְיָה בַּסָּלוֹן.
כְּשֶׁאֲנִי הוֹלֶכֶת לִישׁוֹן חַיֶּיבֶת לִהְיוֹת דְּמָמָה וְחַיָּיב לִהְיוֹת חוֹשֶׁךְ, וְלָכֵן אֵין לִי שִׁימּוּשׁ בַּטֵּלֵוִויזְיָה בַּחֲדַר הַשֵּׁינָה".

נִקְשֶׁרֶת לַחֲפָצִים אוֹ נִפְטַרְתָּ בְּקַלּוּת?

"אֲנִי נִפְטַרְתָּ מֵחֲפָצִים לְגַמְרֵי בְּקַלּוּת".


------------------
עֶקְרוֹן הַמִּלְחָמָה הָעַתִּיק שֶׁלְּפִיו הַהַתְקָפָה הִיא הַהֲגָנָה הַטּוֹבָה בְּיוֹתֵר תָּקַף גַּם לְגַבֵּי הָאִיּוּמִים הַנִּשְׁקָפִים מֵהָאֲוִויר לָעוֹרֶף הַיִּשְׂרְאֵלִי.
אֵין סָפֵק שֶׁהָעֶלְיוֹנוּת הַמּוֹדִיעִינִית, הָאֲוִוירִית וְהַטֶּכְנוֹלוֹגִית שֶׁל יִשְׂרָאֵל מַעֲנִיקָה לְצַהַ"ל בִּכְלָל וּלְחֵיל הָאֲוִויר בִּפְרָט כֵּלִים רַבֵּי עוֹצְמָה כְּדֵי לְשַׁתֵּק בִּמְהִירוּת אֲתָרֵי שִׁיגּוּר שֶׁל רָקֵטוֹת, כָּטְבָּ"מִים, טִילֵי שִׁיּוּט וְרַחֲפָנִים.

------------------
נְקַנַּח בְּעוֹד קְלָאסִיקָה, הַפַּעַם שֶׁל מִשְׂחַק אַסְטְרָטֶגְיָה שֶׁיָּצָא לַחֲנוּיוֹת בְּפֶבְּרוּאָר 99 (לִפְנֵי יוֹתֵר מִ-20 שָׁנָה) וַעֲדַיִין נֶחֱשַׁב לְאֶחָד הַמִּשְׂחָקִים הַטּוֹבִים בַּשּׁוּק, כְּזֶה שֶׁאֲנַחְנוּ פָּשׁוּט לֹא מְסוּגָּלִים לִמְחוֹק מֵהַמַּחְשֵׁב שֶׁלָּנוּ.
Heroes of Might and Magic III, אוֹ בִּשְׁמוֹ הַקָּצָר - הִירוֹז 3, הוּא מִשְׂחַק אַסְטְרָטֶגְיָה בְּתוֹרוֹת בּוֹ תַּפְקִידְכֶם לִכְבּוֹשׁ אֶת מַפַּת הַמִּשְׂחָק בְּעֶזְרַת הַגִּיבּוֹרִים שֶׁטִּיפַּחְתֶּם.

------------------
כְּשֶׁנִּכְנַסְנוּ לָגוּר יַחַד כִּמְעַט לֹא הִכַּרְנוּ, וְזֶה לֹא פָּשׁוּט לְהַכְנִיס לַבַּיִת, לִפְרָטִיּוּת, מִישֶׁהוּ שֶׁבְּקוֹשִׁי מַכִּירִים אוֹתוֹ.
חוּץ מֵהַבֵּן שֶׁלִּי וְחֶבְרָה שֶׁהָיְיתָה לִי, לָקַח לָנוּ מִסְפַּר יָמִים לְהִתְאַקְלֵם.
צָרִיךְ לִמְצוֹא אֶת הָאִיזּוּן בֵּין הַפְּרָטִיּוֹת לְבֵין בְּיַחַד.
לְהַרְגִּישׁ חוֹפְשִׁי לָרֶדֶת עִם טְרֵנִינְג אוֹ מִכְנָסַיִים קְצָרִים, בְּעֶצֶם לֹא לְהַרְגִּישׁ שֶׁיֵּשׁ אוֹרֵחַ בַּבַּיִת אֶלָּא בֶּן בַּיִת".


------------------
לְגוּשׁ הַיָּמִין יֵשׁ 58 חַבְרֵי כְּנֶסֶת, וּבְכָל זֹאת הוּא מְבַקֵּשׁ לְעַצְמוֹ פָּרִיטֵטִיּוֹת, זְכוּיוֹת וֵטוֹ, מִסְפָּר זֵהֶה שֶׁל שָׁרִים וְעוֹד תּוּפִינִים מֵהַסּוּג הַזֶּה.
זֹאת חוּצְפָּה שֶׁעוֹד לֹא הָיְיתָה כְּמוֹתָהּ בָּעוֹלָם הַפּוֹלִיטִי שֶׁלָּנוּ שֶׁטֶּבֶל וְהַרְבֵּה שֶׁרָצִים בְּיָדוֹ.

------------------
גַּם אֲחוּזֵי הַתְּמוּתָה מִסַּרְטָן הָיוּ נְמוּכִים מַשְׁמָעוּתִית בְּקֶרֶב אֵלּוּ שֶׁהָיוּ בִּקְבוּצַת הַסִּיכּוּן הַנָּמוּךְ.
מַסְקְנוֹת הַמֶּחְקָר קָבְעוּ כִּי תּוֹצְאוֹתָיו מְהַוּוֹת תְּמִיכָה מַשְׁמָעוּתִית לַהֲבָנָה כִּי נִיתָּן לְהַפְחִית תַּחֲלוּאָה בְּסַרְטָן בְּאוֹפֶן מַשְׁמָעוּתִי בְּאֶמְצָעוּת שִׁינּוּיֵי אוֹרְחוֹת חַיִּים מְתוּנִים יַחֲסִית.

מֶחְקָרִים קוֹבְעִים: אוֹרַח חַיִּים בָּרִיא יָכוֹל לְעַכֵּב הִתְקַדְּמוּת וַחֲזָרַת מַחֲלָה

------------------
הַהַחְלָטָה הִתְקַבְּלָה עַל דַּעַת פְּרַקְלִיט הַמְּדִינָה הַיּוֹצֵא מֹשֶׁה לְדוֹר, שֶׁאִימֵּץ אֶת עֶמְדַּת פְּרַקְלִיטוּת הַמָּחוֹז, וְכֵן עַל דַּעַת פְּרַקְלִיט הַמְּדִינָה הַמְּכַהֵן שַׁי נִיצָּן.

בְּהוֹדָעַת מִשְׂרַד הַמִּשְׁפָּטִים בַּנּוֹשֵׂא נִכְתַּב כִּי "בְּתִיק זֶה קִייְּמָה הַמִּשְׁטָרָה חֲקִירָה יְסוֹדִית וּמַקִּיפָה בַּחֲשָׁד לְבִיצּוּעַ עֲבֵירוֹת מִין שׁוֹנוֹת עַל יְדֵי מַר רוֹזֵן, בְּמִסְפַּר מִתְלוֹנְנוֹת, לְאוֹרֶךְ תְּקוּפָה מְמוּשֶּׁכֶת.

------------------
הַגָּעָתָם לָעִיקּוּל הֶחָבִיב עֲלֵיכֶם, סוֹבַבְתֶּם אֶת הַהֶגֶה לַכִּיווּן הַמַּתְאִים בְּהֶתְאֵם לְקֶצֶב הַנְּהִיגָה וּלְרַדְיוּס הַפְּנִיָּיה.
אִם אַתֶּם מַכִּירִים אוֹתָהּ, זֶה יִהְיֶה מְאוֹד מְדוּיָּק וְלֹא תִּדְרְשׁוּ לְתַקֵּן.
אִם אַתֶּם לֹא מַכִּירִים אוֹתָהּ, אוּלַי תִּיקּוּן עָדִין כְּדֵי לְנַטְרֵל יוֹתֵר מִדַּי הַפְנָיָיה - אוֹ לְהֵיפֶךְ.

עַל קֶרַח וְשֶׁלֶג, וְלֹא מְשַׁנֶּה כַּמָּה אַתֶּם מְיוּמָּנִים, יֵשׁ הַרְבֵּה יוֹתֵר תְּנוּעוֹת תִּיקּוּן.

------------------
הִתְאַכְזַבְתִּי מֵהֲמוֹן אֲנָשִׁים, לֹא כָּאֵלֶּה שֶׁקְּרוֹבִים אֵלַיי בְּרָמָה יוֹם-יוֹמִית, אֲבָל מֵהַרְבֵּה אֲחֵרִים".

בְּמַהֲלַךְ הַזּוּגִיּוּת שֶׁלָּכֶם, אֱלִירָז הִסְפִּיק לַעֲבוֹר סוּג שֶׁל מַהְפָּךְ.
מִכּוֹכַב רֵיאָלֵיטִי הוּא פִּתְאוֹם הָפַךְ לְמִישֶׁהוּ שֶׁאֲנָשִׁים רוֹצִים לְהַקְשִׁיב לַדֵּעוֹת הַפּוֹלִיטִיּוֹת שֶׁלּוֹ.
אֵיךְ אֶת מַסְבִּירָה אֶת הַשִּׁינּוּי הַזֶּה?
"זֶה נֶחְמָד שֶׁאֲנָשִׁים קוֹרְאִים לְזֶה מַהְפָּךְ אֲבָל אֱלִירָז הָיָה כָּכָה תָּמִיד.

------------------
אוּצִ'י סִיפֵּר שֶׁבַּאֲרוּחָה הוּא הִתְבַּקֵּשׁ לְהַצְבִּיעַ נֶגֶד רֵגוּלַצְיוֹת בְּשׁוּק הַתִּקְשׁוֹרֶת שֶׁמֶּמְשַׁלְתּוֹ הַשַּׁמְרָנִית בִּיקְּשָׁה לְהַעֲבִיר.
בִּתְמוּרָה, אָמְרוּ לוֹ, יִזְכֶּה לְ"יְחָסִים מְיוּחָדִים", סִיקּוּר תִּקְשׁוֹרְתִּי חִיּוּבִי, וּבְקִיצּוּר: "אֲנַחְנוּ נִדְאַג לְךָ".

אוּצִ'י זָכָר שֶׁבְּאוֹתָהּ מִסְעָדָה סַעַד בְּאוֹתָהּ שָׁעָה גַּם לֶכְלָּן מַרְדּוּק, בְּנוֹ שֶׁל רוּפֵּרְט.

------------------
מוּצְרֵי הַטִּיפּוּחַ שֶׁל הַחֶבְרָה עֲדַיִין לֹא נִמְכָּרִים בַּסּוֹפֵר פָארְם אֲבָל הַבָּאזְז שֶׁלָּהֶם מְחַיֵּיב הֶיכֵּרוּת:

הֵסֵבָּה אֶת תְּשׂוּמֶת לִיבֵּנוּ מַסֵּיכַת הַפָּנִים Pink Clay, הַמְּנַקָּה וּמְסַלֶּקֶת רְעָלִים מֵהָעוֹר.
הִיא מוּמְלֶצֶת לְשִׁימּוּשׁ בֵּין פַּעַם לְפַעֲמַיִים בְּשָׁבוּעַ לְאַחַר הֲסָרַת הָאִיפּוּר, וְהַאֲמִינוּ לָנוּ, הִיא עוֹשָׂה פְּלָאִים!


------------------
"מֵיְיק-אַפּ רֶבוֹלוּשֵׁן" שָׂמָה לְעַצְמָהּ מַטָּרָה לִהְיוֹת אַחַת מֵהַחֲבֵרוֹת הַלָּלוּ.
הַמּוּתָג, שֶׁבְּדוֹמֶה לְ"פִּיקְסִי" ו"ג'ו מְלוֹן לוֹנְדוֹן", הִגִּיעַ אֵלֵינוּ כָּל הַדֶּרֶךְ מֵהַמַּמְלָכָה הַבְּרִיטִית, רוֹאָה עַצְמוֹ כְּמַגְדִּיר מֵחָדָשׁ אֶת גְּבוּלוֹת הַיּוֹפִי תַּחַת הַסִּיסְמָא "We are all shades" (אֲנַחְנוּ כָּל הַגְּוָונִים) לְלֹא קֶשֶׁר לְגִיל, מִגְדָּר אוֹ צֶבַע עוֹר - כּוּלָּם מוּזְמָנִים.

------------------
בָּרוּר לְגַמְרֵי שֶׁהַמֶּשֶׁק הַכַּלְכָּלִי כּוּלּוֹ לֹא יָכוֹל לְתַפְקֵד בְּאוֹפֶן נוֹרְמָלִי בִּזְמַן מִלְחָמָה, אֲבָל הָאוּכְלוּסִיָּיה צְרִיכָה לְקַבֵּל חַשְׁמַל, מַיִם וּמָזוֹן, בָּתֵּי הַחוֹלִים וְהַשֵּׁירוּתִים הָרְפוּאִיִּים וְהַמִּשְׁטָרָה צְרִיכִים לְתַפְקֵד, וְכָךְ גַּם שֵׁירוּתֵי הַחֵירוּם וְהַהַצָּלָה שֶׁל פִּיקּוּד הָעוֹרֶף.


------------------
בְּצַהַ"ל הִדְגִּישׁוּ: "כּוֹחוֹת פִּיקּוּד הַצָּפוֹן נִמְצָאִים בִּדְרִיכוֹת גְּבוֹהָה לְכָל תַּרְחִישׁ, צַהַ"ל רוֹאֶה בַּמִּשְׁטַר הַסּוּרִי אַחְרַאי לְכָל פְּעוּלָּה הַנַּעֲשֵׂית בְּשִׁטְחוֹ וְלֹא יְאַפְשֵׁר כָּל פְּגִיעָה בָּרִיבּוֹנוּת שֶׁל מְדִינַת יִשְׂרָאֵל".

------------------
"שָׁבוּעַ אַחַר כָּךְ בָּאתִי בַּבּוֹקֶר לַחֲנוּת, וּבַעַל הַבַּיִת הָיָה שָׁם.
אָמַרְתִּי לוֹ שֶׁאֲנִי רוֹצֶה לַעֲבוֹד עַל הַקּוּפָּה, וְהוּא צָחַק וְאָמַר שֶׁזֶּה קָשֶׁה וְשֶׁצָּרִיךְ לִלְמוֹד אֶת זֶה, שֶׁלֹּא כָּל אֶחָד יָכוֹל.
אָמַרְתִּי לוֹ שֶׁכְּבָר לָמַדְתִּי וְהוּא יָכוֹל לִבְדּוֹק אוֹתִי.
שְׁעָתַיִים עָמַדְתִּי בְּיַחַד אִיתּוֹ עַל הַקּוּפָּה, וְהוּא הָיָה בְּהֶלֶם וְאָמַר לִי שֶׁיִּרְשׁוֹם אוֹתִי בְּשִׁיבּוּץ הָעֲבוֹדָה.

------------------
מִנֶּגֶד סוֹנִי יְדָעָהּ לְיַיצֵּר לְעַצְמָהּ מַסְפִּיק מִשְׂחָקִים יִיחוּדִיִּים לִפְלַטְפוֹרְמָה שֶׁלָּהּ כְּדֵי שֶׁהַשַּׂחְקָנִים יַשְׁקִיעוּ וְיִרְכְּשׁוּ אֶת הַפְּלֶיְיסְטֵיְישְׁן כְּדֵי לֵיהָנוֹת מֵהֶם.


------------------
בְּעִקְבוֹת אֵירוּעֵי קַיִץ 2006 הִתְחַדְּדָה הַשְּׁאֵלָה מִי אַחְרַאי לִתְקוּפוֹת מִלְחָמָה וְחֵירוּם וְעַל הָרְצִיפוּת הַתִּפְקוּדִית שֶׁל הָעוֹרֶף הָאֶזְרָחִי וְהָעוֹרֶף הַצְּבָאִי.
נַעֲשָׂה בָּרוּר לְגַמְרֵי שֶׁצָּרִיךְ בַּעַל בַּיִת, שֶׁיְּכַוֵּון וְיִתְעַדֵּף אֶת פְּעִילוּת הָרָשׁוּיוֹת הָאֶזְרָחִיּוֹת וְהַצְּבָאִיּוֹת וְיִתְאַם בֵּינֵיהֶן
יְחִידַת הַחִילּוּץ וְהַהַצָּלָה שֶׁל פִּיקּוּד הָעוֹרֶף.
כְּשִׁירוּת נְמוּכָה בְּמִילּוּאִים (צִילּוּם: דּוֹבֵר צַהַ"ל)

------------------
יָכוֹל לִהְיוֹת שֶׁשְּׁמַעְתֶּם לָאַחֲרוֹנָה עַל הַמְּכַשֵּׁף (The Witcher) בִּזְכוּת סִדְרַת הַפַנְטַזְיָה הַחֲדָשָׁה שֶׁל נֶטְפְלִיקְס בַּעֲלַת כּוֹתָר זֵהָה, אַךְ לֹא בָּטוּחַ שֶׁיְּדַעְתֶּם שֶׁמְּדוּבָּר בְּסִדְרָה הַמְּבוּסֶּסֶת עַל סִפְרִי פַנְטַזְיָה מֵאֵת הַסּוֹפֵר הַפּוֹלָנִי אַנְדְּזֵ'יי סַפְּקוֹוֹסְקִי (Andrzej Sapkowski), שֶׁהָפְכָה גַּם לְסִדְרַת מִשְׂחֲקֵי וִידֵאוֹ מַצְלִיחָה.

------------------
הַמֵּרוֹץ יִתְקַיֵּים בְּיוֹם שִׁישִּׁי, הָ־29 בְּיוּלִי.
הַרְשָׁמָה וּפְרָטִים: מָרָתוֹן יִשְׂרָאֵל.
------------------
אֵיךְ זֶה לְהִיכָּנֵס לְנַעֲלַיִים שֶׁל אֶסְתִּי גִּינְזְבּוּרְג? בְּכָל זֹאת, הִיא הָיְיתָה הַפְּרֵזֶנְטוֹרִית בְּמֶשֶׁךְ 8 עוֹנוֹת.
"אֲנִי לֹא מִתְחַבֶּרֶת לַמּוּנָּח שֶׁל 'לְהִכָּנֵס לְנַעֲלַיִים שֶׁל מִישֶׁהוּ'.
גַּם לֹא לַנַּעֲלַיִים שֶׁלִּי.
זוֹ הָעֲבוֹדָה שֶׁלָּנוּ, אִם זֶה לֹא הָיָה עוֹבֵד כָּכָה - לֹא הָיִינוּ עוֹבְדִים.
כָּל אַחַת עוֹשֶׂה כַּמָּה וְכַמָּה עוֹנוֹת וְאַחַר כָּךְ יֵשׁ רִעֲנוּן וְתַחֲלוּפָה.
מַחֲלִיפִים אוֹתִי, הֶחֱלַפְתִּי וְיַחֲלִיפוּ אוֹתִי גַּם בֶּעָתִיד.

------------------
כֵּן מַעֲרֶכֶת בְּחִירוֹת שְׁלִישִׁית אוֹ לֹא מַעֲרֶכֶת בְּחִירוֹת שְׁלִישִׁית, הַהִידַּרְדְּרוּת הִיא אֶפְשָׁרוּת שֶׁחַיָּיבִים לְהִתְכּוֹנֵן אֵלֶיהָ בִּטְוָוח הַזְּמַן הַמִּיָּידִי.
לָכֵן אַצְבִּיעַ עַל הַלִּיקּוּיִים הָעִיקָּרִיִּים בְּשִׁישָּׁה תְּחוּמִים שֶׁבָּהֶם תְּלוּיָה מוּכָנוּת מְדִינַת יִשְׂרָאֵל וְהָעוֹרֶף לַמִּלְחָמָה.

------------------
אֲנִי לֹא אוֹהֶבֶת לְהִישָּׁאֵר עַל אוֹתוֹ עִיצּוּב לְאוֹרֶךְ זְמַן וְלִשְׁמוֹר דְּבָרִים לְאוֹרֶךְ זְמַן".

אָרִיק וְדוּבִּי: "צָרִיךְ לְכַבֵּד אֶת הַמֶּרְחָב הָאִישִׁי אֶחָד שֶׁל הַשָּׁנִי"

אָרִיק וְדוּבִּי הִגִּיעוּ שְׁנֵיהֶם לַתּוֹכְנִית כְּדֵי לִמְצוֹא זוּגִיּוּת פֶּרֶק ב'.
אָרִיק (59) הוּא קַבְּלַן בִּנְיָין מֵחוֹלוֹן, וּמִנִּשּׂוּאָיו הָרִאשׁוֹנִים יֵשׁ לוֹ שְׁנֵי בָּנִים - אֲבִיעָד וְאֶלְדָּר.
הוּא הִתְגָּרֵשׁ לִפְנֵי יוֹתֵר מֵעָשׂוֹר וְשׁוֹמֵר עַל קֶשֶׁר טוֹב עִם אִשְׁתּוֹ לְשֶׁעָבַר.

------------------
זֶה אוּלַי נִשְׁמָע קְצָת נַרְקִיסִיסְטִי, אֲבָל אֶפְשָׁר לְהָבִין לָמָּה אֲנִי מִתְכַּוֶּונֶת.
אֲנָשִׁים הָיוּ מַתְחִילִים לְהַגִּיד עִם מִי דִּיבַּרְתִּי, מְפָרְשִׁים כָּל מַבָּט.
רָצִיתִי שֶׁהַפוֹקוּס יִשָּׁאֵר עַל עַל אֱיָיל וְדָנִיאֵל, וְשֶׁלַּיְּלָדִים שֶׁלִּי יִהְיֶה נָעִים.
אִם הַיְּלָדִים הָיוּ אוֹמְרִים לִי, אֲנַחְנוּ צְרִיכִים אוֹתָךְ שָׁם, בָּא לָנוּ שֶׁתָּבוֹאִי, הָיִיתִי בָּאָה, בָּרוּר.
אֲבָל הֵם אָמְרוּ לִי, אִמָּא, זֶה הֲכִי מוּבָן".


------------------
הָעוּבְדָּה הַשְּׁנִיָּיה הִיא שֶׁלְּרוֹב הָאוּכְלוּסִיָּיה בִּצְפוֹן הָאָרֶץ אֵין מֶרְחָבִים מוּגַנִּים כְּלָל.
הַמַּצָּב חָמוּר בִּמְיוּחָד בְּקֶרֶב הָאוּכְלוּסִיָּיה הָעַרְבִית, אֲבָל גַּם בְּיִישּׁוּבִים אֲחֵרִים אֵין מַסְפִּיק מָמָ"דִים וּמִקְּלָטִים.

------------------
אַתָּה עוֹצֵם עֵינֶיךָ בְּאִי נְעִימוּת, יוֹדֵעַ שֶׁגַּם הָאַרְכִי נְבֵלָה שֶׁעִמָּהּ מִתְמוֹדֵד אַתָּה, גִּיבּוֹר-הָעַל הַיִּשְׂרְאֵלִי, אֵינֶנָּה מֵהַסּוּג הַפוֹטוֹגֵנִי.
לֹא.
לְפָנֶיךָ מִתְיַיצֶּבֶת מְדִינִיּוּת מֶמְשַׁלְתִּית אַפְרוּרִית, שֶׁיָּדִי מְחוֹקְקִים רַבִּים מִדַּי חָבְטוּ בָּהּ, וְהִיא שׁוֹלַחַת זְרוֹעוֹת אַפְרוּרִיּוֹת לָרָשׁוּת הַמְּבַצַּעַת: הַפָּקִיד בְּבִיטּוּחַ לְאוּמִּי, בָּרָשׁוּת הַמְּקוֹמִית, בְּקוּפַּת חוֹלִים.
אַתָּה רוֹצֶה לְהִתְמוֹדֵד אִתָּהּ? בְּבַקָּשָׁה.

------------------
צַהַ"ל תָּקַף הַלַּיְלָה (בֵּין רִאשׁוֹן לְשֵׁנִי) חוּלְיָה שֶׁל אַרְבָּעָה מְחַבְּלִים שֶׁהִנִּיחוּ מִטְעֲנֵי חַבָּלָה עַל קַו הַגְּבוּל עִם סוּרְיָה. הַחוּלְיָה מָנְתָה אַרְבָּעָה מְחַבְּלִים, לְפָחוֹת חָמוּשׁ אֶחָד וְסוּכְּלָה בְּעֵת שֶׁנִּיסָּה לְהַנִּיחַ אֶת הַמִּטְעָן. הָאַרְבָּעָה כְּכָל הַנִּרְאֶה נֶהֶרְגוּ. בְּצַהַ"ל צִייְּנוּ כִּי לֹא הָיוּ נִפְגָּעִים לְכוֹחוֹתֵינוּ. 
------------------
אִם אֶחָד הַמּוּעֲמָדִים יִזְכֶּה לְ-61 חֲתִימוֹת שֶׁל חַבְרֵי כְּנֶסֶת, הוּא זֶה שֶׁיְּקַבֵּל אֶת הַמַּנְדָּט לְהַרְכָּבַת הַמֶּמְשָׁלָה.

בְּכָחוֹל לָבָן הִבְהִירוּ כִּי לֹא יִהְיוּ מוּכָנִים לִפְתּוֹחַ מֵחָדָשׁ אֶת הַהֶסְכֵּם בְּכָל הַנּוֹגֵעַ לְמִינּוּי שַׂר הַמִּשְׁפָּטִים וְהַוַּועֲדָה לִבְחִירַת שׁוֹפְטִים.
"אֵין מוּ"מ בֵּין צִוְותֵי הַמּוּ"מ אֲבָל יֵשׁ רַק דִּיבּוּרִים מִסָּבִיב", אָמַר גּוֹרֵם בְּכָחוֹל לָבָן.
