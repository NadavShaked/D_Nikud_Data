וּכְשֶׁאָמְרוּ לוֹ: מַסְפִּיק עִם הַפַּרְצוּף הֶחָמוּץ, אַתָּה נִרְאֶה כְּמוֹ לִימוֹן, אָז הוּא חָשַׁב שֶׁאֲנָשִׁים סְבִיבוֹ הִשְׁתַּגְּעוּ, אֲבָל לְיֶתֶר בִּטָּחוֹן הָלַךְ לִרְאִי כְּדֵי לִבְדוק שֶׁיֵּשׁ לוֹ פַּרְצוּף רָגִיל שֶׁלֹּא נִרְאֶה כְּמוֹ לִימוֹן.
כְּשֶׁהוּא גָּדַל קְצָת יוֹתֵר, הַמִּשְׂחָקִים נִרְאוּ לוֹ מְשׁוּנִּים.

------------------
הוּא רָץ עַל פְּנֵי הַקַּרְקַע כְּשֵזְנַבוֹ הַאָרוֹךְ מִתְנוֹעֵעַ בְּלִי הֶרֶף.
תְּנוּעוֹת הַזָּנָב מְשַׁמְּשׁוֹת לוֹ גַּם לְנִוּוּט וְגַם לְקֶשֶׁר בֵּין נַחֲלִיאֵלִי אֶחָד לַשֵּׁנִי.
לַנַּחֲלִיאֵלִי מָקוֹר אָרֹךְ וְדַק.
מָקוֹר זֶה מַתְאִים לְצַיִד שֶׁל חֲרָקִים קְטַנִּים כְּמָזוֹן.
וּכְשֶׁהַנַּחֲלִיאֵלִי רָעֵב , הוּא מִתְרוֹצֵץ עַל פְּנֵי הַקַּרְקַע לְחַפֵּשׂ מָזוֹן.
מִדֵי פַּעַם הוּא מִתְעוֹפֵף בַּאֲוִיר כְּדֵי לָצוּד לוֹ זְבוּבִים וְיָתוּשִים.

------------------
יָעֵלִי לֹא הֵבִינָה מָה מַצְחִיק כָּאן.
הַיּוֹם הָרִאשׁוֹן
"אֲנִי לֹא מַאֲמִינָה," אָמְרָה יָעֵלִי לָאִשָּׁה שֶׁעָמְדָה מוּלָהּ, "אַתְּ אֲפִלוּ לֹא יוֹדַעַת עִבְרִית?"
"אֲנִי קְצָת עִבְרִית לָדַעַת," אָמְרָה הָאִשָּׁה.
"קְצָת זֶה לֹא מַסְפִּיק," רָטְנָה יָעֵלִי.
הֵם נָסְעוּ לְקַבֵּל אֶת פָּנֶיהָ בִּשְׂדֵה הַתְעוּפָה וְיָעֵלִי קְצָת הֻפְתְעָה כְּשֶׁרָאֲתָה אֶת הָאִשָּׁה הַגְּבוֹהָה, הָרָזָה וְהַיָּפָה שֶׁחִבְּקָה אֶת הַהוֹרִים שֶׁלָּהּ וְאַחַר כָּךְ גַּם אוֹתָהּ.
כָּכָה נִרְאֵית סָבְתָא?

------------------
לְבַסוֹף, כְּשֶׁחָזַר אֶל הַשַּׁיָרָה עָיֵף וְצָמֵא, נֶחְרְדָה טָאצַ’נְגָה לְגַלוֹת שֶׁהַמִּזְחֶלֶת הַמִּטַלְטֶלֶת מֵאֲחוֹרָיו רֵיקָה.
בְּאוֹתוֹ זְמַן שָׁכְבָה הַתִינֹקֶת הַקְּטַנָה בָּעֵשֶׂב וּבָכְתָה.
קוֹל בִּכְיָה הִגִּיעַ לְעֵדֶר בּוּפָלוֹ שֶׁרָעָה בְּסָמוּךְ.
נְקֵבַת בּוּפָלוֹ הֵרִימָה אֶת רֹאשָׁה וְהֵחֵלָה לְהִתְקָרֵב לַמָּקוֹם בְּסַקְרָנוּת כְּדֵי לְבָרֵר מִיהוּ זֶה שֶׁמַּשְׁמִיעַ קוֹלוֹת מְשֻׁנִים כָּל כָּךְ בָּעֵשֶׂב.

------------------
פִּתְאֹם פּוֹנֶה הָרַב הַקָּדוֹשׁ מֵאֲרוֹן הַקֹדֶשׁ אֶל הַקָּהָל וְשׁוֹאֵל "אֵיפֹה הַצַּדִיק אֲשֶׁר נָקַב וְחָדַר מִבַּעַד לְעַנְנֵי הָעוֹפֶרֶת עַד הַשָּׁמַיִם וּפָרַץ דֶּרֶךְ לִתְפִלוֹתֵינוּ לַעֲבוֹר שָׁמָה?"
אֲבָל הַצַּדִיק כְּבָר לֹא הָיָה בְּבֵית הַכְּנֶסֶת.
הוּא בָּרַח מִשָׁם, חָלַץ אֶת נַעֲלָיו, שָׂם אוֹתָן עַל שִׁכְמוֹ וְהוּא צוֹעֵד בַּדֶּרֶךְ אֶל הַכְּפָר…

------------------
בִּשְׁבִיל מָה? כָּל אֶחָד הָיָה יָכוֹל לְעַרְבֵּב קֶמַח עִם קְצָת מַיִם, לְהוֹסִיף אֶת הַתַבְלִין, לְהַכְנִיס לַתַנוּר וּלְהוֹצִיא אֶת הַלֶּחֶם הַטָּעִים בְּיוֹתֵר בָּעוֹלָם.
אַחֲרֵי עוֹד כַּמָּה יָמִים גִּלוּ הָאֲנָשִׁים שֶׁאֵין יוֹתֵר צֹרֶךְ לֶאֱפוֹת שׁוּם דָּבָר.
אֶפְשָׁר לָקַחַת מְלָפְפוֹן, לְפַזֵר עָלָיו מְעַט מֵהַתַבְלִין, וְכָל נְגִיסָה בּוֹ תִּהְיֶה כְּמוֹ בִּקוּר קָצָר בְּגַן עֵדֶן.
הַיְּלָדִים כְּבָר לֹא הֵבִיאוּ כְּרִיכִים לְבֵית הַסֵּפֶר.

------------------
מָאלָה הוֹדְתָה לְכֻלָם עַל קַבָּלַת הַפָּנִים הַנֶּהֱדֶרֶת, אֲבָל בֵּינָה לְבֵין עַצְמָה חָשְׁבָה שֶׁלֹּא כְּדַאי לְהִקָשֵׁר יוֹתֵר מִדַי לָאֲנָשִׁים הַטּוֹבִים אוֹ לַיְּלָדִים הַחֲמוּדִים כְּדֵי שֶׁלֹּא תִּתְפַּתֶה לְהִשָׁאֵר וּלְוַתֵר עַל נְהַר הַשּׁוֹקוֹלָד שֶׁזּוֹרֵם לוֹ אֵי שָׁם וּמְחַכֶּה לָה.
כְּשֶׁהֵאִיר הַשַּׁחַר עָזְבָה מָאלָה אֶת הַכְּפָר הַמַּקְסִים, וּבְלִי לְחַכּוֹת שֶׁתּוֹשָׁבֵי הַכְּפָר יָקוּמוּ מִשְׁנָתָם נִכְנְסָה שׁוּב לְתוֹךְ הַיַּעַר.
הַנָּהָר

------------------
וְאָבִיו רוֹאֶה אוֹתוֹ מִתַּחַת לְטַלִיתוֹ וְנֶאֱנָח מָרָה: "רִבּוֹנוֹ שֶׁל עוֹלָם! זְכרֹ גַּם יְלָדִים כְּמוֹ יַאשֶׁק!"
וְיַאשֶׁק עוֹמֵד וּמִתְבּוֹנֵן.
כֻּלָם בּוֹכִים, כֻּלָם מִתְפַּלְלִים, כֻּלָם קוֹרְאִים לֶאֱלֹהִים בְּחָזְקָה וְגַם הוּא רוֹצֶה לַעֲשׂוֹת כְּמוֹהֶם.
גַם הוּא רוֹצֶה לְהִתְפַּלֵּל.

------------------
"זֶה בְּסֵדֶר," מִלְמֵל אַלוֹן, עֲדַיִן נָבוֹךְ.
לְמָחֳרָת הֵבִיא אֲרִי לַכִּתָה אַרְבַּע פַּחִיוֹת וּבַקְבּוּק.
"תִרְאֶה, אָסַפְתִי בִּשְׁבִילְךָ," אָמַר לְאַלוֹן וְהוֹשִׁיט לוֹ אוֹתָם.
"מָה זֶה?" הִתְעַנְיְנָה נָעֳמִי.
"אַלוֹן אוֹסֵף בַּקְבּוּקִים רֵיקִים בִּשְׁבִיל הַפִּקָדוֹן," עָנָה לָה אֲרִי לִפְנֵי שֶׁאַלוֹן הִסְפִּיק לְהַשְׁתִיק אוֹתוֹ.
"אָה," חִיְכָה נָעֳמִי, "זֶה מָה-זֶה קָשֶׁה! נִסִיתִי פַּעַם, אֲבָל צָרִיךְ הָמוֹן.
גַם אֲנִי אָבִיא לְךָ אִם אֶמְצָא."

------------------
יֵשׁ לִי מַשֶׁהוּ חָשׁוּב מְאֹד לְהַגִּיד לָךְ!"
הֵצַצְתִי בָּעֵינִית.
וּבֶאֱמֶת, לְיַד דֶּלֶת הַכְּנִיסָה לַבַּיִת עָמְדָה לֹא אַחֶרֶת מֵהַמּוֹרָה שֶׁלִּי לְאַנְגְּלִית.
"חַיָּבִים לִפְתֹחַ…" לָחַשְׁתִי לְמִיכָאֵל.
מָצָאתִי אֶת הַמַּפְתֵחַ וְסוֹבַבְתִי פַּעֲמַיִם.
נִיצָה עָמְדָה לְיַד הַדֶּלֶת כֻּלָהּ חִיוּכִים מְשֻׁנִים.
הִיא לָבְשָׁה שִׂמְלָה צִבְעוֹנִית שֶׁמֵעוֹלָם לֹא רָאִיתִי בְּבֵית הַסֵּפֶר.

------------------
יָדִיד שֶׁל סוֹקְרָטֵס בָּא לְבַקֵר אוֹתוֹ.
"הַאִם שָׁמַעְתָ מָה שֶׁמְּסַפְּרִים עַל חֲבֵרְךָ"? שָׁאַל הַיָּדִיד אֶת סוֹקְרָטֵס.
"לִפְנֵי שֶׁאֶשְׁמַע, אֲנִי רוֹצֶה לָדַעַת אִם סִנַנְתָ אֶת הַשְּׁמוּעָה הַזֹּאת בִּשְׁלוֹשׁ הַמְּסַנְנוֹת", אָמַר סוֹקְרָטֵס.
"אֵילוּ מְסַנְנוֹת"? שָׁאַל הַיָּדִיד.
"הַמְּסַנֶנֶת הָרִאשׁוֹנָה הִיא הַמְּסַנֶנֶת שֶׁל הָאֱמֶת", אָמַר סוֹקְרָטֵס.
"הַאִם מָה שֶׁאַתָּה רוֹצֶה לְסַפֵּר לִי עַל חֲבֵרִי הוּא אֱמֶת?"

------------------
אִם בָאָרוֹן הָיוּ מֻנָּחוֹת כְּזוּג מַתְאִים בְּאוֹתָם הַצְּבָעִים ,עַד שֶׁלָּבַשׁ אוֹתָן צָ'אפָּאצ'וּל: לִפְעָמִים גֶּרֶב עִם עִגּוּלִים ברֶגֶל יָמִין וְעַל רֶגֶל שְׂמֹאל- גֶּרֶב עִם פַּסִים אֲדֻמִּים.

לִנְעֹל נַעֲלַיִם- הוּא לֹא אָהַב בִּכְלָל, לָרֹב הֶעֱדִיף לָלֶכֶת יָחֵף אוֹ לְפָחוֹת בַּסַּנְדָּל וְכָךְ הִסְתּוֹבֵב סָבִיב בַּסְּתָו, בַּקַּיִץ וּבָאָבִיב, וְרַק בַּחֹרֶף כְּשֶׁגֶּשֶׁם יָרַד, נָעַל אוֹ מַגָּפַיִם חַמּוֹת בִּמְיֻחָד.


------------------
"אָסוּר לָלֶכֶת לִישֹׁן מְאֻחָר!"
(מְמְמְ… נִרְאֶה)
"אָסוּר לֶאֱכֹל מַמְתַּקִּים בְּלַיְלָה!"
(הִיא מַכִּירָה אוֹתִי, הָאִשָּׁה הַזֹּאת)
"אַתֶּם חַיָּבִים לְהִתְקַשֵׁר עִם כָּל בְּעָיָה קְטַנָה!"
(לְכוּ כְּבָר!)
"שֶׁלֹּא נִמְצָא אֶת אַרְתוּר בַּמִּטָה שֶׁל אֶחָד מִכֶּם!"
(וְאִם הוּא יַעֲלֶה אַחֲרֵי שֶׁנֵרָדֵם?)
"אָסוּר לְהַדְלִיק תַּנוּר כְּשֶׁאֲנַחְנוּ לֹא בַּבַּיִת!"
(לָנוּ תָּמִיד חַם.
רַק בִּגְלַלְכֶם יֵשׁ בַּבַּיִת תַּנוּר)
"אָסוּר לְהִשְׁתוֹלֵל!"
"אָסוּר לִפְתֹחַ טֵלֵוִיזְיָה!"
"אָסוּר!"
"אָסוּר!"

------------------
אֲחוֹתוֹ קָשְׁרָה אֶת כַּדוּר-הָאוֹר בִּשְׂרוֹךְ וְהַיֶּלֶד נֶהֱנָה מְאֹד מֵהַמִּשְׂחָק.
הוּא מָשַׁךְ בָּרְצוּעָה וְגִלְגֵל אֶת כַּדוּר-הָאוֹר.
זֶה הָיָה מִשְׂחָק מְעַנְיֵן.
כַּעֲבֹר זְמַן קָצָר יָצְאוּ הַמַּנְהִיג וּבִתוֹ מֵהָאֹהֶל וְהִשְׁאִירוּ אֶת יְרִיעַת הַכְּנִיסָה פְּתוּחָה.
זוֹ הָיְתָה הַהִזְדַּמְנוּת שֶׁל הָעוֹרֵב.
הוּא קָפַץ הַחוּצָה מֵאָזְנוֹ שֶׁל הַיֶּלֶד בִּדְמוּתוֹ כְּגַרְגִיר אָבָק וְהָפַךְ אֶת עַצְמוֹ שׁוּב לְעוֹרֵב.

------------------
רַק אַחַר כָּךְ, כְּשֶׁהַיֶּלֶד הוֹפִיעַ בַּטֶּלֵוִיזְיָה וְסִפֵּר מָה בֶּאֱמֶת קָרָה, הֵם הֵבִינוּ.
מִילְיוֹן מַצְלֵמוֹת טֵלֵוִיזְיָה סִנְוְרוּ אוֹתִי כְּשֶׁהִצְלִיחוּ לְחַלֵץ אוֹתִי, עָטְפוּ אוֹתִי בִּשְׂמִיכָה וְנָתְנוּ לִי לִשְׁתּוֹת.
הַפֶּה שֶׁלִּי הָיָה יָבֵשׁ וּמָלֵא בְּעָפָר.

------------------
זֶה הָיָה מְשֻׁנֶה לִהְיוֹת רַק שְׁתֵיהֶן בַּבַּיִת, אוֹ בַּבַּיִת הֶחָדָשׁ שֶׁל אַבָּא בַּיָּמִים שֶׁהָיְתָה נִשְׁאֶרֶת אֶצְלוֹ.
הַדַּירָה שֶׁל אַבָּא הָיְתָה קְצָת מְבֻלְגֶנֶת וְלֹא הִרְגִישָׁה כְּמוֹ בַּיִת בִּכְלָל.
עֶרֶב אֶחָד, כְּשֶׁאִמָּא וּמִיכַל הָיוּ לְבַד בַּבַּיִת, מִיכַל שׁוּב שָׁאֲלָה אֶת אִמָּא עַל הַכֶּלֶב.
"טוֹב," אָמְרָה אִמָּא בְּחִיּוּךְ עִם אֲנָחָה, "מָחָר נֵלֵךְ לַכַּלְבִּיָה וְנִקַח כֶּלֶב."
"כַּלְבָּה," תִּקְנָה מִיכַל וְחִבְּקָה אֶת אִמָּא בְּאֹשֶׁר.

------------------
הָיָה לוֹ רֹאשׁ מָלֵא תַּלְתַּלִים אֲבָל הוּא לֹא אָהַב לְהִסְתָּרֵק וְאֶת הַכִּפָּה הָיָה בְּסִכָּה מְהַדֵּק אֲבָל כְּשֶׁהָיָה עִם חֲבֵרָיו מְשַׂחֵק הַכִּפָּה הָיְתָה לִפְעָמִים נוֹפֶלֶת הַרְחֵק! וּכְשֶׁחָזַר הַבַּיְתָה גַּם אֶת הַגָּרְבַּיִם לֹא תָּמִיד לָבַשׁ בְּסֵדֶר, כִּי לֹא פַּעַם הֵן הָיוּ בִּמְקוֹמוֹת שׁוֹנִים בַּחֶדֶר.


------------------
אֲבָל הוּא לֹא רוֹאֶה וְלֹא שׁוֹמֵעַ כְּלוּם.
הַמְּכוֹנִית נוֹסַעַת הָלְאָה – וְעוֹצֶרֶת בַּרַמְזוֹר הַבָּא.
וְאָז שׁוֹאֵל רוֹן אֶת אַבָּא:
"תַגִיד לִי אַבָּא, רַמְזוֹר יוֹדֵעַ לְדַבֵּר? יֵשׁ לוֹ פֶּה?"
"מוּבָן שֶׁרַמְזוֹר לֹא יוֹדֵעַ לְדַבֵּר.
אֵין לוֹ פֶּה," עוֹנֶה אַבָּא.
עַכְשָו יֵשׁ לְרוֹן שְׁאֵלָה חֲשׁוּבָה:
"תַגִיד לִי, אַבָּא,"הוּא שׁוֹאֵל "אִם לָרַמְזוֹר אֵין פֶּה, וְהוּא לֹא יוֹדֵעַ לְדַבֵּר, אִיךְ יוֹדְעִים כָּל הַנֶּהָגִים מָתַי לַעְצוֹר וּמָתַי לִנְסוֹעַ? וְאֵיך

------------------
בָּאוֹטוֹ, בַּדֶּרֶךְ חֲזָרָה, כְּשֶׁכְּבָר הִתְרַחֲקוּ מֵהַמָּקוֹם הֶעָצוּב הַזֶּה, הִתְחִיל אַלְפְרֶד לְלַקֵק אֶת מִיכַל בְּהִתְלַהֲבוּת.
מִיכַל צָחֲקָה.
זֶה הָיָה מְדַגְדֵג.
"תִּרְאִי אִמָּא, הוּא מְחַיֵךְ," קָרְאָה מִיכַל.
"אֵיזֶה יוֹם מְאֻשָׁר," אָמְרָה אִמָּא וְחִיְכָה אֶל מִיכַל דֶּרֶךְ הַמַּרְאָה הַגְּדוֹלָה שֶׁבָּאוֹטוֹ.
מִיכַל הֵשִׁיבָה לָהּ חִיּוּךְ וְאָמְרָה, "הַיּוֹם הֲכִי מְאֻשָׁר בָּעוֹלָם!"
------------------
"אַבָּא תִּקְנֶה לִי מְרִיצָה בַּחֲנוּת הַצַּעֲצוּעִים, וְגַם אוֹטוֹ מָנוֹף
"אַבָּא תִּקְנֶה לִי מְרִיצָה בַּחֲנוּת הַצַּעֲצוּעִים, וְגַם אוֹטוֹ מָנוֹף וּמַשָׂאִית גַּם.
טוֹב אַבָּא?" כָּךְ אוֹמֵר רוֹן שׁוּב וָשׁוּב, וְחוֹזֵר עַל כָּךְ בְּלִי סוֹף
אֲבָל אַבָּא חוֹשֵׁב לוֹ וּבִכְלָל לֹא שָׂם לֵב.
אַחַר כַּךְ הוּא מְדַבֵּר עִם אִמָּא וְלֹא מַקְשִׁיב לַבַּקָשָׁה.
וּכְשֶׁהוּא עוֹנֶה לְרוֹן סוֹף סוֹף, הוּא אוֹמֵר:
"לֹא עַכְשָׁו, רוֹן.

------------------
כְּשֶׁהִגִּיעַ עָלִי בַּבָּא לְבֵיתוֹ, הֶרְאָה אֶת מַטְבְּעוֹת הַזָּהָב לְאִשְׁתּוֹ וְסִפֵּר לָהּ עַל כָּל מַה שֶּׁקָּרָה.
אָמְרָה אִשְׁתּוֹ:
״עָלִי בַּבָּא בַּעֲלִי, נִשְׁתַּמֵּשׁ בִּמְעַט זָהָב בְּכָל יוֹם, וְאַתָּה תָּבִיא זָהָב מִן הַמְּעָרָה רַק כַּאֲשֶׁר יִגָּמֵר הַזָּהָב בְּבֵיתֵנוּ.
וּכְדֵי שֶׁנּוּכַל לָדַעַת בְּדִיּוּק בְּכַמָּה זָהָב הִשְׁתַּמַשְׁנוּ, הָבֵא מֵאָחִיךָ קָאסֶם אֶת הַכְּלִי שֶׁבּוֹ שׁוֹקְלִים וּמוֹדְדִים כַּמֻּיּוֹת."

------------------
- זוֹ סְתָם מַטְבֵּעַ, זֶה סְתָם מִשְׂחָק.
אִם בָּא לְךָ לְהַפְסִיק אָז תַּפְסִיק וְתַחֲזוֹר הַבַּיְתָה.
- זֶה אָמְנָם סְתָם מְשַׂחֵק, אֲבָל מָה אִם הַמַּטְבֵּעַ תּוֹבִיל אוֹתִי לַמָּקוֹם שֶׁאֲנִי צָרִיךְ לְהַגִּיעַ, חֲבָל לְפַסְפֵס אִם הִגַּעְתִּי עַד לַכָּאן.
- אַתָּה לֹא מַגְזִים? הַמַּטְבֵּעַ הַזּוֹ לֹא מוֹבִילָה אוֹתְךָ לְשׁוּם מָקוֹם...
אַתָּה מַחֲלִיט לָלֶכֶת לְפִי הַמַּטְבֵּעַ וְלא הַמַּטְבֵּעַ מַחֲלִיטָה לְאָן תֵּלֵךְ.
