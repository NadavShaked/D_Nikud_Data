יֵשׁ לְצַיֵן, שֶׁבְּסָעוּדְיָה יֶשְׁנָם חֻקִים קָשִׁים בְּיוֹתֵר. לַגַנָבִים כּוֹרְתִים יָדַיִם, לְנָשִׁים אָסוּר לִנְהוֹג בִּמְכוֹנִיוֹת וְאָסוּר לְהֵן לָשֶׁבֶת בְּמִסְעָדָה בְּלִי שֶׁוִּילוֹן עָבֶה מַסְתִיר אוֹתָן. הַנָּשִׁים בְּסָעוּדְיָה חַיָּבוֹת לְכַסוֹת אֶת כָּל גּוּפָן כַּאֲשֶׁר הֵן יוֹצְאוֹת לָרְחוֹבוֹת. גַם הֵן מַתְחִילוֹת לִדְרוֹשׁ יוֹתֵר חֹפֶשׁ וְרוֹצוֹת לָצֵאת לָעֲבוֹדָה כְּמוֹ הַגְּבָרִים.

הַפְּסִיכוֹלוֹגִים מַכִּירִים תּוֹפָעוֹת דּוֹמוֹת אֲחֵרוֹת הַקְּשׁוּרוֹת לְתִקְשֹׁרֶת, כְּמוֹ הַשִּׁמּוּשׁ בָּאִינְטֶרְנֵט. עַל הַתּוֹפָעָה הַזּוֹ, בְּמַאֲמָר אַחֵר.

------------------
אַבְרָהָם בּוּרְג, יו"ר הַכְּנֶסֶת אָמַר: "אֵין טַנְק אוֹ מָטוֹס בָּעוֹלָם שֶׁיָכוֹל לְהָגֵן עַל חֶבְרָה שֶׁנִּפְגַעַת מִבִּפְנִים. אֲנַחְנוּ מִתַּחַת לַקַו הָאָדֹם בַּתְחוּם הַחֶבְרָתִי וַאֲנַחְנוּ חַיָּבִים לְשַׁנוֹת אֶת הַמַּצָב".
הַחוֹקְרִים מָצְאוּ כִּי בָּאָרֶץ יֵשׁ מִסְפָּר קָטָן שֶׁל אֲנָשִׁים הַמַּחְזִיק בְּרֹב הַכְּסָפִים בַּמְּדִינָה. עוֹד קָבַע הַדּוּ"חַ, שֶׁבְּכָל שָׁנָה נוֹסָפוֹת עוֹד 50 אֶלֶף מִשְׁפָּחוֹת לְמַעְגַל הָעֹנִי.

------------------
אֵיךְ זֶה קוֹרֶה? כַּאֲשֶׁר אֲנַחְנוּ מְשַׁקְּרִים, הַגּוּף שֶׁלָּנוּ נִמְצָא בְּמַצָּב לֹא טִבְעִי. הַגּוּף מֵגִיב בְּצוּרָה הַכּוֹלֶלֶת הַזְּרָמַת דָּם לְאֵזוֹר הַפָּנִים. זוֹ אוּלַי הַסִּבָּה שֶׁאֲנַחְנוּ מַסְמִיקִים כַּאֲשֶׁר אֲנַחְנוּ מְשַׁקְּרִים. הַזְּרָמַת הַדָּם הַזּוֹ מַגִּיעָה גַּם לָאַף שֶׁמִּתְרַחֵב וּמִתְאָרֵךְ בִּמְעַט.
אָז חֲבֵרִים, לֹא כְּדַאי שֶׁתְּשַׁקְּרוּ, וְאִם אַתֶּם עוֹשִׂים זֹאת, תַּסְתִּירוּ מְעַט אֶת הָאַף, כִּי אַחֶרֶת יֵדְעוּ שֶׁאַתֶּם מְשַׁקְּרִים.

------------------
בְּקָרוֹב, תִּהְיֶה לָכֶם אֶפְשָׁרוּת נוֹסֶפֶת – לְהַדְּפִּיס אֶת הַצַּלָחוֹת. הַכַּוָנָה לְהַדְּפִּיס צַלָחוֹת אֲמִתִיוֹת, וְלֹא עַל הַנְּיָר. חֶבְרָה יִשְׂרְאֵלִית הִמְצִיאָה אֶת הַמַּדְפֶּסֶת הַמְּשֻׁכְלֶלֶת, שֶׁיְכוֹלָה לְהַדְּפִּיס קֻפְסָאוֹת, נַעֲלַיִם, מִבְרְשׁוֹת שִׁנַיִם וְכִמְעַט כָּל דָּבָר הָעוֹלֶה עַל רוּחֲכֶם.
אִם כֵּן, כָּל מַה שֶׁנִּשְׁאַר לָנוּ לַעֲשׂוֹת, הוּא לְהַזְּמִין מַדְפָּסוֹת כְּאֵלֶה וּלְהַתְּחִיל לַחְגוֹג.

הַדּוֹלָר בִּבְעָיוֹת
הַדּוֹלָר נֶחְשַׁב לַמַּטְבֵּעַ הֶחָזָק בְּיוֹתֵר בָּעוֹלָם. גַם הַפּוֹשְׁעִים יוֹדְעִים אֶת הָעֻבְדָה הַזּוֹ. לָכֵן, הֵם מִתְאַמְצִים מְאֹד לְזַיֵּף דּוֹלָרִים. בְּקוֹלוּמְבְּיָה הִתְגַלָה בֵּית דְּפוּס לְיִצוּר דּוֹלָרִים. הַפּוֹשְׁעִים הִצְלִיחוּ לְהַדְּפִּיס שְׁטָרוֹת בִּסְכוּם שֶׁל 20 מִילְיוֹן דּוֹלָרִים, עַד שֶׁנִּתְפְּסוּ.
בַּתְּמוּנָה: שׁוֹטֵר מְקוֹמִי בּוֹדֵק אֶת הַכַּמוּת הָעֲצוּמָה שֶׁל הַדּוֹלָרִים הַמְּזֻיָפִים.

מִסִימָנֵי הַמַּצָב הַכַּלְכָּלִי הַקָּשֶׁה
100 אֶלֶף מִשְׁפָּחוֹת נֻתְקוּ מֵהַחַשְׁמַל
חֶבְרַת הַחַשְׁמַל נִתְקָה כ-100 אֶלֶף מִשְׁפָּחוֹת מֵהַחַשְׁמַל בַּשָׁנָה הָאַחֲרוֹנָה. הַנִּתוּקִים בֻּצְעוּ לְאַחַר שֶׁאוֹתָן מִשְׁפָּחוֹת לֹא שִׁלְמוּ אֶת הַתַּשְׁלוּם הַחָדְשִׁי.
בְּיָמִים אֵלֶה, בּוֹדְקִים הַצָּעָה בַּכְּנֶסֶת, לֶאֱסוֹר עַל חַבְרַת הַחַשְׁמַל לְנַתֵק לָאֲנָשִׁים אֶת הַזֶּרֶם.

------------------
שַׂר הַבִּטָּחוֹן הֶחְלִיט, שֶׁהַמַּאֲחָזִים הַלֹּא חֻקִּיִּים צְרִיכִים לְהֵהָרֵס, וְהוּא שָׁלַח אֶת הַצָּבָא לְפַנּוֹת אֶת הָאֲנָשִׁים הַנִּמְצָאִים בָּהֶם וְלַהֲרֹס אֶת הַמִּבְנִים הַנִּמְצָאִים שָׂם.
הַמִּתְנַחֲלִים, הַחַיִּים בַּמַּאֲחָזִים, מְסָרְבִים לַעֲזֹב אֶת הַמָּקוֹם וּמִתְנַגְּדִים בְּכֹחַ לִפְעֻלּוֹת הַפִּנּוּי שֶׁל הַצָּבָא וְהַמִּשְׁטָרָה.
בְּחֵלֶק מֵהַמַּאֲחָזִים, נַעֲשָׂה פִּנּוּי הַמִּתְנַחֲלִים בְּשֶׁקֶט וּלְלֹא אַלִּימוּת.

------------------
גַם אֶת הַתַּפְרִיט אִי אֶפְשָׁר לִקְרוֹא וְהַמֶּלְצָרִים מְסַפְּרִים לַסוֹעֲדִים הָרְעֵבִים מַה נִתַן לְהַזְּמִין. כַּאֲשֶׁר מְקַבְּלִים אֶת הָאֹכֶל, הַמֶּלְצַר מַסְבִּיר לַסוֹעֵד, שֶׁאֵינוֹ רוֹאֶה דָּבָר, מַה יֵשׁ בַּצַלַחַת, וְאֵיךְ לֶאֱכוֹל מִבְּלִי לְהִתְלַכְלֵךְ אוֹ לִשְׁפּוֹךְ אֶת כּוֹסוֹת הַיַּיִן.

------------------
קֲצִין מִשְׁטָרָה בְּשֵׁם הָאנְק רַפֵּרְטִי, הַלֹּא הוּא סְטִיב זָ'אן, עוֹצֵר אֶת מַרְטִין לוֹרֵנְס, בַּחֲשָׁד שֶׁהוּא פָּרַץ לָרֶכֶב שֶׁלוֹ. לוֹרֵנְס, שְׁחוּם הָעוֹר, מְנַסֶּה לְשַׁכְנֵעַ אֶת קֲצִין הַמִּשְׁטָרָה הָעַצְבָּנִי, שֶׁהוּא טוֹעֶה. וְכָאן נִכְנֶסֶת הַדְּבוֹרָה, שֶׁמְּשַׁנָה לַשְׁנַיִם אֶת הַחַיִּים. הָאנְק, שֶׁרוֹצֶה לְהִפָּטֵר מֵהַדְּבוֹרָה, מְנוֹפֵף בְּיָדָיו כְּדֵי לְסַלֵק אוֹתָה. אוּלָם, מֵרָחוֹק זֶה נִרְאָה כְּאִלּוּ הוּא מַכֶּה אֶת הָאִישׁ הַשָּׁחֹר שֶׁלְיָדוֹ. 
------------------
כְּדֵי שֶׁלֹּא יִקְרֶה מַצָּב שֶׁלְּתוֹשָׁבֵי הַמְּדִינָה הַחֲדָשָׁה יִמָּאֵס מֵהַנָּשִׂיא הַמּוּזָר שֶׁלָּהֶם, הוּא קָבַע שֶׁהוּא יִהְיֶה נָשִׂיא לְכָל יְמֵי חַיָּיו. בְּהַחְלָטָה זוֹ תָּמְכוּ כָּל חַבְרֵי הַפַּרְלָמֶנְט שֶׁל טוּרַקְמָנִיסְטָאן.

------------------
לְמַעֲשֶׂה, דרק וְאִמּוֹ, לזלי, אֵינָם חוֹלִים כְּלָל. הֵם מִשְׁתַּתְּפִים בְּנִסּוּי שֶׁאוּלַי יָבִיא לְכָךְ, שֶׁכָּל אָדָם בָּעוֹלָם יִשָּׂא מִתַּחַת לָעוֹר שֶׁלּוֹ שְׁבָב מְיֻחָד עִם כָּל הַפְּרָטִים הָרְפוּאִיִּים שֶׁלּוֹ. אֹמֶר דר' ווּלְקָן, מִי שֶׁמְּבַצֵּעַ אֶת הַשְּׁתָּלוֹת הַשְּׁבָבִים: "לִפְעָמִים, מַגִּיעִים לְבֵית הַחוֹלִים אֲנָשִׁים שֶׁלֹּא יְכוֹלִים לְדַבֵּר. אֲנַחְנוּ לֹא יוֹדְעִים עֲלֵיהֶם דָּבָר וְקָשֶׁה לָנוּ לְטַפֵּל בָּהֶם. 
------------------
הַפִּתְגָם הָרִאשׁוֹן לָקוּחַ מהתנ"ך. כַּאֲשֶׁר אָדָם יוֹצֵא לְמִלְחָמָה, הוּא חוֹגֵר אֶת הַחֲגוֹר שֶׁלוֹ. (חֲגוֹר – בֶּגֶד צְבָאִי מְיֻחָד). לְאַחַר שֶׁהוּא חוֹזֵר מֵהַקְּרָב, הוּא פּוֹתֵחַ אֶת הָרְצוּעוֹת שֶׁל הַחֲגוֹר וְהוֹלֵךְ לָנוּחַ.
הַפִּתְגָם בָּא לְהַזְּהִיר אֶת הַלּוֹחֵם, שֶׁלֹּא יַחְגֹג לִפְנֵי הַקְּרָב, אֶלָא רַק לְאַחַר שֶׁהוּא סִיֵם אֶת הַמִּלְחָמָה.

------------------
הַיְּלָדִים, שֶׁאֵינָם יְכוֹלִים לְהָגֵן עַל עַצְמָם, נִלְקָחִים לַעֲבוֹדוֹת קָשׁוֹת. הַתַּשְׁלוּם שֶׁהֵם מְקַבְּלִים קָטָן בְּיוֹתֵר וּפְעָמִים רַבּוֹת הֵם אֲפִלּוּ לֹא מְקַבְּלִים תַּשְׁלוּם כְּלָל. חֵלֶק מֵהֶם נִפְגַּע בַּעֲבוֹדוֹת וְנִשְׁאַר נָכֶה לְכָל חַיָּיו. הַסִּכּוּיִים שֶׁל יְלָדִים אֵלֶּה, שֶׁלֹּא לוֹמְדִים, לְהַצְלִיחַ בְּחַיִּים, קָטָן מְאֹד. הָאוּ"ם מְנַסֶּה לְהָגֵן עַל אוֹתָם יְלָדִים, שֶׁרוּבָּם נִמְצָאִים בְּמִזְרַח אַסְיָה וּבְאַפְרִיקָה.

נוֹשֵׂא אַחֵר שֶׁיִשְׂרָאֵל מוֹבִילָה בּוֹ הוּא הַטֶּלֶפוֹנִים הַנַּיָדִים. עַל פִּי מִמְצְאֵי הַסֶּקֶר, ל-80% מֵהַיִּשְׂרְאֵלִים יֵשׁ טֶלֶפוֹנִים נַיָּדִים. יִשְׂרָאֵל נִמְצֵאת בַּמָּקוֹם הָרְבִיעִי בָּעוֹלָם בִּתְחוּם זֶה. לְפָנֶיהָ נִמְצָאוֹת אִיטַלְיָה, הוֹנְג-קוֹנְג וְלוּקְסֶמְבּוּרְג עִם 96%. זֹאת אוֹמֶרֶת, שֶׁכִּמְעַט לְכָל תוֹשָׁב שָׁם יֵשׁ טֶלֶפוֹן נַיָד.
