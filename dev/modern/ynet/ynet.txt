מְקוֹם הַקְּבוּרָה הַזֶּה הוּא בַּעַל מַשְׁמָעוּת גְּדוֹלָה מְאוֹד לִיהוּדֵי אִירָאן וְאַף עֲבוּר תּוֹשָׁבֵי הַמָּקוֹם שֶׁאֵינָם יְהוּדִים, שֶׁמּוֹצְאִים בּוֹ קְדוּשָּׁה. ד"ר תָּמָר עֵילָם גִּינְדִין, חוֹקֶרֶת אִירָאן הַקְּדוּמָה, מְתָאֶרֶת בָּאֲתָר שֶׁלָּהּ הַפְגָּנָה גְּדוֹלָה שֶׁל סְטוּדֶנְטִים שֶׁהִתְקַיְּימָה לְאוֹרֶךְ תְּקוּפָה לְיַד צִיּוֹן הַקֶּבֶר שֶׁל מָרְדְּכַי וְאֶסְתֵּר. 
------------------
בְּצַעַד בּוֹטֵחַ עָלְתָה אָן נוֹיְבֶּרְגֵּר לְדוּכַן הַנּוֹאֲמִים בַּחֲדַר הַתִּדְרוּכִים שֶׁל הַבַּיִת הַלָּבָן בַּשָּׁבוּעַ שֶׁעָבַר. הַתִּדְרוּךְ, שֶׁבּוֹ עִדְכְּנָה אֶת הַצִּיבּוּר הָאָמֵרִיקָנִי עַל מִתְקָפוֹת הָאקֵרִים רוּסִים נֶגֶד אָמֵרִיקָה, הוֹעֳבַר בְּשִׁידּוּר חַי בְּרִשְׁתוֹת הַחֲדָשׁוֹת וְזָכָה לְכוֹתָרוֹת. 
------------------
"שְׁנֵינוּ גּוֹלְשֵׁי גַּלִּים, וְהִכַּרְנוּ בַּשֵּׁירוּת הַצְּבָאִי שֶׁבּוֹ הָיִינוּ בְּיַחַד בַּצֶּוֶות בְּחֵיל הַיָּם", מְסַפֵּר פַרְבְּשְׁטֵיְין. "נִיסִּינוּ לַחֲשׁוֹב אֵיךְ לְנַצֵּל אֶת הַהִזְדַּמְּנוּת שֶׁשְּׁנֵינוּ פְּנוּיִים כְּדֵי לְהִתְחַבֵּר וְלִיזוֹם מַשֶּׁהוּ, וּבְסֶגֶר הָרִאשׁוֹן הָיוּ גַּלִּים מְטוֹרָפִים וְאָסוּר הָיָה לָצֵאת לִגְלוֹשׁ. 
------------------
פֵייד אָאוּט לְמוּזִיקָה.
בְּמַהֲלַךְ שְׁנוֹת הָאַלְפַּיִים הַמּוּקְדָּמוֹת (בָּאֵזוֹר 2010) מַגִּיעוֹת מַעַצְמוֹת הָעָל שֶׁל הָעוֹלָם לַמַּסְקָנָה הַמִּתְבַּקֶּשֶׁת.
נִמְאַס לָהֶן לִבְחוֹשׁ בְּצָרוֹת שֶׁל אֲחֵרִים וּלְשַׂחֵק אוֹתָהּ מוֹשִׁיעֵי הָעוֹלָם.
אַרְצוֹת הַבְּרִית וְרוּסְיָה מַחֲלִיטוֹת שֶׁאֶת כָּל הַכּוֹחַ שֶׁלָּהֶן יַשְׁקִיעוּ בַּהֲגָנָה עַל עַצְמָן וְעַל עַצְמָן בִּלְבַד.
דַּי לִתְמִיכָה בְּכָל מִינֵי מְדִינוֹת קְטַנּוֹת וְחַלָּשׁוֹת, בִּמְיוּחָד כָּאֵלֶּה מֵהַמִּזְרָח הַתִּיכוֹן.

------------------
בְּעָיָה נוֹסֶפֶת הַמּוּכֶּרֶת הֵיטֵב מִנְּגִיף הַקּוֹרוֹנָה הִיא הִתְפַּתְּחוּת הַזָּנִים הָעֲמִידִים: עֲמִידוּת לְאַנְטִיבִּיּוֹטִיקָה מוּכֶּרֶת גַּם בְּהִתְמוֹדְדוּת עִם חַיְידַּקִּים וְהוֹפֶכֶת לִבְעָיָה מַשְׁמָעוּתִית בַּשָּׁנִים הָאַחֲרוֹנוֹת, אֶלָּא שֶׁנְּגִיפִים מְפַתְּחִים מוּטַצְיוֹת מַהֵר הַרְבֵּה יוֹתֵר וּמְגַלִּים עֲמִידוּיוֹת לִתְרוּפוֹת אַנְטִי וִירָאלִיּוֹת. 
------------------
זֶה מַשֶּׁהוּ שֶׁבָּחַרְתִּי לַעֲשׂוֹת וַאֲנִי כְּלָל לֹא מִתְחָרֶטֶת עַל כָּךְ.
זֶה גָּרַם לִי אוֹשֶׁר לִרְדּוֹף אַחֲרֵי מַשֶּׁהוּ שֶׁהָיְיתָה לִי כָּזוֹ תְּשׁוּקָה אֵלָיו".

הִיא הוֹסִיפָה עוֹד: "הָלַכְתִּי בְּעִקְבוֹת הַמַּטָּרוֹת שֶׁלִּי.
אֲנִי יוֹדַעַת שֶׁנָּתַתִּי כָּל טִיפָּה שֶׁל אֵנֶרְגִּיָּה שֶׁהָיְיתָה לִי, וְלֹא יָכוֹלְתִּי לַעֲשׂוֹת שׁוּם דָּבָר יוֹתֵר טוֹב.
אֵין לִי יוֹתֵר אֶת הַתְּשׁוּקָה לְאַתְלֵטִיקָה אוֹ לַקְּרָב 7 כִּי אֲנִי יוֹדַעַת שֶׁאֵין לִי לְאָן לְהִתְקַדֵּם.

------------------
לָאַחֲרוֹנָה עָלְתָה לְסֵדֶר הַיּוֹם בְּאַרְהָ"ב סוּגְיַית הַשְׁמָנַת הַיֶּתֶר, בְּעִיקָּר בְּקֶרֶב בְּנֵי נוֹעַר.
סְקָרִים שׁוֹנִים הִצְבִּיעוּ עַל כָּךְ שֶׁהָאָמֵרִיקָנִים הֵם הָעָם הַשָּׁמֵן בְּיוֹתֵר בָּעוֹלָם.
סְקָרִים אֲחֵרִים הִצְבִּיעוּ עַל עֲלִיָּיה בְּשִׁיעוּר מִסְפַּר חוֹלֵי הַלֵּב בְּאַרְצוֹת הַבְּרִית.
הַדִּיּוּן בַּנּוֹשֵׂא הֵבִיא לְלַחַץ גּוֹבֵר וְהוֹלֵךְ שֶׁהוּפְעַל עַל רִשְׁתוֹת הַמָּזוֹן הַמָּהִיר מִצַּד אִרְגּוּנֵי בְּרִיאוּת הַצִּיבּוּר.

------------------
נָכוֹן שֶׁהַמָּנוֹחַ הָיָה אִישׁ מְכוּבָּד וּמוּכָּר, אֲבָל זֶה לֹא אוֹמֵר שֶׁצָּרִיךְ לְקַיֵּים הַלְווֹיָה כָּל כָּךְ גְּדוֹלָה".
שִׁלְשׁוֹם הִתְקַיְּימָה חֲתוּנָּה בִּכְפָר בַּרְטָעָה, שֶׁמַּחֲצִיתוֹ בְּשִׁטְחֵי הָרָשׁוּת הַפָלַסְטִינִית וּמַחֲצִיתוֹ בָּאָרֶץ. הָאוּלָם בַּבַּרְטָעָה, שֶׁבּוֹ מִתְקַיֶּימֶת חֲתוּנָּה כִּמְעַט מִדֵּי יוֹם, נִמְצָא בְּפַאֲתֵי הַכְּפָר וְהוּא מְהַוֶּוה מוֹקֵד מְשִׁיכָה לְתוֹשָׁבֵי הַיִּישּׁוּבִים הָעַרְבִיִּים מֵהַצָּפוֹן, מֵהַמְּשׁוּלָּשׁ וּמִוּוֹאדִי עֵרָה.

------------------
אַחֲרֵי נִישּׂוּאֶיהָ, וּבְעִידוּדוֹ שֶׁל בַּעֲלָהּ, עו"ד בְּמִקְצוֹעוֹ, הִתְקַבְּלָה לְבֵית הַסֵּפֶר לְמִנְהַל עֲסָקִים בְּאוּנִיבֶרְסִיטַת קוֹלוֹמְבִּיָּה הַיּוּקְרָתִית. בֵּין לֵידַת בִּתָּהּ הַבְּכוֹרָה לְהַתְקָפוֹת הַטֵּרוֹר שֶׁל הַ-11 בְּסֶפְּטֶמְבֶּר, הִצְלִיחָה נוֹיְבֶּרְגֵּר לְהַשְׁלִים תּוֹאַר שֵׁנִי בְּמִינְהַל עֲסָקִים וְכֵן בִּיחָסִים בֵּינְלְאוּמִּיִּים בְּקוֹלוֹמְבִּיָּה.


------------------
בְּכָל בַּשְּׁלָן יֵשׁ אֶת הַיֶּלֶד הָרַע, אוֹתוֹ יֶלֶד הַמִּתְעַקֵּשׁ לְהָזִין אֶת שׂוֹנְאֵי הַכּוּסְבָּרָהּ דַּוְוקָא בְּכוּסְבָּרָהּ וּמִשְׁתּוֹקֵק לְרַגֵּשׁ וּלְהַקְסִים מֵהַמָּקוֹם הַבִּלְתִּי צָפוּי.

לִפְנֵי כְּעָשׂוֹר מָצָאתִי אֶת עַצְמִי דּוֹהֵר בְּנוֹרְמַנְדִּי בִּגְרוּטָאָה אֲדוּמָּה, דּוֹפֵק עַל דְּלָתוֹת שֶׁל יַצְרָנֵי סַיְידֵּר מְקוֹמִי.
כֵּן, סַיְידֵּר, אוֹתוֹ מַשְׁקֶה תַּעֲשִׂיָּיתִי הַמּוּכָּר לָנוּ מֵהַמַּכּוֹלֶת אֲבָל בְּגִרְסָה מוּפְלָאָה - עֲכוּרָה, תּוֹסֶסֶת וְאֶלְכּוֹהוֹלִית.


------------------
הַשּׁוּתָּפוֹת בִּפְּרוֹיֶקְט יָם תֶּטִיס, קְבוּצַת דֶּלֶק וְנוֹבֵּל אֵנֶרְגִּ'י, הִקְטִינוּ חַד-צְדָדִית אֶת הֶיקֵּף הַתַּמְלוּגִים שֶׁהֵן מְשַׁלְּמוֹת לַמְּדִינָה.
הַחֲבָרוֹת, שֶׁשִּׁילְּמוּ מֵרֵאשִׁית הַשָּׁנָה רַק כִּשְׁלִישׁ מֵהַתִּמְלוּגִים שֶׁנִּדְרְשׁוּ לְשַׁלֵּם עַל יְדֵי הַמְּדִינָה, הַפְחִיתוּ אֶת הַתַּשְׁלוּמִים בִּכְ-10 מִילְיוֹן שֶׁקֶל לְחוֹדֶשׁ.


------------------
סִדְרַת בְּדִיקוֹת שֶׁהִיא עָבְרָה לֹא מָצְאָה סִיבָּה רְפוּאִית לַפִּרְכּוּסִים.
כָּל הַבְּדִיקוֹת הָיוּ תַּקִּינוֹת.
כְּשֶׁהָרוֹפְאִים אָמְרוּ לָהּ שֶׁהֵם חוֹשְׁבִים שֶׁהַבְּעָיָה שֶׁלָּהּ נַפְשִׁית, ג'וֹ קִיבְּלָה זֹאת.
בָּרֶגַע שֶׁהִיא הֵבִינָה שֶׁהַפִּרְכּוּסִים הָיוּ דֶּרֶךְ מִילּוּט מֵהַמַּצָּב הַקָּשֶׁה שֶׁאֵלָיו נִקְלְעָה, הַפִּרְכּוּסִים פָּסְקוּ.
בְּעִקְבוֹת זֹאת הֶחֱלִיטָה ג'וֹ לְהִתְעַלֵּם מֵהַצִּיפִּיּוֹת שֶׁל הָאֲחֵרִים מִמֶּנָּה, לִנְטוֹשׁ אֶת הַצִּילּוּם וּלְשַׁנּוֹת כִּיווּן מִקְצוֹעִי.

------------------
שְׁמוֹנֶה הֲפָקוֹת שֶׁאוֹרְכָן עַד 50 דַּקּוֹת יִתְמוֹדְדוּ הַשָּׁנָה בְּתַחֲרוּת הַפֶסְטִיבָל.
הַהַצָּגָה "אִישׁ צִיפּוֹר" שֶׁכָּתַב אַלּוֹן מַרְקוּס וּבִייֵּם עִידָּן בְּלַנְק, הִיא אַגָּדָה גְּרוֹטֶסְקִית עַל בְּרִיאָה מֵחָדָשׁ שֶׁל הֶחָיָה הַשַּׂחְקָנִית שֶׁיּוֹצֵאת לַחוֹפְשִׁי אַחֲרֵי תְּקוּפָה אֲרוּכָּה בַּכְּלוּב.

------------------
אִישּׁוּם נֶגֶד מַדְבִּיר, שֶׁהִשְׁאִיר חוֹמֶר מְסוּכָּן בְּפִינַּת בּוּבּוֹת בְּגַן יְלָדִים

מְקוֹרָן בְּבִיקּוֹרֶת דּוֹמָה מְאֹד לְזוֹ שֶׁיֵּשׁ גַּם לְרַבִּים בַּצִּיבּוּר הַחִילּוֹנִי עַל הַטֵּלֵוִויזְיָה.
רַק הַפִּתְרוֹן שׁוֹנֶה.
 
"סַכָּנַת הַמַּחְשֵׁב" עָלְתָה לְדִיּוּן כְּשֶׁהִתְבָּרֵר שֶׁהַמַּחְשֵׁב יָכוֹל לְשַׁמֵּשׁ לִצְפִיָּיה בִּסְרָטִים, וְשֶׁהַכְנָסָתוֹ לְבָתִּים חֲרֵדִיִּים רַבִּים בְּעֶצֶם הַכְּנִיסָה אֲלֵיהֶם גַּם אֶת הַטֵּלֵוִויזְיָה בַּדֶּלֶת הָאֲחוֹרִית.

------------------
בְּעוֹד שֶׁמֶּנְקֶּר טָעַן בִּתְבִיעָתוֹ כִּי הַהֲפָקָה הוֹעַלְתָה בְּנִיגּוּד לַחוֹזֶה מִחוּץ לְאוּלַם הַבַּיִת לְלֹא יְדִיעָתוֹ אוֹ הַסְכָּמָתוֹ, טָעֲנוּ בַּקָּאמֶרִי לָהָדָ"ם וְהִדְגִּישׁוּ כִּי כָּל הַגָּשַׁת הַתְּבִיעָה יְסוֹדָהּ בִּשְׁאִיפַת נָקָם מִצַּד הַבַּמַּאי.


------------------
אֶת טִיּוּלֵי הַקָּרָוָואנִים אֶפְשָׁר לְהַזְמִין בָּאִינְטֶרְנֵט, אַךְ כַּיּוֹם פּוֹעֲלוֹת בָּאָרֶץ חֲבֵרוֹת שֶׁמִּתְמַחוֹת בְּאִרְגּוּנָם, בָּהֶן מוֹטוֹרְהוֹם‬ וּבַנְדָּנָה קָרָוָואנִים, שֶׁמְּפַשְּׁטוֹת תַּהֲלִיכִים בַּעֲבוּר הַנּוֹסֵעַ הַיִּשְׂרְאֵלִי.
לִפְנֵיכֶם הַמֵּידָע שֶׁחָשׁוּב לָדַעַת לִפְנֵי שֶׁיּוֹצְאִים לַדֶּרֶךְ.

 
כְּשֶׁמְּתַכְנְנִים טִיּוּל קָרָוָואנִים צָרִיךְ לְהִשְׁתַּחְרֵר מֵ"הַסְּפּוֹרְט הַלְּאוּמִּי" שֶׁל הַזְמָנוֹת בָּרֶגַע הָאַחֲרוֹן.

------------------
בִּצְעִירוּתָהּ בִּתְקוּפַת הַשֵּׁפֶל בְּאַרְצוֹת הַבְּרִית, הִיא עָסְקָה בְּכָל עֲבוֹדָה אֶפְשָׁרִית, מֵרְחִיצַת חַלּוֹנוֹת וְעַד שְׁמַרְטַפּוּת, כְּדֵי לְסַיֵּיעַ לְכַלְכָּלַת הַמִּשְׁפָּחָה.
בְּגִיל 42 הִתְאַלְמְנָה מִבַּעֲלָהּ, וְנוֹתְרָה מְטוּפֶּלֶת בְּאַרְבַּעַת יְלָדֶיהָ, כְּשֶׁכּוּלָּם חַיִּים בְּדִירַת חֶדֶר אַחַת.
בִּשְׁנוֹת הַשִּׁישִּׁים הָפְכָה לִפְעִילָה בַּתְּנוּעָה לִזְכוּיוֹת אָדָם.

------------------
אַחֲרֵי פִּעְנוּחַ הַטֵּקְסְט, מִתְגַּלֶּה לְעִיתִּים כִּי הַטֵּקְסְט הַמְּפוֹרָשׁ אֵינֶנּוּ אֶלָּא פְּתִיחָתָהּ הָאֲרוּכָּה שֶׁל הָאִיגֶּרֶת, שֶׁבָּהּ אֵין שׁוּם תּוֹכֶן מְעַנְיֵין. לְעוּמַּת זֹאת, בְּמִקְרִים אֲחֵרִים הַחוֹקֵר הוּא כְּמוֹ בַּלָּשׁ הַחוֹשֵׂף סוֹדוֹת.


------------------
הַמִּתְבַּגְּרִים שֶׁלָּכֶם עוֹבְרִים אֶת אַחַת הַתְּקוּפוֹת הַסּוֹעֲרוֹת בְּחַיֵּיהֶם, וְהֵם זְקוּקִים לָכֶם כְּאִי שֶׁל וַדָּאוּת בְּתוֹךְ הַיָּם הַזֶּה אוֹ כְּקִיר חָזָק וְיַצִּיב שֶׁהֵם יְכוֹלִים לְהִישָּׁעֵן עָלָיו בְּבִיטָּחוֹן. 
------------------
הַקִּיצּוּץ הַנּוֹכְחִי יְאוּשַּׁר, בַּמִּקְרֶה הַטּוֹב, לֹא לִפְנֵי אֶמְצַע הַחוֹדֶשׁ הַבָּא.
חוֹדְשַׁיִים לְאַחַר מִכֵּן, בְּאֶמְצַע חוֹדֶשׁ יוּלִי, יַתְחִיל כְּבָר הָאוֹצָר בַּדִּיּוּנִים עַל תַּקְצִיב 2003, שֶׁגַּם הוּא יִהְיֶה כָּרוּךְ, מִן הַסְּתָם בְּקִיצּוּץ מַשְׁמָעוּתִי.
לָכֵן יִהְיֶה עַל צַמֶּרֶת הָאוֹצָר לִפְעוֹל בַּיָּמִים הַקְּרוֹבִים בִּמְהִירוּת וּבִנְחִישׁוּת - עֵצָה שֶׁקַּל לָתֵת, אֲבָל הַרְבֵּה יוֹתֵר קָשֶׁה לְבַצֵּעַ.
הַשְּׁאֵלָה הַגְּדוֹלָה הִיא, כַּמּוּבָן, כַּמָּה צָרִיךְ לְקַצֵּץ.

------------------
סֶלַע הַקִּירְטוֹן הָרַךְ בִּשְׁפֵלַת יְהוּדָה מְאַפְשֵׁר חֲצִיבָה נוֹחָה אֲשֶׁר אִפְשְׁרָה אֶת הֲכָנַת מְעָרוֹת הַמִּסְתּוֹר.

מְעָרוֹת הַמִּסְתּוֹר הָיוּ מוּכָּרוֹת בְּמֶשֶׁךְ שָׁנִים רַבּוֹת, אַךְ רַק בַּמֵּאָה הַ-20 הוּבַן הַקֶּשֶׁר בֵּינָן לְבֵין אוֹתוֹ מֶרֶד שֶׁפָּרַץ בְּ-132 לַסְּפִירָה.

------------------
לְשִׂמְחָתוֹ הַשִּׂיא הַשְּׁלִילִי הַזֶּה נִלְקַח עַתָּה מִמֶּנּוּ...
 
אֲבָל מִי שֶׁחוֹשֵׁב שֶׁ-67 הַיָּמִים שֶׁל סֶם אָלֶרְדַּיְיס בְּתַפְקִיד הַמְּאַמֵּן זֶה שִׂיא עוֹלָמִי אוֹ מַשֶּׁהוּ שֶׁקָּרוֹב לְזֶה, כְּדַאי שֶׁיַּחֲשׁוֹב שׁוּב.
אוֹ עָדִיף, שֶׁיִּקְרָא אֶת הַסִּיפּוּרִים הַמּוּזָרִים - וְהַמְּאוֹד קְצָרִים – הָאֵלּוּ.

בְּרַאיָין קְלָאף הָיָה אֶחָד הַמֶּנֶגֶ'רִים הַגְּדוֹלִים בַּהִיסְטוֹרְיָה שֶׁל הַכַּדּוּרֶגֶל הָאַנְגְּלִי.

------------------
הֲתַרְחִישׁ הַזֶּה נִשְׁמַע קִיצוֹנִי בְּיוֹתֵר, וְהוּא מְאוֹד לֹא סָבִיר גַּם עַכְשָׁיו, אֲבָל הִתְבַּצְּרוּתוֹ שֶׁל טְרַאמְפּ בְּעֶמְדָּתוֹ וְסֵירוּבוֹ לְהַכִּיר בַּהֶפְסֵד גּוֹרֶמֶת לְאָמֵרִיקָנִים לְכָל הַפָּחוֹת לְהַרְהֵר בָּאֶפְשָׁרוּת שֶׁיִּתְמַמֵּשׁ

כָּךְ אוֹ כָּךְ, לְפִי הַדִּיוּוּחִים, בְּאַרְהָ"ב כְּבָר נֶעֱרָכִים לַבַּיִת הַלָּבָן הַדֵּמוֹקְרָטִי. 
------------------
"אֲנַחְנוּ נִכְנָסִים לַחוֹרֶף וְיֵשׁ חֲשָׁשׁ שֶׁמִּסְפַּר הַנִּדְבָּקִים יִגְדַּל", צִייֵּן גַּמְּזוּ וְהוֹסִיף: "אֲנַחְנוּ עֲדַיִין בְּמַגֵּפָה, חַיָּיבִים לְהִיזָּהֵר". רֹאשׁ הַמּוֹעָצָה, פַארֶס בִּדְחִי, דִּיבֵּר עַל הַצּוֹרֶךְ בְּחִידּוּשׁ מַעֲרֶכֶת הַחִינּוּךְ וְגַמְּזוּ הֵשִׁיב לוֹ כִּי "חַיָּיבִים לִמְצוֹא אֶת הַפִּתְרוֹן לְכָךְ. זֶה לֹא פָּשׁוּט, אֵין פִּתְרוֹן בָּרוּר, אֲבָל אֲנַחְנוּ מְנַסִּים לְהוֹבִיל דְּבָרִים".


------------------
סְפָרִים נִכְתָּבִים בַּמַּחְשֵׁב, יְשִׁיבוֹת וּבָתֵּי-סֵפֶר מְנוֹהָלִים בְּאֶמְצָעוּת מַחְשֵׁב, הַנְהָלַת הַחֶשְׁבּוֹנוֹת בְּמוֹסָדוֹת חֲרֵדִיִּים נַעֲשֵׂית בְּתוֹכְנוֹת מַתְאִימוֹת".
 
הַבְּעָיָה, הוּא מַסְבִּיר, אֵינָהּ עִם הַטֶּכְנוֹלוֹגְיָה אֶלָּא עִם הַתְּכָנִים: "גַּם בַּצִּיבּוּר הַחִילּוֹנִי יֵשׁ אֲנָשִׁים שֶׁמּוּדְאָגִים מֵהַהַשְׁפָּעָה שֶׁל מִשְׂחָקִים וְאִינְטֶרְנֵט עַל הַיְּלָדִים, וּמִשְׁתַּדְּלִים לְהָגֵן עֲלֵיהֶם.

------------------
"הִרְגַּשְׁנוּ שֶׁאִיזּוּן זֶה בְּדִיּוּק מָה שֶׁאֲנָשִׁים צְרִיכִים בַּתְּקוּפָה הַזֹּאת, גַּם מֵעֵבֶר לַקֶּטַע הַפִיזִי. הִתְחַלְנוּ לַעֲבוֹד בַּנַּגָּרִיָּיה שֶׁל אַבָּא שֶׁל דוּנְיוֹ, שִׂיחַקְנוּ עִם הָעֵץ וְהַצְּבָעִים וְאַחֲרֵי שֶׁהֵכַנּוּ אֶת הַבּוֹרְד הָרִאשׁוֹן הִסְתַּכַּלְנוּ וְאָמַרְנוּ 'ווֹאוּ'. מַמָּשׁ אָהַבְנוּ אֶת הַמּוּצָר".
------------------
הַדָּבָר הָרִאשׁוֹן שֶׁעָשָׂה הַצֶּוֶות הַמְּטַפֵּל, לְדִבְרֵיהֶם לְפִי הַנְחָיוֹת שֶׁנִּיתְּנוּ לְכָל בָּתֵּי הַחוֹלִים, הָיָה לְהַפְרִיד בֵּין בְּנֵי הַזּוּג. אָבִי נִלְקַח לְטִיפּוּל בְּמַחְלֶקֶת "מִיּוּן קוֹרוֹנָה", וְאִילּוּ אִימִּי, שֶׁלֹּא שָׁהֲתָה בְּמֶרְחָק שֶׁל יוֹתֵר מִשְּׁנֵי מֶטְרִים מִמֶּנּוּ כְּבָר חוֹדָשִׁים אֲרוּכִּים, נִשְׁלְחָה לְחַכּוֹת בַּחוּץ. אַף אֶחָד לֹא יָצָא לִשְׁאוֹל עַל מַצָּבוֹ הַכְּלָלִי, עַל הַתְּרוּפוֹת שֶׁהוּא מְקַבֵּל. כָּכָה עוֹבְדִים הַנְּהָלִים הַחֲדָשִׁים. 
------------------
"אֲנִי מַדְהִים אֶת עַצְמִי מִדֵּי יוֹם", אָמַר וֶוסְטְבְּרוּק לְעִיתּוֹנָאִים בַּסִּיּוּם.
"אֲנִי לֹא יוֹדֵעַ מָה יִקְרֶה בַּמִּשְׂחָקִים שֶׁלָּנוּ לְטוֹב אוֹ לְרַע, אֲבָל הָעֲבוֹדָה שֶׁלִּי הִיא לְשַׂחֵק בָּרָמָה הַגְּבוֹהָה בְּיוֹתֵר וּבוֹרַכְתִּי לְשַׂחֵק אֶת הַמִּשְׂחָק הַזֶּה.
בְּכָל עֶרֶב אֲנִי מַשְׁאִיר הַכֹּל עַל הַפַּרְקֵט.
לִפְעָמִים דְּבָרִים טוֹבִים קוֹרִים וְלִפְעָמִים רָעִים.
זֶה הַטֶּבַע שֶׁל הַמִּשְׂחָק, אֲבָל אֲנִי תָּמִיד מְרוּצֶּה מֵהַתּוֹצָאוֹת".
 

------------------
הָאִינְטֶרְנֶט עֲדַיִין אֵינוֹ נָפוֹץ בַּבָּתִּים, וּבְחֶלְקָם הוּא מְשַׁמֵּשׁ לְדוֹאַר אֶלֶקְטְרוֹנִי בִּלְבַד, אֲבָל בַּאֲחֵרִים נוֹהֲגִים גַּם לִגְלוֹשׁ - לְלֹא הַגְבָּלָה, אוֹ בְּעֶזְרַת תּוֹכְנוֹת Parental Control מֵהַסּוּג שֶׁכָּל סַפָּקֵי הָאִינְטֶרְנֶט מַצִּיעִים הַיּוֹם לְדָתִיִּים, לַחֲרֵדִים וְלִסְתָם הוֹרִים מוּדְאָגִים.


------------------
לְמָחֳרָת, בִּשְׁעוֹת הַבּוֹקֶר, הִבְחִינוּ כַּמָּה פָּעוֹטוֹת בַּקּוּפְסָה, וּשְׁנַיִים מֵהֶם אַף הִצְלִיחוּ לִפְתּוֹחַ אוֹתָהּ וְנָגְעוּ בְּחוֹמֶר הַהַדְבָּרָה שֶׁהָיָה בְּתוֹךְ הַקּוּפְסָה. מִשֶּׁהִבְחִינָה הַגַּנֶּנֶת בַּמִּתְרַחֵשׁ, הִיא הִרְחִיקָה אֶת הַיְּלָדִים מֵהַחוֹמֶר וּמֵהַקּוּפְסָה, וְהֵם הוּפְנוּ לִבְדִיקָה רְפוּאִית. לְמַרְבֵּה הַמַּזָּל, הִתְבָּרֵר כִּי לֹא אֵירַע נֶזֶק בְּרִיאוּתִי לִילָדִים שֶׁשִּׁיחֲקוּ בַּחוֹמֶר. 

------------------
שִׂיחַ פָּתוּחַ, אֶמְפָּתִי וְאוֹתֶנְטִי, יְקָרֵב אֶתְכֶם אֶל הַמִּתְבַּגְּרִים וִיאַפְשֵׁר לָהֶם לְהַרְגִּישׁ שֶׁאַתֶּם מַקְשִׁיבִים לָהֶם, מְבִינִים אוֹתָם וְלֹא שִׁיפּוּטִיִּים. זֶהוּ מַרְכִּיב עִיקָּרִי וְחִיּוּנִי בְּתִקְשׁוֹרֶת יְעִילָה, בְּכָל מַעֲרֶכֶת יְחָסִים וּבִמְיוּחָד בַּקֶּשֶׁר בֵּין הוֹרֶה לַיֶּלֶד.
