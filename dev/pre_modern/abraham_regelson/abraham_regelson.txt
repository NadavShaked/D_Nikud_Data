שׁוֹכֵב הוּא כָּל הַיָּמִים עַל עַרְשׂוֹ, לֹא חַי וְלֹא מֵת, וְשָׁבְתוּ תַעֲנוּגוֹת-אַהֲבָה מִן הָעוֹלָם, כִּי אֵין הוּא מוֹרֶה אֶת חִצָּיו עוֹד.
אַתְּ, שֶׁהִנָּךְ כֹּה לְמוּדָה בְּהִלְכוֹת-אַהֲבָה, מַה נִבְעַרְתְּ אַתְּ כֶּשֶׁהַדְּבָרִים נוֹגְעִים לִבְנֵךְ.
בְּעֵינַיִךְ עוֹד נַעַר הוּא, כִּמְעַט יֶלֶד, אֲבָל בֶּאֱמֶת כְּבָר גָּדַל וַיְהִי לְבָחוּר, וְנַפְשׁוֹ חָשְׁקָה בְּרַעְיָה.

------------------
הַגִּלְיוֹנוֹת נֶאֶסְפוּ, וְכָל נַעַר, בְּהִקָּרֵא שְׁמוֹ, קָם וְעָמַד לִפְנֵי מַכְתֵּבָה אַחַת, אֶצְלָהּ יָשַׁב גּוֹי גָּבוֹהַּ וּבָרִיא, לָבוּשׁ בְּהִדּוּר. אַחֲרֵי חֲלִיפֵי מִלִּים מֻעָטוֹת עִם הַנַּעַר הָעוֹמֵד לְפָנָיו, רָשַׁם הָאִישׁ תָּו בַּגִּלָּיוֹן, וְהַנַּעַר, עָבַר עִם גִּלְיוֹנוֹ אֶל מַכְתֵּבָה שְׁנִיָּה, אֶצְלָהּ יָשַׁב מַזְכִּיר יְהוּדִי קָטָן, בַּעַל שָׂפָם וּמִשׁקָפַיִם, וְאַף הוּא רָשַׁם מַשֶּׁהוּ בַּגִּלָּיוֹן. 
------------------
עָמַד וְלָקַח אוֹתוֹ הָעוֹר, נִעֲנַע אוֹתוֹ הָעוֹר, גֵּרַד אוֹתוֹ הָעוֹר וְשִׁפְשֵׁף אוֹתוֹ הָעוֹר, שַׁפְשֵׁף וּמַלֵּא אוֹתוֹ, כָּל כַּמָּה שֶׁיָּכִיל בְּתוֹכוֹ, פֵּרוּרֵי-עֻגָה יְשָׁנִים-נוֹשָנִים, יְבֵשִׁים וַעֲבֵשִׁים, פְּסוּלִים וּקְמוּלִים, כְּתוּשִׁים וּקְרוּשִׁים, וְדֻמְדְּמָנִיּוֹת שְׂרוּפוֹת וַחֲרוּכוֹת, כִּי יָדַע כִּי יֵש בָּאֵלֶּה כְּדֵי לְדַגְדֵּג. אַחַר טִפֵּס וְעָלָה בְרֹאשׁ-הַדֶּקֶל אֲשֶׁר לוֹ, וְחִכָּה לַקַּרְנָף שֶׁיֵּצֵא מִן הַמַּיִם וְיִלְבְּשֵׁהוּ.
וְהַקַּרְנָף כֵּן עָשָׂה. 
------------------
וְכִי תִיעַפְנָה אֵלֶּה הַשְּׁתַיִם, וַתֹאחֲזֵן תְּנוּמָה, מִיָּד שְׁתֵּי עֵינַיִם אֲחֵרוֹת נִפְקָחוֹת רַעֲנַנּוֹת.
יוֹמָם הָיָה אַרְגּוֹס נוֹתֵן לְיוֹ הַפָּרָה לִרְעוֹת בַּשָּׂדֶה, וְלַיְלָה הָיָה כוֹפֵת אוֹתָהּ בְּחָבֶל.
פַּעַם בְּלַחֲכָהּ בָּעֵשֶׂב, הָלוֹךְ וְלָחוֹךְ, יָרְדָה עַד בֵּית-אָבִיהָ, אֲשֶׁר עִם הַנָּהַר.
אֲבָל לֹא הוֹרֶיהָ וְלֹא אַחֶיהָ וְאַחְיוֹתֶיהָ הִכִּירוּהָ.

------------------
אֲבָל בְּנַפְשָׁם פַּחַד סָתוּם מִפְּנֵי נְבוּאָה כִּי עֲתִידָה יַבַּשְׁתָּם לִצְלוֹל תְּהוֹמָה וְכָל מַמְלַכְתָּם תִּהְיֶה כְּלֹא הָיָתָה; רַק פִּסְגוֹת-הָרִים אֲחָדוֹת, אֲשֶׁר תִּבְלֹטְנָה מֵעַל פְּנֵי-הָאוֹקְיָנוֹס, יְסַפְּרוּ לִמְעִזֵּי הַפְלִיג אֶל מַיִם אַדִּירִים כִּי שָׁם הָיְתָה יַבָּשֶׁת.

------------------
וְהֶרְמֶס, בְּמַקְלוֹ-זָהָב, נוֹהֵג אֶת הַפָּרוֹת אֶל מְעָרַת-הַמִּכְלָא אֲשֶׁר לְיַד נְהַר-אַלְפֵיאוֹס – בַּחֲגוֹרָתוֹ נוֹצֵץ עֲנַף-הָעֹשֶר מְשֻׁלַּשׁ-הֶעָלִים, וְנַפְשׁוֹ שְׂמֵחָה בְּמִמְכָּרוֹ-וּמִקָּחוֹ בַּיּוֹם הַהוּא.
בְּשׂוֹרַת גְּדֻלָּה 

------------------
וּבַלַּיְלָה הַהוּא, כְּשָׁכְבָהּ עַל יְצוּעָהּ בַּחֲדַר-כְּלוּלוֹתֶיהָ, וּבְהַרְהֲרָהּ בְּאֶפֶס שֵׁנָה בְּכָל הַנִּפְלָאוֹת אֲשֶׁר קָרוּהַּ, וְהִנֵּה קוֹל לוֹחֵשׁ בְּלִטּוּף עַל אָזְנָהּ; וַתִּבָּהֵל, אַךְ מְהֵרָה פָּג פַּחְדָּהּ, כִּי דִבְרֵי-הַלַּחַשׁ – כִּסּוּפִים וּדְבֵקוּת וְאַהֲבָה אֵין-קֵץ.
וַתֵּדַע הַנַּעֲרָה כִּי בַּעְלָהּ, אֲדוֹן-הָאַרְמוֹן, הוּא זֶה.
וַיִּקָּחֶנָּה בִזְרוֹעוֹתָיו וַתְּהִי לוֹ לְאִשָּׁה, וַיִדְבְּקוּ נֶפֶשׁ בְּנֶפֶשׁ.

------------------
וְהֶפֵיסְטוֹס פָּתַח לְפָנָיו בִּתְלוּנָה: "הֲזֶה הַגְּמוּל הַטּוֹב, גְּמָלַנִי אָרֶס עַל אֲשֶׁר שִׁחְרַרְתִּי אוֹתוֹ מִדּוּד-כִּלְאוֹ? וּמִי יִתֵּן וְהִנַּחְתִּיו לִשְׁכַּב שָׁם מִתְעַלֵּף עַד נְקוֹב עַצְמוֹתָיו אֶת עוֹרוֹ הַצּוֹפֵד!" וַיִּגְעַר בּוֹ הַשַּׁלִּיט בָּאֵלִים: "מַה-לְּךָ כִּי תְגַלֶּה עֶלְבּוֹנְךָ בָּרַבִּים?" וְאֶל הָאֵלִים אָמַר: "וְלֹא תִתְבַּיְּשׁוּ לַחֲזוֹת אֶת אֲחִיכֶם וַאֲחוֹתְכֶם בְּקַלְקָלָתָם?" בְּבֹשֶׁת-פָּנִים הִתְפַּזְּרוּ הָאֵלִים אִישׁ לְעֶבְרוֹ.

------------------
הִתְחִיל הֶחָתוּל מְקוֹנֵן, כְּאִלּוּ בֶּאֱמֶת צַר לוֹ, וְאָמַר: “הַאִם לְעוֹלָם לֹא אָבוֹא אֶל הַמְּעָרָה פְּנִימָה? הַאִם לְעוֹלָם לֹא אֵשֵׁב אֵצֶל הָאֵשׁ הַחַמָּה? הַאִם לְעוֹלָם לֹא אֶשְׁתֶּה אֶת הֶחָלָב הֶחָם וְהַלָּבָן? הִנָּךְ חֲכָמָה מְאֹד וְיָפָה מְאֹד.
חָלִילָה לָךְ מְהִתְאַכְזֵר אֲפִילוּ לְחָתוּל.”
אָמְרָה הָאִשָּׁה: “יָדַעְתִּי שֶׁחֲכָמָה אָנֹכִי, אֲבָל לֹא יָדַעְתִּי שֶׁיָּפָה אָנֹכִי.
עַל כֵּן זֹאת אֶעֱשֶׂה לָךְ.

------------------
וַתָּחֶל לָצאֵת לַיְלָה-לַיְלָה בָּאַשְׁמֹרֶת הָאַחֲרוֹנָה, בִּהְיוֹת כָּל אַנְשֵׁי-הַבַּיִת יְשֵׁנִים, אֶל סְבִיבוֹת הָאָחוּ וּשְׂפַת-הַבִּצָּה אֲשֶׁר חֲזִירֵי-הַמֶּלֶךְ רוֹעִים שָׁם, וְאֶל גֵּיא-הֶהָרִים מֵעֵבֶר לָאָחוּ; וּבְיָדֶיהָ שְׁנֵי לַפִּידֵי-אֹרֶן לְהָאִיר דַּרְכָּה וְחִפּוּשֶׂיהָ.

------------------
אַפְרוֹדִיטֵי מִטַּהֶרֶת וְאַפְרוֹדִיטֵי הָלְכָה אֶל פָּפוֹס, עִיר-הַפְּאֵר עַל חוֹף קַפְרִיסִין הָאִי, אֲשֶׁר שָׁם לָהּ הֵיכָל, וַתִּרְחַץ בְּקֶצֶף-הַגַּלִּים, וַיָּשׁוּבוּ אֵלֶיהָ טָהֳרָהּ וְזֹךְ-בְּתוּלֶיהָ כְּקַדְמָתָהּ.
הָדֶס בָּא בְּבַקָּשָׁה לִפְנֵי זֶאוּס, וּבַקָּשָׁתוֹ נִרְצֵית 
הָדֶס, מֶלֶךְ טַרְטָרוֹס וְכָל מְדוֹרֵי-שְׁאוֹל, בָּא לִפְנֵי זֶאוּס בְּרוּם-אוֹלִימְפּוֹס וַיֹּאמַר: "זֶאוּס אָחִי, נְגִיד-הַשְּׁחָקִים, שַׁלִּיט בָּאֵלִים! 
------------------
וְאַתֵּינֵי הִצִּיגָה לִפְנֵי הָאֵלִים שְׁתִיל-זַיִת רַךְ, וַתֹּאמֶר: "זֹאת מַתְּנָתִי אֲנִי לִבְנֵי-אָדָם, וּבְבִרְכָתָהּ פְּרִי וָשֶׁמֶן לְמַאֲכַל דַּלִּים וְלִמְשִׁיחַת-מְלָכִים".
וַיִּשְׁקֹל זֶאוּס בְּלִבּוֹ שַׁי נֶגֶד שַׁי, וַיַּחְתֹּךְ אֹמֶר: "טוֹבָה מַתְּנַת אַתֵּינֵי בִּתִּי מִמַּתְּנַת פּוֹסֵידוֹן אָחִי".
וַתַּעֲמוֹד הָעִיר הַמְּהֻלֶּלֶת לְאַתֵּינֵי עַד עוֹלָם.

------------------
עוֹד הִיא מַבִּיטָה אֶל הָאֶשֶׁד הַמִּתְרַתֵּחַ וְשׁוֹאֵג, וְתוֹהָה מַה תַּעֲשֶׂה, וְזֶאוּס, הָרוֹאֶה כֹל, שָׁלַח לָהּ אֶת נִשְׁרוֹ.
וְהַנֶּשֶׁר נָחַת עַל יָדָהּ, וַיֹּאמֶר לָהּ: "שְׁלִיחַ-זֶאוּס אָנִי.
לֹא יִתֵּן זֶאוּס לְנֶפֶשׁ תַּמָּה כִּי תֹאבֵד.
תְּנִי לִי אֶת הַצְּלוֹחִית, אֵשֶׁת-אֶרוֹס, וַאֲמַלְאֶנָּה".
וַתִּתֵּן לוֹ פְּסִיכֵי אֶת הַצְּלוֹחִית.
וְהַנֶּשֶׁר עָף אֶל לֵב-הַצּוּר מִמַּעַל, אֲשֶׁר שָׁם מוֹצָא-הָאֶשֶׁד, וַיִּשְׁאַב מִן הַמָּיִם; וַיֵּרֶד וַיָּבֵא אֶת הַכְּלִי הַמָּלֵא אֶל הָעַלְמָה.

------------------
הִתְחִילוּ הַבְּרִיּוֹת רוֹאִים אוֹתוֹ כְּאֶחָד קָדוֹשׁ, נִשְׁכַּח מוֹצָאוֹ וְשֵׁם מִשְׁפַּחְתּוֹ, וְהָיוּ קוֹרְאִים שְׁמוֹ: “יוֹנִי זֶרַע-תַּפּוּחַ”.
מִכְתָּבִים
מֶה עָלְתָה לָהֶם לַזֵּרְעוֹנִים אֲשֶׁר זָרַע וְלַשְּׁתִילִים אֲשֶׁר שָׁתַל?
בְּשׁוּב יוֹנָתָן אֶל פִּיטְסְבּוּרְג בִּשְׁבִיל לִצְבֹּר זֵרְעוֹנִים חֲדָשִׁים, הָיָה מוֹצֵא מִכְתָּבִים הַמְּחַכִּים לוֹ. 
------------------
אַתָּה יָכוֹל לִשְׁכַּב עַל הַסְּלָעִים הָעֵירֻמִּים וְלִהְיוֹת נִרְאֶה כְּפֶלַח אֶבֶן-פַּשְׁטִידָה.
אַתָּה יָכוֹל לִשְׁכַּב עַל עָנָף רַב-עָלִים וְלִהְיוֹת נִרְאֶה כְאֶפֶס-וְלֹא-כְלוּם-לָאו-דַוְקָא.
חֲשֹׁב עַל זֹאת וִירַנְּנוּ בְּנֵי-מֵעֶיךָ!
– אִם אֲנִי כָּל זֶה – אָמַר הַנָּמֵר – מַדּוּעַ לֹא אִוִּיתָ לְךָ חֲבַרְבּוּרוֹת גַּם אַתָּה?
הוֹי, לְכוּשׁוֹנִי שֶׁכְּמוֹתִי אֵין טוֹב מִן הַשָּׁחוֹר הַפָּשׁוּט – אָמַר הַכּוּשִׁי.

------------------
וַיִּשְׁאַל נוֹדְדֵי יַבָּשָׁה וְנוֹדְדֵי יָם, וְאִישׁ לֹא יָכֹל לְהַגִּיד לוֹ אַיֵּה מְקוֹם אָרֶס.
וַתֵּצֵא אֶרִיבּוֹיָה לִקְרַאת הֶרְמֶס, וַתֹּאמֶר לוֹ: "שַׁגְרִיר-הָאֵלִים, אַל תְּבַקֵּשׁ עוֹד.
בְּבֵיתִי נִמְצָא אָרֶס.
בְּעַל כָּרְחִי וְשֶׁלֹּא בִרְצוֹנִי הָפְקַדְתִּי שׁוֹמֶרֶת עָלָיו, וְהוּא כָּלוּא בְּדוּד-נְחֹשֶׁת.
בָּנַי הַחוֹרְגִים, תְּאוֹמֵי-הֶעָתָק, כָּפוּ עָלַי אֶת הַדָּבָר בְּאִיּוּמֵי-רֶצַח, וַאֲנִי לֹא יוֹלַדְתָּם, וְאַלּוֹאֵאוֹס בַּעְלִי לֹא מוֹלִידָם.

------------------
הַכֶּלֶב הָיָה מֻכְרָח לְהִכָּנֵעַ לְסֵדֶר לֹא-לוֹ, לְדִין וְתַקָּנָה שֶׁלֹא לַהֲנָאָתוֹ נַעֲשׂוּ.
וְתוּ לֹא? – יֶשְׁנוֹ גַם הַסֵּדֶר שֶׁל מַרְאֵה-עֵינָי.
הַמּוּסָךְ, הֶחָצֵר, הַבַּיִת, הָרְחוֹב כֻּלּוֹ עַל שׁוּרוֹת בָּתָּיו מְשֻׁפְּעֵי הַגַּג, וְהַשִּׁקְמוֹת הָעֵרֻמּוֹת, הַמֻּגְלָדוֹת לִפְנֵיהֶם, – מִתְרַכְּזִים אֵיךְ-שֶׁהוּא בְּכַדּוּרֵי-עֵינַי וּמְהַוִּים בָּהֶם תְּמוּנָה אַחַת, אִרְגּוּן אֶחָד, מִשְׁטָר אֶחָד.
וַהֲלֹא דָּבָר הוּא.

------------------
וַיְהִי כִּי בָאָה אַפְרוֹדִיטֵי אֶל אוֹלִימְפּוֹס לִתְפּוֹס אֶת מְקוֹמָהּ בֵּין הָאֵלִים, וַיַּחְשֹׁק בָּהּ פּוֹסֵידוֹן שַׂר-הַיָּם, וַיִּתְאַו אוֹתָהּ הָדֶס, אֲדוֹן-שְׁאוֹל.
רַק זֶאוּס נִשְׁמַר מִפְּנֵי שִׁעְבּוּד נַפְשׁוֹ לְיָפְיָהּ, כִּי אָמַר: "אִם אֶכָּנַע לִפְנֵי יָפְיָהּ, אֶהְיֶה עֶבֶד לָהּ, וְאֶחְדַּל לִהְיוֹת שַׁלִּיט בָּאֵלִים".
מֶטִיס הָרָה, וְזֶאוּס בּוֹלֵעַ אוֹתָהּ 
אֲבָל חֲגוֹרַת-אַפְרוֹדִיטֵי הָיְתָה גַם לוֹ לְמַרְגִּיז-נֶפֶשׁ וְלִמְעוֹרֵר-תְּשׁוּקוֹת.

------------------
הַפְסָקָה לִטְעִימָה קַלָּה וּלְרִוּוּי-צִמָּאוֹן בְּמַשְׁקֵה-הֲדָרִים, וּלְסִיּוּר בַּצְּרִיפִים הַקְּטַנִּים בְּרֹאשׁ-הַגִּבְעָה, שֶׁעֲדַיִן לֹא נֶחְנְכוּ בְּלִינָה.

------------------
וַיִּתְחַנְנוּ הָאֵלִים לִפְנֵי הֶפֵיסְטוֹס: "אָנָּא, שַׁחְרֵר אֶת הַגְּבֶרֶת, וְיִנָּתֵן לְךָ כָּל אֲשֶׁר תְּבַקֵּשׁ נַפְשֶׁךָ".
סֵרַב הֶפֵיסְטוֹס, כִּי מָתְקָה לוֹ נִקְמָתוֹ, וְדִמְעוֹת-אִמּוֹ לֹא נָגְעוּ עַד לִבּוֹ.
אַךְ לְבַסּוֹף נִמְלַךְ בְּדַעְתּוֹ וַיֹּאמַר: "הִשָּׁבְעוּ לִי כִּי תִנָּתֶן-לִי בַּקָּשָׁתִי".
וַיִּשָּׁבְעוּ לוֹ בְּאָלָה חֲמוּרָה בְּשֵׁם הַנָּהָר סְטִיכְּס.
וַיֹּאמֶר הֶפֵיסְטוֹס: "תִּנָּתֵן לִי אַפְרוֹדִיטֵי הַיָּפָה לְאִשָּׁה".
בְּעַל כָּרְחָם וּלְרַעַץ-רוּחָם נֵאוֹתוּ לוֹ הָאֵלִים.

------------------
וּכְשׁוֹב הָאִישׁ וְהַסּוּס וְהַכֶּלֶב מִצּוּד צַיִד וְהֵם שָׁאֲלוּ אוֹתָן הַשּׁאֵלוֹת אֲשֶׁר מִקֹּדֶם, עָנְתָה הָאִשָּה וְאָמְרָה: “לֹא פָּרַת-הַבָּר יִקָּרֵא עוֹד שְׁמָהּ, כִּי אִם מְפַרְנֶסֶת הַבַּיִת הַטּוֹב.
הִיא תִתֵּן לָנוּ אֶת הֶחָלָב הֶחָם וְהַלָּבָן לְעוֹלָם וּלְעוֹלְמֵי-עוֹלָמִים, וַאֲנִי אַשְׁגִּיחַ עָלֶיהָ, בְּעֵת שֶׁאַתָּה וְהַיְּדִיד הָרִאשׁוֹן וְהָעוֹזֵר הָרִאשׁוֹן יוֹצְאִים לָצוּד צָיִד.”

------------------
זֹאת הִיא הַצָּרָה! הֵם בּוֹזְזִים לְעַצְמָם אֶת כָּל-שְׁמוֹתֵינוּ הַיָּפִים. עוֹד מְעַט וְנִצְטָרֵךְ לִקְרוֹא לִילָדֵינוּ אָנוּ בְּשֵׁמוֹת יְהוּדִיִּים… שְׁלֹמֹה’לִי תַּחַת וַיְזָתָא, משֶׁה’לִי תַּחַת אֲרִידָתָא… וְכֻלָּם מַהְפְּכָנִים, מוֹרְדִים, קוֹשְׁרֵי-קֶשֶׁר, רוֹצִים לְהַנְהִיג חֶבְרָה שֶׁל שִׁוְיוֹן, שֶׁלֹּא יִהְיֶה הֶבְדֵּל בֵּין שַׂר וּבֵין פַּלָּח…
אֲחַשְׁוֵרוֹשׁ:

------------------
וּבִרְבוֹת הַיָּמִים הָיְתָה לְרֹאשַׁת-פֻּלְחָן וְגוֹנֶנֶת לְעִיר אֲשֶׁר נִקְרְאָה עַל-שְׁמָהּ: אַתֵּינָה; הִיא הָעִיר אַתֵּינָה הַמְּהֻלֶּלֶת בַּחֲכָמֶיהָ וּפַסָּלֶיהָ וּמְשׁוֹרְרֶיהָ, מְגַלְּמֵי חֶזְיוֹנוֹתֵיהֶם בִּדְרָמוֹת וּמַקְהֲלוֹת-שִׁיר לִפְנֵי אַלְפֵי מַקְשִׁיבִים בְּאַמְפִיתֵיאַטְרוֹן.
וַיְחַבֵּב זֶאוּס אֶת אַתֵּינֵי יוֹתֵר מִכָּל זַרְעוֹ, אֲשֶׁר יֻלְּדוּ לוֹ אַחֲרֵי-כֵן.
כִּי כֻלָּם יְלִידֵי-רֶחֶם הָיוּ, וְהִיא יְלִידַת-מֹחוֹ, אַחֲרֵי אֲשֶׁר בָּלַע אֶת אִמָּהּ מֶטִיס.

------------------
אֵם-הַבָּנִים עָמְדָה עַל יַד שֻׁלְחַן-הָעֵץ הַחַף מִצֶּבַע, וְקָלְפָה בִּמְתִינוּת שָׁרְשֵׁי-סֶלֶק, כְּאֶגְרוֹף גָּדְלָם, הִיא הָיְתָה לְבוּשָׁה פִּיגַ’מַת-מֶשִׁי שְׁחוֹרָה עִם שׁוּלַיִם אֲדֻמִּים לִקְצוֹת בָּתֵּי-הָרֶגֶל, חֲגוֹרָה אֲדֻמָּה לַמָּתְנַיִם, וְזֵר אָדֹם מִסָּבִיב לַצַּוָּאר. 
------------------
וּמִשְתַּקְּפִים בְּשִׁירֵיהֶם וְסִפּוּרֵיהֶם יְפִי הָעִיר, וּשְׁמֵי אֶרֶץ-יִשְׂרָאֵל, וְגִבְעוֹתֶיהָ וַהֲרָרֶיהָ, וְיַמָּהּ, וְהַחַי וְהַצּוֹמֵחַ אֲשֶׁר בָּהּ, וְרִנַּת עוֹבְדֶיהָ בַּשָּׂדֶה וּבְבֵית-הַחֲרשֶׁת, וְשִׂמְחַת אִמָּהוֹת בְּרִיאוֹת בְּהֵרָיוֹן וּבִצְמִיחַת תִּינוֹקוֹת.
שֵׁבֶט אֶחָד הֵם הַסּוֹפְרִים עִם כָּל הָעוֹבְדִים, הַבּוֹנִים אֶת בֵּית ה' עַל אַדְמָתוֹ בְּצֶדֶק וּלְתִפְאֶרֶת”.
------------------
וַיַּזְהֵר הַמֶּלֶךְ מִידַס אֶת הַגַּלָּב שֶׁלּוֹ, לֵאמֹר:"אִם תְּגַלֶּה אֶת סוֹד מַרְאֵה-אָזְנַי לִבְרִיָּה אַחַת, יוּסַר רֹאשְׁךָ מֵעָלֶיךָ.
שָׁמוֹר וְזָכוֹר!"
סוֹף סוֹד לָצֵאת 
וְהַגַּלָב, מִיִּרְאַת הַמֶּלֶךְ וּמֵחֻמְרַת מִצְוָתוֹ, שָׁמַר אֶת הַסּוֹד בְּלִבּוֹ.
לֹא הִגִּידוֹ לְאָבִיו וְאִמּוֹ, לֹא הִגִּידוֹ לְאֶחָיו וְאַחְיוֹתָיו, לֹא הִגִּידוֹ לְבָנָיו וּבְנוֹתָיו, וְגַם לְאִשְׁתּוֹ לֹא הִגִּידָהוּ.

------------------
צִפּוֹר רַמְזָנִית בּוֹ בָרֶגַע עוֹפְפָה צִפּוֹר מִמַּעַל, וְנִפְנְפָה בִּכְנָפֶיהָ אָנֶה וְאָנָה.
וַיָּבֶן אַפּוֹלוֹן אֶת רִמְזֵי-הַצִּפּוֹר, וַיֵּדַע יַד-מִי הָיְתָה בְּפָרוֹתָיו.
וַיְכַוֵּן רַגְלָיו אֶל הַר-קִילֶנֵי, וַיָּבוֹא אֶל הַמְּעָרָה, וַיִּקְרָא קוֹל גָּדוֹל: "גַּנָּב מִלֵּדָה, אֵיפֹה תָּחַבְתָּ אֶת פָּרוֹתַי?"
"עַל מִי הֵרַמְתָּ קוֹלֶךָ?" שָׁאֲלָה מַיָּה, "וּפֹה אֵין בִּלְתִּי אִם אֵם מֵינֶקֶת וְתִינֹוק קָטָן שֶׁלָּהּ".

------------------
וְאוּלָם לִפְנוֹת עֶרֶב, כְּשֶׁבְּנֵי-אָדָם וְכָל הַיְּצוּרִים נַעֲשִׂים חִדְלֵי-מַרְגּוֹעַ וַעֲיֵפִים, בָּא הָאָדָם (עִם בִּתּוֹ הַקְּטַנָּה?) – כֵּן, עִם בִּתּוֹ הַקְּטַנָּה הָאֲהוּבָה-בְּיוֹתֵר וְהִיא יוֹשֶׁבֶת עַל שִׁכְמוֹ, וְאָמַר: “מַה הַמִּשְׂחָק הַזֶּה, זְקַן-אַשָּׁפִים?” עָנָה וְאָמַר זְקַן-הָאַשָּׁפִים: “הוֹי בֶּן-אָדָם, זֶהוּ הַמִּשְׂחָק שֶׁל תְּחִלַּת-הַתְּחִלּוֹת; וְאוּלָם אַתָּה חָכָם מִדַּי מִלְּשַׂחֵק זֶה הַמִּשְׂחָק.” וְהָאָדָם הֶחֱוָה קִדָּה

------------------
הִנֵּה אָהֳלֵיהֶם הַקְּטַנִּים, הַמְּאֻבָּקִים, תְּקוּעִים בְּפִזּוּר עַל שִׁפּוּעָהּ שֶׁל גִּבְעָה.
בְּפִנָּה שֶׁלְּרַגְלֵי הָאֳהָלִים – בִּצְבּוּצִים זְעִירִים שֶׁל נִסּוּיֵי-מֶשֶׁק.
וּבְרֹאשׁ-הַגִּבְעָה – צְרִיפִים אֲחָדִים, רַעֲנַנֵּי-צֶבַע.
וַהֲרֵי הֵם עַצְמָם, הַצְּעִירִים.
צוֹחֲקִים וּפְרוּעֵי-רֹאשׁ הֵם רָצִים לִקְרָאתֵנוּ.
אָנוּ פּוֹרְצִים בְּ*"הֵבֵאנוּ שָׁלוֹם עֲלֵיכֶם* ".
מִמִּי אָנוּ מְבִיאִים שָׁלוֹם עֲלֵיכֶם? אוּלַי מִתֵּל-אָבִיב, אוּלַי מִבְּאֵר-שֶׁבַע.
אַךְ יוֹתֵר מִזֶּה:

------------------
אָז צָחַק גַּם הָאִישׁ הַזָּר; אָז צָחַק טֶגּוּמֵי עַד שֶׁנָּפַל פְּרַקְדָּן עַל שְׂפַת הַנָּהָר; אָז צָחֲקוּ כָל אַנְשֵׁי הַשֵּׁבֶט, וְהוֹסִיפוֹ לִצְחוֹק קוֹל גָּדוֹל.
רַק טֶשּׁוּמֵי וְהַגְּבִירוֹת מִבְּנוֹת תְּקוּפַת הָאֶבֶן הַחֲדָשָׁה, לֹא צָחָקוּ.
הֵן נָהֲגוּ דֶּרֶךְ-אֶרֶץ גָּדוֹל בְּבַעֲלֵיהֶן, וְקָרְאוּ לָהֶם “שׁוֹטִים” ו“טִפְּשִׁים” הַרְבֵּה-הַרְבֵּה פְּעָמִים.

------------------
וּנְמָלָה קְטַנָּה בְּשׁוּלֵי-הַדֶּשֶׁא אֲשֶׁר לַחֲצַר-הַהֵיכָל, רָאֲתָה בְּצַעַר הָעַלְמָה הַיָּפָה, וַתַּחְמֹל עָלֶיהָ, וַתִּקְרָא לָהּ בְּקוֹל דַּק מִתּוֹךְ דֶּשֶׁא-הָעֵשֶׂב: "אַל תִּדְאֲגִי, הַיָּפָה! הִנְנִי וְאֶקְרָא לְאַחְיוֹתַי, וְתָבֹאנָה לְעֶזְרָתֵךְ".
וַתִּקְרָא הַבְּרִיָּה הַזְּעִירָה טוֹבַת-הַנְּשָׁמָה לְאַחְיוֹתַיהָ הֶחָרוּצוֹת וַתָּבֹאנָה לְמֵאוֹת וְלַאֲלָפִים.

------------------
פְּסִיכֵי לְבַדָּהּ בְּרֹאשׁ-הַכֵּף וַיְהִי לְעֵת-עֶרֶב, בְּהִטָּלֵא שַׁחַק מוֹרִיק עָבֵי-חַכְלִיל אֲרֻכִּים, וַיְלַוּוּהָ אָבִיהָ וְאִמָּהּ, וְאִתָּם רַבִּים מִן הָעָם נוֹשְׂאֵי לַפִּידֵי-כְלוּלוֹת, אֶל כֵּף נִשָּׂא אֲשֶׁר בִּקְצֵה גִבְעָה.
וְשָׁם, הַרְחֵק מִתַּחַת, יַהַרְסוּ מִשְׁבְּרֵי-יָם בֵּין סְלָעִים בְּמִרְתַּח-רָעַשׁ; וְהִיא לְבוּשָׁה פְּאֵר וּמְעֻטֶּרֶת כְּכַלָּה.

------------------
ד) מְכַבְּדִים שֵׁם חָבֵר-וְגָדוֹל בְּצוּרַת-הַנְצָחָה הַחוֹפֶפֶת אֶת שֶׁטַח-פְּעֻלָּתוֹ שֶׁל הָאִישׁ, וְאִם אֶפְשָׁר, בִּמְקוֹם-מְגוּרָיו הָאַחֲרוֹן, וְאֵין מַרְכִּיבִים סְתָם שֵׁם לִסְתָם מִפְעָל עַל יְסוֹד “אַסְטְרוֹלוֹגִי” – דְּהַיְנוּ, שֶׁפְּטִירָתוֹ שֶׁל זֶה נָפְלָה בְּמִקְרֶה בִּסְמִיכוּת-זְמַן לְיִסּוּדוֹ שֶׁל זֶה.
ה) סוֹפְרִים וָאֳמָנִים – מְלַאכְתָּם הִיא הַנְצָחָתָם.
אֲדָמִי – אִדְמִית – אֲדָמִית
לְאוֹר הַבְּחִינוֹת הַלָּלוּ, לֹא יַעֲמוֹד “אַדְמוֹת יִצְחָקִי” 
------------------
"מָה הַשַּׁחַק, וּמָה אַתֶּם, הַמְּאוֹרוֹת הַדָּרִים בְּמֶרְחָבָיו? שְׁבִיבִים קְטַנִּים, דְּלוּקֵי חֵשֶׁק נִסְתָּר; טִפּוֹת קְטַנּוֹת, יִדְחָפֵן וְיַזְרִימֵן לֵב-הַטֶּבַע בְּעוֹרְקִים דַּקִּים וְעִוְרִים. 
“מָה הוּא הַשַּׁחַק? אֵגֶל-טַל אֶחָד, רְסִיס-שַׁחַר זַעֲרוּר, זוֹרֵחַ לְפִי-תֻמּוֹ עַל אֵיזֶה פֶּרַח הַפּוֹקֵחַ עֵינַיִם רַכּוֹת בְּעוֹלָם שַׂגִּיא מֵהַשָּׂגָתֵנוּ. 
------------------
וְאוּלָם, בִּזְמַן שֶׁהַלִּוְיָתָן עֲדַיִן הָיָה שׂוֹחֶה, לָקַח זֶה הַסַּפָּן, שֶׁהָיָה אִישׁ רַב-בִּינוֹת עַד אֵין סוֹף וַחֲכַם-תַּחְבּוּלוֹת עַד אֵין חֵקֶר, אֶת אוֹלָרוֹ, עָמַד וְחִתֵּךְ אֶת רַפְסוֹדָתוֹ לִשְׂבָכָה קְטַנָּה וּמְרֻבַּעַת, קִשֵּׁר אוֹתָהּ קַשֵּׁר הֵיטֵב בְּקִשּׁוּרֵי-מִכְנָסָיו (וְעַתָּה תָבִינוּ לָמָּה לֹא נְתַתִּיכֶם לִשְׁכּוֹחַ אֶת קִשּׁוּרֵי הַמִּכְנָסַיִם), סָחַב אֶת הַשְּׂבָכָה אֶל תּוֹךְ לוֹעַ הַלִּוְיָתָן, שָׁם הֶעֱמִיד אוֹתָהּ וְהִדֵּק אוֹתָהּ יָפֶה-יָפֶה, וְשָׁם הִיא דְבוּקָה
------------------
עֲנַף-תַּפּוּחַ קָטַפְנוּ, טוֹבֵל בְּלַבְנוּנִית-פְּרִיחָה, אַגָּס בְּלִבְלוּבוֹ שָׁדַדְנוּ, בְּזֵרֵי-לוֹט קְשַׁרְנוּם וּבְמִבְחַר כּוֹכָבִיּוֹת כְּחַלְחַלּוֹת: מַתְּנַת עֶקְרוֹן הִיא בִּידֵי שְׁלוּחֶיהָ אֵלַיִךְ, דְּלִילָה, חֶבֶל אָסַר שִׁמְשׁוֹן וּמְשָׁכָהוּ בָּטֵל לְחֶרְפָּתוֹ.
סַרְנֵי גַת:

------------------
וְלֹא יִהְיוּ אֵיפָה וְאֵיפָה, אַמָּה וְאַמָּה, אֶבֶן-מִשְׁקָל וְאֶבֶן-מִשְׁקָל; אַךְ מִקָּח כִּי קָם, אֵין חוֹזְרִים מִמֶּנּוּ.
הֶרְמֶס מַמְצִיא קֻבִּיּוֹת-מִשְׂחָק 
וּמֶה עָשָׂה הֶרְמֶס בְּשָּׁעָה הַהִיא? שִׁמְעוּ וּתְמָהוּ.
עָשָׂה עַצְמוֹ כְּמַעֲמִיק הַקְשָׁבָה לְדִבְרֵי זֶאוּס אָבִיו, אֲבָל מִתַּחַת לְשֻּׁלְחָן עִצֵּב בְּיָדָיו צַעֲצוּעַ חָדָשׁ, זוּג קֻבִּיוֹת מִקֶּרֶן-תַּיִשׁ.

------------------
יִשָּׁמֵר, אֵפוֹא, הַשֵּׁם שֶׁלַח בִּשְׁבִיל וְלַד-יִשּׁוּב שֶׁעֲדַיִן לֹא נוֹלַד, וּכְבָר הוּא בְּרֶחֶם-הַתִּכְנוּן; אֲזַי יַרְגִּישׁוּ הַנִּשְׁלָחִים שָׁמָּה לִמְלָאכָה וְלִשְׁמִירָה בְּיָפְיוֹ וְיָאוּתוֹ שֶׁל אוֹתוֹ הַשֵּׁם לְמִן תְּחִלַּת-הַהַתְחָלוֹת.
וְהַ“שֵּׁיזָפִים” שֵׁיזָפִים יִשָּׁאֵרוּ.
בִּגְלַל אָבוֹת
עַד שֶׁלֹּא יָצָאנוּ בַּחֲזָרָה אֶל בְּאֵר-שֶׁבַע, עָטַף עֲרָפֶל סָגֹל אֶת מֶרְחַקֵּי-סִינַי.

------------------
– חטֶם זֶה מַתָּנָה הוּא לִי מִן הַתַּנִּין עַל גְּדוֹת נְהַר-הַלִּימְפּוֹפּוֹ הַגָּדוֹל, הַיָּרֹק-אָפֹר וְהַמְּשֻׁמָּן, – אָמַר הַפִּילוֹן – שָׁאַלְתִּי אוֹתוֹ מַה הוּא נוֹהֵג לֶאֱכֹל לִסְעֻדַּת-צָהֳרַיִם שֶׁלּוֹ, וְהוּא נָתַן לִי אֶת זֶה לִצְמִיתוּת.
– הוּא מְכֹעָר מְאֹד לְמַרְאֶה, – אָמַר דּוֹדוֹ הַשָּׂעִיר הַבַּבּוּן.
– צָדַקְתָּ, – אָמַר הַפִּילוֹן – אֶלָּא שֶׁהוּא גַם מוֹעִיל מְאֹד, – וְהוּא תָפַס רֶגֶל אַחַת שְׂעִירָה שֶׁל דּוֹדוֹ הַשָּׂעִיר, הַבַּבּוּן, וּזְרָקוֹ אֶל תּוֹךְ קַן-צְרָעוֹת.

------------------
 כָּל רִצְפַּת הַמְּעָרָה, עַד שֶׁעָיַף וְנִרְדַּם כְּשֶׁהֶחָתוּל בִּזְרוֹעוֹתָיו.
– וְעַתָּה, – אָמַר הֶחָתוּל, – אֲזַמֵּר לַתִּינוֹק זֶמֶר אֲשֶׁר יַאֲרִיךְ אֶת שְׁנָתוֹ שָׁעָה אֶחָת.
וְהִתְחִיל רוֹטֵן חֶרֶשׁ וּבְקוֹל, בְּקוֹל וְחֶרֶשׁ, עַד שֶׁשָּקַע הַתִּינוֹק בְּשֵׁנָה עֲמֻקָּה.
חִיְּכָה הָאִשָּה בְּהַבִּיטָּה עַל שְׁנֵיהֶם וְאָמֱרָה: “יָפֶה נַעֲשָׂה הַדָּבָר.
אָכֵן, פִּקֵּחַ אַתָּה מְאֹד, הוֹ חָתוּל!”

------------------
– “וְלֹא תֵדַע”, עָנְתָה אִשְׁתִּי, “כִּי חַזָּן הַנֶּהֱנֶה מִתְּפִלַּת עַצְמוֹ, גַּם הַקָּדוֹשׁ בָּרוּךְ-הוּא נֶהֱנֶה מִתְּפִלָּתוֹ?” הִיא נָשְׂאָה עֵינֶיהָ הַצְּלוּלוֹת, וַתּוֹסֶף: “כָּל דְּבֵקוּת-נְשָׁמָה שֶׁל-אֱמֶת הִיא דְּבֵקוּת בֵּאלֹהִים. אֲמִתּוּתָהּ שֶׁל הַדְּבֵקוּת הִיא הַקּוֹבַעַת אֶת אֱלֹהוּתָהּ, וְלֹא הַדָּבָר אֲשֶׁר דְּבֵקִים בּוֹ; כִּי אֵין עֶצֶם, אֵין חֵשֶׁק וְאֵין תְּנוּעָה, שֶׁאֵינוֹ יָכוֹל לְהֵעָשׂוֹת חַלּוֹן לְגִלּוּי; וְגַם זִמְרָה כָּךְ, וְגַם אוֹפֵּירָה, וְגַם שַׂחֲקָנוּת”.

------------------
יוֹם אֶחָד לָקַח קֶמַח וּמַיִם וְדֻמְדְּמָנִיּוֹת וּשְׁזִיפִים וְסֻכָּר וּשְׁאָר יְרָקוֹת, וְעָשָׂה לוֹ עֻגָה, שְׁנֵי טְפָחִים אָרְכָּהּ וּשְׁלֹשָׁה טְפָחִים עָבְיָהּ וְהִיא הָיְתָה טוֹבָה לְמַאֲכָל מֵאֵין כָּמוֹהָ וְהוּא שָׂם אוֹתָהּ עֻגָה בַּתַּנּוּר, כִּי הָיְתָה לוֹ רְשׁוּת לְבַשֵּׁל וְלֶאֱפוֹת בְּאוֹתוֹ תַּנּוּר. וְהוּא אָפָה אוֹתָהּ וְאָפָה אוֹתָהּ עַד שֶׁהָיְתָה עֲשׂוּיָה כֻלָּהּ חוּמָה, וְרֵיחָהּ מְעוֹרֵר אֶת הָרֶגֶשׁ בְּיוֹתֵר. 
------------------
מִשְׁבָּר שֶׁהוּא רָחוֹק מִן הַחוֹף, הוּא גַם בַּגְּדוֹלִים שֶׁבָּהֶם, יָרֹק הוּא מִתְרוֹמֵם, נִצָּב לְשׁוֹן-אִזְמָרַגְדְּ, וְעַד שֶׁצּוּרָתוֹ שְׁלֵמָה, הוּא כּוֹשֵׁל, מִתְמוֹטֵט, קְוֻצָּתוֹ נוֹפֶלֶת עַל גוּפוֹ בְּאֶשֶׁד-קְצָפִים גּוֹלֵשׁ; עוֹד עַמּוּד-קֶצֶף נִזְרָק מִמֶּנּוּ כְּלַפֵּי מַעְלָה, וְהוּא כְּבָר הִתְעַגֵּל, הָשְׁבַּר וְאֵינֶנוּ, וּקְצָפָיו מִתְפַּזְּרִים וְנִשָּׂאִים עַל גַּל חָדָש הַקָּם. 
------------------
וּמְלַטְּפֵנִי רֶגֶשׁ חָמִים: קָם לְפָנַי וָחַי הָעֵדֶר שֶׁהָיָה נוֹהֵר מֵעֲיָרָתִי אֶל הַשָּׂדוֹת בְּקַדְמוּת-יַלְדוּתִי.
מִטָּתִי פְּרוּעָה, וְעֵינַי מִתְעַצְּמוֹת; אֲבָל בּוֹשׁ אֲנִי מֵהִכָּנֵס אֶל בֵּין הַשְּׂמִיכוֹת בְּשָׁעָה שֶׁאַנְשֵׁי הַכְּפָר יוֹצְאִים לִמְלַאְכתָּם הַתַּמָּה.
------------------
וַיִּמְצְאוּ אֶת אָרֶס מִתְעַלֵּף מֵחֹסֶר אֲוִיר וּמָזוֹן; וְלוּלֵי הָיָה בֶּן-אַלְמָוֶת, אֲזַי מִכְּבָר מֶת.
וַיִּצְּקוּ נֶקְטַר אֶל פִּיו, וַיַּאֲכִילוּהוּ אַמְבְּרוֹסְיָה, וַתָּשָׁב אֵלָיו רוּחוֹ.
אָז עָלוּ שְׁלֹשֶׁת הָאֵלִים אֶל אוֹלִימְפּוֹס, וְכָל דָּרֵי-אוֹלִימְפּוֹס קִבְּלוּ פְּנֵיהֶם בְּשִׂמְחָה וּבְמִשְׁתֶּה.
עֵינֶי אָרֶס בְּאַפְרוֹדִיטֵי 

------------------
שְׁמוֹ הָיָה טֶגּוּמֵי בֶּפְּסוּלֵי, וּפֵרוּשׁוֹ שֶׁל שֵׁם זֶה: “הָאִישׁ-שֶׁאֵינוֹ-דּוֹחֵק-אֶת-הַשָּׁעָה.” וְאוּלָם אֲנַחְנוּ, הוֹ חֲבִיבַי וַאֲהוּבַי, נִקְרָא לוֹ לְשֵׁם קִצּוּר, טֶגּוּמֵי, וְשֵׁם אִשְׁתּוֹ הָיָה טֶשּׁוּמֵי טֶוִּינְדְרוֹ, וּפֵרוּשׁוֹ שֶׁל שֵׁם זֶה: “הָאִשָּׁה-הַשּׁוֹאֶלֶת-שְׁאֵלוֹת-הַרְבֵּה-עַד-אֵין-שִׁעוּר.” וְאוּלָם אֲנַחְנוּ, הוֹ חֲבִיבַי וַאֲהוּבַי, נִקְרָא לָהּ לְשֵׁם קִצּוּר טֶשּׁוּמֵי.

------------------
– לֹא – אָמַר הָאִישׁ. – פַּעַם אַחַת בַּיּוֹם וּפַעַם אַחַת בַּלַּיְלָה, הַיָּם גּוֹאֶה וְעוֹלֶה אֶל נְהַר-פֵּרָק וְדוֹחֵף אֶת הַמַּיִם הַמְּתוּקִים בַּחֲזָרָה אֶל תּוֹךְ הַיַּעַר, עַד שֶׁבֵּיתִי כֻּלּוֹ נִרְטָב. פַּעַם בַּיּוֹם וּפַעַם בַּלַּיְלָה הוּא יוֹרֵד אֶל הַנָּהָר וּמוֹשֵׁךְ אֶת כָּל הַמַּיִם אַחֲרָיו, עַד שֶׁלֹא נוֹתָר בִּלְתִּי אִם בֹּץ, וְהַדּוּגִית שֶׁלִּי נֶהְפֶּכֶת. הַאִם זֶה הַמִּשְׂחָק שֶׁצִּוִּיתָ עָלָיו לְשַׂחֵק?
– לֹא – אָמַר זְקַן-הָאַשָּׁפִים. – זֶהוּ מִשְׂחָק חָדָשׁ וָרַע.

------------------
עִם כָּל זֶה, לֹא אָמַר מְאוּמָה וְנָחָם אֶל שְׂפַת הַנָּהָר, וְשָׁם מָצְאוּ אֶת טָפִי כְשֶׁהִיא עוֹשֶׂה זֵרֵי מַרְגָּנִיּוֹת, וְאֶת טֶגּוּמֵי כְּשֶׁהוּא נוֹעֵץ חֲנִיתוֹ הַמְּתֻקֶּנֶת בִּזְּהִירוּת בִּדְגֵי-שִׁבּוּט קְטַנִּים.
– אָכֵן, הָיִיתָ זָרִיז מְאֹד, הַזָּר! – אָמְרָה טָפִי.
– וְאוּלָם לָמָּה אַתָּה מֵבִיא אִתְּךָ אֶת כָּל הָעָם הָרַב הַזֶּה? אַבָּא יַקִּירִי, זֹאת הִיא הָאַפְתָּעָה שֶׁלִּי.
הַאִם אַתָּה מֻפְתָּע, אַבָּא?

------------------
הוֹרִיד הַפִּילוֹן אֶת רֹאשׁו סָמוּךְ לְפִי הַתַּנִּין הָרֵיחָנִי הַשִּׁנָּנִי, וְהַתַּנִּין תְּפָסוֹ בְּחָטְמוֹ הָקָּטָן, אֲשֶׁר עַד הַשָּׁבוּעַ הַהוּא וְהַיּוֹם הַהוּא, וְהַשָּׁעָה הַהִיא, וְהָרֶגַע הַהוּא לֹא הָיָה גָדוֹל מִכַּנַעַל, אִם כִּי לֹא הַרְבֵּה יוֹתֵר מוֹעִיל מִנַּעַל.
– כְּסָבוּר אֲנִי, – אָמַר הַתַּנִּין, וְהוּא אָמַר זֹאת מִבֵּין שִׁנָּיו, כָּכָה – כְּסָבוּר אֲנִי שֶׁהַיּוֹם אֶפְתַּח בְּפִילוֹן!

------------------
וּבְאֵיזֶה שֶׁפַע שֶׁל צוּרוֹת שָׁבִים הַמַּיִם אָחוֹר. בְּכֹחַ הֵם שָׁבִים, סוֹרְקִים וְסוֹחֲפִים הַחוֹלוֹת; וְהֵם עֲשׂוּיִים זְרָמִים וּפְלָגִים וְקִלּוּחִים, מִתְנַגְּשִׁים זֶה עִם זֶה, מִתְכַּנְּסִים אִישׁ לִרְשׁוּת חֲבֵרוֹ, סַבִּים בְּסִבּוּבִים סְחַרְחָרִיִּים, תּוֹסְסִים וְרָצִים. צֶדֶף כִּי יִתְעַקֵּשׁ וְלֹא יַחְפֹּץ לָרוּץ אִתָּם, הֲרֵי הֵם נוֹטְלִים הַחוֹל מִסָּבִיב לוֹ, חוֹרְצִים חָרִיץ בְּהֶקֵּפוֹ, עַד שֶׁהוּא מַתְחִיל שׁוֹקֵעַ. 
------------------
אַגְדִּיסְטִיס, יְצוּר רַב-אֵימוֹת; הָאֵלִים אֲשֶׁר בְּאוֹלִימְפּוֹס חִדְלֵי-כֹחַ מוּל אַגְדִּיסְטִיס פַּעַם שָׁכַב זֶאוֹס עַל אֶבֶן לָנוּח: אֶבֶן-אַגְדּוֹס שְׁמָהּ בְּפִי אַנְשֵׁי-הַסְּבִיבָה.
וַתִּתְחַמֵּם הָאֶבֶן מִכֹּח זֶאוּס, וַתֵּלֶד יְצוּר רַב-אֵימוֹת, אַגְדִּיסְטִיס שְׁמוֹ.
גַּם זָכָר וְגַם נְקֵבָה הַיְּצוּר הָאֶחָד; וְרַעֲבוֹנוֹ כָפוּל, צִמְאוֹנוֹ כָפוּל, וְגַעַשׁ-מִשְׁנֶה כָּל יְצָרָיו.
בְּחֶדְוַת-אַכְזְרִיּוּת יָצָא אַגְדִּיסְטִיס לָשֹׁד, לִרְצוֹח, לְהַשְׁמִיד וּלְהַשְׁחִית.

------------------
הֵם בַּדֶּרֶךְ אֶל מִרְעֵה-הַבָּקָר בְּהַר-פְּיֶרִיָּה, וַיּוֹצֵא הֶרְמֶס אֶת כְּלִי-הֶזֶּמֶר שֶׁלּוֹ – אֶת קְלִפַּת-הַצָּב אֲשֶׁר עָשָׂה אוֹתָהּ לְלִירָה – וַיָּחֶל לְנַגֵּן, נַגֵּן וְתֵת קוֹלוֹ בְּשִׁיר.

------------------
בַּשָּׁעָה הַהִיא עָזַב הֶפֵיסְטוֹס אֶת בֵּית-מְלַאכְתּוֹ, וַיָּבוֹא פִּתְאֹם אֶל חֲדַר אַפְרוֹדִיטֵי אִשְׁתּוֹ, וְהִנֵּה הִיא צְווּתָה עִם אֵל-הַמִּלְחָמָה.
חָרְדוּ הַשְּׁנַיִם לָקוּם, אֲבָל מָצְאוּ כִּי חֲבוּשִׁים וּכְבוּשִׁים הֵם תַּחַת רֶשֶׁת-הַזָּהָב, זוּ טָמַן לָהֶם הֶפֵיסְטוֹס.
וַיִּקְרָא הֶפֵיסְטוֹס לְכָל הָאֵלִים דָּרֵי-אוֹלִימְפּוֹס כִּי יִהְיוּ עֵדִים לִבְגִידַת אַפְרוֹדִיטֵי וְחַטַּאת אָרֶס.
אָרֶס וְאַפְרוֹדִיטֵי בְּמִשְׁבַּתָּם, פּוֹסֵידוֹן וְהֶרְמֶס בִּתְשׁוּקָת 

------------------
בְּיוֹם סִיוָן, עֵת צַבָּרֵי הַגְּדֵרוֹת מִתְפָּרְחִים בִּגְבִיעִים צְהֻבִּים וַאֲדַמְדַּמִּים, וְעִשְּׂבֵי-מִרְעֶה מִתְחָרְכִים-מִשְׁתַּחֲמִים, וּמַטְלִיּוֹת-אוֹר נִקְווֹת אַרְגָּמָן בִּדְגָנִים רוֹטְטִים, – נָסְעָה קְבוּצַת הוֹרִים וִידִידִים (מֵעָרֵי-הַחוֹף, מִן הָעֵמֶק, מִירוּשָׁלַיִם) נִכְחָם דָּרוֹמָה אֶל חֲבִיבֵיהֶם, בָּנִים וּבָנוֹת, בִּמְקוֹם-הֵאָחֲזוּתָם בְּתַחְתִּית-הַנֶּגֶב.

------------------
וּכְבָר רָאָה עוֹרוֹתֵיהֶן שֶׁל שְׁתֵּי פָּרוֹת מִתְיַבְּשִׁים עַל סָלַע.

"מֶה עָשִׂיתָ בַּשְּׁתַּיִם?" שָׁאַל אַפּוֹלוֹן אֶת הֶרְמֶס.
שְׁנֵים-עָשָׂר גְּדוֹלֵי אוֹלִימְפּוֹס 
"בָּתַרְתִּי אוֹתַן לִשְׁנֵים-עָשָׂר בְּתָרִים לְמַעַן הֱיוֹתָן קָרְבָּן לִשְׁנֵים-עָשָׂר הָאֵלִים הַגְּדוֹלִים אֲשֶׁר בְּאוֹלִימְפּוֹס".

------------------
זֶאוּס מִתְגַּלֶּה לְסֶמֶּלֵי בִּמְלֹא-כְבוֹדוֹ, וְהִיא נִשְׂרָפֶת; העֻבָּר נִצָּל 
וַתִּפָּת סֶמֶּלֵי.
וּבַלַּיְלָה הַהוּא, כְּהִתְרָאוֹת הַנֶּאֱהָבִים יַחַד, אָמְרָה סֶמֶּלֵי: "מַה לְּאוֹהֲבִי כִּי בְּמִתְחַפְּשִׂים יְשַׂחֵק עִמָּדִי.
הֵיטֵב יָדַעְתִּי כִּי לֹא רוֹעֶה מֵעַם-הָאָרֶץ אָתָּה.
הַרְאֵנִי אֶת מְלֹא כְבוֹדְךָ הָאֲמִתִּי, וְאֵדַע כִּי בֶּאֱמֶת אֲהַבְתָּנִי".
וַיַּפְצֵר בָּהּ זֶאוּס: "בִּתִּי, אַל תְּבַקְשִׁי זֹאת מִיָּדִי.

------------------
הִנְנִי הוֹלֵךְ אֶל נְהַר-הַלִּימפּוֹפּוֹ הַגָּדוֹל הַיָּרוֹק-אָפֹר וְהַמְּשֻׁמָּן, שֶׁהוּא מֻקָּף עֲצֵי קַדַּחַת וְאַשְׂבִּיעַ אֶת תַּאֲבוֹנִי לָדַעַת מַה יֹּאכַל הַתַּנִּין לִסְעֻדַּת-צָהֳרַיִם שֶׁלוֹ.” וְהֵם הִלְקוּהוּ עוֹד פַּעַם לְסִמָּן טוֹב וּלְמַזָּל טוֹב, אַף-עַל-פִּי שֶׁהוּא בִּקֵּשׁ מֵהֶם בְּנִמּוּס וּבְדֶרֶךְ-אֶרֶץ שֶׁיֶּחְדְּלוּ מִכָּךְ.

------------------
מָה רָעָה אֶעֱשֶׂה, אִם אֶקַּח מְעַט מִמֶּנָּה גַם לְעַצְמִי, וְאִיף בְּעֵינֵי בַעְלִי כַּאֲשֶׁר אֶמְצָאֵהוּ?" וַתִּפְתַּח אֶת הַקֻּפְסָה, וַתָּסָךְ אֶת בְּשָׂרָהּ מִן הַמִּשְׁחָה, מַעֲנִיקַת יְפִי-הַנֶּצַח.
וַתֵּהָפֵךְ הַמִּשְׁחָה לְאֵד-עֲרָפֵל, עוֹטֵף אֶת כָּל גּוּפָהּ; וְעַל נַפְשָׁהּ נָפְלָה תַרְדֵּמַת-עִלָּפוֹן, דּוֹמָה לְמָוֶת.
כִּי זֶה פַּח, טָמְנָה אַפְרוֹדִיטֵי לָעַלְמָה, גַּם אִם תַּצְלִיחַ לָצֵאת חַיִּים מִשְּׁאוֹל.

הֶרָה מַפְגִּיעָה בְּאַפְרוֹדִיטֵה בְּעַד פְּסִיכֵי 

------------------
הַנַּעַר אֶלְשָׂרָף שָׁכַב לַיְלָה אֶחָד עַל יְצוּעוֹ הַדַּל בַּחֲדַר-עֲלִיָּתוֹ, וְהִבִּיט אֶל הָאֶשְׁנָב הַפָּתוּחַ – מִסְגֶּרֶת לְרִבּוּעַ-שָׁמָיִם כָּחֹל-אַפְלוּלִי, מְשֻׁבָּץ וְרָצוּף חֲבוּרַת כּוֹכָבִים, יוֹצְקֵי כֶסֶף וִירַקְרַק-אוֹרָה.
הִתְבּוֹנֵן וְרָאָה כִּי אַחַד הַכּוֹכָבִים גָּדוֹל וְעָגֹל הוּא מִכָּל אֶחָיו: אוֹר לָבָן לוֹ, וְהוּא נוֹצֵץ בִּבְרַק-רָז מַבְטִיחַ מָה. נִקְשְׁרוּ עֵינֵי הַנַּעַר אֶל הַכּוֹכָב הַהוּא, עַד בְּלִי שְׂטוֹת מֵעָלָיו.

------------------
וּבַיָּמִים הַתְּמִימִים הָהֵם לֹא חָשְׁבוּ מְלָכִים לִפְחִיתוּת-כָּבוֹד לָהֶם לַחֲרוֹשׁ וְלִזְרוֹעַ וְלָדוּשׁ עִם כָּל אַנְשֵׁי-הַשָּׂדֶה – כְּמוֹ שָׁאוּל אֲשֶׁר, אַחֲרֵי הֱיוֹתוֹ מָשׁוּחַ מֶלֶךְ, הָלַךְ אַחַר הַבָּקָר וְהַמַּחֲרֵשָׁה; וּכְמוֹ אוֹדִיסֶס, אֲשֶׁר שִׁנָּה טַעְמוֹ, וְחָרַשׁ בַּחֲמוֹר וָשׁוֹר יַחַד, וּבַמַּעֲנִית הַחֲרוּשָׁה זָרַע מֶלֶח תַּחַת גַּרְעִינִים, לְעֵינֵי שְׁלִיחֵי אֲגֵמֶמְנוֹן, כְּדֵי לְהִשְׁתַּמֵּט מִנְּסוֹעַ לַמִּלְחָמָה עַל אִילְיוֹן, הִיא טְרוֹיָה, בִּגְלַל אֵשֶׁת-אִישׁ
------------------
וַתֵּלֶךְ פְּסִיכֵי אֶל הַצּוּר, וְהַצּוּר גָּבוֹהַּ וְתָלוּל וְחָלָק, וְהַמַּיִם יוֹצְאִים מֵאֶמְצָעִיתוֹ בֵּין הַשָּׁמַיִם וּבֵין הָאָרֶץ, וּמִשְׁתַּפְּכִים מַטָּה בְּרַעַם וָקֶצֶף,
וְהֵם שׁוֹטְפִים בְּנָהָר אָפֵל עַל הָאָרֶץ, עַד שֶׁהֵם נִכְנָסִים וְנֶעְלָמִים בְּחֵיק-מְעָרָה, הַמּוֹלִיכָה אֶל עֻמְקֵי-שְׁאוֹל.
וַתֵּרֶא פְּסִיכֵי כִּי לֹא תוּכַל עֲלוֹת בַּצּוּר מִבְּלִי הַחֲלֵק וְנָפֹל מַטָּה אֶל מוֹתָהּ.

------------------
רָצִיתִי לְהָשִׁיב לוֹ, כִּי גַּם הַשֵּׂכֶל הָעִיּוּנִי דּוֹרֵשׁ אִמּוּן, הִתְמַחוּת וְהִתְמַכְּרוּת, וְיֵשׁ לוֹ חֲלוּצִים וּקְדוֹשִׁים מִשֶּׁלוֹ. אַךְ בּוֹ בָרֶגַע תָּקְעָה הַחֲצוֹצְרָה מִן הַדֶּרֶךְ, לְאוֹת כִּי תֻּקְּנָה הַמְּכוֹנִית. לֹא חָפַצְתִּי לְעַכֵּב אֶת חֲבֵרַי-לִנְסִיעָה, וְנִפְרַדְתִּי מִן הַזָּקֵן הַמְּסַקֵּל מִתּוֹךְ “שָׁלוֹם” חֲטוּפָה.
------------------
כָּל הַלַּיְלָה וְכָל יוֹם-הַמָּחֳרָת הָגָה הַנַּעַר בְּבַת-הַפְּלָאִים, וְלִבּוֹ יוֹצֵא אֵלֶיהָ. בָּאַחֲרוֹנָה אָמַר: לֹא בְרַק אֲבָנִים טוֹבוֹת וְלֹא בֹשֶׂם פִּרְחֵי-חֵן יִתְּנוּ לָהּ אֶת הַחַיִּים, חָשָׁקָה. אֵין זֹאת כִּי אִם תִּתְאַו לְמָעוֹן יוֹתֵר חַי וְלוֹהֵט מֵאֲשֶׁר בָּנִיתִי לָהּ – לְמַעַן תִּתְקַשֵּׁר אֶל הָאָרֶץ, כִּי עַל כֵּן לֵב-כּוֹכָב מוֹשָׁבָהּ עַד כֹּה.
וּבַלַּיְלָה הַבָּא לָקַח אִתּוֹ חֶרֶב אֶל מִטָּתוֹ.

------------------
פְּלַג-רָקִיעַ זוֹעֵף, בְּמִצְחוֹ בְּרָקִים מְעוֹפְפִים (וְרַעַם לֹא נִשְׁמָע); וּכְנֶגְדּוֹ, כַּעֲטָרָה מֻפְלָאָה מִתְנוֹצֶצֶת עַל אַפְסֵי-אֹפֶק – מְאוֹרוֹת חֵיפָה-עַכּוֹ-נַהֲרִיָּה; וּלְמַטָּה, מִקָּרוֹב, בְּבֶקַע-הָר, מַבְלִיחִים נֵרוֹת-אֵלוֹן.

------------------
וְהָעוֹבֵר בֵּין חַצְרוֹת הַשִּׂיחַ הַנּוֹתְנִים רֵיחָם בְּטַל-הַלַּיְלָה, יִשְׁמַע תָּמִיד קוֹל הֶמְיָה וּנְגִינָה.
כִּי אֲנָשִׁים מְלֻמָּדִים מָצְאוּ וְהֶחֱיוּ אֶת כְּלֵי-הַזֶּמֶר הַיִּשְׂרְאֵלִיִּים הָעַתִּיקִים – נֵבֶל וּמִינִים וְעָשׂוֹר וְשׁוֹשַׁנִּים וַעֲלָמוֹת – וְהַצְּעִירִים שׁוֹקְדִים עֲלֵיהֶם.
אַף קָמוּ מְחַבְּרֵי-מַנְגִּינָה לַכֵּלִים הָעַתִּיקִים-הַמְּחֻדָּשִׁים, שֶׁהֵם כָּל-כָּךְ מַתְאִימִים לַאֲוִיר-הָאָרֶץ.

------------------
רְאוּ אֵת אֲשֶׁר עוֹלַלְתֶּם לָעוֹלָם! אֲבָל אֶת הַנַּעֲשֶׂה אֵין לְהָשִׁיב.
הָבָה נָצִיץ בִּזְהִירוּת בַּפִּטָּס, וְנִרְאֶה אִם הַכֹּל עָרַק-פָּרַח, אוֹ אִם יֵשׁ עוֹד-מָה כָּלוּא שָׁם.
מַה שְּׁנִּשְׁאַר בְּתַּחְתִּית הַפִּטָּס 
וְהוּא לְאַט-לְאַט הֵסִיר אֶת כִּסּוּי-הַפִּטָּס וַיַּבֵּט פְּנִימָה.
בְּתַחְתִּית הַפִּטָּס עוֹד רִפְרְפָה בְּרִיָּה אַחַת וִיחִידָה, זְעִירָה וְרָפָה, אֲשֶׁר דּוֹמֶה וְלֹא עָצְרָה כֹּחַ לָצֵאת עִם כָּל אַחֶיהָ וְאַחְיוֹתֶיהָ.

------------------
וַיְהִי כְּבוֹא אַפְרוֹדִיטֵי, וּבְמָצְאָהּ כִּי הִנֵּה עָשְׂתָה פְּסִיכֵי גַם אֶת תַּפְקִידָהּ זֶה בִּשְׁלֵמוּת, רָגְזָה הָאֵלָה עַד-מְאֹד: "גַּם הַפַּעַם עָמְדוּ לָךְ כְּשָׁפַיִךְ הַגְּדוֹלִים.
בֶּאֱמֶת, נַעֲרָה מְאֹד חֲכָמָה אָתְּ.
וְרַק שְׁלִיחוּת עוֹד אַחַת – וְהִיא קַלָּה לְעֻמַּת הַקּוֹדְמוֹת – אַטִּיל עָלָיִךְ.
תַּעֲשִׂי אוֹתָהּ לְמַעֲנִי, וְאֶקְרָא לָךְ דְּרוֹר מִשִּׁעְבּוּדֵךְ".
תַּפְקִיד מִשׁחַת-פֶּרְסֶפוֹנֵי 

------------------
שְׁנַיִם-שְׁלשָׁה נִדְלְקוּ אוֹרִים בְּתוֹךְ טוּרֵי-הַחַלּוֹנוֹת הַמְּאֻנָּכִים שֶׁל מְקַרְצְפֵי-הַשְּׁחָקִים, וְכִמְעַט-רֶגַע וְהֵם נוֹגְהִים לַעֲשָׂרוֹת, מְפֻזָּרִים וְאֵין מִשְׁטָר. שָׁם אוֹר אָדֹם – גֻּלַּת-כּוֹתֶרֶת לְאַחַד הַבִּנְיָנִים, וְצֵלַע שֶׁל שֵׁנִי מוּצֶפֶת נְהָרָה לְבָנָה עַד הֱיוֹת לָהּ מַרְאֵה גָבִישׁ. הֻדְלַק הַלַּפִּיד בְּיַד אֵלַת-הַדְּרוֹר, וְקוֹמָתָהּ נִשְׁתַּרְבְּבָה הֲפוּכָה וּמְדֻבְלֶלֶת בַּמְּצוּלָה לְרַגְלֶיהָ. 
------------------
וַיִּמְצָא אֶת הָדֶס אֲפֵל-הַפָּנִים יוֹשֵׁב עַל כִּסְאוֹ בְּטַרְטָרוֹס, וּפֶרְסֶפוֹנֵי (הִיא קוֹרֵי) יוֹשֶׁבֶת לְצִדּוֹ, וְהִיא כֻלָּהּ תּוּגָה.
וַיֹּאמֶר הֶרְמֶס אֶל הָדֶס: "כֹּה אָמַר אֵלַי זֶאוּס: רֵד אֶל טַרְטָרוֹס וְהָשֵׁב אֶת קוֹרֵי, אֲשֶׁר שְׁמָהּ הוּסַב פֶּרְסֶפוֹנֵי, אֶל נְאוֹת-הָאוֹר וְאֶל דֵּימֶטֶר אִמָּהּ".
וַיֹּאמֶר הָדֶס: "חַס לִי מִמְּרוֹד בְּמִצְוַת אָחִי זֶאוּס, הַשַּׁלִּיט בָּאֵלִים.
קַח אֶת הַנַּעֲרָה.

------------------
וְרָמִים, צְדוּדֵי אַרְבַּע פָּנִים וּצְנוּפֵי שֵׁשׁ כְּנָפַיִם, וְעַל-יָדָם קְלֻלִּים וְחַשְׁמַלִּים וּשְׁבִיבִים, הִסְתַּדְּרוּ מַחֲנוֹת-מַחֲנוֹת לְהַצִּיג בְּפֹעַל חֶזְיוֹנוֹת אֲשֶׁר חָזָה הַמְּשׁוֹרֵר בְּרַעְיוֹנוֹ.
אֲבָל הַמְּשׁוֹרֵר לֹא שָׁת לִבּוֹ אֶל הַתְּכוּנָה הָרַבָּה הַזֹּאת. אַט-אַט צָעַד, עַד הַגִּיעוֹ אֶל רַגְלֵי כִּסֵּא-הַכָּבוֹד, עָלָיו יוֹשֵׁב נוֹרָא אֲדוֹן כָּל הַצְּבָאוֹת, וְעָנָן וַעֲרָפֶל יְסֻכּוּהוּ. 
------------------
הַבְּרוֹשִׁים הַחַדּוּדִיִּים (מִתַּחַת לָהֶם חוֹסִים צְרִיפִים אֲחָדִים, וּפַח-גַּגּוֹתֵיהֶם נוֹצֵץ כְּסַפְסַפִּי), וְכֵן אֹרֶן צָעִיר עֲגֹל-גוּף אֲשֶׁר הִתְעָרָה בְּפִנָּתוֹ שֶׁל מִגְרָשׁ רֵיק, פֹּה בִּשְׁכוּנַת נָוֶה-שַׁאֲנָן, – הִנֵּה הֵם נִצָּבִים דֹּם, קַשּׁוּבִים לְזִמְזוּם סְאוֹן הָרְסִיסִים וּלְרֶנֶן קִלּוּחִים מִצִּנּוֹרוֹת, וְרַק הִסְתַּכְּלוּת מְאֻמֶּצֶת מְגַלָּה כִּי אֵלּוּ הָעֵצִים מִתְנוֹעֲעִים נַעְנוּעַ כָּלְשֶׁהוּ בְּקֶצֶב, כְּמוֹ מְטֻטָּלוֹת רְשֵׁלוֹת.

------------------
יֵשׁ וְהַשּׁוֹמֵר מִשֵּׁש מִבַּחוּץ אֶת כִּיסָיו שֶׁל צָעִיר חָשׁוּד אוֹ בִּקֵּשׁ מִנַּעַר אֶחָד שֶׁיּוֹצִיא יָדָיו מִכִּיסָיו וְיַרְאֶה מַה-בָּהֶן. הַדָּבָר נַעֲשָׂה מַהֵר. כְּשֶׁבָּא תוֹרִי לַעֲבוֹר עַל פְּנֵי הַשּׁוֹמֵר הִפְסִיק אוֹתִי וּמִמֵּילָא גַם אֶת כָּל הַשּׁוּרָה שֶׁמֵּאֲחוֹרַי, וְשָׁאַל: “מַה בְּכִיסֶיךָ?” – “סְפָרִים” אָמַרְתִּי, וְכִמְעַט בָּכִיתִי מֵעֶלְבּוֹן. הוּא לֹא הֶ
בַּבַּיִת אָמְרָה אִמָּא: “רַק יוֹם אֶחָד עָבַדְתָּ, וּכְבָר יֵשׁ סִימָנִים כְּחֻלִּים תַּחַת עֵינֶיךָ”.

------------------
הַצְּעִירָה יַמְבֵּי וְהַיְּשִׁישָׁה בּוֹבּוֹ מְנַסּוֹת לְשַׂמֵּחַ אֶת דֵּימֶטֶר וְהַיֶּלֶד דֵּימוֹפוֹאוֹן, בִּהְיוֹתוֹ נָתוּן בְּחֵיק-הָאֵלָה וּמִתְבַּשֵּׂם מֵרֵיחַ-אַפֶּיהָ, גָּדַל לְעֹז וְתִפְאֶרֶת כְּאֶחָד מִבְּנֵי-הָאֵלִים.
וַתִּשְׂמַח הַמַּלְכָּה עַל הָאוֹמֶנֶת הַחֲדָשָׁה.

------------------
– אַה, – אָמַר הֶחָתוּל הַמַּקְשִׁיב – אֲבָל מַה הַתִּינוֹק מְחַבֵּב?
– הוּא מְחַבֵּב דְּבָרִים שֶׁהֵם רַכִּים וְהֵם מְדַגְדֵּגִים – אָמַר הָעֲטַלֵּף.
– הוּא מְחַבֵּב דְּבָרִים חַמִּים לְהַחֲנִיקָם בְּתוֹךְ זְרוֹעוֹתָיו, כְּשֶׁהוּא אוֹמֵר לִישׁוֹן, הוּא מְחַבֵּב שֶׁיְשַׂחֲקוּ אִתּוֹ.
חֲבִיבִים עָלָיו כָּל הַדְּבָרִים הָאֵלֶּה.
– אַה, – אָמַר הֶחָתוּל הַמַּקְשִׁיב – הִנֵּה הִגִּיעָה שַׁעְתִּי.

------------------
"כָּל הָאָרֶץ עֲשׂוּיָה נְשָׁמוֹת, לְעוֹלָם אֵין נשָׁמָה אוֹבֶדֶת. מִתְוַסֶּפֶת הִיא עַל מָהוּתָהּ וְטִבְעָהּ, עַל אֶפְשָׁרוּיוֹתֶיהָ וְכֹחוֹתֶיהַ, שֶׁל הָאָרֶץ. וּלְעוֹלָם כָּל נְשָׁמָה בָאָה עַל שֶׁלָּהּ. הַאִם מַה שֶׁאֲנִי חוֹטֵף וְלוֹקֵחַ מִן הָעוֹלָם הוּא שֶׁלִּי? לֹא כֵן. רַק מַה שֶׁאֲנִי נוֹתֵן לָעוֹלָם הוּא בֶאֱמֶת שֶׁלִּי. סִמָּן לִבְעָלוּת – הַנְּתִיּנה. וְהַנָּתוּן, דַּרְכּוֹ לַחֲזוֹר אַחַר בְּעָלָיו הַטִּבְעִיִּים.
