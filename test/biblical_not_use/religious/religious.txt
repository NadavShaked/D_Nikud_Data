לֹא נַפְסִיק לָשִׁיר!
עִם חֲבִיבָה וְעִם כֹּל הַקַּיְטָנָה נָסַעְנוּ לְפַּארְק מַיִם גָּדוֹל בַּדָּרוֹם, שָׂחִינוּ בַּבְּרֵכָה , הִתְגַּלַּשְׁנוּ בַּמַּגְלֵשׁוֹת מַיִם, וְאָכַלְנוּ הָמוֹן הָמוֹן מַמְתַּקִּים.

------------------
חֲבֵרוֹת שֶׁלִּי מֵהַגַּן וְגַם מֵהַשְּׁכוּנָה בָּאוֹת אֵלֵינוּ הַבַּיְתָה, וְיַחַד אֲנַחְנוּ מִתְפַּלְּלוֹת תְּפִלַּת "קַבָּלַת שַׁבָּת", וְשָׂרוֹת שִׁירֵי שַׁבָּת בְּיַחַד, וְאִמָּא מְסַפֶּרֶת לָנוּ פָּרָשַׁת שָׁבוּעַ וּמְחַלֶּקֶת עֲלוֹנֵי פָּרָשַׁת שָׁבוּעַ וְשַׂקִּיוֹת מַמְתַּקִּים לְכָל הַיְּלָדוֹת.

------------------
כִּי פִּתְאוֹם, מִשּׁוּם מָקוֹם, הִגִּיעָה לְשָׁם מִשְׁפָּחָה דָּתִית.
אַבָּא, אִמָּא, וּשְׁלוֹשָׁה יְלָדִים.
כְּמוֹ מַלְאָכִים הֵם הָיוּ.
הַמַּדְרִיכָה דִּבְּרָה אִתָּם, הִסְבִּירָה לָהֶם שֶׁיֵּשׁ כָּאן נַעֲרָה דָּתִיָּה שֶׁרוֹצֶה לְהַגִּיעַ לָאַכְסַנְיָה לִפְנֵי שַׁבָּת, וְהֵם, הַמַּלְאָכִים הָהֵם, פָּשׁוּט לָקְחוּ אוֹתִי אִתָּם וְהֵסִיעוּ אוֹתִי לָאַכְסַנְיָה בָּה אֲמוּרִים הָיִינוּ לִהְיוֹת.

------------------
רַבָּה שֶׁל פָּלוֹנָאָה הִתְקָרֵב בַּחֲשַׁאי אֶל הָאוּלָם שֶׁבּוֹ נֶעֶרְכָה הַחֲתֻנָּה, וְנִצְמַד אֶל הַחַלּוֹן שֶׁהִקִּיף פְּנִימָה.
הוּא רָאָה אֶת הַקָּהָל מַקִּיף יְהוּדִי בְּגִיל הָעֲמִידָה, שֶׁפָּנָיו מַקְרִינוֹת אֲצִילוּת, וּמַאֲזִין בִּשְׁקִיקָה לִדְבָרָיו.
"זֶהוּ הַבַּעַל שֵׁם טוֹב!", הִרְהֵר הָרַב וְנִיסָה לִשְׁמֹעַ אֶת דְּבָרָיו.
לְתַדְהֵמָתוֹ, שָׁמַע כִּי הַבַּעַל שֵׁם טוֹב מְדַבֵּר עָלָיו.

------------------
*שֶׁכְּמוֹ שֶׁבּוֹרֵא עוֹלָם בָּרָא אֶת הָעוֹלָם בְּשִׁשָּׁה יָמִים וְנָח בַּיּוֹם הַשְּׁבִיעִי, כָּךְ גַּם אֲנַחְנוּ עוֹבְדִים וְטוֹרְחִים וַעֲמֵלִים בַּיָּמִים רִאשׁוֹן שֵׁנִי שְׁלִישִׁי רְבִיעִי חֲמִישִׁי וְשִׁשִּׁי, וּבְשַּׁבָּת אֲנַחְנוּ נָחִים, לֹא עוֹבְדִים וְלֹא עֲמֵלִים, וְנֶהֱנִים מִמַּה שֶׁהוּכַן בַּמַּהֲלָךְ שֶׁשֶת יְמֵי הַשָּׁבוּעַ.
*בְּשַּׁבָּת לוֹבְשִׁים בְּגָדִים חֲגִיגִיִּים וְהוֹלְכִים בִּצְּעָדִים רְגוּעִים וְשַׂלְוִים, כְּמוֹ בְּנֵי מְלָכִים!
מוֹצָאֵי הַשַּׁבָּת

------------------
אֲבָל וְאַחַר כָּךְ הִגִּיעַ אִישׁ מְאוֹד מְאוֹד מַפְחִיד.
הוּא הָיָה אָדָם גָּבוֹהַּ וְגָדוֹל, וּפָשׁוּט הִתְחִיל לִצְעֹק עָלֵינוּ.

------------------
הָיִיתִי מֵרִים אֶת הַחֻלְצָה מֵעַל לָרֹאשׁ בְּבֵיִת הַסֵּפֶר כְּדֵי לוֹמַר בְּרָכוֹת בְּכִּסּוּי רֹאשׁ.
בְּשַּׁבָּת הַטֶּלֶוִיזְיָה הָיְתָה דּוֹלֶקֶת וַאֲנִי בַּחֶדֶר הָיִיתִי עוֹשֶׂה לְעַצְמִי קִדּוּשׁ וְאוֹכֵל אוֹכֶל מִשֶׁלִי...
בְּשַּׁבָּת בַּבֹּקֶר הָיִיתִי חוֹמֵק וְהוֹלֵךְ לַבֵּית הַכְּנֶסֶת לַתְּפִלָּה, וּבַצָּהֳרַיִם, כְּשֶׁהָיוּ כֻּלָּם נוֹסְעִים לַיָּם, הָיִיתִי מִתְחַמֵּק וְאוֹמֵר שֶׁאֲנִי לֹא מַרְגִּישׁ טוֹב...
כָּךְ הָיָה כְּשֶׁהָיִיתִי נַעַר, יֶלֶד קָטָן בְּעֶצֶם.

------------------
וּבְכָל זֹאת, הָרֶגֶשׁ בַּלֵּב שֶׁלִּי יָכוֹל לְמַצּוֹת אֶת הַמִּצְוָה אוֹ חָלִילָה לְהַחְמִיץ אֶת הַמַּטָּרָה!
אָז אֵיךְ אֲנִי יָכוֹל לִבְדֹּק אֶת עַצְמִי? אֵיךְ אֵדַע אִם זוֹ גַּאֲוַת הֶחָמֵץ אוֹ שִׂמְחַת הַמַּצָּה? אֶתֵּן לָכֶם סִימָן: נַנִּיחַ שֶׁדּוּדִי שׁוּב מִתְקַשֶּׁה בַּהֲבָנַת הַמִּשְׁנָה וְשׁוּב אֲנִי רוֹצֶה לַעֲזֹר לוֹ, אֲבָל הַפַּעַם מַגִּיעַ פִּתְאוֹם יוֹסִי לְפָנַי, מִתְיַשֵּׁב לְיַד דּוּדִי וְעוֹשֶׂה בְּדִיּוּק אֶת מָה שֶׁתִּכְנַנְתִּי אֲנִי לַעֲשׂוֹת.

------------------
וְכָךְ, בְּעוֹדוֹ עוֹמֵד וּמִתְיַסֵּר בְּהִתְרַגְּשׁוּת עֲצוּמָה, הֵסֵב פִּתְאוֹם הַבַּעַל שֵׁם טוֹב אֶת פָּנָיו אֶל הַחַלּוֹן וְקָרָא: "רַבָּה שֶׁל פָּלוֹנָאָה, אִם רְצוֹנְךָ לְהַגִּיעַ לְתִקּוּן שָׁלֵם – בּוֹא אֵלַי!".
בְּאוֹתוֹ לַיְלָה הָיָה רַבִּי יַעֲקֹב יוֹסֵף, רַבָּה שֶׁל פָּלוֹנָאָה, לְאַחַד מֵחֲסִידָיו הַגְּדוֹלִים, הַנֶּאֱמָנִים וְהַקְּרוֹבִים בְּיוֹתֵר שֶׁל הַבַּעַל שֵׁם טוֹב.

------------------
אַךְ כַּעֲבֹר זְמַן מַה הֵם מָצְאוּ סֵפֶר מְעַנְיֵן בַּבַּיִת, וּמֵאָז הַבֵּן הַגָּדוֹל מַקְרִיא לָאַחִים שֶׁיּוֹשְׁבִים וּמַאֲזִינִים לוֹ בְּשֶׁקֶט .
אֲחוּזֵי סַקְרָנוּת נִכְנְסוּ הַהוֹרִים לַחֶדֶר הַיְּלָדִים וְנִדְהֲמוּ לְגַלּוֹת אֶת ה"קִצּוּר שֻׁלְחָן עָרוּךְ" בִּידֵי בְּנָם הַבְּכוֹר, שֶׁיָּשַׁב וְהִקְרִיא לְאֶחָיו הַקְּטַנִּים.
כֹּה שְׁקוּעִים בַּסֵּפֶר הָיוּ הַשְּׁלוֹשָׁה עַד שֶׁלֹּא הִבְחִינוּ כְּשֶׁנִּכְנְסוּ הַהוֹרִים לַחֶדֶר.

------------------
סַבָּא וְסַבְתָּא שֶׁלָּנוּ מִקִּרְיַת רִמּוֹנִים הִתְאָרְחוּ בְּאוֹתָהּ שַׁבָּת בְּדִּירָה שֶׁל שְׁכֵנִים שֶׁלָּנוּ, כְּדֵי שֶׁלֹּא יִהְיֶה לָהֶם רָחוֹק מִדַּי, כִּי הֵם גָּרִים בַּשְּׁכוּנָה קְצָת רְחוֹקָה.

------------------
שָׁלוֹם יְלָדִים חֲמוּדִים, וּבְרוּכִים הַבָּאִים! אֲנִי כָּל כָּךְ שְׂמֵחָה שֶׁבָּאתֶם לְבַקֵּר אוֹתָנוּ שׁוּב, כָּאן בְּקִרְיַת רִמּוֹנִים! אַתֶּם וַדַּאי זוֹכְרִים אוֹתִי עוֹד כְּשֶׁהָיִיתִי קְטַנָּה, בְּגַן חוֹבָה, וְסִפַּרְתִּי לָכֶם עַל הַמִּשְׁפָּחָה הַחֲמוּדָה שֶׁלָּנוּ, עַל הַתֻּכִּי שֶׁלָּנוּ תּוּקְתּוּק, וְעַל הַחֲבֵרָה שֶׁלִּי הִלָּה וְאֵיךְ הִכַּרְנוּ אֶת הַמִּשְׁפָּחָה שֶלָּה בִּזְכוּת תּוּקְתּוּק שֶׁהָלַךְ לְאִבּוּד.

------------------
אוֹתוֹ שׁוֹאֵב מַיִם הוּא צַדִּיק נִסְתָּר, שֶׁעָשָׂה אֶת דַּרְכּוֹ אֶל חֲבֵרָיו הַצַּדִּיקִים, שֶׁאֵינָם נוֹפְלִים בִּגְדֻלָּתָם מִן הָרַב".
רַבָּה שֶׁל פָּלוֹנָאָה עָמַד נִדְהַם, אַךְ הוּא לֹא הִסְפִּיק לְהַאֲרִיךְ בְּמַחְשְׁבוֹתָיו וּכְבָר סִפֵּר הַבַּעַל שֵׁם טוֹב מַעֲשֶׂה נוֹסָף.
"בְּלֵיל תִּשְׁעָה בְּאָב נוֹתַר הָרַב לְבַדּוֹ בְּבֵית הַכְּנֶסֶת, מְקוֹנֵן עַל חֻרְבַּן הַבַּיִת בִּבְכִיָּה עֲצוּמָה.
קִינָתוֹ, שֶׁיָּצְאָה מֵעֹמֶק לִבּוֹ, בָּקְעָה רְקִיעִים.

------------------
גִּלָּה הַמַּזְכִּירָה רִכְּזָה אֶת הַטֶּלֶפוֹנִים שֶׁל הַמִּשְׁפָּחוֹת שֶׁרָצוּ לְהִתְעַנְיֵן וְלִשְׁמֹעַ, וְאִמָּא שֶׁל הִלָּה רִכְּזָה אֶת הַמִּתְנַדְּבוֹת שֶׁאָפוּ חַלּוֹת בֵּיתִיוֹת לִכְבוֹד שַׁבָּת, וְאֶת הַמִּתְנַדְּבוֹת שֶׁחִלְּקוּ חַלּוֹת לַמִּשְׁפָּחוֹת.

------------------
הֵם נֶהֱנוּ מְאוֹד, אַבָּא דִּבֵּר עִם הַבַּעַל שֶׁל גִּילָה, וְיָאִיר וְהַיֶּלֶד שֶׁל גִּלָּה שִׂחֲקוּ כָּל הַזְּמַן, וְגִלָּה וְאִמָּא שֶׁלִּי הֲרֵי כְּבָר מַכִּירוֹת וָתִיקוֹת! בְּמוֹצָאֵי שַׁבָּת שָׁאֲלוּ גִּלָּה וּבַעֲלָהּ אֵיךְ אֶפְשָׁר לִשְׁמֹר שַׁבָּת גַּם אֶצְלָם בַּבַּיִת.
כָּל כָּךְ נֶהֱנוּ מֵהָאֲוִירָה הַמְּיֻחֶדֶת וּמֵהַרְגָּשַׁת הַקְּדֻשָּׁה וְהַשַּׁלְוָה שֶׁיֵּשׁ בְּשַׁבָּת קֹדֶשׁ!
כָּעֲבוֹר שָׁבוּעַ הִגִּיעַ תִּשְׁעָה בְּאָב.

------------------
אִמָּא חִכְּתָה שֶׁאֲסַיֵּם אֶת תְּפִלָּתִי, וּמִשָּׁם נָסַעְנוּ יָשָׁר הַבַּיְתָה.
הִרְגַּשְׁתִּי הֲקָלָה.
הִרְגַּשְׁתִּי שֶׂה' שָׁמַע אֶת תְּפִלָּתֵנוּ.
הִרְגַּשְׁתִּי שֶׂה' נָתַן לָנוּ אוֹר יָרֹק.
הִרְגַּשְׁתִּי שֶׁהַדֶּרֶךְ סְלוּלָה.
שֶׂה' סָלַל אוֹתָהּ.
שֶׁעַכְשָׁו בְּעֶזְרַת ה' הַכֹּל יָרוּץ.
וְאָמְנָם, הָחֳדָשִׁים הַבָּאִים עָבְרוּ בִּמְהִירוּת הַבָּזָק.
הַהַתְחָלָה הָיְתָה קְצָת קָשָׁה כִּי עוֹד לֹא יָדַעְנוּ מַה וְאֵיךְ בְּדִיּוּק לַעֲשׂוֹת.

------------------
"כֵּן, הַיְּלָדוֹת כְּבָר חָזְרוּ הַבַּיְתָה, וְשִׁירָה נָחָה", אָמַרְתִּי.
"לֹא, לֹא לְזֶה אֲנִי מִתְכַּוֶּנֶת", אָמְרָה סַבְתָּא, "שָׁקֵט פֹּה מִבַּחוּץ.
תָּמִיד הָיְתָה כָּאן הֲמֻלָּה בְּשַׁבָּת.
תָּמִיד הָיָה רַעַשׁ שֶׁל מְכוֹנִיּוֹת".
"תָּמִיד?" שָׁאַלְתִּי.
"לֹא.
לֹא תָּמִיד." דִּיְּקָה סַבְתָּא, "כְּשֶׁקִּרְיַת רִמּוֹנִים נִבְנְתָה, הַשַּׁבָּת נִשְׁמְרָה בִּמְלוֹאָהּ.
אַבָּא וְאִמָּא שֶׁלִּי, כְּשֶׁהֵם עָלוּ לָאָרֶץ, הָיוּ שׁוֹמְרֵי מִצְווֹת, וּכְמוֹתָם רֹב הַמִּתְיַשְּׁבִים בְּקִרְיַת רִמּוֹנִים.

------------------
"זֶה סִפּוּר אָרֹךְ," אָמַר אַבָּא, "הָיוּ לָנוּ שְׁכֵנִים.
מִשְׁפַּחַת סוֹפֵר, שֶׁהֵם שׁוֹמְרֵי מִצְווֹת.
לִפְעָמִים הָיִיתִי מַגִּיעַ אֵלֵיהֶם לְהִתְאָרֵחַ בְּשַׁבָּת וְהִרְגַּשְׁתִּי אֶת הַהֶבְדֵּל בָּאֲוִירָה.
לִפְעָמִים הָיִיתִי מַגִּיעַ אִתָּם לְבֵית הַכְּנֶסֶת....
וְכָךְ הִתְרַגַּלְתִּי לְהַגִּיעַ לַתְּפִלּוֹת, וְהִתְחַלְתִּי גַּם לְהַקְשִׁיב לַדְּרָשׁוֹת שֶׁל הָרַב...
