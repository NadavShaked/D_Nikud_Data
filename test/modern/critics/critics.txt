"יוֹם אֶחָד, בְּעֵת עִימּוּת עִם פְּסִיכוֹלוֹגִית בֵּית הַסֵּפֶר שֶׁל הַבֵּן שֶׁלִּי, בַּאֲשֶׁר לִיכוֹלוֹתָיו הָעֲתִידִיּוֹת שֶׁל הַבֵּן שֶׁלִּי לְהִשְׁתַּלְּבוּת בַּחֶבְרָה, אָמַרְתִּי לָהּ שֶׁאֲנִי לֹא דּוֹאֶגֶת בִּכְלָל. אֲנָשִׁים מֵהַסּוּג הַזֶּה, אָמַרְתִּי, פָּגַשְׁתִּי בְּכָל פִּינָּה בְּלִימּוּדַיי בָּאָקָדֶמְיָה. הֵם מְמַלְּאִים אֶת הַפָקוּלְטוֹת שֶׁל מָתֵמָטִיקָה וּמַדְּעֵי הַמַּחְשֵׁב. וַאֲנִי מַנִּיחָה שֶׁגַּם מַחְלָקוֹת אֲחֵרוֹת. 
------------------
הוּא לֹא בָּנָה מַמָּשׁ רוֹמָן, יוֹתֵר אוֹסֵף עֲלִילוֹת נִפְרָדוֹת שֶׁמַּשִּׁיקוֹת זוֹ לְזוֹ בְּצוּרָה רְפוּיָה וּמְסוּפָּרוֹת לְסֵירוּגִין. בַּסּוֹף הוּא גַּם מְצָרֵף אַחֲרִית דָּבָר שֶׁמְּסַכֶּמֶת אֶת דַּעְתּוֹ בְּנוֹשֵׂא חִידּוּשִׁים גֵּנֵטִיִּים - אֵין לָתֵת זְכוּיוֹת פָּטֵנְט עַל גֵּנִים, יֵשׁ לָתֵת לָאָדָם זְכוּת עַל רִקְמוֹת גּוּפוֹ, יֵשׁ לְפַרְסֵם בְּפוּמְבִּי נְתוּנֵי מֶחְקָר וְלֹא לֶאֱסוֹר עַל מֶחְקָרִים גֵּנֵטִיִּים כִּי הַמְּנִיעָה לֹא תַּעֲשֶׂה כְּלוּם. 
------------------
אֲנִי זוֹכֶרֶת אֶת הַמְּנִיעִים שֶׁגָּרְמוּ לִי לִקְפּוֹץ מֵהַמַּרְזֵב, וְאֶת הַתְּחוּשָׁה הַמּוּזָרָה כְּשֶׁהַשֵּׂיעָר שֶׁלִּי נִשְׂרַף כְּשֶׁעָמַדְתִּי קָרוֹב מִדַּי לְכִירָיִים בְּבַיִת שֶׁל חֲבֵרָה וְהַחֲבֵרָה רָאֲתָה וְלֹא אָמְרָה כְּלוּם "כִּי זֶה הָיָה נוֹרָא יָפֶה".

כָּכָה זֶה יְלָדִים בְּנֵי שֶׁבַע, סַכָּנוֹת מְהַלְּכוֹת. יֵשׁ לָהֶם אֶפֶס אֶפְשָׁרוּיוֹת תִּכְנוּן עֲתִידִי. וְכַמּוּבָן, יֵשׁ לָהֶם גַּם מַלְאָךְ שֶׁשּׁוֹמֵר מִלְּמַעְלָה, אַחֶרֶת יִהְיֶה רַע מְאוֹד.
