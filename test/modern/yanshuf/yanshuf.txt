הַאִם, בָּעִדָן שֶׁל הַיּוֹם, יֵשׁ סִכּוּי שֶׁנִּפֹּל בַּפַּח עִם סִפּוּרִים מוּזָרִים וְלֹא אֲמִתִיִים? יֵשׁ לְהַאֲמִין שֶׁכֵּן. הַצִּבּוּר רוֹצֶה לִשְׁמוֹעַ סִפּוּרִים מוּזָרִים וּמְעַנְיְנִים וְלִפְעָמִים זֶה לֹא מְעַנְיֵן אוֹתוֹ, שֶׁהֵם לֹא אֲמִתִיִים. מְעַנְיֵן לִרְאוֹת, אִם כֵּן, מַה תִּהְיֶה הַמְּתִיחָה הַבָּאָה.

------------------
לְאַחַר כְּשִׁשִׁים שָׁנָה, הוֹדוּ אֵלְסִי רָיְיט וּפְרָנְסִיס גְּרִיפִית, שֶׁהַכֹּל הָיָה שֶׁקֶר. "רָצִינוּ לִמְתוֹחַ אֶת הַמִּשְׁפָּחָה וְהַחֲבֵרִים", סִפְּרוּ הַשְּׁתַיִם. "לֹא תֵּאַרְנוּ לְעַצְמֵנוּ שֶׁהָעִנְיָן יְגִיעַ לִמְמַדִּים כָּאֵלֶּה. לֹא יָדַעְנוּ אֵיךְ לְהַפְּסִיק אֶת זֶה".
בְּעוֹלַם הַקּוֹלְנוֹעַ אָהֲבוּ אֶת הָרַעְיוֹן שֶׁל הַפֵּיוֹת וְאַף עָשׂוּ סֶרֶט עַל שְׁתֵי הַיְּלָדוֹת הָאַנְגְּלִיוֹת. יוֹצְרֵי הַסֶּרֶט הִתְעַלְמוּ מֵהָעֻבְדָה שֶׁהַכֹּל הָיָה שֶׁקֶר אֶחָד גָדוֹל.

------------------
מִפְלֶגֶת הָעֲבוֹדָה הוֹדִיעָה, שֶׁהִיא תַעֲזֹר לַמֶּמְשָׁלָה לִפְעוֹל לְפִי הַמְּדִינִיוּת הַחֲדָשָׁה שֶׁלָה.
בַּשָׁבוּעוֹת הַקְּרוֹבִים, צְפוּיוֹת הַפְּגָנוֹת רַבּוֹת נֶגֶד הַחְלָטַת הַמֶּמְשָׁלָה.

------------------
יִשְׂרָאֵל רוֹצָה אַסְטְרוֹנָאוּט יִשְׂרְאֵלִי נוֹסַף
לַמְרוֹת אֲסוֹן מַעְבֹּרֶת הֶחָלָל, מִתְכַּוֶנֶת יִשְׂרָאֵל לְשַׁגֵר אַסְטְרוֹנָאוּט נוֹסָף לֶחָלָל. בְּיָמִים אֵלֶה, פָּנְתָה יִשְׂרָאֵל לָאָמֶרִיקָאִים כְּדֵי לִבְדוֹק אֶפְשָׁרוּת לִשְׁלוֹחַ אַסְטְרוֹנָאוּט יִשְׂרְאֵלִי נוֹסַף.
אִם הַתְּשׁוּבָה תִּהְיֶה חִיוּבִית, יַתְחִילוּ הַחִפּוּשִׂים אַחַר מֻעֲמָד מַתְאִים. כְּכָל הַנִּרְאֶה, גַם הַפַּעַם יִהְיֶה זֶה טַיָּס לְשֶׁעָבַר שֶׁל חֵיל הָאֲוִיר.

------------------
"הַמַּדְפֶּסֶת הַחֲדָשָׁה שִׁנְתָה לָנוּ אֶת כָּל שִׁיטַת הָעֲבוֹדָה", אוֹמְרִים ב"אָדִידָס". "תַהֲלִיכֵי הַיִּצוּר מְהִירִים וּמְדֻיָקִים יוֹתֵר".
גַם חֶבְרוֹת הַמְּכוֹנִיוֹת מִשְׁתַּמְשׁוֹת בַּמַּדְפֶּסֶת, כְּדֵי לִבְנוֹת דִגְמֵי מְכוֹנִיוֹת קְטַנִים וּלְפַתֵחַ חֲלָקִים מְיֻחָדִים שֶׁל תַעֲשִׂיַת הָרֶכֶב.
וּבְכָל זֹאת, אִם בָּא לָכֶם לִקְנוֹת מַדְפֶּסֶת כָּזוֹ, שֶׁהַגֹּדֶל שֶׁלָה כְּגֹדֶל מְכוֹנַת כְּבִיסָה, אַתֶם מֻזְמָנִים לְהוֹצִיא חֲצִי מִילְיוֹן שֶׁקֶל. 
------------------
עָרַב הַסָּעוּדִית – הַסַּכָּנָה הַבָּאָה בַּמִּזְרָח הַתִּיכוֹן
עֹשֶׁר עָצוּם מוּל עֹנִי וּפְגִיעָה קָשָׁה בִּזְכֻיוֹת הַנָּשִׁים עֲלוּלִים לְהָבִיא לַמַּהְפֵּכָה בְּסָעוּדְיָה. מַהְפֵּכָה כָּזוֹ יְכוֹלָה לְהַשְּׁפִּיעַ עַל כָּל מְדִינוֹת עָרַב וּלְשַׁנוֹת אֶת הַמִּזְרָח הַתִּיכוֹן. הָאֲנָשִׁים שֶׁל בֵּן-לָאדֶן מְחַכִּים לַהוֹרָאָה לִפְתוֹחַ בְּמִתְקָפָה נֶגֶד הַשִּׁלְטוֹן הַמֻּשְׁחָת

------------------
עוֹלַם הַמַּדָע וְהַתִּקְשֹׁרֶת יוֹצֵא לֹא פַּעַם בְּכוֹתְרוֹת עֲנָק עַל תַגְלִית מַדָעִית חֲשׁוּבָה אוֹ עַל תוֹפַעַת טֶבַע יוֹצֵאת מֵהַכְּלָל. מַדְעָנִים מִכָּל רַחֲבֵי הָעוֹלָם נִפְגָשִׁים כְּדֵי לָדוּן בְּתַגְלִיוֹת הַמְּעַנְיְנוֹת, מְנַסִּים לִמְצוֹא לָהֶם תְּשׁוּבָה וּמְבַזְבְּזִים שָׁנִים בְּמֶחְקָרִים. אוּלָם, לֹא פַּעַם מִתְבָּרֶרֶת הָאֱמֶת – אוֹתָן תוֹפָעוֹת מְעַנְיְנוֹת לֹא קָרוּ בַּמְּצִיאוּת, אֶלָא הָיוּ תַרְמִית שֶׁל בְּנֵי אָדָם.

------------------
הַחִפּוּשִׂים הֵחֵלוּ מִיָּד, מִתוֹךְ חֲשָׁשׁ שֶׁגַּנָבִים הִצְלִיחוּ לִגְנוֹב אוֹתָם. "אָמְנָם הַחַיוֹת מְאֻלָפוֹת, אֲבָל הֵן בְּכָל זֹאת מְסֻכָּנוֹת" אָמַר מְנַהֵל הַקִּרְקָס לַשׁוֹטְרִים.
לְאַחַר בְּדִיקָה קְצָרָה הִתְבָּרֵר, שֶׁגַּם מְאַלֶפֶת הָאַרְיוֹת נֶעֶלְמָה וְאִתָּהּ גַם הַבֵּן הַמְּבֻגָר (20) שֶׁל בַּעַל הַקִּרְקָס. הַשְּׁנַיִם הִתְאַהֲבוּ תוֹךְ כְּדֵי עֲבוֹדָתָם הַמְּשֻׁתֶפֶת וּבָרְחוּ עִם הַחַיוֹת הַמְּסֻכָּנוֹת כְּדֵי לְהִתְחַתֵן.

------------------
הָאָמֶרִיקָאִים מַמְצִיאִים נֶשֶׁק חָדָשׁ לַמִלְחָמָה בָּאוֹיֵב:
פִּצְצוֹת סִרָחוֹן
בְּעוֹד מִסְפַּר שָׁנִים, יֵרָאֶה שְׂדֵה הַקְּרָב בְּצוּרָה שׁוֹנָה לְגַמְרֵי. בִּמְקוֹם יְרִיוֹת וַהֲרוּגִים, יִהְיוּ פִּצְצוֹת סִרָחוֹן וְגֶ'ל שֶׁגוֹרֵם לַטַנְקִים לְהַחְלִיק. הַאִם הַמִּלְחָמָה הַבָּאָה תִּסְתַּיֵם עוֹד לִפְנֵי הַיְּרִיָה הָרִאשׁוֹנָה?

------------------
הַכַּתָבוֹת הַמְּעַנְיְנוֹת בְּיוֹתֵר
לַמַּעֲרֶכֶת שָׁלוֹם,
אָנוּ, תַלְמִידֵי כִּתָה ו' בְּבֵית הַסֵּפֶר "הַחַדְשָׁנִי" בְּדָלְיַת אֶל כַּרְמֶל, הֶחְלַטְנוּ לִבְחוֹר, בְּיַחַד עִם הַמּוֹרָה שֶׁלָּנוּ תָמָר חַמָאדֵה, אֶת שָׁלֹשׁ הַכַּתָבוֹת שֶׁהֲכִי מְעַנְיְנוֹת אוֹתָנוּ בָּעִתּוֹן הַכָּפוּל, מִסְפָּר 32-33.
עָרַכְנוּ הַצְּבָּעָה וְאֵלֶה הַתּוֹצָאוֹת:
הַכַּתָבוֹת הַקְּשׁוּרוֹת לְסִדְרַת הַטֶּלֶוִיזְיָה "הַמּוֹרְדִים" – 32 נְקֻדוֹת.

------------------
גַּם בַּחֲלָקִים אֲחֵרִים בָּעוֹלָם יֶשְׁנָם יְלָדִים רַבִּים הַנֶּאֱלָצִים לַעֲבוֹד, מִפְּנֵי שֶׁהַמְּבֻגָּרִים מַכְרִיחִים אוֹתָם.
בְּיִשְׂרָאֵל יֶשְׁנוֹ חֹק הָאוֹסֵר עַל הַעֲסָקַת יְלָדִים, וּבֶאֱמֶת מִסְפַּר הַיְּלָדִים הָעוֹבְדִים בְּאֶרֶץ קָטָן בְּיוֹתֵר. 
אִרְגּוּן יוּנִיצֶ"ף שַׁיָּךְ לָאוּ"ם וְתַפְקִידוֹ לְטַפֵּל בִּזְכִיּוֹת הַיֶּלֶד בָּעוֹלָם. בָּאַחֲרוֹנָה, יָצָא הָאִרְגּוּן בְּסִדְרַת מוֹדָעוֹת, לְטוֹבַת הַיְּלָדִים.

------------------
אִם הַפְּרוֹפֶסוֹר הַמְּכֻבָּד הָיָה קוֹרֵא אֶת הַדְּבָרִים שֶׁל שְׁלֹמֹה הַמֶּלֶךְ, אוּלַי הוּא הָיָה יָכֹל לַחְסוֹךְ לְעַצְמוֹ אֶת הַמַּאֲמָץ שֶׁל הַמֶּחְקָר הַקָּשֶׁה שֶׁעָרַךְ.

------------------
מֶחְקָר עוֹלָמִי קוֹבֵעַ:
מְקוֹם הָעֲבוֹדָה חָשׁוּב מְאֹד לַיִשְׂרְאֵלִים
הַיִּשְׂרְאֵלִים אוֹהֲבִים מְאֹד אֶת מְקוֹם הָעֲבוֹדָה שֶׁלָּהֶם. מֶחְקָר בֵּינְלְאֻמִי קוֹבֵעַ, שֶׁהַיִּשְׂרְאֵלִי נִמְצָא בַּמָּקוֹם הָרִאשׁוֹן בַּמְּחֻיָּבוֹת לִמְקוֹם הָעֲבוֹדָה שֶׁלוֹ. הוּא גַם נֶהֱנֶה מֵעֲבוֹדָתוֹ וּמַרְגִישׁ שֶׁהוּא מוֹעִיל לָעֵסֶק בּוֹ הוּא עוֹבֵד.

------------------
לֹא הִרְגַּשְׁתִּי שֶׁאֲנִי מְדַבֶּרֶת כָּל כָּךְ הַרְבֵּה. לְמַזָּלִי, הַהוֹרִים שֶׁלִּי הֵבִינוּ אֶת הַמַּצָּב וְלָקְחוּ אוֹתִי לְטִפּוּל. עַכְשָׁו אֲנִי נִזְהֶרֶת יוֹתֵר וּמְדַבֶּרֶת הַרְבֵּה פָּחוֹת", הִיא מְחַיֶּכֶת. "בְּעֶצֶם, הַהוֹרִים לָקְחוּ לִי אֶת הפלאפון וְאֵין לִי אֶפְשָׁרוּת לְדַבֵּר. בַּהַתְּחָלָה כָּעַסְתִּי עֲלֵיהֶם, אֲבָל כַּאֲשֶׁר הֵבַנְתִּי שֶׁזֶּה חֵלֶק מֵהַטִּפּוּל נִרְגַּעְתִּי".

------------------
