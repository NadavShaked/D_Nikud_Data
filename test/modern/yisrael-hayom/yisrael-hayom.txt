אֵקוֹפֶּרְסוּרָה לֹא מַצְרִיכָה מְחָטִים לְטִיפּוּל וּמְאַפְשֶׁרֶת לָכֶם בְּצוּרָה קַלָּה לַעֲשׂוֹת מִין טִיפּוּל בֵּייתִי כְּזֶה, רַק קַבְּלוּ הַנְחָיוֹת מֵהַמְּטַפֵּל שֶׁלָּכֶם בִּרְפוּאָה סִינִית וְתַעֲשׂוּ הַשְׁלָמָה שֶׁל הַטִּיפּוּל בַּיּוֹם יוֹם, כְּדֵי לְחַזֵּק אֶת הַטִּיפּוּל. כְּמוֹ כֵן, מַסָּאז', רֵיְיקִי, טִיפּוּלִים טִבְעִיִּים אֲחֵרִים, נַעֲשׂוּ יוֹתֵר פּוֹפּוּלָרִיִּים וּמְאַפְשְׁרִים עוֹד דְּרָכִים לְטַפֵּל בַּכֶּלֶב.


------------------
חִיבּוּרִים רִגְשִׁיִּים נוֹצָרִים גַּם בַּקּוֹלְנוֹעַ. סֶרֶט קוֹלְנוֹעַ יִשְׂרְאֵלִי, "שְׂרוֹכִים", בְּבִימּוּיוֹ שֶׁל יַעֲקֹב גּוֹלְדְווֹסֶר, שֶׁתּוּרְגַּם בְּיוֹזְמַת שַׁגְרִירוּת יִשְׂרָאֵל עֲבוּר פֶסְטִיבָל הַקּוֹלְנוֹעַ בְּטוֹקְיוֹ, נִרְכַּשׁ עַל יְדֵי מֵפִיץ מְקוֹמִי וְיַעֲלֶה בְּמֵאוֹת בָּתֵּי קוֹלְנוֹעַ בְּרַחֲבֵי הַמְּדִינָה. לִכְאוֹרָה, הַהַצְלָחָה הַזּוֹ הִיא הַפְתָּעָה גְּמוּרָה, שֶׁכֵּן הָעֲלִילָה מִתְרַחֶשֶׁת בְּיִשְׂרָאֵל.

------------------
בְּאוֹגוּסְט ‭ 2014‬ הִסְתַּיְּימָה מִלְחֶמֶת צוּק אֵיתָן. חֲבֵרַיי  וַאֲנִי, מִשְּׂדֵרוֹת וּמֵעוֹטֵף עַזָּה, חָזַרְנוּ לְבָתֵּינוּ וְהִתְחַלְנוּ לְשַׁקֵּם אֶת חַיֵּינוּ הַמְּטוּלְטָלִים. נְתַנְיָהוּ יָשַׁב לְצַד גַּנְץ  הָרַמַטְכָּ"ל וְיַעֲלוֹן שַׂר הַבִּיטָּחוֹן וְדִיבֵּר עַל "חַלּוֹן הִזְדַּמְּנוּיוֹת מְדִינִי", שֶׁנִּפְתַּח. לְמַרְבֵּה הַצַּעַר, הוּא לֹא נִיצֵּל אוֹתוֹ.

בְּכִירֵי מַעַרְכוֹת הַבִּיטָּחוֹן מַסְבִּירִים שׁוּב וָשׁוּב שֶׁכּוֹחַ מוּגְבָּל. 
------------------
בְּדוּ"חַ אַחֵר נִכְתַּב שֶׁחָשׁוּב לְוַודֵּא שֶׁהַנְּתוּנִים יִשְׁתַּקְּפוּ בַּדּוּ"חַ הַשְּׁנָתִי שֶׁל מַזְכָּ"ל  הָאוּ"ם בְּנוֹשֵׂא הֲפָרוֹת זַכּוּיוֹת יְלָדִים - אוֹתוֹ דּוּ"חַ שֶׁאָמוּר  לְמַקֵּם אֶת צַהַ"ל לְצַד אִרְגּוּנֵי טֵרוֹר.


------------------
מִשְׂרַד הַבְּרִיאוּת הוֹכִיחַ שֶׁהוּא לֹא בָּשֵׁל לְהוֹבִיל שִׁינּוּי דְּרָמָטִי כְּזֶה, מִשּׁוּם שֶׁאֲנָשָׁיו דְּבֵקִים בְּקוֹנְסֶפְּצִיּוֹת יְשָׁנוֹת. הַפִּתְרוֹן יָבוֹא רַק מִוַּועֲדַת מוּמְחִים, שֶׁתַּגְדִּיר מֵחָדָשׁ אֶת צוֹרְכֵי הַבְּרִיאוּת שֶׁל יִשְׂרָאֵל, תַּצִּיג חֲלוּפוֹת אַסְטְרָטֶגְיוֹת וְתָבִיא אֶת הַמְלָצוֹתֶיהָ לְאִישּׁוּר הַמֶּמְשָׁלָה. 

------------------
הָאַחֲרוֹן הוּא פָּחוֹת חָשׁוּב וּמַשְׁפִּיעַ, וְלָכֵן פָּחוֹת נִתְקָלִים בּוֹ. הוּא מוּכֶּה וְנִרְדָּף עַל יְדֵי פַתַ"ח. בְּקֶרֶב הַנְהָגַת חָמָאס בַּגָּדָה יֵשׁ מִי שֶׁמְּבַטְּאִים נְכוֹנוֹת לְהַכִּיר בְּיִשְׂרָאֵל בִּגְבוּלוֹת 67' לַטְּוָוח הָאָרוֹךְ, וְזֶה מְרַתֵּק וְלֹא מוּבָן מֵאֵלָיו. 'מַנְהִיגֵי חָמָאס בַּגָּדָה רוֹאִים בְּיִשְׂרְאֵלִים בְּנֵי אָדָם', אָמַר לִי מָקוֹר מֵהָאִרְגּוּן כִּבְשׂוֹרָה מְטַלְטֶלֶת. 
------------------
צִמְחֵי מַרְפֵּא וְתוֹסָפִים טִבְעִיִּים, כְּמוֹ לְמָשָׁל כּוּרְכּוּם, יוֹתֵר וְיוֹתֵר נְפוֹצִים הַיּוֹם גַּם לַכְּלָבִים וְלֹא רַק לַאֲנָשִׁים. חָשׁוּב לְהִתְיַיעֵץ עִם אִישׁ מִקְצוֹעַ לִפְנֵי שֶׁנּוֹתְנִים צִמְחֵי מַרְפֵּא, בֶּטַח אִם הַכֶּלֶב אוֹ הֶחָתוּל סוֹבְלִים מִבְּעָיָה בְּרִיאוּתִית וְנוֹטְלִים תְּרוּפוֹת. אֶפְשָׁר לָתֵת צִמְחֵי מַרְפֵּא לְכֶלֶב בָּרִיא, כְּדֵי לִשְׁמוֹר עַל הַבְּרִיאוּת שֶׁלּוֹ וְלִמְנוֹעַ מַחֲלוֹת וְאֶפְשָׁר כַּמּוּבָן לְטַפֵּל בִּבְעָיוֹת בְּרִיאוּת קַיָּימוֹת.

------------------
עֲבוּרָם, מְקוּבָּל שֶׁהַדֵּיְיט מִתְקַיֵּים בְּבַיִת שֶׁל אֶחָד הַמְּשׁוּדָּכִים אוֹ בְּלוֹבִּי שֶׁל בֵּית מָלוֹן. דָּוִד, גָּרוּשׁ מֵאַשְׁדּוֹד, מְסַפֵּר כִּי אֵירֵחַ דֵּיְיט בְּגַג בֵּיתוֹ: "הַפוֹרְמָט שֶׁל דֵּיְיט בַּבַּיִת מוּכָּר לַחֲרֵדִים הַרְבֵּה לִפְנֵי הַקּוֹרוֹנָה, אַךְ מְדוּבָּר בִּנְקוּדָּה רְגִישָׁה עֲבוּר גְּרוּשִׁים חֲרֵדִים מוֹדֶרְנִיִּים שֶׁמְּחַפְּשִׂים פֶּרֶק ב'. 
------------------
אֶת סִפְרוֹ הִקְדִּישׁ חֵמוֹ לְאָבִיו, חַיִּים זַ"ל, "הָאָדָם שֶׁהַשֶּׁטַח הָיָה טִבְעוֹ הַשֵּׁנִי וֶאֱנוֹשִׁיּוּת - הָרִאשׁוֹן". הָאָב, שֶׁעָבַד 44 שָׁנִים כְּטֶכְנַאי בַּמָּכוֹן הַגֵּיאוֹלוֹגִי, נִפְטַר מִמַּחֲלַת הַסַּרְטָן לִפְנֵי שְׁנָתַיִים, בְּגִיל 66, בְּמַהֲלַךְ כְּתִיבַת הַסֵּפֶר - שֶׁנֶּעֶשְׂתָה בֵּין הַיֶּתֶר לְצַד מִיטָּתוֹ בְּבֵית הַחוֹלִים.

------------------
בְּתוֹךְ כָּךְ, יוֹ"ר הָאוֹפּוֹזִיצְיָה חַ"כּ  יָאִיר לַפִּיד (יֵשׁ עָתִיד) הוֹדִיעַ אֶתְמוֹל  כִּי בְּיוֹם רְבִיעִי הַבָּא, 2 בְּדֵצֶמְבֶּר, בְּכַוּוֹנָתוֹ לְהָבִיא לְאִישּׁוּר מְלִיאַת הַכְּנֶסֶת בִּקְרִיאָה טְרוֹמִית אֶת הַצָּעַת הַחוֹק  לְפִיזּוּר הַכְּנֶסֶת. לַפִּיד, שֶׁבַּשָּׁבוּעוֹת הָאַחֲרוֹנִים מָשַׁךְ פַּעֲמַיִים אֶת הַצָּעַת הַחוֹק בָּרֶגַע הָאַחֲרוֹן, הִבְטִיחַ בִּישִׁיבַת הַסִּיעָה: "לֹא נִמְשׁוֹךְ אוֹתוֹ בָּרֶגַע הָאַחֲרוֹן וְלֹא נְנַהֵל עַל זֶה מוּ"מ. הַחוֹק  יַעֲלֶה לַהַצְבָּעָה. 
------------------
הָרוֹב הַמּוּחְלָט מַעֲדִיף לְהַגִּיעַ לַחֲנוּיוֹת הַפִיזִיּוֹת, לִבְחוֹר אֶת הַחוּלְצוֹת, לִמְדּוֹד אוֹתָן וּלְאַחַר מִכֵּן לְהַדְפִּיס אֶת סֵמֶל בֵּית הַסֵּפֶר. בָּאִינְטֶרְנֵט זֶה קְצָת מְסוּרְבָּל וְגַם אִיטִּי בְּיַחַס לַחֲנוּיוֹת הַפִיזִיּוֹת. אֲנָשִׁים אוֹהֲבִים לְהַרְגִּישׁ אֶת חֲוָויַית הַקְּנִיָּיה הַפִיזִית".

------------------
דִּיקּוּר סִינִי
שִׁיטַת הַטִּיפּוּל הַזּוֹ נִמְצֵאת כָּאן כְּבָר שָׁנִים רַבּוֹת לִבְנֵי אָדָם, אֲבָל גַּם כְּלָבִים וַחֲתוּלִים כְּבָר מְטוּפָּלִים בְּדִיקּוּר סִינִי. אֶפְשָׁר לְהַגִּיד שֶׁדִּיקּוּר סִינִי הָפַךְ לְטִיפּוּל לְגַמְרֵי מְקוּבָּל בָּעוֹלָם וְגַם בָּאָרֶץ. 

------------------
"בְּ-2004 פָּגַשְׁתִּי צְעִירָה פָלַשְׂתִּינִית מִמַּחֲנֵה גָ'בָאלִיָּה בְּעַזָּה, שֶׁנִּתְפְּסָה עִם חֲגוֹרַת נֶפֶץ בְּמַחְסוֹם אֶרֶז, כְּשֶׁהַיַּעַד שֶׁלָּהּ הָיָה לְהִתְפּוֹצֵץ בְּבֵית הַחוֹלִים סוֹרוֹקָה בִּבְאֵר שֶׁבַע", הוּא מְסַפֵּר. "כְּשֶׁהִתְיַישַּׁבְתִּי מוּלָהּ הִבְחַנְתִּי עַד כַּמָּה הִיא צְעִירָה, וְגַם שֶׁיֵּשׁ כְּוִויָּיה גְּדוֹלָה מְאוֹד עַל צַוּוֹארָהּ. 
------------------
סֶט שֶׁל שְׁתֵּי מַסֵּיכוֹת עִם הֶדְפֵּס שֶׁל מוּתָג YOLO נִמְכַּר בְּ-39.90, וְאוֹתוֹ נִיתָּן לְהַשִּׂיג בְּרֶשֶׁת וּבְטֶרְמִינָל X. מַסֵּיכָה שֶׁל הַמּוּתָג FILA מוּצַּעַת בְּ-47 שְׁקָלִים, וְרֶשֶׁת H&O מַצִּיעָה זוּג מַסֵּיכוֹת צִבְעוֹנִיּוֹת וְסַסְגּוֹנִיּוֹת לִילָדִים בִּשְׁמוֹנָה שְׁקָלִים בִּלְבַד.

בִּתְחוּם הַהִיגְיֵינָה, שֶׁחָשׁוּב בְּהִתְמוֹדְדוּת עִם נְגִיף הַקּוֹרוֹנָה, אֶפְשָׁר לִמְצוֹא פְּרִיטִים מְתוּחְכָּמִים. 
------------------
כִּשְׁבוּעַיִים אַחֲרֵי שֶׁרִיבּוֹנוּתָהּ הוּחְזְרָה לָהּ בְּ-28 בְּאַפְּרִיל 1952, הוֹדִיעַ שַׂר הַחוּץ הַיַּפָּנִי כִּי מֶמְשַׁלְתּוֹ הֶחֱלִיטָה לְקַבֵּל אֶת הַצָּעַת יִשְׂרָאֵל לְכוֹנֵן יְחָסִים דִּיפְּלוֹמָטִיִּים מְלֵאִים בֵּין שְׁתֵּי הַמְּדִינוֹת. צִיר יִשְׂרָאֵל בְּטוֹקְיוֹ הָיָה לַנָּצִיג הַדִּיפְּלוֹמָטִי הָרִאשׁוֹן שֶׁל הַמְּדִינָה הַיְּהוּדִית בְּיַבֶּשֶׁת אַסְיָה כּוּלָּהּ.


------------------
חָשׁוּב לִי לִרְאוֹת פִיזִית אֶת הָאָדָם שֶׁאֲנִי יוֹצֵאת אִיתּוֹ, הָיָה לִי קָשֶׁה עִם הַפָּן הַטֶּכְנוֹלוֹגִי. לְאַחַר שֶׁאוּשְּׁרוּ הַהֲקַלּוֹת, הֲדֶיְיטִים הִתְחַדְּשׁוּ וְעָבְרוּ לְפַארְקִים צִיבּוּרִיִּים אוֹ לִמְקוֹמוֹת שֶׁאֶפְשָׁר לָשֶׁבֶת בָּהֶם בִּפְרָטִיּוּת בָּאֲוִויר הַפָּתוּחַ. 
------------------
מִי יוֹדֵעַ, אוּלַי עַל הַקְּשָׁרִים הַמִּתְהַדְּקִים בֵּין יִשְׂרָאֵל לַעֲרַב הַסָּעוּדִית עוֹד יִהְיוּ כַּמָּה שַׁגְרִירִים לְשֶׁעָבַר וְאוּלַי חַגַּי אֶלְעָד אוֹ מִישֶׁהוּ בִּדְמוּתוֹ שֶׁיַּגִּישׁוּ תְּלוּנָה בְּהָאג. הָאָמֵרִיקָנִים לֹא אוֹהֲבִים אֶת הַסָּעוּדִים. פַּעַם הַדִּיפְּלוֹמָטִים וְאַנְשֵׁי הַבִּיטָּחוֹן מְאוֹד אָהֲבוּ אוֹתָם, וְלֹא רַק בִּגְלַל הַנֵּפְט אֶלָּא בִּגְלַל הַקַּרְיֵירָה הַשְּׁנִיָּיה שֶׁהֵם אִפְשְׁרוּ לַחֲלִיפוֹת הַחֲנוּטוֹת מֵהַבִּיצָּה שֶׁל ווֹשִׁינְגְּטוֹן.
