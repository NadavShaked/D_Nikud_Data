הוֹרִים רַבִּים רוֹצִים לִהְיוֹת הַחֲבֵרִים הֲכִי טוֹבִים שֶׁל הַמִּתְבַּגְּרִים שֶׁלָּהֶם. בָּרָמָה הָרִגְשִׁית, אֲנִי יָכוֹל לְהָבִין וּלְהִתְחַבֵּר לָרָצוֹן הַזֶּה. הֲרֵי מָה יוֹתֵר כֵּיף וּמְמַלֵּא מִזֶּה שֶׁהַיְּלָדִים רוֹאִים אוֹתָנוּ כַּ"בֶּסְטִיז" שֶׁלָּהֶם? הַשְּׁאֵלָה הִיא הַאִם הַחֲבֵרוּת הַזּוֹ נְכוֹנָה עֲבוּרָם? מָה הַהֶבְדֵּל בֵּין לִהְיוֹת מְבוּגָּר מַשְׁמָעוּתִי לְבֵין לְשַׁמֵּשׁ כְּחָבֵר טוֹב? וַהֲכִי חָשׁוּב - מָה הֵם בֶּאֱמֶת צְרִיכִים מֵאִיתָּנוּ?

הַשִּׂמְלָנִית (חֲצָאִית) שֶׁלָּהּ הָעֲשׂוּיָה מִבַּד וָרוֹד, רְדִידָהּ הַלָּבָן הַמְּרוּקָּם בְּזָהָב, צְעִיפָהּ הַמְּכוּתָּר בְּיַהֲלוֹמִים. כִּיסּוּי רֹאשָׁהּ הֶעָשׂוּי מֶשִׁי... כּוּלָּם חוֹבְרִים יַחְדָּיו כְּדֵי לְגַדֵּל וּלְהַלֵּל אֶת גּוּפָהּ הַפַּתְיָינִי".

אֲפִילּוּ בְּקֶרֶב הַיְּהוּדִים הָאוֹרְתּוֹדוֹכְּסִים בְּאַרְהָ"ב, נִיכָּר שִׁינּוּי בַּטּוֹן כְּלַפֵּי נִישּׂוּאִים מְעוֹרָבִים, וּבִמְקוֹם נִידּוּי שֶׁל חַבְרֵי קְהִילָּה שֶׁנִּישְּׂאוּ לְלֹא-יְהוּדִים, נִיכֶּרֶת נְכוֹנוּת רַבָּה יוֹתֵר לְקַבֵּל בִּבְרָכָה בְּנֵי זוּג שֶׁאֶחָד מֵהֶם אֵינוֹ יְהוּדִי, בְּתִקְוָוה שֶׁבֶּן הַזּוּג הַלֹּא יְהוּדִי יִתְגַּיֵּיר בְּסוֹפוֹ שֶׁל דָּבָר.
גַּם יַחֲסָהּ שֶׁל הַחֶבְרָה הָאָמֵרִיקָנִית כְּלַפֵּי הַמִּיעוּט הַיְּהוּדִי הִשְׁתַּנָּה לְטוֹבָה עִם הַזְּמַן.

כָּעֵת הִיא מַגִּיעָה לְמִשְׂחָק אֲפִילּוּ יוֹתֵר קָשֶׁה מוּל קְבוּצָה שֶׁלֹּא אִיבְּדָה נְקוּדּוֹת בַּבַּיִת מֵאָז חוֹדֶשׁ סֶפְּטֶמְבֶּר.
הַמְּנַצַּחַת תִּזְכֶּה לִפְרָס קָטָן: יִתְרוֹן בְּמַאֲזַן הַנִּיצְחוֹנוֹת בְּכָל הַזְּמַנִּים, שֶׁכֵּן הַקְּבוּצוֹת נִמְצָאוֹת בְּתֵיקוֹ 13:13.
לָמָּה בֵּיתָ"ר: אֲפִילּוּ לֹא סָפְגָה בְּטֶדִי מֵאָז הַתְּבוּסָה לְמַכַּבִּי ת"א.
לָמָּה קִרְיַית שְׁמוֹנָה: הַרְבֵּה זְמַן לֹא הָיְיתָה הַפְתָּעָה בַּלִּיגָה.

קָשֶׁה לְהַאֲמִין שֶׁאָלֶרְדַּיְיס גֵּאֶה בַּהֶישֵּׂג הַזֶּה שֶׁלּוֹ, כִּי מָה שֶׁכּוּלָּם יִזְכְּרוּ וְיַזְכִּירוּ לוֹ זוֹ הָעוּבְדָּה שֶׁהוּא שָׂרַד בְּתַפְקִיד 67 יָמִים בִּלְבַד, כְּלוֹמַר קְצָת לְמַעְלָה מֵחוֹדְשַׁיִים.
סְטִיב מֶקְלָארֶן, עוֹד מְאַמֵּן כּוֹשֵׁל, הָיָה עַד עַתָּה הַשִּׂיאָן הָאַנְגְּלִי, כִּי הֶחֱזִיק בְּתַפְקִיד הַמְּאַמֵּן הַלְּאוּמִּי בְּמֶשֶׁךְ 18 חוֹדָשִׁים וּ-18 יָמִים, בָּהֶם נִיהֵל אֶת הַנִּבְחֶרֶת בְּ-18 מִשְׂחָקִים.

------------------
אֲנָשִׁים שֶׁסּוֹבְלִים מִכִּיב קֵיבָה (אוּלְקוּס) אוֹ בַּעֲלֵי נְטִיָּיה לְדִימּוּמִים בְּמַעֲרֶכֶת הָעִיכּוּל צְרִיכִים לָדוּן עַל הַסִּיכּוּן עִם הָרוֹפֵא הַמְּטַפֵּל, שֶׁיַּמְלִיץ עַל תּוֹסֶפֶת תְּרוּפוֹת שֶׁמַּפְחִיתוֹת אֶת חוּמְצִיּוּת הַקֵּיבָה (כְּמוֹ אוּמְפְּרָאזוֹל אוֹ לוּסָאק).
אִם גַּם אָז הָאַסְפִּירִין יִגְרוֹם לְדִימּוּם אוֹ לִכְאֵבִים, יַמְלִיץ הָרוֹפֵא עַל תְּרוּפָה אַחֶרֶת לְדִילּוּל הַדָּם.
 
הַתְּשׁוּבָה לַשְּׁאֵלָה הַזּוֹ הִיא לָרוֹב שְׁלִילִית.

הַהִתְפָּרְצוּת הַנּוֹכְחִית הֵחֵלָּה לְאַחַר שֶׁפָּעוֹט בֶּן שָׁנָה וָחֵצִי מֵת בִּדְרוֹם מְחוֹז גְּוַואנְגְּדּוֹנְג.
מֵאָז דּוַּוח עַל יוֹתֵר מִ-3,000 יְלָדִים נוֹסָפִים נְגוּעִים בַּנְּגִיף בִּמְחוֹזוֹת אַנֶּהֱוֵי וְזָגִ'יאַנְג.
הַהִתְפָּרְצוּת הֵחֵלָּה לְהִתְפַּשֵּׁט בָּעִיר פוֹיַאנְג בְּמֵרְץ הָאַחֲרוֹן, אוּלָם דּוְּוחָה רַק בַּשָּׁבוּעַ שֶׁעָבַר.
מוּמְחִים הִבִּיעוּ חֲשָׁשׁ שֶׁהַוִּוירוּס הִתְפַּשֵּׁט מִדְּרוֹם הַמְּדִינָה לִצְפוּנָה.

בְּצַלַּחַת הַגָּשָׁה מַנִּיחִים אֶת הֶחָצִיל, מְתַבְּלִים בְּשֶׁמֶן זַיִת, מִיץ לִימוֹן וְעָלִי זַעְתָּר, מְנִיחִים אֶת הַשְּׁקֵדִים הַמְּטוּגָּנִים עַל הֶחָצִיל, סוֹחֲטִים מִיץ לִימוֹן, שֶׁמֶן זַיִת, מֶלַח, פִּלְפֵּל וּמַגִּישִׁים.

לָמָּה רַבִּין נָהַג כָּךְ? הוּא כְּכָל הַנִּרְאֶה סָבַר שֶׁפִּינּוּי הִתְנַחֲלוּיוֹת יַעֲלֶה לוֹ בִּמְחִיר פּוֹלִיטִי רַב מִדַּי. לְיָמִים הוּא שִׁילֵּם בְּחַיָּיו מִבְּלִי שֶׁפִּינָּה אֲפִילּוּ הִתְנַחֲלוּת אַחַת
אַחֲרֵי הַחֲתִימָה עַל הֶסְכֵּם אוֹסְלוֹ הִשְׁקִיעָה מֶמְשֶׁלֶת רַבִּין מִילְיַארְדֵּי שְׁקָלִים בְּמַעֲרֶכֶת תַּשְׁתִּיּוֹת וּכְבִישִׁים עוֹקְפִים בַּשְּׁטָחִים. 
------------------
הוּשְּׂגוּ הַסְכָּמוֹת עֶקְרוֹנִיּוֹת בֵּין שָׁאוּל אַלוֹבִיץ' לְנָתִי סַיְידוֹף לִמְכִירַת הַבַּעֲלוּת עַל חֶבְרַת יוּרוֹקוֹם לְסַיְּידוּף.
עַל פִּי הַהַסְכָּמוֹת, יִרְכּוֹשׁ סַיְידוֹף אֶת מְלוֹא הַבַּעֲלוּת עַל הַחֶבְרָה שֶׁמַּחֲזִיקָה בִּשְׁלִיטָה בְּבֶזֶק.
הַבַּנְקִים הַמְּמַנִּים שֶׁל יוּרוֹקוֹם הֵם הַפּוֹעֲלִים, דִּיסְקוֹנְט וְהַבֵּינְלְאוּמִּי.
עַתָּה מַמְתִּינִים הַצְּדָדִים לְאִישּׁוּרָם.

הַחוֹקְרִים מָצְאוּ שֶׁ-99.5% מֵהַנֶּחְקָרִים פִּיתְּחוּ נוֹגְדָנִים בִּתְגוּבָה לַחִיסּוּן, אַךְ הַתְּגוּבָה הָיְיתָה פְּחוּתָה בְּקֶרֶב אֲנָשִׁים שֶׁסּוֹבְלִים מֵהַשְׁמָנַת יֶתֶר.

מֶחְקָרִים קוֹדְמִים שֶׁפּוּרְסְמוּ הַשָּׁנָה מָצְאוּ שֶׁהַשְׁמָנַת יֶתֶר – מַדַּד מַסַּת גּוּף (BMI) 30 וּמַעֲלָה –  מַגְדִּילָה אֶת הַסִּיכּוּן לִתְמוּתָה מִקּוֹרוֹנָה בְּכִמְעַט 50%, וְכֵן מַעֲלָה אֶת הַסִּיכּוּן לְהִתְאַשְׁפֵּז בְּבֵית חוֹלִים כְּתוֹצָאָה מִסִּיבּוּכֵי הַנְּגִיף בְּ-113%.

חַבָּ"ד וְחוּגֵי הַמַּחְזִירִים בִּתְשׁוּבָה, גִּילּוּ כְּבָר מִזְּמַן אֶת הַפּוֹטֶנְצְיָאל שֶׁל הָאִינְטֶרְנֶט בַּעֲשִׂיַּית נְפָשׁוֹת, וְעוֹשִׂים בּוֹ שִׁימּוּשׁ מְגוּוָּן.
אֲבָל גַּם בַּחוּגִים הָאֵלֶּה לֹא תִּמְצְאוּ רַבִּים שֶׁחוֹשְׁבִים שֶׁהָאִינְטֶרְנֶט טוֹב לִילָדִים.
 
הַנָּחַת הָעֲבוֹדָה הִיא שֶׁ"הַמְּבוּגָּרִים יְכוֹלִים לְהִיזָּהֵר, אֲבָל עַל הַצְּעִירִים צָרִיךְ לְהָגֵן".
הַמּוֹטִיב הַחוֹזֵר הוּא הַסַּכָּנָה לִילָדִים וּלְנוֹעַר אוֹתָהּ הַהוֹרִים לֹא יוּכְלוּ לַעֲצוֹר.

אוֹבְדַן אָדָם אָהוּב הוּא נְקוּדַּת הַתְחָלָה שְׁכִיחָה לְמַחֲלָה פְּסִיכוֹסוֹמָטִית, בְּיִיחוּד כְּשֶׁהָאוֹבְדָן הָיָה טְרָגִי אוֹ כָּרוּךְ בְּרִגְשׁוֹת אָשָׁם.

מַדְבִּיר, שֶׁהִשְׁאִיר קוּפְסָה עַל רַעַל בְּפִינַּת הַבּוּבּוֹת בְּגַן יְלָדִים בְּאֵזוֹר הַשְּׁפֵלָה בְּתוֹךְ קוּפְסָה שֶׁל תַּחֲלִיף חָלָב לְתִינוֹקוֹת, הוֹאֳשַׁם בְּשׁוּרָה שֶׁל עֲבֵירוֹת. בְּנוֹסָף הִגִּישׁ הַמִּשְׂרָד לַהֲגָנַת הַסְּבִיבָה כְּתַב אִישּׁוּם נֶגֶד חֶבְרַת הַהַדְבָּרָה וּמְנַהֲלֶיהָ. כְּתַב הָאִישּׁוּם הוּגַּשׁ בְּתוֹם חֲקִירָה שֶׁל הַמִּשְׁטָרָה הַיְּרוּקָּה.


------------------
כְּמוֹ דַּרְדַּקִּים שֶׁהַמְּלַמֵּד שֶׁלָּהֶם הִתְפַּטֵּר מִתַּפְקִידוֹ, שְׁאַר מְדִינוֹת הָעוֹלָם מַתְחִילוֹת בְּחֶדְוָוה לְהִילָּחֵם בֵּינָן לְבֵין עַצְמָם.
אֲבָל מִכֵּיוָון שֶׁהָעֲלוּיוֹת לְמִלְחָמָה בָּעֵת הַחֲדָשָׁה מַגִּיעוֹת לְמִסְפָּרִים אַסְטְרוֹנוֹמִיִּים, הִגִּיעוּ הַצְּדָדִים הַלּוֹחֲמִים לְמַסְקָנָה שֶׁצָּרִיךְ לִנְקוֹט בַּחֲשִׁיבָה שׁוֹנָה לַחֲלוּטִין.

------------------
ב', 43, טֶכְנַאי מַחְשְׁבִים יְרֵא שָׁמַיִים שֶׁעוֹבֵד עִם הַמִּגְזָר הַחֲרֵדִי הַיְּרוּשַׁלְמִי, אוֹמֵר שֶׁהַשִּׁימּוּשׁ בָּאִינְטֶרְנֵט הוּא בִּבְחִינַת מִפְלָט אַחֲרוֹן "נָדִיר לִרְאוֹת הוֹרִים חֲרֵדִים שֶׁמְּאַפְשְׁרִים לְיַלְדֵיהֶם לְהִשְׁתַּמֵּשׁ בָּאִינְטֶרְנֵט, וְגַם אָז בְּדֶרֶךְ-כְּלָל מְדוּבָּר בִּילָדִים שֶׁנֶּחֱשָׁבִים 'בְּעָיָיתִיִּים', וְזוֹ דֶּרֶךְ יְעִילָה לְהַשְׁאִיר אוֹתָם בַּבַּיִת שֶׁבּוֹ הֵם מַרְגִּישִׁים שֶׁיִּהְיוּ בְּטוּחִים יוֹתֵר.

------------------
הַגִּרְסָה שֶׁלִּי לְדוּבְדְּבָנִים בְּסִירוֹפּ מָתוֹק וְאַלְכּוֹהוֹלִי כּוֹלֶלֶת גַּם מַקֵּל וָנִיל (אוֹ תַּמְצִית טוֹבָה‭.(‬ אֶת הַדּוּבְדְּבָנִים שׁוֹמְרִים בְּצִנְצְנוֹת זְכוּכִית מְעוּקָּרוֹת.
הָעִיקּוּר מִתְבַּצֵּעַ עַל-יְדֵי בִּישּׁוּל הַצִּנְצָנוֹת בְּסִיר עִם מַיִם רוֹתְחִים בְּמֶשֶׁךְ חָמֵשׁ דַּקּוֹת, לְחִיטּוּי.

אָז זֶה הַזְּמַן לַעֲשׂוֹת לַדּוּבְדְּבָנִים מְסִיבַּת פְּרֵידָה, כִּי בְּעוֹד שְׁבוּעַיִים-שְׁלוֹשָׁה הֵם נֶעֱלָמִים, וּמַשְׁאִירִים אֶת הַקַּצֶּפֶת לְבַד.

הַמַּרְכִּיבִים:


------------------
הֵם צְרִיכִים אֶתְכֶם כְּהוֹרִים - בְּוַודָּאוּת    
נָכוֹן שֶׁהַמִּתְבַּגְּרִים עוֹשִׂים הַכּוֹל כְּדֵי לְהַרְאוֹת לָכֶם שֶׁהֵם לֹא צְרִיכִים אֶתְכֶם, אֲבָל אַל תִּטְעוּ. הֵם צְרִיכִים אֶתְכֶם עַכְשָׁיו יוֹתֵר מִמָּה שֶׁנִּדְמֶה לָכֶם, יוֹתֵר מִמָּה שֶׁנִּדְמֶה לָהֶם וּבֶטַח יוֹתֵר מִמָּה שֶׁהֵם מְסוּגָּלִים לְהַרְאוֹת לָכֶם, בַּהֲנָחָה שֶׁאַתֶּם לֹא "קוֹרְאִים בְּגִלְגּוּלֵי עֵינַיִים". 


------------------
אוֹסֶף הָאִיגְּרוֹת מִתְּקוּפַת הַגְּאוֹנִים הוּא אֶחָד הַמְּקוֹרוֹת הַמְּעַנְיְינִים וְהַקְּדוּמִים שֶׁבָּהֶם עוֹסְקִים הַחוֹקְרִים שֶׁל מִפְעַל הַמִּילּוֹן הַהִיסְטוֹרִי בָּאָקָדֶמְיָה לַלָּשׁוֹן הָעִבְרִית. 



------------------
נוֹיְבֶּרְגֵּר, כַּיּוֹם בַּת 45, הָיְיתָה הַמּוֹעֳמֶדֶת הַטִּבְעִית. בְּגִיל 31 נָחֲתָה כְּמֵטֵאוֹר בְּלִשְׁכָּתוֹ שֶׁל שַׂר הַהֲגָנָה רוֹבֵּרְט גֵּיְיטְס בַּפֶּנְטָגוֹן, בְּגִיל 39, עָמְדָה בְּרֹאשׁ מִינְהֶלֶת הֲגַנַּת הֲסַייְבֶּר שֶׁל הַסּוֹכְנוּת לְבִיטְחוֹן לְאוּמִּי (NSA), שֶׁאַחְרָאִית עַל מַעֲקָב אֵלֶקְטוֹנִי וְצִיתוּתִים.


------------------
"תָּמִיד חָשַׁבְנוּ שֶׁמִּצְווֹת וְטֶכְנוֹלוֹגְיָה הוֹלְכוֹת יַחַד", הוּא אוֹמֵר, "הֵקַמְנוּ עָרוּץ לַוְויָין מִשֶּׁלָּנוּ בּוֹ צוֹפִים מֵאוֹת אַלְפֵי בְּנֵי אָדָם, וְנִדְמֶה לִי שֶׁכַּמָּה עֲרוּצֵי טֵלֵוִויזְיָה יִתְקַנְּאוּ בַּרֶיְיטִינְג הַזֶּה.

------------------
בְּרַחֲבֵי הָעוֹלָם נֶעֱרָכִים נִיסּוּיִים בְּעֶשְׂרוֹת תְּרוּפוֹת נוֹסָפוֹת, בֵּינְתַיִים לְלֹא עֵדוּיוֹת לְהַצְלָחוֹת מַרְשִׁימוֹת. כְּמוֹ בְּמַגֵּפוֹת וִירָאלִיּוֹת אֲחֵרוֹת, נִרְאֶה שֶׁלְּפָחוֹת בַּשָּׁנִים הַקְּרוֹבוֹת, הַדֶּרֶךְ הַיְּחִידָה לְנַצֵּחַ אֶת הַמַּגֵּפָה תִּהְיֶה בְּעִיקָּר בְּאֶמְצָעוּת הַחִיסּוּנִים.
------------------
אָמְנָם יוֹקֶר הַמִּחְיָה עֲדַיִין נוֹשֵׁף בְּעָרְפֵנוּ וּמַקְשֶׁה עַל הַהִתְנַהֲלוּת הַשּׁוֹטֶפֶת, אַךְ סוֹף מַעֲשֶׂה בְּמַחְשָׁבָה תְּחִילָּה.
תִּיעֲדוּף נָכוֹן שֶׁל הַהוֹצָאוֹת וּוִיתּוּר הֵיכָן שֶׁלֹּא צָרִיךְ אָמְנָם יַרְגִּישׁ לָנוּ עַכְשָׁיו קְצָת לֹא נָח, אַךְ לִטְוָוח אָרוֹךְ – נִהְיֶה בִּצְמִיחָה.
------------------
כְּשֶׁאֲחוֹתִי וַאֲנִי הִגַּעְנוּ לַמָּקוֹם אָבִי כְּבָר הָיָה מֵאֲחוֹרֵי דֶּלֶת נְעוּלָה, בְּיַחַד עִם עוֹד אֲנָשִׁים שֶׁהַצֶּוֶות הָרְפוּאִי חֲשָׁד כִּי יִיתָּכֵן שֶׁיֵּשׁ לָהֶם קוֹרוֹנָה. 
לְאַחַר יָמִים שֶׁל דְּאָגָה וּמַחְשָׁבוֹת הֵבַנְתִּי שֶׁשִּׁגְרַת הַמַּחֲלָה כְּבָר הִקְהֲתָה אֶת חוּשֵׁינוּ בֵּין סֶגֶר לְמַגֵּפָה. נוֹחַ לָנוּ לְהִתְעַלֵּם מִמִּסְפַּר הַחוֹלִים הַגָּדוֹל וּלְנַסּוֹת לְהַמְשִׁיךְ אֶת חַיֵּינוּ כְּמוֹ קוֹדֶם

------------------
אֲבִי אָדָם עַצְמָאִי וּמְתַפְקֵד, אָמְנָם מִתְקַשֶּׁה בַּהֲלִיכָה וְנִזְקַק לִתְרוּפוֹת, כִּמְעַט כְּמוֹ כָּל אָדָם בְּגִילוֹ, אֲבָל גַּם מַרְצֶה, כּוֹתֵב מַאֲמָרִים וּבְהֶחְלֵט אָדָם חִיּוּנִי. בְּיַחַד עִם אִימִּי, שֶׁאִיתָּהּ הוּא מְבַלֶּה 24 שָׁעוֹת בִּימָמָה מֵאָז שֶׁהַקוֹרוֹנָה נִכְנְסָה לְחַיֵּינוּ, הוּא הוּרַד מֵהָאַמְבּוּלַנְס בַּחֲדַר הַמִּיּוּן.

כָּךְ לְמָשָׁל, הַכְּנִיסָה שֶׁל קֶמַח חִיטָּה מוֹעֳשָׁר בְּבַרְזֶל לְיַרְדֵּן הִפְחִיתָה אֶת שִׁיעוּר הָאָנֶמְיָה בְּקֶרֶב יְלָדִים בַּמְּדִינָה כְּתוֹצָאָה מִמַּחְסוֹר בְּבַרְזֶל כִּמְעַט לְמַחֲצִית. הַעֲשָׁרַת מְזוֹנוֹת שִׁימְּשָׁה גַּם לִמְנִיעַת מַצָּבִים אֲחֵרִים הַנִּגְרָמִים מִמַּחְסוֹר בִּתְזוּנָה, כּוֹלֵל רַכֶּכֶת וּמוּמִים מוּלָדִים.

בַּנְק יִשְׂרָאֵל שָׂם דָּגֵשׁ עַל חִיזּוּק מְעוֹרָבוּת הַדִּירֶקְטוֹרְיוֹן וְהַהַנְהָלָה בִּתְחוּם זֶה, "בְּנִיגּוּד לְגִישׁוֹת שֶׁרָוְוחוּ בֶּעָבָר וְשֶׁגָּרְסוּ כִּי הָאַחְרָיוּת בִּתְחוּמִים אֵלּוּ מוּטֶּלֶת אַךְ וְרַק עַל פוּנְקְצְיַית טֶכְנוֹלוֹגְיַית הַמֵּידָע בְּתַאֲגִיד הַבַּנְקָאִי", אָמְרָה יַעֲקֹבִּי.
