וְנִשְׁפְּכוּ אֶל הַכְּבִישׁ נֶהָגִים וְנוֹסְעִים, אֲנָשִׁים וְנָשִׁים, צְעִירִים וּצְעִירוֹת בְּמַדִּים, עַרְבִים עֲטוּפֵי-רֹאשׁ שְׁנַיִם שְׁלשָׁה – מִי בִּשְׁמוּעָה, מִי בִּתְלוּנָה וּמִי בְּעֵצָה; עַד שֶׁבָּאָה הַמִּשְׁטָרָה וְעָשְׂתָה סְדָרִים. זֶה קָרָה מֵעֵבֶר לְמִגְדַּל-כְּרָמִים, אַחֲרֵי הִתְגַּלּוּת הָרֵי הַגָּלִיל הַתַּחְתּוֹן וּבִקְעוֹתָיו, בְּסָמוּךְ לִגְבוּל אֵזוֹר קִרְיַת-בְּיַאלִיק. 
------------------
כְּקֶדֶם תַּשֵּׁב מַעַרְכוֹת קָמֶיךָ; תְּנַגְּחֵם בְּקַרְנֵי-אוֹן וְתַצְמִיתֵם. סַלְעֵי-מִדְבָּר יֵדְעוּ קַדַּחַת-צַעֲדֶךָ. אָז תִּפְקְדֵנִי גַּם אָנִי, שִׁמְשׁוֹן. שׁוּב תַּעַקְדֵנִי בְּנַהֲרַת מַחְלְפוֹתֶיךָ, וְתַרְקִיד מַבָּטֶיךָ זַהֲרוּרִים-זַהֲרוּרִים עַל גִּבְעוֹת-גוּפִי. תְּחַמֵּם נוֹזְלֵי-לִבִּי מִכִּמּוּרֵי רַחֲמֶיךָ עָלָי… בֵּינְתַיִם אֵרֵד אֶל גְּבוּל עַמְּךָ, אֲלַוֶּךָ אֶל קֶבֶר מָנוֹחַ אָבִיךָ, וְאֶאֱבַל יַחַד עִם אַחֶיךָ. 
------------------
יָשַׁב לוֹ הַפִּילוֹן עַל עַכּוֹזוֹ הַקָּטָן וּמָשַׁךְ, וּמָשַׁךְ, וּמָשַׁךְ, וְחָטְמוֹ הִתְחִיל מִתְמַשֵּׁךְ וְהַתַּנִין גִּלְגֵּל עַצְמוֹ אֶל תּוֹךְ הַנָּהָר וְחָבַט בִּזְנָבוֹ חֲבִיטוֹת גּדוֹלוֹת עַד שֶׁהִקְצִיפוּ וְהִלְבִּינוּ הַמַּיִם כְּשַׁמֶּנֶת. וְגַם הוּא מָשַׁךְ, וּמָשַׁךְ וּמָשַׁךְ.

------------------
הַשָּׁמַיִם קְלוּיִים, מִן הַנָּכוֹן, לוֹמַר – קְלוּלִים, מִבַּעַד לִצְעִיפֵי-עָנָן דַּקִּים, רַחֲפָנִיִּים.
יֵשׁ לְשַׁעֵר, כִּי עֲנָנִים אֵלֶּה לֹא שֶׁל-מַיִם הֵם כִּי-אִם שֶׁל קִמְחֵי-חוֹל קַלִּים, נִשָּׂאִים בָּאֲוִירִים הָעֶלְיוֹנִים.
וְאֵימָתַי הִבַּטְתִּי גּוֹן-שָׁמַיִם כָּזֶה? בִּשְׁעַת לִקּוּי-חַמָּה, כְּשֶׁאַפְלוּלִית אֲדַמְדֶּמֶת נוֹחֶלֶת לְתִשְׁבֹּרֶת-שָׁעָה אֶת מֶמְשֶׁלֶת תְּכֵלֶת-הַיּוֹם.
כְּכֹל שֶׁמַּדְרִימִים, הַצִּמְחִיָּה הַשְּׁדוּפָה, הַסְּמַרְמֶרֶת, מַעֲשִׁירָהּ.

------------------
תִּהְיֶה הָאָח הַמְּחַמֶּמֶת לִי לְמִשְׁכָּן, וּבְכָל בַּיִת וָבַיִת אֵאָהֵב וְאֵעָבֵד".
וַתְּהִי הֶסְטִיָּה לְאֵלַת אֵשׁ-הָאָח; וּמִצְּרִיף-עֲנִיִּים וְעַד אַרְמוֹן-מְלָכִים יֻתַּן לָהּ יְקָר.
וּמְקוֹמָהּ בְּאוֹלִימְפּוֹס נִתַּן לְדִיאוֹנִיסוֹס.
דִּיאוֹנִיסוֹס מְשַׂמֵּחַ הָאֵלִים בְּיֵינוֹ
מַתָּנָה לוֹ מֵאַפְרוֹדִיטֵי וַיְהִי כַּעֲלוֹת דִּיאוֹנִיסוֹס בָּרִאשׁוֹנָה אֶל אוֹלִימְפּוֹס, וַיָּבֵא שְׁנֵי כַּדֵּי יַיִן עַל כְּתֵפָיו, וַיְשַׂמַּח אֶת לֵב הָאֵלִים בְּמַשְׁקֵה-הָאַרְגָּמָן הַצָּלוּל.

------------------
בִּשְׁנַת 1801 לִסְפִירַת הָאֻמּוֹת, עֶשְׂרִים וְחָמֵשׁ שָׁנָה אַחֲרֵי הָכְרָזַת עַצְמָאוּת אַרְצוֹת-הַבְּרִית, נִרְאָה בָּעֵבֶר הַמַּעֲרָבִי שֶׁל נְהַר אוֹהַיוֹ עֶלֶם שְׁחוֹר-עֵינַיִם, לָבוּשׁ בִּגְדֵי עוֹר-צְבִי כְּאַחַד הָאִינְדְּיָנִים. הָעֶלֶם הָיָה רוֹכֵב עַל סוּס אֲשֶׁר שְׁנֵי שַׂקִּים תְּלוּיִים לוֹ בְּגַבּוֹ מִזֶּה וּמִזֶּה. מָצָא הָעֶלֶם חֶלְקַת-יַעַר מְבֹרָאָה, יָרַד מֵעַל הַסּוּס, הוֹצִיא מִן הַשַּׂקִּים מְלוֹא חָפְנָיו זֵרְעוֹנֵי-תַפּוּחִים, וְזָרַע אוֹתָם בַּחֶלְקָה.

------------------
וַיְנַגֵּן הֶרְמֶס בַּחֲלִילוֹ אֶת כִּסּוּפֵי אֶנְדִּימְיוֹן, בֶּן הַנִּימְפָה קַלִּיקֵי, אֶל סֶלֶנֵי, אֵלַת-הַלְּבָנָה.
עֵת יָשֵׁן הָעֶלֶם אֶנְדִּימְיוֹן בִּמְעָרָה שֶׁבְּהַר-לֶטְמוֹס, רָאֲתָה אוֹתוֹ סֶלֶנֵי וַתִּשְׁלַח אֵלָיו מִקַּרְנֵי-כַסְפָּהּ וַתְּנַשֵּׁק אֶת שְׁתֵּי עַפְעַפָּיו.
וַיִּתְעוֹרֵר אֶנְדִּימְיוֹן, וַיִּתְהַלֵּךְ כָּל הַיָּמִים חוֹלֵה אַהֲבַת-הַלְּבָנָה.

------------------
הָאִינְדְּיָנִי, בְּעָקְבוֹ אַחַר צַיִד, הֶחֱלִיק עַל עֲרֵמַת-עָלִים רְטֻבָּה, וְהָרוֹבֶה שֶׁלּוֹ (כִּי “אֲדֻמֵּי-הָעוֹר” הָיוּ מוֹכְרִים פַּרְווֹת שֶׁל חַיּוֹת בִּמְחִיר הַנֶּשֶׁק שֶׁל “לִבְנֵי-הַפָּנִים”) נִלְחַץ הִדְקוֹ וְיָרָה מֵאֵלָיו. יוֹנָתָן הוֹצִיא הַכַּדּוּר שֶׁבָּרֶגֶל, חָבַשׁ הַפֶּצַע, וְהִשְׁגִּיחַ עַל הָאִינְדְּיָנִי יָמִים אֲחָדִים עַד שֶׁהַלָּה יָכֹל לָקוּם וְלָלֶכֶת. הֱבִיאָהוּ הָאִינְדְּיָנִי הַצָּעִיר אֶל אֹהֶל-שִׁבְטוֹ, וְסִפֵּר לְאֶחָיו אֶת הַחֶסֶד שֶׁעָשָׂה עִמּוֹ הָאִישׁ הַלָּבָן.

------------------
כְּרוּבֵי-אֵימָה, שׁוֹמְרֵי אוֹפַנִּים וָרֶכֶב פְּלִיאֵי-שֵׁם, הִדְּקוּ יְדֵיהֶם עַל חַרְבוֹתָם הַבּוֹרְקוֹת, אַךְ עמְדוּ קוֹפְאִים מֵרֹב-גְּאוֹן הַמְּשׁוֹרֵר וּמִכְּבוֹד חֻפָּתוֹ. וּלְפָנָיו תִּהֲלַךְ רְצוּעָה-שֶׁל-אֵשׁ, מְסַקֶּלֶת מִדַּרְכּוֹ נִגְפֵי-סוֹד וּמִכְשׁוֹלֵי-סְתוּמוֹת, כְּגַל-אוֹקְיָנוֹס בַּלַּיְלָה הַקָּם וּמוֹחֵק, בִּתְנוּפָה אַחַת, צַלְמֵי-כּוֹכָבִים מְפֹרָרִים.

------------------
 מִיָּדָם, וְיִזְדָּרֵז לְהַתְחִיל בְּקִדּוּחַ.
עַד שֶׁהַצְּעִירִים שׁוֹגִים בִּפְלֻגְתּוֹת וְדִמְיוֹנוֹת, בָּא “קוֹמַנְדְּקַר” לְהַסִּיעָם מִשָּׁם אֶל מָקוֹם אַחֵר…
וּבָזֶה הָעִנְיָן הָאַחֲרוֹן שֶׁבַּמַּעַרְכוֹן יֵשׁ עֹקֶץ אַקְטוּאָלִי.
כִּי גְזֵרָה יָצְאָה שֶׁמַּנִּיחֵי הַיְּסוֹדוֹת לַמֶּשֶׁק בְּשַׁדְמוֹת-שֵׁיזָף, לֹא יֵשְׁבוּ פֹּה לִצְמִיתוּת, אֶלָּא יְפַנּוּ מְקוֹמָם לְמִתְיַשְּׁבִים קְבוּעִים אֲשֶׁר יָבוֹאוּ אַחֲרֵיהֶם.

------------------
וַיְהִי כִּרְאוֹת דֵּימֶטֶר אֶת בִּתָּהּ, וַתְּזַנֵּק אֵלֶיהָ, וַתְּחַבְּקֶנָּה, וְהָאֵם וְהַבַּת בָּכוּ בְּגִיל אִשָּׁה עַל צַוְּארֵי רְעוּתָהּ.

נְאוּם הֶרְמֶס אֶל דֵּימֶטֶר: "הִנֵּה בִּתֵּךְ עַל-יָדֵךְ, וְיָדַעַתְּ כִּי שָׁלוֹם לָהּ.
הָסִירִי, אֵפוֹא, אֶת הַמְּאֵרָה מֵעַל הָאָרֶץ, וְתָשׁוּב הָאֲדָמָה לָתֵת יְבוּלֶיהָ הַטּוֹבִים, וְלֹא יֵעָצְבוּ עוֹד בְּהֵמוֹת וַאֲנָשִׁים וְאֵלִים".
וַתִּשְׁאַל דֵּימֶטֶר: "וְלֹא יָבוֹא עוֹד הָדֶס כְּשׁוֹדֵד לַחְטוֹף אֶת קוֹרֵי שֶׁלִּי?"

------------------
וּלְכָל צִפּוֹר אֲשֶׁר בַּגָּן וַאֲשֶׁר בַּהֵיכָל פְּנִימָה – נֶקֶב מְיֻחָד בְּכָתְלֵי הַחֲדָרִים אוֹ בְרִצְפּוֹתָם, אֲשֶׁר מִשָּׁם תַּעֲלֶה זִמְרַת הַצִּפּוֹר אֶל הַקּוֹסֵם הַיּוֹשֵׁב בַּעֲלִיָּתוֹ. וְכָל צִנּוֹר עָשׂוּי לְכַתְּחִלָּה, בְּחָכְמַת הַקּוֹסֵם, לְפִי אָפְיָהּ, כֹּחָהּ וְאֹרַח-שִׁירָתָהּ שֶׁל צִפּוֹר-צִפּוֹר. וְצִנּוֹר הַמַּעֲבִיר זִמְרָתָהּ שֶׁל הַצִּפּוֹר הָאַחַת, לֹא יוּכַל לְהַעֲבִיר זִמְרָתָהּ שֶׁל הַצִּפּוֹר חֲבֶרְתָהּ. 
------------------
וּפְסִיכֵי פָּקְחָה עֵינֶיהָ, וַתֵּרֶא אֶת אֲהוּבָהּ.
וַתָּקָם, וּתִּדְבַּק בּוֹ, וַתֵּבְךְּ בִּזְרוֹעוֹתָיו.
וְהוּא נִרְפָּא כָּלִיל, וְלִכְוִיָּתוֹ אֵין סִימָן וָזֵכֶר.
וְהַפְּרִיחָה עָלְתָה בִלְחָיָיו, וְנוֹצוֹת-כְּנָפָיו רָטְטוּ בְּגִיל וַיִּרְעֲפוּ חַכְלִילוּת.
וְאֶרוֹס אָמַר אֶל פְּסִיכֵי: "אִתִּי אֶל אוֹלִימְפּוֹס תָּבוֹאִי, אֲהוּבָה, וְלֹא אֶעֶזְבֵךְ עוֹד לְעוֹלָם.
וְאֶת הַקֻּפְסָה עִם מִשְׁחַת-פֶּרְסֶפוֹנֵי תָּבִיאִי מִנְחָה אֶל אִמִּי, לְאוֹת כִּי מִלֵּאת שְׁלִיחוּתֵךְ.

------------------
וּגְבֶרֶת-הָאֵלִים, הֶרָה אֵשֶׁת-זֶאוּס, אֲשֶׁר מִתַּפְקִידֶיהָ לְהַשְׁכִּין שָׁלוֹם בֵּין אִישׁ לְאִשְׁתּוֹ, נִגְּשָׁה אֶל אַפְרוֹדִיטֵי בְּאוֹלִימְפּוֹס, וַתְּדַבֵּר אֵלֶיהָ לֵאמֹר: "מִמָּרוֹם הִשְׁקַפְתִּי וְרָאִיתִי אֵת אֲשֶׁר עָשִׂית לִפְסִיכֵי הַצְּעִירָה.
מֵרָצוֹן בָּאָה לְהַכְנִיעַ עַצְמָהּ לְפָנַיִךְ, וְעָמְדָה בְּכָל הַנִּסְיוֹנוֹת הַקָּשִׁים אֲשֶׁר שַׁתְּ עָלֶיהָ; וְגַם אֶל שְׁאוֹל יָרְדָה לְמַעֲנֵךְ, וַתָּעַל.

------------------
– הֵם כּוֹעֲסִים, – אָמְרָה אֵשֶׁת הַפַּרְפָּר בְּנַפְנְפָהּ בִכְנָפֶיהָ – כּוֹעֲסִים עַל לֹא דָבָר, אֲבָל עָלֵינוּ לְהַחֲנִיף לָהֶם, – הוֹ מַלְכָּה! אֵין הֵם מִתְכַּוְּנִים לַחֲצִי-הַדְּבָרִים שֶׁהֵם אוֹמְרִים. אִם טוֹב בְּעֵינֵי בַעְלִי לְהַאֲמִין שֶׁאֲנִי מַאֲמִינָה שֶׁהוּא יָכֹל בִּרְקִיעַת-רֶגֶל לַעֲשׂוֹת שֶׁיֵּעָלֵם אַרְמוֹנוֹ שֶׁל שְׁלֹמֹה בֶן דָּוִד, מָה אִכְפַּת לִי? הוּא יִשְׁכַּח אֶת כָּל הָעִנְיָן עַד מָחָר.

------------------
הַיּוֹם נִמְלְאוּ אַרְבַּע שָׁנִים לִפְטִירַת אָבִי-מוֹרִי, עָלָיו הַשָּׁלוֹם, אִישׁ רָצוּי לַבְּרִיּוֹת וִירֵא-אֱלֹהִים.
אֶכְשַׁר בְּעֵינֵיכֶם לַעֲבוֹר לִפְנֵי הַתֵּבָה וְלִהְיוֹת מֵלִיץ בַּעַדְכֶם לֵאלֹהִים”.
וַיְהִי הַס בָּעֵדָה.

------------------
 לְמִין-הָאָדָם; חוֹבַת-הָעֲבוֹדָה וְחוֹבַת-הַמְּנוּחָה, שֶׁשְׁתֵּיהֶן בְּיַחַד עוֹשׂוֹת אֶת הָאָדָם לְבֶן-חוֹרִין.
זֹאת הִיא הַהִתְגַּלּוּת בַּמִּדְבָּר.
אֲבָל הַמִּדְבָּר לֹא לְעוֹלָם יִשָּׁאֵר מִדְבָּר.
“לֹא תֹהוּ בְרָאָהּ, לָשֶׁבֶת יְצָרָהּ”.
גְּזֵרָה וְצַו הֵם שֶׁהַשְּׁמָמָה תִּוָּשֵׁב, תִּהְיֶה לִכְפָרִים, לְעָרִים, לְמֶרְכְּזֵי-מְדִינָה.

------------------
שְׁלֹמֹה בֶּן דָּוִד הָיָה חָכָם. הוּא שָׁמַע אֶת שְׁפַת הַחַיּוֹת וְהָעוֹפוֹת וְהַדָּגִים וְהָרְמָשִׂים. הֵבִין מַה שֶּׁאָמְרוּ הַסְּלָעִים עָמֹק מִתַּחַת הָאֲדָמָה, כְּשֶׁהֵם הִתְכּוֹפְפוּ זֶה לִקְרַאת זֶה וְנָהָקוּ; וְהוּא הֵבִין מַה שֶּׁאָמְרוּ הָעֵצִים בְּרַשְׁרְשָׁם לְרוּחַ הַבֹּקֶר. אֶת הַכֹּל הֵבִין, לְמִן הַהֶגְמוֹן עַל כִּסֵּא הֶגְמוֹנוּתוֹ וְעַד הָאַגְמוֹן שֶׁבָּאֲגָם; וּבַלְקִיס רִאשׁוֹנָה בְּמַלְכוֹתָיו, הַיָּפָה שֶׁבַּמְּלָכוֹת בַּלְקִיס, הָיְתָה חֲכָמָה כִּמְעַט כָּמוֹהוּ.

------------------
נֻקַּב נֶקֶב בְּצָמִיג אֶחָד שֶׁל הַמְּכוֹנִית הַהוֹלֶכֶת מִירוּשָׁלַיִם אֶל תֵּל-אָבִיב. עָמְדוּ חֲבֵרַי-לִנְסִיעָה וְהִתְיָעֲצוּ, כֵּיצַד מְרִימִים אֶת הַמְּכוֹנִית וּמַעֲלִים עָלֶיהָ צָמִיג חָדָשׁ. אֲנִי, בִּהְיוֹתִי הֶדְיוֹט גָּמוּר בְּהִלְכוֹת-מְכוֹנָה, עָמַדְתִּי מִנֶּגֶד. בִּקַּשְׁתִּי מֵחֲבֵרַי רְשׁוּת לְטַיֵּל מְעַט בִּסְבִיבוֹת הַמָּקוֹם, וּלִכְשֶׁתִּהְיֶה הַמְּכוֹנִית מְתֻקֶּנֶת, יִקְרְאוּנִי בִּתְקִיעַת חֲצוֹצְרָה. נָתְנוּ לִי.
הַמָּקוֹם מְקוֹם הָרִים שׁוֹמְמִים, כֻּלּוֹ סְלָעִים וַאֲבָנִים. 
------------------
"אַתָּה מְכַלְכֵּל צוּרוֹת-הוֹד: הָאָרֶץ, וְכָל בְּנֵי-לִוְיָתָהּ; כַּדּוּרִים חַיִּים, הַמְּאַכְלְסִים שִׁמְמוֹת מְצוּלוֹתֶיךָ; עוֹלָמוֹת יְרֻקִּים הַמַּחֲלִיקִים וְחוֹפְזִים לְדַרְכָּם; וְכוֹכָבִים פְּזִיזִים, שַׂעֲרוֹתֵיהֶם – נֹגַהּ; וּלְבָנוֹת מְאִירוֹת, קָרוֹת מִכָּל קָר; וּשְׁמָשׁוֹת אַדִּירִים בַּל יַגִּיעֵם לָיְלָה: פְּרָדֵי אוֹר-שֶׁבְּאוֹר! 

------------------
יוֹם אֶחָד יָרַד טֶגּוּמֵי דֶרֶךְ בִּצַּת הַבִּיבָרִים1 אֶל הַנָּהָר לִדְקוֹר דְּגֵי-שִׁבּוּט לִסְעֻדַּת-צָהֳרַיִם, וְגַם טָפִי הָלְכָה עִמּוֹ.
חֲנִיתוֹ שֶׁל טֶגּוּמֵי הָיְתָה עֲשׂוּיָה עֵץ וְשִׁנֵּי כָּרִישׁ2 בְּקָצֵהוּ, וְטֶרֶם צָד אַף דָּג אֶחָד, קָרָה מִקְרֵהוּ לִשְׁבּוֹר אֶת הַחֲנִית, כִּי הִכָּה אוֹתָהּ בְּחָזְקָה בְּקַרְקָעִית הַנָּהָר.
וְהֵם הָיוּ רְחוֹקִים מְאֹד מִבֵּיתָם (כַּמּוּבָן, לָקְחוּ אִתָּם צֵידָה לַדָּרֶךְ בְּתוֹךְ שַׂק קָטָן), וְטֶגּוּמֵי שָׁכַח הַפַּעַם לְהָבִיא חֲנִיתוֹת נוֹסָפוֹת.

------------------
הִכּוּ אֶת הַפֶּלַּסְגִּים עַד הִשָּׁאֵר אַךְ מְעַטִּים מֵהֶם בְּקִמְטֵי-הֶהָרִים צָפוֹנָה.
וַיִּבְנוּ הַיְּוָנִים עָרִים וַיְכוֹנְנוּ מַמְלָכוֹת.
וְגַם אֶת הָאִיִּים הַקְּרוֹבִים כָּבְשׁוּ, וְיִסְּדוּ בָּהֶם מַמְלָכוֹת וְעָרִים.
הֵמָּה הֵבִיאוּ אִתָּם סִפּוּרִים חֲדָשִׁים עַל קוֹרוֹת רֵאשִׁית-עוֹלָם.
מַתְחִילִים סִפּוּרֵיהֶם בַּאֲדָמָה, צָפָה בְּתֹהוּ.
וַתַּאֲצֵל אֲדָמָה אֲשֶׁר גֵּיאָה שְׁמָהּ, אֶת רְקִיעַ-שָׁמַים, אֲשׁר שְׁמוֹ אוּרָנוֹס; מֵעַצְמָהּ וְגוּפָהּ הֶאֱצִילַתְהוּ.

------------------
אֲבָל בִּגְתָן וְתֶרֶשׁ, שֶׁקָּשְׁרוּ קֶשֶׁר לְהָרְגֵנִי, לֹא יְהוּדִים הָיוּ… וְדַוְקָא יְהוּדִי, אֵינֶנִּי זוֹכֵר שְׁמוֹ – נִדְמֶה לִי, מָרְדְּכַי – הִצִּילַנִי.
הָמָן: (כִּנְשׁוּךְ-נָחָשׁ)

------------------
אָז שָׁלַח הַשַּׁלִּיט בָּאֵלִים אֶת אִמּוֹ, אֶת רֵיעָה, אֲשֶׁר הִיא גַם אֵם-דֵּימֶטֶר, וַיָּשֶׂם בְּפִיהָ הַצָּעַת-פְּשָׁרָה.
רֵיעָה אֵם-הָאֵלִים, בְּדִבְרֵי-פִיוּס, וְדֵימֶטֶר נַעֲנֵית לָהּ 
וַתֵּרֶד רֵיעָה אֶל דֵּימֶטֶר, וַתֹּאמֶר לָהּ: "כֹּה אָמַר זֶאוּס: עֵקֶב אֲשֶׁר אָכְלָה קוֹרֵי מִגַּרְעִינֵי-הַרִמּוֹן בְּטַרְטָרוֹס, תִּמְלֹךְ מַלְכָּה לְיַד הָדֶס – אֲבָל רַק לִשְׁלֹשֶׁת יַרְחֵי-שָׁנָה; וּמַרְבִּית-הַשָּׁנָה, תִּשְׁעָה יְרָחִים, תִּהְיֶה עִם אִמָּהּ".

------------------
וְהַמֶּלֶךְ, אֲבִי-פְסִיכֵי, בִּרְאוֹתוֹ אֶת יְגוֹן הַנַּעֲרָה וּרְזוֹנָה, נִבְהַל מְאֹד.
וַיֵּלַךְ אֶל דְּבִיר-אַפּוֹלוֹן אֲשֶׁר בְּמִלֶּטוֹס, וְאַחֲרֵי הַקְרִיב קָרְבָּנוֹת וְהִתְפַּלֵל כַּדָּת, שָׁאַל בְּאַפּוֹלוֹן, מֶה עָתִיד נָכוֹן לְבִתּוֹ הַצְּעִירָה, אֲשֶׁר יָפְיָהּ נוֹבֵל בְּבֵית-הוֹרֶיהָ, לְלֹא הֱיוֹתָהּ אִשָּׁה לְאִישׁ וּלְלֹא חֲבוֹק יָלֶד.
וַיַּעַן לוֹ אַפּוֹלוֹן עַל יְדֵי כֹּהַנְתּוֹ אֲשֶׁר בַּדְּבִיר, לֵאמֹר:
לֹא יְלוּד-אִשָּׁה יִהְיֶה חָתָן לְבִתְּךָ.
הַעֲלֵה אוֹתָהּ עֲדוּיַת-פְּאֵר

------------------
אָז יָצְאוּ הַחוּצָה, וְחָבְטוּ אֶת הַתֵּבוֹת הָעֲקוּרוֹת בַּקַּרְקַע, כְּדֶרֶךְ שֶׁיְּהוּדִים חוֹבְטִים עַרְבֵי-“הוֹשַׁעְנָא”.
וָאִבָּהֵל וָאֶקְרָא: הַאִם כְּלָיָה יַעֲשׂוּ אֵלֶּה בַּסֻּכָּה, וְיִגְמְרוּ הֵם אֶת הַהֶרֶס אֲשֶׁר לֹא הִסְפִּיקוּ לִגְמוֹר סוּפָה זְדוֹנִית וְחַיָּה רָעָה?
וָאֶשְׁמַע בַּת-קוֹל לֵאמֹר: “תָּנוּחַ דַּעְתֶּךָ.
לֹא הֶרֶס, חַס וְשָׁלוֹם, מְבִיאִים אֵלֶּה לַסֻּכָּה, כִּי אִם הַצָּלָה וּבִנְיָן וּתְחִיָּה.

------------------
הַאִם לֹא תִרְאֶינָה עֵינֵיכֶם כִּי אַחַד-הָאֵלִים הוּא?" וַיּלְעֲגוּ רַב-הַחוֹבְלִים וְהַמַּלָּחִים לְאִישׁ-הַהֶגֶה וְלַצָּעִיר הַשַּׁבוּי בִּידֵיהֶם.
פְּלָאִים בָּאֳנִיָּה
וְהָאֳנִיָּה אָז נוֹסַעַת, וּמִפְרָשֶׂיהָ מְלֵאִים רוּחַ חֲזָקָה.
פִּתְאֹם עָמְדָה הָאֳנִיָּה מִלֶּכֶת, הַתֹּרֶן הָיָה לְגֶפֶן עָבָה, פּוֹרֶצֶת אַשְׁכֹּלוֹת-אַשְׁכְּלוֹת עֲנָבִים, זַלְזַלֵּי-גֶפֶן הִשְׂתָּרְגוּ עַל הַחֲבָלִים, וְסִפּוּן-הָאֳנִיָּה הָלַךְ נַחֲלֵי-יָיִן.

------------------
– בְּנִי, בְּנִי! – אָמְרָה אִמָּא-יָגוּאָר כַּמָּה וְכַמָּה פְּעָמִים, כְּשֶׁהִיא מְנַעֲנַעַת זְנָבָהּ בְּחֵן – קִפּוֹד סוֹמֵר הוּא קִפּוֹד סוֹמֵר וְאֵינוֹ יָכוֹל לִהְיוֹת שׁוּם דָּבָר חוּץ מִקִּפּוֹד סוֹמֵר. וְצָב לְעוֹלָם הוּא צָב וְאֵינוֹ יָכוֹל לִהְיוֹת שׁוּם יְצוּר זוּלַת צָב. 
– אֲבָל הַלָּלוּ לֹא קִפּוֹד וְלֹא צָב הֵם, אֶלָּא כָּל אֶחָד יֵשׁ בּוֹ מְעַט מִסְּגֻלּוֹת שְׁנֵי יְצוּרִים אֵלֶּה וְלֹא אֵדַע מַה שֵּׁם יִקָּרֵא לָהֶם. 

------------------
הַפַּעַם נִגַּשׁ אֶל הַמְּנַהֵל, וְאָמַר: “קְנֵה לִי אֶת הַכִּנּוֹר הַזֶּה”.
הַמְּנַהֵל הִבִּיט עָלָיו מִשְׁתּוֹמֵם, וְחָפֵץ לִדְחוֹתוֹ.
אֲבָל הוּא רָאָה חֵשֶׁק עַז בְּעֵינֵי הַיֶּלֶד.
הוּא חָשַׁב: “מֵילָא.
לֹא יַזִּיק אִם יִהְיֶה כִּנּוֹר בַּמּוֹסָד.
נְנַסֶּה לְלַמֵּד אֶת יִצְהָל.
מִמַּה נַפְשָׁךְ.
אִם לֹא יִצְלַח הוּא לִנְגִינָה, יִמָּצֵא בַּמּוֹסָד יֶלֶד אַחַר בַּעַל-כִּשְׁרוֹן, וִיפָאֵר לָנוּ נְשָׁפִים וַחֲגִיגוֹת”.

------------------
אָמַר לָהּ הַקּוֹסֵם: “בִּתִּי, מָה אַתְּ מַעלָה אָבָק, וְסוֹתֶמֶת אֶת צִנּוֹרוֹת-הַזִּמְרָה? וְלֹא זִמְרָתֵךְ בִּלְבָד אתְּ מְקַלְקֶלֶת אֶלָּא שֶׁלָּךְ וְשֶׁל כָּל חַבְרוֹתָיִךְ”.
אָמְרָה לוֹ הַצִּפּוֹר: “לְכָל הַצִּפֳּרִים יֵשׁ בֶּן-זוּג. אֲנִי בּוֹדֵדָה נוֹלַדְתִּי, וּבוֹדֵדָה אֲנִי חַיָּה, ולֹא מָצָאתִי בֶּן-מִינִי בְּכָל הַגָּן הַזֶּה. עַל-כֵּן אֲנִי כוֹעֶסֶת מְאֹד, וַאֲנִי מַעלָה אָבָק, וְסוֹתֶמֶת הַצִּנּוֹרוֹת, וְלֹא אִיכְפַּת לִי אִם כָּל הַגָּן הַזֶּה וְהַהֵיכָל הַזֶּה יֵהָפְכוּ לְאַיִן גָּמוּר”.

------------------
וְכָל הָאֵלִים וְהָאֵלוֹת נָתְנוּ לָהּ מַתָּנוֹת; אַפְרוֹדִיטֵי הֶעֱנִיקָה לְגוּפָהּ קֶסֶם-חֶמְדָּה; פּוֹסֵידוֹן תָּלָה לְצַוָּארָה מַחֲרֹזֶת מִן הַנִּפְלָאוֹת בִּפְנִינֵי-יַמָּיו; הֶרָה נָתְנָה בְּרֹאשָׁהּ כְּלִיל-מַלְכוּת; אַתֵּינֵי הֵאִירָה עֵינֶיהָ בְּטַל-טֹהַר; נִימְפוֹת בְּנוֹת-אֵלִים אָרְגוּ לָהּ כָּתְנוֹת-מֵשִׁי, דַּקּוֹת כְּקֶצֶף לָבָן בְּצֵלַע גַּל יְרַקְרַק; וְהַמּוּסוֹת בָּאוּ מִמְּעוֹנָן בְּהַר-הֶלִּיקוֹן וּבֵרְכוּהָ בְּזֹךְ-זִמְרָה וּמֹתֶק-מַנְגִּינָה וּמְשׁוּבַת-מָחוֹל.

------------------
הֶלְיוֹס עִם מִרְכֶּבֶת-הַשֶּׁמֶשׁ שֶׁלּוֹ יָרַד מֵאֲחוֹרֵי הַר בִּפְיֶרִיָּה.
וּמִצַּד הָהָר מִזֶּה, בַּצֵּל הַפָּרוּשׂ, רָעוּ הַבָּקָר אֲשֶׁר לְאַפּוֹלוֹן.
קָטַע הֶרְמֶס בֶּן-מַיָּה, זֶה הַגַּנָּב הַקָּטָן, חֲמִשִּׁים פָּרוֹת מִן הָעֵדֶר, וְנִהֵל אוֹתָן אֲחוֹרַנִּית, פַּרְסוֹתֵיהֶן הַקְּדוֹמַנִּיּוֹת מֵאָחוֹר וּפַרְסוֹתֵיהֶן הָאֲחוֹרִיּוֹת לְפָנִים.
הִגִּיעוּ אֶל אַדְמַת-חוֹל.
קָטַף תִּינוֹק-הָאֵלִים עַנְפֵי אֵשֶׁל וָמוֹר, קְלָעָם כְּמוֹ-שֶׁהֵם, וּקְשָׁרָם לְרָגְלָיו.

------------------
וְרֵיעָה נִשְּׂאָה לְרֹאשׁ-הַטִּיטָנִים, כְּרוֹנוֹס, וַתֵּלֶד לוֹ שָׁלֹשׁ בָּנוֹת וּשְׁלֹשָה בָנִים, הֵמָּה: הֶסְטִיָּה, הַמְּמֻנָּה עַל אֵשׁ-הָאָח אֲשֶׁר בַּבַּיִת, דֵּימֶטֶר, אֵם-הַדָּגָן, וְהֶרָה, אֲשֶׁר בְּיָדֶיהָ מַפְתְּחוֹת עֲמַל-נָשִׁים וְעִצְּבוֹנָן, וְהָדֶס, פּוֹסֵידוֹן וְזֶאוּס, אֵלִים שַׂגִּיאים.
כְּרוֹנוֹס בּוֹלֵע אֶת יְלָדָיו 
אַךְ בְּטֶרֶם נִכְנְסוּ בְּנֵי כּרוֹנוֹס אֶל מִמְשָׁל וּמִשְׂרָה, הָיָה מִשְׁכָּנָם חֹשֶךְ, וְהֻצְרְכוּ לַעֲשׂוֹת מִלְחָמָה אֶת דּוֹר-אֲבִיהֶם.

------------------
עַל כֵּן, בִּהְיוֹתָן מְרִיבוֹת יוֹתֵר מִדַּי, הָיָה מִתְהַלֵּךְ לְבַדּוֹ בְּחֶלְקָה אַחַת שֶׁבְּגִנּוֹת-הָאַרְמוֹן, וְהָיָה אוֹמֵר: – אָרוּר יוֹם אִוָּלֶד בּוֹ.
יוֹם אֶחָד, לְאַחַר שֶׁרָבוּ נָשָׁיו שְׁלֹשָה שָׁבוּעוֹת – כָּל תְּשַׁע מֵאוֹת וְתִשְׁעִים וָתֵשָׁע הַנָּשִׁים יַחְדָּו – יָצָא שְׁלֹמֹה לְבַקֵּשׁ שָׁלוֹם וּמְנוּחָה, כְּמִשְׁפָּטוֹ; וּבֵין הַזֶּהָבִיּוֹת פָּגַשׁ בְּבַלְקִיס, הַיָּפָה בַּנָּשִׁים, וְהִיא כֻלָּהּ עֲצוּבָה עַל שֶׁמַּלְכָּהּ כָּל-כָּךְ מָלֵא דְאָגָה. 
------------------
רָאֲתָה הַצְּעִירָה רַסְמִיָּה כָּךְ, וְנִצְטָרְפָה גַם הִיא עִמָּהֶן. רָאוּ הַיְּלָדוֹת הָעַרְבִיּוֹת כָּךְ, וְנוֹסְפוֹ גַם הֵן אֶל הַשּׁוּרָה הַמְּרַקֶּדֶת; הַלָּלוּ, כָּל-כַּמָּה שֶׁקְּטַנּוֹת הֵן, הֶרְאוּ כֹּחַ כִּמְחוֹלְלוֹת-בְּחֶסֶד. אַחֲרֵי-כֵן נִתְפַּתַּח מַעְגַּל-רִקּוּד כְּלָלִי, אֲשֶׁר גְּדוֹלִים וּקְטַנִים, יְהוּדִים וְעַרְבִים, מְעֹרָבִים בּוֹ. 
------------------
וְכֵן לְהֵפֶךְ: אִלּוּ אָמְרוּ עַנְנֵי-מַעְלָה: “טוֹב לָנוּ פֹּה בַּשְּׁחָקִים; נְרַחֵף לָנוּ בַּנְּעִימִים וְנָצוּר מֵעַצְמֵנוּ צוּרוֹת-נוֹי; וּמַה-לָּנוּ הֶעָפָר הֶעָכוּר מִתָּחַת?” – כִּי-אָז מֵעוֹלָם לֹא הָיוּ הַמַּיִם הָעֶלְיוֹנִים נִכְנָסִים לְתַפְקִידִים פְּרוֹלִיטָרִיִּים שֶׁל הַפְרָחַת הָרִים וּבְקָעוֹת. 
------------------
– הוֹי, בֶּן-אָדָם, הַאִם כָּל הַחַיּוֹת מְצַיְּתוֹת לָךָ?
– כֵּן – אָמַר הָאִישׁ.
– הַאִם כַּדּוּר הָאָרֶץ מְצַיֵּת לְךָ?
– כֵּן – אָמַר הָאִישׁ.
– הַאִם הַיָּם מְצַיֵּת לְךָ?

------------------
– אִם אַתָּה אוֹמֵר מִלָּה נוֹסֶפֶת עַל מַה שֶּׁאָמָרה אִמָּא שֶׁלִּי – הֵשִׁיב הַיָּגוּאָר. אֲבָל טֶרֶם הִסְפִּיק לִגְמֹר אֶת פְּסוּקוֹ, קָפַץ הַצָּב הָאִטִּי-הַמּוּצָק אֶל תּוֹךְ נְהַר –הָאֲמַזּוֹן הַמְּרֻפָּש, שָׂחָה לְאֹרֶך-דֶּרֶךְ מִתַּחַת לַמַּיִם, וְיָצָא אֶל הַחוֹף אֲשֶׁר שָׁם חִכָּה לוֹ הַקִּפּוֹד הַסּוֹמֵר-דּוֹקֵר. 
– הָיָה זֶה מִפְלָט בְּעוֹר-שִׁנַּיִם – אָמַר הַקִּפּוֹד הַסּוֹמֵר-דּוֹקֵר – הַיָּגוּאָר רַב-הַגְּוָנִים אֵינוֹ חָבִיב עָלַי כָּל-עִקָּר. מָה הִגַּדְתָּ לוֹ עַל זֶהוּתְךָ? 

------------------
רַק הָאֶרִינְיוֹת הַנּוֹרָאוֹת נִשְׁאֲרוּ מְהַלְּכוֹת עַל פְּנֵי הָאָרֶץ לַעֲשׂוֹת שְׁפָטִים בְּשׁוֹפְכֵי דְמֵי-נְקִיִּים.
חִתּוּנֵי הַטִּיטָנִים אֶת הַטִּיטָנִיּוֹת וְהַיְּלָדִים אֲשֶׁר יֻלְּדוּ לָהֶם 
וַיִּתְחַתְּנוּ שֵׁשֶׁת הַטִּיטָנִים אֶת שֵׁשׁ אַחְיוֹתֵיהֶן הַטִּיטָנִיּוֹת.
וּמְסַפּרִים הַמְּשׁוֹרְרִים כִּי הַטִּיטָנִיּוֹת הֵן אֲשֶׁר בָּחֲרוּ בְּבַעְלֵיהֶן, אָח אֶחָד לְכָל אָחוֹת.

------------------
וַיַּבֵּט כֹּה וָכֹה, וּבִרְאוֹתוֹ כִּי אֵין נֶפֶשׁ חַיָּה הִקְרִיב אֶת פִּיו אֶל חֲלַל-הַגֻּמָּץ, וַיִּלְחַשׁ אֶל-תּוֹכוֹ כִּלְחֹשׁ אֶל אֹזֶן; "לַמֶּלֶךְ מִידַס אָזְנֵי-חֲמוֹר!" וְחִישׁ סָתַם אֶת חֲלַל-הַגֻּמָּץ בַּבֹּץ.
וַיֵּלֶךְ הַבַּיְתָה קַל-לֵבָב כִּי רָוַח לוֹ מִן הַסּוֹד הַמֵּצִיק לוֹ, אַחֲרֵי אֲשֶׁר הוֹצִיאָהוּ הַחוּצָה.
לְיָמִים צָמְחָה חֻרְשָׁה שֶׁל קְנֵי-סוּף עַל שְׂפַת הַנָּהָר.

------------------
אָז שָׂם אֶת כַּפָּתוֹ הָרַכָּה אֶל תוֹךְ פִּיו, וְהָעֳקָצִים הִכְאִיבוּ לוֹ פִּי כַּמָּה וְכַמָּה. וְכֵיוָן שֶׁחָזַר אֵלָיו כֹּחַ-הַדִּבּוּר, אָמַר: 
– עַכְשָׁו יוֹדֵעַ אֲנִי שֶׁהוּא כְּלָל-וּכְלָל אֵינֶנּוּ צָב. אֲבָל – הוֹסִיף בְּגָרְדוֹ אֶת רֹאשׁוֹ בַּכַּפָּה הַנְּקִיָּה מֵעֳקָצִים: מִנַּיִן לִי שֶׁהַשֵּׁנִי הוּא צָב? 
– אֲבָל אֲנִי בֶּאֱמֶת צָב – אָמַר אִטִּי-מוּצָק. – אִמָּא שֶׁלְּךָ בְּהֶחְלֵט צָדְקָה. הִיא אָמָרה לְךָ כִּי עָלֶיךָ לִרְדּוֹת אוֹתִי בְּכַפָּתְךָ. הַתִחֵל! 

------------------
– אִם כֵּן, – אָמַר הַסּוּס, – יֶשְׁנוֹ יְצוּר בְּמִדְבָּרְךָ (וְהוּא עַצְמוֹ בַּעַל-בֶּכִי), וְצַוָּארוֹ אָרֹךְ וְרַגְלָיו אֲרֻכּוֹת, וְהוּא לֹא נָקַף פַּרְסָה לִמְלָאכָה מֵאָז הַיּוֹם הָרִאשׁוֹן בַּבֹּקֶר. זוּז לֹא יָזוּז.
– פְיוּ, – אָמַר הַגִּ’ין, כְּשֶׁהוּא שׁוֹרֵק בִּשְׂפָתָיו. – הֲלֹא זֶה הוּא גְמַלִּי, כֹּה יִתֵּן וְכֹה יוֹסִיף לִי אֱלֹהִים! וּמַה הוּא אוֹמֵר לְכָךְ?
– הוּא אוֹמֵר דֻּבְּשְׁ, – אָמַר הַכֶּלֶב, – וְלֹא יַעֲשֶׂה כָל מְלָאכָה.
– הַאִם הוּא אוֹמֵר עוֹד מַשֶּׁהוּ?

------------------
מֵאַלְפֵי רִבְבוֹת הַשְּׂרָפִים וְהַשִּׁנְאַנִּים וְהַמַּלְאָכִים חֲצוּבֵי-הַלַּהַב, וְהַמַּלְאָכִים בְּלוּלֵי הַקֶּרַח וְהָאֵשׁ, וְזַמְרָנֵי-שַׁחֲרִית וְזַמְרָנֵי-עַרְבִית, לֹא הָיָה אַף אֶחָד וְיַבִּיט עַל הַמְּשׁוֹרֵר בְּרֹךְ. 
------------------
 לְמַיָּה כִּי הִיא וְאַחְיוֹתֶיהָ יַחַד תִּהְיֶינָה כִּימַת-כּוֹכָבִים בָּרָקִיעַ, לְהִתְנוֹצֵץ עֲדֵי-עַד.
אוֹיָה! – שָׁר הֶרְמֶס בְּרָגֶשׁ – רַק כּוֹכָבִים וְשׁוֹכְנֵי-אוֹלִימְפּוֹס יִחְיוּ לָעַד, וְדָרֵי-מַטָּה, אִם אָדָם וְאִם בְּהֵמָה, יַזְקִינוּ וְיִבֹּלוּ, חוּץ מִן הַנָּחָשׁ הַמַּחֲלִיף עוֹרוֹ וּמְחַדֵּשׁ נְעוּרָיו.
כִּי זֶאוּס, בְּרֶגַע שֶׁל רַחֲמִים עַל דָּרֵי-מַטָּה שָׁלַח לָהֶם חֲמוֹר, עָמוּס נְעוּרִים מְלוֹא-הַשַּׂק.
וַיָּבוֹא הַחֲמוֹר עַד מַעְיָן אֶחָד, וַיֵּט אֵלָיו לִשְׁתּוֹת, כִּי צָמֵא.

------------------
וַיְדַבֵּר הַמְּשׁוֹרֵר אֶל אֵל עוֹשֵׂהוּ, לֵאמֹר: “אֱלֹהַי, אַתָּה אָצַלְתָּ לִי מִתִּפְאַרְתְּךָ, וַתִּתֶּן-לִי אֶת אַדֶּרֶת-הַשִּׁיר וְאֶת כְּלִילוֹ. לְךָ הָאַדֶּרֶת וְהַכְּלִיל, וְלֹא לִי. רְאֵה חַיַּי בְּדַלּוּתָם! הַחַיִּים הָאֵלֶּה אֲנִי עֲשִׂיתִים, וְאַתָּה עָשִׂיתָ הַשִּׁיר. מַעֲשֵׂי-יָדַי אֵינָם אֱלֹהִיִּים!”

------------------
וְעַתָּה הָיָה הָאִישׁ הַזָּר בֶּאֱמֶת נִרְגָּז.
הֵן מִלְּאוּ אֶת שַׂעֲרוֹתָיו טִיט.
גִּלְגְּלוּ אוֹתוֹ עַל אַבְנֵי-חָצָץ חַדּוֹת.
הֵן יָשְׁבוּ עָלָיו בְּטוּר אָרֹךְ, שֶׁבֶת שֵׁשׁ נָשִׁים יַחַד.
הֵן הִרְבִּיצוּ בּוֹ עַד שֶׁכִּמְעַט לֹא יָכוֹל לִנְשׁוֹם עוֹד, וְאַף-עַל-פִּי שֶׁלֹּא הֵבִין אֶת שְׂפָתָם, יָדַע כִּי מַרְבִּיצִים בּוֹ גַם קְלָלוֹת וַחֲרָפוֹת נוֹרָאוֹת.

------------------
נִתְיָאַשְׁתִּי, וְהָלַכְתִּי אֶל הַמָּקוֹם הַשֵּׁנִי, אַף הוּא קוֹמוֹת אֲחָדוֹת לְמַעְלָה; וְגַם שָׁם הֲמוֹנֵי נְעָרִים, מִילָדִים קְטַנִּים קִצְרֵי-מִכְנָסַיִם וְעַד בַּחוּרִים אַרְכָנִים שֶׁשְּׂפָמָם מְבַצְבֵּץ.
אָז יָדַעְתִּי כִּי לֹא עַל דֶּרֶךְ זוֹ תִּמָּצֵא לִי עֲבוֹדָה. יוֹתֵר טוֹב לַעֲנוֹת לְשֶׁכָּאֵלֶּה הַדּוֹרְשִׁים מִכְתָּבִים כְּתוּבִים. וְזוֹ צָרָה: כָּל הַמּוֹדָעוֹת הַדּוֹרְשׁוֹת מִכְתָּבִים אוֹמְרוֹת: “סַפֵּר מַה-נִּסְיוֹנוֹתֶיךָ”. 
------------------
אַחֲרֵי לֶכְתָּן, תַּאַסְפִי אֶת שַׂעֲרוֹת-הַצֶּמֶר הָאֵלֶּה, וְהָיוּ בְיָדֵךְ לִפְקָעַת".

וַתַּעַשׂ פְּסִיכֵי כַּעֲצַת הָקָּנֶה.
וּכְבוֹא אֵלֶיהָ אַפְרוֹדִיטֵי, מָצְאָה כִּי פְקֻדָּתָהּ נִמְלְאָה עַל יְדֵי הַצְּעִירָה, וְהִיא לֹא נִזּוֹקָה.
וְלֹא יָדְעָה הָאֵלָה אֵיךְ הָיָה הַדָּבָר.
וַתֹּאמַר: "אָכֵן כְּשָׁפִים עִמָּךְ, וְנִצַּלְתְּ הַפָּעַם.

------------------
וּמֵאֶרֶץ-חוּץ אֵין קוֹנִים עַד שֶׁתִּתְחַיֵּב אוֹתָהּ אֶרֶץ לִקְנוֹת בָּהּ בַּמִּדָּה מִמְּלֶאכֶת הַיִּשׁוּב הָעִבְרִי, אִם מַעֲשֵׂי-שָׂדֶה וְאִם מַעֲשֵׂי-חֲרשֶׁת.
לְמִשְׁקֵי-הַחַוּוֹת הָעִבְרִיִּים, יֵשׁ מְשַׁלֵּם הָאוֹצָר בְּכֶסֶף, וְיֵשׁ בְּרָהִיטִים, מַכְשִׁירִים, כְּלֵי-בַּיִת וּמַלְבּוּשִׁים, מֵהַנַּעֲשִׂים בְּבָתֵּי-הַחֲרשֶׁת אֲשֶׁר בִּגְבוּלוֹת הָעִיר.
"בָּתֵּי-הַחֲרשֶׁת נִבְדָּלִים מֵעַל בָּתֵּי-הַדִּירָה בְּרֹבַע מְיֻחָד.

------------------
– אָמַר טֶגּוּמֵי, וְהוֹסִיף לִמְשׁוֹךְ וְלִמְתּוֹחַ אֶת גִּידֵי-הַצְּבִי, כְּשֶׁפִּיו מָלֵא חוּטִים וּרְצוּעוֹת.
וְהַזָּר – טֶוָּרִי בֶּן טֶוָּרִים הָיָה – יָשַׁב לוֹ עַל הָעֵשֶׂב, וְטָפִי הֶרְאֲתָה לוֹ אֶת מַעֲשֵׂה אָבִיהָ.
הַזָּר חָשַׁב: “אָכֵן, יָלְדָּה נִפְלָאָה הִיא זוֹ.
הִיא רוֹקַעַת כְּנֶגְדִי בְּרַגְלָהּ, וּמְעַוָּה אֶת פָּנֶיהָ.

------------------
וְלֹא עוֹד: שׁוּמָה עָלֵינוּ לְהַעֲלוֹת אֶת דַּרְגַּת הַנְּתִינָה שֶׁל הָאָדָם – מֵעֲבוֹדַת מֻרְדְּפֵי-רָעָב, מְפָרֶכֶת הַגּוּף וּמְיַגַּעַת הַנֶּפֶשׁ, אֶל חֶדְוַת בְּנֵי-חוֹרִין בִּנְטִיעָה וּבִנְיָן וְעִצּוּב וְהַמְצָאָה, אֶחָד הַמְּצָאַת שִׁיר-מִזְמוֹר וְאֶחָד הַמְּצָאַת כְּלֵי-כָנָף לִמְעוּף בַּשַּׁחַק אוֹ עֲדֶשֶׁת-זְגוּגִית לְקָרֵב אֵלֵינוּ אֶת הַכּוֹכָבִים. וְגַם הַעֲלָאָה זוֹ טְעוּנָה הַרְבֵּה נְתִינָה וּמְסִירוּת-נֶפֶשׁ. 
------------------
זֶאוּס הָיָה לְאַיִל, אַפּוֹלוֹן לְעוֹרֵב, דִּיאוֹנִיסוֹס לְתַיִשׁ, הֶרָה לְפָרָה לְבָנָה, אַרְטֶמִיס לְחֲתוּלָה, אַפְרוֹדִיטֵי לְדָג, אָרֶס לַחֲזִיר-בָּר, הֶרְמֶס לְרָחָם, וְכֵן כֻּלָּם.
רַק אַתֵּינֵי עָמְדָה עַל מְקוֹמָהּ מִבְּלִי מֹרֶךְ וַתִּקְרָא אֶל זֶאוּס: "הַחֶרְפָּה לְךָ, שַׁלִּיט-הָאֵלִים, שֶׁאַתָּה הוֹפֵךְ עֹרֶף וְנָס! הוֹפַע בְּעֹז-דְּמוּתְךָ וְהִלָּחֵם!"

זִרְקִי עֲלֵיהֶם טִפּוֹת מִן הַמַּיִם אֲשֶׁר בַּדְּלַעַת, וְיֵרָפְאוּ.
אַךְ קֹדֶם תְּרַפְּאִי אֶת הַגָּדֵר, אֲשֶׁר גַּם הִיא חָלָתָה".
וַתִּזְרֹק הַבַּת מִן הַמַּיִם עַל הַגָּדֵר, וַתְּהִי גֶדֶר קוֹצִים פְּשׁוּטִים כְּמוֹ לִפְנֵי הֱיוֹת בָּהּ מַגַּע-מִידַס לְשַׁנּוֹתָהּ.
אַחַר הָלְכוּ מִוֶּרֶד אֶל וֶרֶד, מִן הַפְּרָחִים אֲשֶׁר קָרְשׁוּ וְהָיוּ לְזָהָב בִּהְיוֹת בָּהֶם יַד-מִידַס; וַתִּזְרֹק בָּהֶם הַיַּלְדָּה מִן הַמַּיִם אֲשֶׁר בַּדְּלַעַת, וַיִּחְיוּ הַפְּרָחִים וַיַּרְטִיבוּ בְּטָל.

אַף הָיוּ בַּחֲדָרִים תֵּבוֹת-פְּנִינִים, מְלֵאוֹת כָּל עֲדִי וְתַכְשִׁיט אֲשֶׁר יִשַׂמַּח לְבַב אִשָּׁה, וַאֲרוֹנוֹת שִׁטָּה וְהָבְנֶה, צוֹפְנֵי מַלְבּוּשֵׁי-מֶשִׁי וְרִקְמוֹת זָהָב וּתְכֵלֶת.

וְאֵין שׁוֹעֵר וְאֵין שׁוֹמֵר וְאֵין נוֹטֵר לְאַרְמוֹן-הַפְּלָאִים וְלָרְכוּשׁ הָרַב אֲשֶׁר בְּתוֹכוֹ.

פַּעַם בְּעֵת חֹרֶף נִפְגַּשׁ עִם אִישׁ נוֹדֵד אֲשֶׁר בִּגְדוֹ קָרוּעַ וְרַגְלָיו יְחֵפוֹת. פָּשַׁט יוֹנָתָן מְעִילוֹ וּנְעָלָיו וְנָתַן אוֹתָם לָאִישׁ. מִנִּי אָז הָיָה הוּא עַצְמוֹ לוֹבֵש שַׂק-קָפֶה, עִם שְׁנֵי חוֹרִים בִּשְׁבִיל הַזְּרוֹעוֹת. וּלְרַגְלָיו הָיָה נוֹעֵל סַנְדָּלִים, שֶׁהִתְקִין אוֹתָם מִקְּלִפַּת-עֵץ. יֵשׁ שֶׁהָיָה מְהַלֵּךְ יָחֵף בַּשֶּׁלֶג וְלֹא חָשַׁשׁ. בְּתוֹרַת כּוֹבַע הָיָה חוֹבֵשׁ לְרֹאשׁוֹ סִיר-בַּרְזֶל קָטָן, בּוֹ הָיָה מְבַשֵּׁל וְקוֹלֶה אֲרוּחוֹתָיו. 
------------------
הֵנָּה בֵּרְכוּנִי, וַתְּעַטֵּרְנָה רֹאשִׁי שַׂעֲרוֹת-גְּבוּרָה וְנָתוֹן אוֹתִי גְּבִיר-עָם, לְהַבְרִיחַ גְּדוּדֵי-אֹפֶל, וּלְהַפְרוֹת אֶרֶץ וְהַעֲשֵׁר יוֹשְׁבֶיהָ – וְהֵנָּה לִהֲטוּ תְשׁוּקָתִי לִשְזוֹר בְּמַיִם כּוֹזְבִים זָהֳרֵי-קְוֻצּוֹתַי וְלִשְׁפּוֹךְ אוֹנִי עַל סְלָעִים לֹא-יָצִיצוּ. 
------------------
רְצוֹנְךָ – אֶעֱשֶׂה קֶסֶם וְאֶתֵּן לְךָ אֶת הַמִּסְפָּרַיִם הָאֵלֶּה, וְהֵם חַדִּים מְאֹד וַחֲזָקִים מְאֹד, וְתוּכְלוּ אַתָּה וְצֶאֱצָאֶיךָ לֶאֱכֹל אֱגוֹזֵי-קוֹקוּס כָּאֵלֶּה בְּמֶשֶׁךְ כָּל הַיּוֹם כֻּלּוֹ כְּשֶׁאַתֶּם עוֹלִים מִן הַיָּם אֶל הַיַּבָּשָׁה; אוֹ שֶׁתּוּכַל לַחְפֹּר פּוּסַט-טַאסֶק לְעַצְמְךָ בַּמִּסְפָּרַיִם שֶׁבִּרְשׁוּתְךָ, כְּשֶׁאֵין אֶבֶן אוֹ חוֹר מִקָּרוֹב; וְאִם הַקַּרְקַע קָשָׁה מִדַּי, תּוּכַל בְּעֶזְרַת הַמִּסְפָּרִים הַלָּלוּ לָרוּץ וּלְטַפֵּס עַל עֵץ.

------------------
וְהַמֶּלֶךְ מִידַס בָּא הַבַּיְתָה כְּמִתְגַּנֵּב, נִכְנַס אֶל חֶדֶר לִפְנִים מֵחֶדֶר, וַיַּחֲבֹשׁ אֶת רֹאשׁוֹ בַּעֲטִיפָה, הַדֵּק אֶת אָזְנָיו הֵיטֵב אֶל קָדְקֳדוֹ לְבַל תֵּרָאֶינָה; וְאֶת הָעֲטִיפָה לֹא הֵסִיר לְעֵין-רוֹאִים כָּל הַיָּמִים.
וְאֶל שָׂרָיו וַעֲבָדָיו אָמַר: "רֹאשִׁי חֲבוּר-מַכְאוֹבִים, עַל כֵּן חֲבַשְׁתִּיו".
וְלֹא יָדַע אִישׁ אֶת סוֹדוֹ, חוּץ מִן הַגַּלָּב, הַמֵּיטִיב אֶת זְקַן הַמֶּלֶךְ וּשְׂפָמוֹ, וּמְסַפֵּר אֶת שַׂעֲרוֹת-רֹאשׁוֹ.

------------------
עָבְרוּ שָׁנִים רַבּוֹת, אוּלַי שִׁבְעִים שָׁנָה וְאוּלַי מֵאָה, וְעַל חוֹף-יַמָּהּ שֶׁל אֶרֶץ-יִשְׂרָאֵל הָיָה יָרִיד גָּדוֹל.
הָיוּ מֻצָּגִים בַּיָּרִיד אֲרִיגִים, כֵּלִים וְצַעֲצוּעִים מִכָּל הָעוֹלָם.
בָּאוּ לְשָׁם הֲמוֹנֵי-אֲנָשִׁים.
בֵּין הַהֲמוֹנִים, בָּא גַם מְנַהֵל בֵּית-יְתוֹמִים אֶחָד עִם כָּל הַמּוֹרִים וְהַיְּלָדִים.

------------------
 וַחֲכַם-תַּחְבּוּלוֹת עַד אֵין חֵקֶר.
שָׂחָה וְשָׂחָה הַלִּוְיָתָן בְּכָל כֹּחוֹ וּבְכָל מְאֹדוֹ עַד שֶׁהִגִּיעַ לְקַו-הָרוֹחַב חֲמִשִּׁים צָפוֹנָה, וּלְקַו-הָאוֹרֶךְ אַרְבָּעִים מַעֲרָבָה. 
------------------
אֲבָל, בְּכָל הָעָם הָרַב אֲשֶׁר עָבְרוּ דֶרֶךְ מֵצַר-הַיַּבָּשָׁה, בְּרִבְבֵי-הָרְבָבוֹת אֲשֶׁר הִתְכַּנְּסוּ הֲלוֹם, אֶל שְׂדֵה-קוֹרִינְתְּ, לְהַאֲזִין הַשִּׁירִים, לַחֲזוֹת בַּהֲצָּגוֹת וּבַמִּשְׂחָקִים, לִכְבוֹד אֵל-הַיָּם, – אֵיכָה תִּמָּצֵא הַיָּד הַפּוֹשָׁעַת? לֵב-מִי יְנַחֵשׁ אִם בִּידֵי אַנְשֵׁי-שֹׁד נַעֲשָׂה הַמַּעֲשֶׂה וְאִם בִּידֵי מְשַׂנְאֵי-סֵתֶר? רַק הָרוֹאֶה כֹּל, הַשֶּׁמֶשׁ, הַסּוֹד כָּמוּס עִמּוֹ. יִתָּכֵן כִּי הָאִישׁ מִסְתּוֹבֵב בְּתוֹכְכֶם, בְּנֵי-יָוָן הַנֶּאֱסָפִים פֹּה! 
------------------
הוּא אַךְ נִגַּשׁ לֶאֱכֹל אוֹתָה וְהִנֵּה מִפְּנִים-הָאִי שֶׁהָיָה בִּלְתִּי-נוֹשָׁב בְּהֶחְלֵט, יָצָא קַרְנָף אֶחָד, וְלוֹ קֶרֶן אַחַת עַל חָטְמוֹ, שְׁתֵּי עֵינֵי-חֲזִירוֹן וְנִמּוּסִים מְעַטִּים. בַּיָּמִים הָהֵם הָיָה עוֹר הַקַּרְנָף מְהֻדָּק לִבְשָׂרוֹ יָפֶה יָפֶה; לֹא הָיוּ בוֹ שׁוּם קְמָטִים בְּשׁוּם מָקוֹם. וְעִם כָּל זֶה הָיָה חֲסַר-נִמּוּס, וְהֹוֶה חֲסַר-נִמּוּס וְיִהְיֶה חֲסַר-נִמּוּס לְעוֹלָם. הוּא הוֹצִיא מִפִּיו הֲבָרָה אַחַת: “מָה!” 
------------------
גָּמָל בּוֹדֵד אִם יֵרָאֶה חֲטוּב-דַּר נֶגֶד הָרָקִיעַ, וּבְקִרְבָתוֹ בֶּדְוִי שַׁח לְקוֹשֵׁשׁ קוֹצִים – הֲרֵי זֶה חִדּוּשׁ.
גַּם צִפּוֹר לֹא תֵרָאֶה בַּיְּשִׁימוֹן הַזֶּה.
אָכֵן, רָאִיתִי זוּג עוֹפוֹת-טֶרֶף – כְּנָפַיִם לְבָנוֹת, וְעַל קְצוֹתָן פַּסֵּי-שְׁחוֹר – נוֹחֲתִים בִּדְאִיָּה אִטִּית.
אִם אֵין נִחוּשִׁי מַטְעֵנִי, הוּא-הוּא הָרָחָם הַנִּזְכָּר בֵּין הָעוֹפוֹת הַלֹּא-טְהוֹרִים בַּחֻמָּשׁ “וַיִּקְרָא”.

------------------
עוֹמֵד הָאָדָם עֵירֹם וְעֶרְיָה מִקִּנְיְנֵי-שָׁוְא וְעִטּוּפֵי-הֶבֶל, עָנָו בְּחֵיק גִּלּוּיֵי טֶבַע-שַׁדַּי, יַצִּיב בְּפַשְׁטוּתוֹ הָאֱנוֹשִׁית, אַדִּיר בְּשִׁוְיוֹנוֹ עִם כָּל בְּנֵי-הָאָדָם, מִתְנַהֵר בְּנִיצוֹץ-יוֹצְרִים שֶׁבְּכֻלָּם.
אָז הוּא מוּכָן לַחְצוֹב לוּחוֹת-אֱמֶת מִנַּפְשׁוֹ.

------------------
סִיַּמְתִּי אֶת בֵּית הַסֵּפֶר הַתִּיכוֹנִי, וּמִפְּנֵי דֹחַק הַפַּרְנָסָה בַּבַּיִת, יָצָאתִי מִיָּד לְבַקֵּשׁ עֲבוֹדָה. בְּעִתּוֹן-הַבֹּקֶר רָאִיתִי כִּי בִּרְחוֹב-קַנַּל, מָקוֹם הַיָּדוּעַ בַּחֲנֻיּוֹת-בְּגָדִים יְהוּדִיּוֹת, דּוֹרְשִׁים שְׁנֵי תַכְשִׁיטָאִים – הַיְנוּ, מוֹכְרֵי שְׁעוֹנִים, אֲבָנִים טוֹבוֹת וּכְלֵי כֶסֶף וְזָהָב – אִישׁ-אִישׁ 2 - 3 נְעָרִים עוֹזְרִים. הַמְּקוֹמוֹת הָיוּ סְמוּכִים זֶה לָזֶה. 
------------------
אֲבָל אֲנִי לֹא רָאִיתִי אֶת זֶה אֶלָּא מִתּוֹךְ עִרְפּוּל-מַחֲשָׁבָה וְכֵהוּת-עֵינַיִם, וְנִגַּשְׁתִּי אֵלֶיהָ וְהִתְחַלְתִּי לְפַזְמֵן פִּזְמוֹן יָשָׁן: בְּאַמֶּרִיקָה… וְכָאן…
“בְּאַמֶּרִיקָה”, אָמַרְתִּי, "כְּבָר מַחֲרִימָה דַעַת-הַקָּהָל אֶת בָּתֵּי-הַכְּנֶסֶת הַקְּרוּיִים פִּטְרִיּוֹת, אֵלֶּה הַצָּצִים בְּאוּלַמֵּי רִקּוּד וְקוֹלְנוֹעַ עֶרֶב הַחַג וְנֶעְלָמִים כִּכְלוֹת הֶחָג; וְכָאן יֵשׁ הַתְחָלָה לָהֶם. תֹּאמְרִי, ‘מַה לְּךָ וְלָזֶה? 
------------------
קָפַץ וְהִתְנַפֵּל עַל קוֹל-הָרִשְׁרוּשׁ, וְהִנֵּה רֵיחוֹ רֵיחַ זֵיבְּרָה, וּמִשְׁמוּשוֹ מִשְׁמוּשׁ זֵיבְּרָה, וּכְשֶׁהִפִּילוֹ לָאָרֶץ, בָּעַט בְּרַגְלָיו כְּמוֹ זֵיבְּרָה, אֲבָל בְּשׁוּם פָּנִים לֹא יָכֹל לִרְאוֹתוֹ.
וַיֹּאמֶר לוֹ: שֶׁקֶט, הוֹ נֶפֶשׁ שֶּׁאֵין לָהּ דְמוּת.
יָשֹׁב אֵשֵׁב עַל רֹאשְׁךָ עַד אוֹר הַבֹּקֶר, כִּי יֵשׁ מַשֶּׁהוּ בַּנּוֹגֵעַ לָךְ שֶׁאֵינוֹ מוּבָן לִי כָל צָרְכּוֹ.

------------------
פּוֹסֵידוֹן שָׁאַג, בְּדָפְקוֹ בִּשְׁלֹשֶׁת שִׁנִּי קִלְּשׁוֹנוֹ בְּרִצְפַת הֵיכַל-הַמִּשְׁפָּט: "אָמְנָם, מִתְּחִלָּה נִקְרְאָה הָעִיר עַל שֵׁם אַתֵּינֵי, יְלִידַת מֶצַח זֶאוּס.
אֲבָל שֵׁם נִתָּן לְהִשְׁתַּנּוֹת.
מַה שֶּׁיָּצְרוּ מַעֲרִיצֵי הָאֵלָה בְּדוֹרָם לֹא עִיר הָיְתָה כִּי-אִם עֲיָרָה, וְלֹא עֲיָרָה כִּי-אִם כְּפָר קָטָן וְנִדָּח.

------------------
מֵרוּחוֹת-הָרִים הִדְבִּיק אָבִיב אֶת-חֹרֶף. לְבָנָה קִרְיָתֵנוּ בַּנִּצָּה, וְרֻדָּה בַּשּׁוֹשַׁנִּים. נֵפֶץ נִשְׁמַע בַּדֻּבְדְּבָן, לֹא עָמְדוּ כַּפּוֹתָיו מִלַּחַץ נִצָּנִים כִּי כָּבֵדוּ. הוֹלֵךְ רֵיחַ הַחֲבַצָּלוֹת לְמֵרָחוֹק, מַרְחִיב נְחִירֵי הַגְּמַלִּים. גֶּשֶׁם נְבָטִים צְהַבְהַבִּים נִתָּךְ-נוֹטֵף מִן הַשְּׁזִיפִים, מְרַחֵף בָּרוּחַ, נִצְבָּר שִׁכְבוֹת-זָהָב עַל יַד הַגְּדֵרוֹת. 
------------------
שָׁאֲלָה אַפְרוֹדִיטִי: "בַּמֶּה אֶעֶזְרֵךְ?" וַתֹּאמֶר הֶרָה: "הַשְׁאִילִינִי-נָא אֶת חֲגוֹרָתֵךְ, חֲגוֹרַת הַפְּנִינִים, אֲשׁר כָּל הַלּוֹבַשְׁתָּהּ תְּאֹהַב".
וַתֵּעָתֶר לָהּ אַפְרוֹדִיטֵי.
בְּרִיחַת הֶרָה 
וְלִפְנוֹת עֶרֶב, כְּהַטּוֹת הֶלְיוֹס, אֵל-הַשֶּׁמֶשׁ, אֶת רִכְבּוֹ-אֵשׁ וְסוּסָיו-אֵשׁ מִמַּעֲרָב צָפוֹנָה, רָאָה זֶאוּס אֶת הֶרָה עֲטוּרַת-זִיו, וַיֶּאֱהָבֶנּה.
אַךְ לֹא בְקַלּוּת נַעַנְתָה הֶרָה לְזֶאוּס.

------------------
וַתְּמַן הֶרָה דְרָקוֹנִית, דֶּלְפִינֵי שְׁמָהּ, לְהַכּוֹת בְּלֶטוֹ; וְדֶלְפִינֵי מִמָּתְנֶיהָ וָמַעְלָה אִשָּׁה, מִמָּתְנֶיהָ וָמַטָּה נָחָשׁ, שְׁנֵי גוּפֵי-נְחָשִׁים דַּקִּים וַאֲרֻכִּים לָהּ לִזְרוֹעוֹת, וּלְכָל זְרוֹע-נָחָשׁ לֹא רֹאשׁ אֶחָד כִּי אִם חֲמִשָּׁה רָאשִׁים קְטַנִּים תַּחַת אֶצְבָּעוֹת.
קָמָה אַרְטֶמִיס לְהָגֵן עַל אִמָּהּ; בַּת שְׁלֹשָה יָמִים הָיְתָה אַרְטֶמִיס עֵת שָׁלְחָה חֵץ בְּדֶלְפִינֵי וַתְּמִיתֶהָ.
הַתִּינוֹק אַפּוֹלוֹן מַכְרִיז אֶת אֲשֶׁר הוּא עָתִיד לַעֲשׂוֹת 

------------------
בַּמִּצְטוֹפְפִים עַל אַפְסֵי-הַקָּהָל, מִסְפַּר מִשְׁפָּחוֹת עַרְבִיּוֹת, נִכָּרוֹת בַּעֲטִיפוֹת-לְבוּשֵׁיהֶן. מִתּוֹכָן מִתְבַּלְּטוֹת שְׁתֵּי צְעִירוֹת, יְדִידוֹת אוֹ אֲחָיוֹת, הָאַחַת בְּשִׂמְלָה כְּחֻלָּה מְקֻשֶּׁטֶת בִּפְרָחִים אֲדֻמִּים, הַשְּׁנִיָּה בְּשִׂמְלָה אֲדֻמָּה מְקֻשֶּׁטֶת בִּפְרָחִים כְּחֻלִּים. נִגַּשׁ עַרְבִי אֶחָד, גָּבוֹהַּ וּלְבֶן-זָקָן, וְקָרָא לְנָשִׁים שְׁתַּיִם אֲשֶׁר שָׁם, שֶׁתֵּלַכְנָה אַחֲרָיו הַבָּיְתָה.

------------------
אַף קוֹל אֵבֶל וְשִׁבְרוֹן-לֵב וְגַעְגּוּעִים כּוֹאֲבִים נָתַן לוֹ הַכִּנּוֹר, וּנְהִימוֹת-אשֶׁר מִתְנַשְּׂאוֹת הַשָּׁמַיְמָה כְּכַנְפֵי-זָהָב אֲשֶׁר לְמַלְאָכִים נִסְתָּרִים.
וְזֶה מוּזָר: מוֹרֵה-הַנְּגִינָה נִסָּה לְלַמֵּד בַּכִּנּוֹר יְלָדִים אֲחֵרִים מֵחֲנִיכָיו, אֲשֶׁר חֲשָׁבָם טוֹבֵי אֹזֶן וְאֶצְבַּע, אֲבָל הַכִּנּוֹר הָיָה כֵּהֶה בִּידֵיהֶם, אָטוּם, מְשַׁעֲמֵם, לֹא נוֹתֵן כְּלוּם.
רַק בִּידֵי הַיֶּלֶד יִצְהָל, הָיָה מַפְלִיא נַגֵּן.

------------------
וַיִּכָּרֵךְ דִּיאוֹנִיסוֹס הַצָּעִיר בְּעִקְּבֵי סִילֶנוֹס הַזָּקֵן כִּסְיָח בְּעִקְּבֵי הַסּוּסָה אִמּוֹ.
יַחְדָּו צָהֲלוּ וַיִשְׁתּוֹלָלוּ.
סִילֶנוֹס מְדַבֵּר "תַּכְלִית" אֶל דִּיאוֹנִיסוֹס 
וּפַעַם אָמַר סִילֶנוֹס לְדִיאוֹנִיסוֹס: "יְדִידִי הַצָּעִיר, הִגִּיעַ זְמַן שֶׁתִּדְאַג לְאַחֲרִיתֶךָ.
זֶאוּס הַגָּדוֹל הוּא אָבִיךָ, אֲבלָ אִמְּךָ בַּת-תְּמוּתָה.
אִם לִהְיוֹת אֵל רָצִיתָ, וּלְהִתְעַלּוֹת לִשְׁכּוֹן בְּאוֹלִימְפּוֹס, תְּחִלָּה עָלֶיךָ לִזְכּוֹת לְהַעֲרָצָה וַעֲבוֹדָה מֵאֵת בְּנֵי-הָאָדָם.
