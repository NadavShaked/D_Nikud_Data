"כֵּן, אֲנִי יוֹדֵעַ, "קָטַע אוֹתוֹ לָיוֹנֵל. "טוֹב, אֶקְרָא אֶת כֻּלָּם. אֲנִי אוֹהֵב לִקְרֹא. אֲנִי כָּל כָּךְ שָׂמֵחַ שֶׁלָּמַדְתִּי לִקְרֹא."
"אִם יֻרְשֶׁה לִי לְיַעֵץ לְהוֹד מַלְכוּתוֹ," אָמַר רֹאשׁ הַמֶּמְשָׁלָה, "לֹא כְּדַאי שֶׁתִּקְרָא אֶת הַסְּפָרִים הָאֵלֶּה. סַב-סַב -" "כֵּן?" שָׁאַל לָיוֹנֵל בִּמְהִירוּת.
"הוּא הָיָה מֶלֶךְ טוֹב מְאוֹד – הוֹ, כֵּן! בֶּאֱמֶת מֶלֶךְ נַעֲלֶה בְּדַרְכּוֹ, אַךְ הוּא הָיָה קְצָת, וּבְכֵן, מוּזָר."
"מְשֻׁגָּע?" שָׁאַל לָיוֹנֵל בַּעֲלִיצוּת.

------------------
כֻּלָּם הִגִּיעוּ הַבַּיְתָה אֵיכְשֶׁהוּ, וְכֻלָּם חַיִּים בְּאשֶׁר וּבְעשֶׁר עַד עֶצֶם הַיּוֹם הַזֶּה.
כְּשֶׁהַמֶּלֶךְ שָׁאַל אֶת הַמַּנְטִיקוֹרָה אֵיפֹה הוּא רוֹצֶה לִחְיוֹת, הוּא הִתְחַנֵּן שֶׁיֻּרְשֶׁה לוֹ לַחְזֹר לְתוֹךְ הַסֵּפֶר. "אֲנִי לֹא מְעֻנְיָן בְּחַיִּים צִבּוּרִיִּים," אָמַר.

------------------
 בִּשְׁתַּיִם – אֲבָל, הִיא לֹא עָשְׂתָה לֹא זֶה וְלֹא זֶה, אֶלָּא פָּנְתָה מִיָּד וְסָבְבָה עַל שֶׁשֶׁת עֲקֵבֶיהָ, בְּלֹא הַפְסֵד זְמַן, וְהִתְחִילָה רָצָה לְאָחוֹר – לְהַסְפִּיק וּלְהָבִיא עוֹד גַּרְגֵּר אֶחָד, בְּטֶרֶם חֹשֶׁךְ.

------------------
זֶהוּ. עַכְשָׁו אַתֶּם יוֹדְעִים כָּל מָה שֶׁצָּרִיךְ לָדַעַת כְּדֵי לִקְרֹא אֶת הַסִּפּוּר.

בְּאוֹתָם יָמִים רְחוֹקִים, בַּמָּקוֹם הַזֶּה מַמָּשׁ, חַיָּה אִשָּׁה פְּשׁוּטָה אַחַת שֶׁרָצְתָה רַק דָּבָר אֶחָד – לָלֶדֶת בַּת. אֲבָל רָצָה הַגּוֹרָל, אוֹ אַלְלָה, אוֹ אַף אֶחָד בִּמְיֻחָד, וְלֹא נוֹלְדָה לָהּ בַּת וְלֹא בֵּן וְלֹא כְּלוּם, וְיוֹם יוֹם הָיְתָה יוֹשֶׁבֶת עַל מִפְתַּן בֵּיתָהּ, צוֹפָה בַּשֶּׁמֶשׁ הַשּׁוֹקַעַת וּמִתְפַּלֶּלֶת לְבַת. 
------------------
אֶת אוֹתוֹ הַדָּבָר טוֹמִי עוֹשֶׂה עִם הַלַּחְמָנִיָּה הַשְּׁנִיָּה, הַשְּׁלִישִׁית, הָרְבִיעִית וְהַחֲמִישִׁית, וּלְאוֹת תּוֹדָה הוּא מֵנִיד בְּרֹאשׁוֹ, וְעֵינָיו הַקְּטַנּוֹת מִצְטַמְצְמוֹת עוֹד יוֹתֵר בִּשְׂבִיעוּת רָצוֹן. הַיַּלְדָּה צוֹחֶקֶת בְּשִׂמְחָה.
לְאַחַר שֶׁכָּל הַלַּחְמָנִיּוֹת נֶאֶכְלוּ, נַדְיָה עוֹרֶכֶת הֶכֵּרוּת בֵּין בֻּבּוֹתֶיהָ לַפִּיל:
"תִּרְאֶה, טוֹמִי, הַבֻּבָּה הַמְּגֻנְדֶּרֶת הַזֹּאת הִיא סוֹנְיָה. הִיא יַלְדָּה טוֹבָה מְאוֹד, אֲבָל קְצָת מְפֻנֶּקֶת וְלֹא רוֹצָה לֶאֱכֹל מָרָק. 
------------------
אֲבָל זֶה לֹא קָרָה, וְהַיְּלָדִים הֵחֵלּוּ לַחְשׁשׁ שֶׁהֵם לֹא כָּל כָּךְ חֲכָמִים אַחֲרֵי הַכֹּל…

------------------
הֶחָתוּל הֶעֱמִיד פָּנִים כְּאִלּוּ הוּא עָצוּב וְאָמַר, "אָז אַף פַּעַם לֹא אוּכַל לְהִכָּנֵס לַמְּעָרָה? וְאַף פַּעַם לֹא אוּכַל לָשֶׁבֶת לְיַד הָאֵשׁ הַחֲמִימָה? וְאַף פַּעַם לֹא אוּכַל לִשְׁתּוֹת אֶת הֶחָלָב הַלָּבָן הֶחָמִים? אַתְּ חֲכָמָה מְאוֹד וְיָפָה מְאוֹד. אַל תִּהְיִי רָעָה כָּל כָּךְ, אֲפִלּוּ לֹא לְחָתוּל."

------------------
מוֹלִי הַקְּטַנָּה הַיְּרֻקָּה וְהַזּוֹעֶמֶת צָחֲקָה וְרָצָה, אֲבָל לְמוֹלִי הָיוּ רַגְלַיִם אֲרֻכּוֹת יוֹתֵר, וְהִיא הִשִּׂיגָה אוֹתָהּ. הִיא הִפִּילָה אֶת מוֹלִי הַקְּטַנָּה הַיְּרֻקָּה וְהַזּוֹעֶמֶת וְהִתְיַשְּׁבָה עָלֶיהָ, וְאָז לָקְחָה בַּחֲזָרָה אֶת הַכּוֹבַע הָאָדֹם.
אֲבָל מוֹלִי הַקְּטַנָּה הַיְּרֻקָּה וְהַיּוֹקֶדֶת הִמְשִׁיכָה לִצְחֹק, וְזֶה עוֹרֵר בְּמוֹלִי הָרְגִילָה תְּחוּשָׁה מַפְחִידָה מְאוֹד.

------------------
אֲנִי אֶתֵּן לָכֶם שָׁבוּעַ לְהִתְעַשֵּׁת לִפְנֵי שֶׁאֶבְחַן אֶת הִתְנַהֲגוּתְכֶם. הִיא הַרְבֵּה יוֹתֵר מִדַּי טוֹבָה בִּשְׁבִיל כָּל אֶחָד וְאֶחָד מִכֶּם." וּמִיָּד פָּרְחָה לְדַרְכָּהּ בִּרְכִיבָה עַל שַׁרְבִיטָהּ וְהִשְׁאִירָה אֶת אוֹרְסוּלָה לְהִתְחַבֵּב עַל חֲבֵרֶיהָ הַחֲדָשִׁים כְּמֵיטַב יְכָלְתָּהּ. אֶלָּא שֶׁאוֹרְסוּלָה לֹא יָכְלָה לְהִתְחַבֵּב עֲלֵיהֶם כְּלָל וּכְלָל, כְּפִי שֶׁגִּלְּתָה עַד מְהֵרָה.

------------------
אֵיזוֹ רִיצָה וְחִפָּזוֹן! כָּל הָעֲרֵמָה הָיְתָה חֲפוּזָה בְּחִפָּזוֹן! שִׁבְעָתַיִם מִמַּה שֶׁנֶחְפַּז בַּכְּבִישׁ הַגָּדוֹל וּבַמְּכוֹנִיּוֹת הַמַּבְרִיקוֹת הָהֵן. הֲגַם לַנְמָלִים יוֹם שִׁשִּׁי אַחַר הַצָּהֲרַיִם?
לֹא. אֲבָל הָעֲנָנִים הַוְרֻדִּים הָלְכוּ וְנַעֲשׂוּ סְגֻלִּים, וַאֲחָדִים כְּבָר הָיוּ אֲדֻמִּים לְגַמְרֵי, אַף כִּי אֲחָדִים הִשְׁתַּנּוּ וְנַעֲשׂוּ זְהֻבִּים כְּפִרְחֵי חַרְצִית – וְכֻלָּם עֲשׂוּיִים לְהִתְקַשֵּׁר כָּעֵת וּלְהַמְטִיר תַּחְתֵּיהֶם מִיָּד כְּשֶׁהַשֶּׁמֶשׁ תִּשְׁקַע. 
------------------
הֵן פָּשְׁטוּ עַל מַדְּפֵי הַמְּזָוֶה וּמֵרֹב לְהִיטוּת הִפִּילוּ אֶת כַּד הַשַּׁמֶּנֶת, וְזוֹ נִשְׁפְּכָה וְהִכְתִּימָה אֶת שִׂמְלָתָהּ שֶׁל הַבֻּבָּה הַצָּרְפָתִיָּה.
∗
הַבֻּבָּה הַהוֹדִית מָצְאָה לַחְמָנִיָּה עֲגַלְגַּלָּה, טָבְלָה אוֹתָהּ בַּדְּבַשׁ וְיָשְׁבָה לִסְעֹד אֶת לִבָּהּ.
צִנְצֶנֶת רִבַּת דֻּבְדְּבָנִים הִתְהַפְּכָה עַל צִדָּהּ, וְהַבֻּבּוֹת לִקְּקוּ וְזָלְלוּ מִמֶּנָּה עַד שֶׁפְּנֵיהֶן נַעֲשׂוּ סְגֻלּוֹת-אֲדַמְדַּמּוֹת.

------------------
הוּא נִרְאָה טִפֵּשׁ מְאוֹד, וּכְשֶׁלָּיוֹנֵל דָּחַף אוֹתוֹ קַלּוֹת וְאָמַר לוֹ, "לֵךְ לְךָ לְהִלָּחֵם בַּדְּרָקוֹן," הוּא הִכְנִיס אֶת זְנָבוֹ בֵּין רַגְלָיו וּפָשׁוּט בָּרַח מִשָּׁם. הוּא הָלַךְ וְהִתְחַבֵּא מֵאֲחוֹרֵי בִּנְיַן הָעִירִיָּה, וּבַלַּיְלָה, כְּשֶׁהָאֲנָשִׁים יָשְׁנוּ, הוּא יָצָא וְאָכַל אֶת כָּל הַחֲתַלְתּוּלִים בָּעִיר. וְאָז הוּא יִלֵּל יוֹתֵר מִתָּמִיד. 
------------------
אֲבָל מִיַּד נִדְחֲפוּ מֵחָדָשׁ, וְהַדְּחִיפָה רָצָה מִן הַקָּרוֹן הָרִאשׁוֹן וְעַד הַקָּרוֹן הָאַחֲרוֹן וְכֻלָּם חָזְרוּ וְהֻסְּעוּ לְאָחוֹר וְהוּרְצוּ אֶל פַּסֵּי מְסִלָּה אַחֶרֶת, וְהָאִישׁ הַקָּטָן, שֶׁשָּׁמר עַל הַמַּחְסוֹם הַגָּדוֹל שֶׁבַּכְּבִישׁ, מִהֵר וְהוֹרִיד אֶת מַחְסוֹמוֹ, עִם שְׁלָטָיו הָאֲדֻמִים, וְקִצֵץ לִשְׁתַיִם אֶת רִיצַת הַמְּכוֹנִיּוֹת וְהָאוֹטוֹבּוּסִים שֶׁבַּכְּבִישׁ, אֵלֶּה מִכָּאן וְאֵלֶּה מִכָּאן, וְכֻלָּם רָתְחוּ אָז וְהִמְתִּינוּ בְּקֹצֶר-רוּחַ עַד שֶׁיַּעַבְרוּ כָּל
------------------
, אֲנַחְנוּ מַגִּיעִים לְאֵיזוֹ עִיר. בְּשָׁעָה שֶׁאֲנַחְנוּ נוֹסְעִים בָּרְחוֹבוֹת, אֲנָשִׁים מַבִּיטִים בָּנוּ וּמַצְבִּיעִים עָלֵינוּ. "תִּרְאוּ מָה זֶה," הֵם אוֹמְרִים, "יוֹשֵׁב אֵיזֶה עֲנָק בָּאוֹטוֹ, שָׂם לוֹ חֲמוֹר עַל הַבִּרְכַּיִם וּמַחְזִיק סִירָה מֵעַל הָרֹאשׁ. חָה! חָה! חָה! וְזֶה שֶׁעַל הַפָּנָס! הוּא לֹא גָּבוֹהַּ יוֹתֵר מִדְּלִי! תִּרְאוּ אֵיךְ הוּא מְקַפֵּץ מִצַּד לְצַד! חָה! חָה! חָה!"

------------------
וּבֶאֱמֶת הָיָה סְבִיבָם קָהָל לֹא קָטָן בִּישִׁימוֹן חַלּוּקֵי הָאֶבֶן: רֹאשׁ הַמֶּמְשָׁלָה וְחַבְרֵי הַמֶּמְשָׁלָה וְשַׂחְקָנֵי הַכַּדּוּרֶגֶל וּבֵית הַיְּתוֹמִים וְהַמַּנְטִיקוֹרָה וְסוּס הַנַּדְנֵדָה וְכָל מִי שֶׁנֶּאֱכַל עַל יְדֵי הַדְּרָקוֹן. עֲלֵיכֶם לְהָבִין, זֶה הָיָה בִּלְתִּי אֶפְשָׁרִי לַדְּרָקוֹן לָקַחַת אֶת כֻּלָּם אִתּוֹ אֶל בֵּין דַּפֵּי הַסֵּפֶר – הַמָּקוֹם הָיָה צַר אֲפִלּוּ לִדְרָקוֹן אֶחָד – כָּךְ שֶׁמּוּבָן שֶׁהָיָה עָלָיו לְהַשְׁאִיר אוֹתָם בַּחוּץ.

------------------
"הֵם לֹא יְכוֹלִים לְהַחְלִיף, יַקִּירִי. כְּשֶׁנָּסִיךְ בּוֹחֵר תַּפּוּחַ זָהָב שֶׁיֵּשׁ נְסִיכָה בְּתוֹכוֹ וְנוֹשֵׁק לוֹ עַד שֶׁהִיא יוֹצֵאת, הוּא לְעוֹלָם לֹא יַחְשֹׁק בִּנְסִיכָה אַחֶרֶת, וְגַם הִיא לְעוֹלָם לֹא תַּחְשֹׁק בְּנָסִיךְ אַחֵר."
בְּעוֹדָהּ מְדַבֶּרֶת הָלְכָה הָאִשָּׁה הַזְּקֵנָה וְהִצְעִירָה עוֹד וָעוֹד וָעוֹד, וּכְשֶׁסִּיְּמָה אֶת דְּבָרֶיהָ כְּבָר הָיְתָה צְעִירָה לְמַדַּי, בַּת חֲמִשִּׁים וְחָמֵשׁ לְכָל הַיּוֹתֵר. וְזוֹ הָיְתָה הָעַלְמָה פִיצְרוֹי רוֹבִּינְסוֹן!

------------------
 לִישֹׁן, כְּשֶׁהוּא מְחַבֵּק אֶת הֶחָתוּל בִּזְרוֹעוֹתָיו.
"עַכְשָׁו," אָמַר הֶחָתוּל, "אֲנִי אָשִׁיר לַתִּינוֹק שִׁיר שֶׁיַּרְדִּים אוֹתוֹ לְמֶשֶׁךְ שָׁעָה שְׁלֵמָה." וְהוּא הִתְחִיל לְגַרְגֵּר, גִּרְגּוּר חָזָק וְחַלָּשׁ, חַלָּשׁ וְחָזָק, עַד שֶׁהַתִּינוֹק שָׁקַע בְּשֵׁנָה עֲמֻקָּה. הָאִשָּׁה חִיְּכָה כְּשֶׁהִבִּיטָה עַל שְׁנֵיהֶם וְאָמְרָה, "עֲבוֹדָה נֶהֱדֶרֶת, אֵין מָה לְדַבֵּר. חָתוּל, אַתָּה חָכָם מְאוֹד."

------------------
"אַל תִּדְאֲגוּ," הִרְגִּיעָה אוֹתָם אִיסִיס, שֶׁבְּחוּשֶׁיהָ הַכִּשּׁוּפִיִּים כְּבָר חָשָׁה כִּי הִגִּיעָה לַמָּקוֹם הַנָּכוֹן. "אַל תִּדְאֲגוּ, אֶצְלִי הוּא יְקַבֵּל טִפּוּל מָסוּר וְסַבְלָנִי, וְאֵין לִי סָפֵק שֶׁהוּא יִתְחַזֵּק."
וְאָמְנָם, אִיסִיס הִתְכַּוְּנָה לְכָל מִלָּה שֶׁאָמְרָה. וְיוֹתֵר מִכָּךְ – הִיא בִּקְּשָׁה לְהַעֲנִיק לַתִּינוֹק חַיֵּי נֶצַח.
אֵיךְ הוֹפְכִים תִּינוֹק חָלוּשׁ לְנִצְחִי? פָּשׁוּט מְאוֹד – מַעֲלִים אוֹתוֹ בְּלֶהָבוֹת כְּשׁוּפוֹת!

------------------
"הַאִם תִּרְצִי אוּלַי לָבוֹא אִתִּי?" אָמַר קוֹל לְיָדָהּ. וּכְשֶׁנָּשְׂאָה אֶת עֵינֶיהָ רָאֲתָה יְשִׁישֹׁנֶת מוּזָרָה בִּגְלִימָה אֲדֻמָּה. הִיא לֹא חָשְׁשָׁה, כִּי הַזְּקֵנָה חִיְּכָה אֵלֶיהָ בַּחֲבִיבוּת וְעֵינֶיהָ הַשְּׁחֹרוֹת הִבְרִיקוּ, אַף שֶׁאַפָּהּ הִתְעַקֵּל מֵעַל סַנְטֵרָהּ הָאָרֹךְ.
"לְאָן אַתְּ רוֹצָה לָקַחַת אוֹתִי?" שָׁאֲלָה הַנְּסִיכָה הַקְּטַנָּה כְּשֶׁהִיא מוֹצֶצֶת אֶת אֲגוּדָלָהּ וְעֵינֶיהָ כִּמְעַט יוֹצְאוֹת מֵחֹרֵיהֶן.

------------------
"פוֹרְטוּנָטוּס אֲהוּבִי, אַתָּה בֶּאֱמֶת חוֹשֵׁב שֶׁנִּתָּן לִבְטֹחַ בָּעַלְמָה פִיצְרוֹי רוֹבִּינְסוֹן? הַאִם אֵי-פַּעַם נִרְאֶה שׁוּב אֶת דֵייזִי שֶׁלָּנוּ?"
וְהַמֶּלֶךְ בָּחַשׁ בְּשַׂעֲרוֹתָיו הַחֲלָקוֹת עַד שֶׁהֵן נֶעֶמְדוּ כְּמוֹ מַקְּלוֹת גְּבִינָה מִתַּחַת לְכִתְרוֹ וְעָנָה –
"יַקִּירָתִי, אַתְּ חַיֶּבֶת לִהְיוֹת סַבְלָנִית; אַתְּ יוֹדַעַת שֶׁקִּבַּלְנוּ עָלֶיהָ הַמְלָצוֹת מְעֻלּוֹת לְלֹא רְבָב."

------------------
הוּא עָלָה רַק שִׁשָּׁה פֵּנִי, וְהָיָה לָהּ אוֹתוֹ רַק כְּחֹדֶשׁ יָמִים, וְהִיא לֹא הִצְלִיחָה לְהַשְׁחִיז בּוֹ דָּבָר, פְּרָט לְעֶפְרוֹנוֹת הַצִּפְחָה שֶׁלָּהּ, אֲבָל אֵיכְשֶׁהוּ הִיא הִצְלִיחָה לַחְתֹּךְ בְּאֶמְצָעוּתוֹ אֶת הַחֲגוֹרָה שֶׁלָּהּ מִקָּדִימָה וְהִשְׁתַּחֲלָה מִחוּצָה לָהּ, מַשְׁאִירָה בֵּין טְפָרָיו שֶׁל הַדְּרָקוֹן לוּלְאַת מֶשִׁי יְרֻקָּה. 
------------------
הוּא אוֹחֵז אֶת הַגֶּרְמָנִי בְּכַפְתּוֹר מְעִילוֹ וּמוֹסִיף בְּקוֹל רוֹעֵד :"וּבְכֵן… אֲנִי, מְקַוֶּה, כַּמּוּבָן, שֶׁיַּלְדָּתִי תַּבְרִיא. אֲבָל… אִם חַס וְחָלִילָה… מָה אִם מַחֲלָתָהּ תִּגָּמֵר רַע… מָה אִם הַיַּלְדָּה תָּמוּת פִּתְאוֹם?… רַק תַּחְשֹׁב: הֲרֵי הַמַּחְשָׁבָה שֶׁלֹּא מִלֵּאתִי אַחַר מִשְׁאַלְתָּהּ הָאַחֲרוֹנָה תְּעַנֶּה אוֹתִי כָּל חַיַּי!"

------------------
לְבַסּוֹף דְּרָקוֹן-הַכּוֹבַע הִנִּיחַ אֶת בִּילִי בְּתוֹךְ תְּלוּלִית שֶׁלֶג וְהֵשִׁיב מִלְחָמָה בִּזְנָבוֹ.
מוֹלִי רָצָה אֶל תְּלוּלִית הַשֶּׁלֶג בְּשִׂיא הַמְּהִירוּת, וְכָךְ עָשְׂתָה גַּם מוֹלִי הַקְּטַנָּה הַיְּרֻקָּה וְהַיּוֹקֶדֶת. מוֹלִי הַקְּטַנָּה הַיְּרֻקָּה וְהַיּוֹקֶדֶת הִגִּיעָה לְשָׁם רִאשׁוֹנָה. הִיא תָּפְסָה אֶת בִּילִי בַּיָּד וּמָשְׁכָה אוֹתוֹ לַכְּבִישׁ וּבְלִי לְהִסְתַּכֵּל בִּכְלָל לִשְׁנֵי הַצְּדָדִים הִיא דָּחֲפָה אוֹתוֹ קָדִימָה לַכְּבִישׁ –
אֲבָל מוֹלִי מָשְׁכָה אוֹתוֹ לְאָחוֹר.

------------------
וְלָכֵן, מִכֵּיוָן שֶׁהָיוּ מוּדָעִים לַדְּבָרִים הָאֵלֶּה, נָהֲגוּ הַהִיפּוֹפּוֹטָמִים לְהִסְתַּעֵר בְּזַעַם עַל הָאֲנָשִׁים הַשְּׁחֹרִים בְּכָל פַּעַם שֶׁהֵרִיחוּ אֶת נִיחוֹחָם הַשַּׁמְנוּנִי, וְאִם בְּמִקְרֶה הִשִּׂיגוּ אֶת אֶחָד מִבְּנֵי הָאוֹיֵב, רִטְּשׁוּ אוֹתוֹ בְּחַטֵּיהֶם הַחַדּוֹת אוֹ רָמְסוּ אוֹתוֹ בְּכַפּוֹת רַגְלֵיהֶם הָעֲנָקִיּוֹת.
מִלְחָמָה תְּמִידִית נִטְּשָׁה בֵּין הַהִיפּוֹפּוֹטָמִים לָאֲנָשִׁים הַשְּׁחֹרִים.

------------------
"אֶל חוֹף הַיָּם, אֶל מָקוֹם שֶׁבּוֹ תּוּכְלִי לְשַׂחֵק בַּחוֹל, מָקוֹם שֶׁבּוֹ יִהְיוּ לָךְ חֲבֵרִים קְטַנִּים וַחֲבֵרוֹת לַמִּשְׂחָק, וְאִישׁ לֹא יִמְנַע מִמְּךָ לְהַרְעִישׁ."
"אֲנִי בָּאָה," אָמְרָה אוֹרְסוּלָה וְזִנְּקָה מִמְּקוֹמָהּ.
"קָדִימָה," אָמְרָה הַזְּקֵנָה בַּעֲדִינוּת, וְהִיא אָסְפָה אֶת אוֹרְסוּלָה בִּזְרוֹעוֹתֶיהָ וְעָטְפָה אוֹתָהּ בִּגְלִימָתָהּ הָאֲדֻמָּה, וְעַד מְהֵרָה הִתְרוֹמְמוּ לָהֶן וּפָרְחוּ מֵעַל גַּגּוֹת הַבָּתִּים.

------------------
"קְחִי אוֹתִי הַבַּיְתָה, טַבּוֹרֶט הַיְּקָרָה," קָרְאָה אוֹרְסוּלָה בְּבֶכִי. "הַחְזִירִי אוֹתִי לְאוֹלִיבֵר וּלְפִילִיפּ וּלְבֵּל. אֲנִי שׂוֹנֵאת אֶת הָאֲנָשִׁים הָאֵלֶּה."
וּמִיָּד הִתְחַדְּשׁוּ דִּמְעוֹתֶיהָ.
טַבּוֹרֶט רַק חִיְּכָה וְלִטְּפָה אֶת רֹאשָׁהּ וְאַחַר כָּךְ הָלְכָה לְדַבֵּר עִם הַמֶּלֶךְ וַאֲנָשָׁיו.

------------------
אֲבָל לְמָחֳרָת הוּא רָצָה לְהָצִיץ שׁוּב, אָז הוּא לָקַח אֶת הַסֵּפֶר הַחוּצָה לְגַן הָאַרְמוֹן, וּכְשֶׁהִתִּיר אֶת הָאַבְזָמִים הַמְּשֻׁבָּצִים אַבְנֵי טוּרְקִיז וְאֹדֶם, הַסֵּפֶר נִפְתַּח לְגַמְרֵי לְבַדּוֹ בַּתְּמוּנָה שֶׁמִּתַּחְתֶּיהָ הָיָה כָּתוּב "דְּרָקוֹן", וְהַשֶּׁמֶשׁ הֵאִירָה יְשִׁירוֹת עַל הַעַמּוּד. 
------------------
הַבְּעָלִים שֶׁל גִּבְעַת הַתִּלְתָּן לֹא הָיָה מוּכָן לִמְכֹּר אֶת אַדְמוֹתָיו בְּשׁוּם אֹפֶן וּבְשׁוּם מְחִיר שֶׁחֶבְרַת "רֵקְס פוֹרְטוּנָטוּס וְשׁוּת'" הָיְתָה יְכוֹלָה לְהַצִּיעַ. לַשָּׁוְא זִמְּנוּ מִשְׁפְּטָנֵי הַחֲבֵרָה אֶת מִשְׁפְּטָנֵי הַבְּעָלִים, לָבְשׁוּ אֶת גְּלִימוֹתֵיהֶם הַטּוֹבוֹת בְּיוֹתֵר, חָגְרוּ חֲרָבוֹת וּמָגִנִּים וְהִזְמִינוּ אוֹתָם לַאֲרוּחַת צָהֳרַיִם, הֶאֱכִילוּ וְהִשְׁקוּ אוֹתָם בּמַטְעַמִּים וּבְמַשְׁקָאוֹת מְשֻׁבָּחִים. גִּבְעַת הַתִּלְתָּן לֹא הָיְתָה לִמְכִירָה.

------------------
וְהַלֵּזוּ, הַשְׁנִיָּה, פָּעֲרָה אָז לִסְתוֹתֶיהָ שֶׁלָּהּ וְתָפְשָׂה בּוֹ, בַּגַּרְגֵּר הַשָׁקֵט הַזֶּה, הַמַּקְרִין, וּכְבָר מִיָּד הָיְתָה אֲחוּזָה וּמְאֻחֶזֶת בּוֹ, וּכְבָר נוֹשֵׂאת אוֹתוֹ כְּאִלּוּ שֶׁלָּהּ הוּא, וּכְאִלּוּ כָּךְ טוֹב וְכָךְ יָפֶה; וְאִלּוּ חוּמִית, שֶׁהָיְתָה צְרִיכָה עַתָּה לָשֶׁבֶת רֶגַע מִן הַצַּד וּלְהִנָּפֵשׁ, לִמְחוֹת זֵעָה, אִם גַּם נְמָלָה מֵזִיעָה, וְלְהָשִׁיב רוּחָהּ וּלְנַפְנֵף – אִם גַּם נְמָלָה מְנַפְנֶפֶת לָהּ – אוֹ שֶׁמָּא לְנַסּוֹת לִתְפּוֹשׂ וְלָשֵׂאת עִם אֲחוֹתָהּ
------------------
דּוֹן שׁוּעָל תָּפַס תַּ'מַּקֵּלַ'לִיכָה שֶׁלּוֹ, הוּא תָּפַס, וּשְׁנֵיהֶם אָצוּ רָצוּ בַּחֲזָרָה לְחֶלְקַתַ'בָּטְנִים; וּכְשֶׁהֵם הִגִּיעוּ, שָׁמָּה הוּא הָיָה, דּוֹן דֹּב, עַל בָּטוּחַ, תָּלוּי עַלָ'עֵץ.
"אָה-הָא! סוֹף-סוֹף תָּפַסְתִּי'תְךָ, הָא?" אָמַר דּוֹן שׁוּעָל; וְלִפְנֵי שֶׁדּוֹן דֹּב יָכַל לַ'סְבִּיר מַשֶּׁהוּ, דּוֹן אַרְנָב קָפַץ מַטָּה-מַ'לָה וְצִפְצֵף בְּקוֹלֵי קוֹלוֹת:
"תַּפְלִיק לוֹ עַלַ'פֶּה, דּוֹן שׁוּעָל! תַּפְלִיק לוֹ עַלַ'פֶּה!"

------------------
"מַ'תָּה עוֹשֶׂה פֹּה, מְנֻוָּל?" זָעַק הַקַּצָּב בְּכַעַס, "לֹא הִבְטַחְתָּ לָשִׂים אֶת הַבְּהֵמָה הַזֹּאת בַּמִּרְעֶה שֶׁל פְּלִימְפְּטוֹן? וְעַכְשָׁו אֲנִי מוֹצֵא אוֹתְךָ פֹּה רוֹכֵב לְךָ עַל הַמִּסְכֵּן כְּמוֹ אֵיזֶה גֶ'נְטְלְמֶן בַּטְלָן!"
"בְּחַיַּי," אָמַר גִ'ים, מֻפְתָּע; "שָׁכַחְתִּי לְגַמְרֵי מֵהַסּוּס!"
*******************************************************************************

------------------
הִיא הִסְתַּכְּלָה סָבִיב-סָבִיב. לֹא הָיָה שָׁם אַף אֶחָד. הִיא נִעֲרָה אֶת הַכִּיסִים, וְזִילְבֵּרְס וּוִילְבֵּרְס נָפְלוּ הַחוּצָה.
"צְצְ…" לָחֲשׁוּ, "אַתְּ לֹא יְכוֹלָה לְהִפָּטֵר מֵאִתָּנוּ סְתָם כָּךְ. רֵאשִׁית, אַתְּ אַחְרָאִית לָנוּ, וְשֵׁנִית, מָה אֲנַחְנוּ אִם לֹא מַשֶּׁהוּ חָזָק, מַשֶּׁהוּ מָהִיר, וְלִקְנוֹת וְלִמְכֹּר אוֹתָנוּ לֹא צָרִיךְ. אוֹתָנוּ מְקַבְּלִים חִנָּם."
מָה יָכְלָה אָמוֹרִי הַמִּסְכֵּנָה לַעֲשׂוֹת? הֵם מִשְׁמְשׁוּ בִּמְחוֹשֵׁיהֶם הַדַּקִּים הַמַּגְעִילִים.

------------------
יָד בְּיָד סָבְבוּ זְאֵבֵי וְאַבָּא אֶל דֶּרֶךְ הֶעָפָר הַהִיא, שֶׁלָּהֶם, הַשְׁלֵוָה, כְּשֶׁנִּזְכַּר זְאֵבֵי : וּמַה קָרָה לַעֲרֵמַת הַגַּרְגְרִים הַהִיא?
אַבָּא טָעַן כִּי צָרִיךְ לְמַהֵר הַבַּיְתָה, וְכִי הָעֲרֵמָה נִשְׁאֲרָה בְּוַדַּאי עַל מְקוֹמָהּ, כְּשֶׁהָיְתָה. אֲבָל טָעָה. וזְאֵבֵי הוֹרָה בְּאֶצְבַּע: הַבֵּט!

------------------
הֵם הֻקְסְמוּ מִן הָרִהוּט, וּמַגַּע יָדָהּ הַמְּעֻדָּן שֶׁל אִשָּׁה שֶׁנִּכַּר בַּכֹּל מָשַׁךְ אוֹתָם כְּמוֹ מַגְנֵט. הֵם שָׂכְרוּ אֶת הַבַּיִת, אַךְ אֶת אֲרוּחוֹתֵיהֶם סָעֲדוּ בָּאַרְמוֹן בְּהַזְמָנָה מְיֻחֶדֶת שֶׁל הַמֶּלֶךְ.

------------------
"מָה עִם הַפִּיל? הוּא בָּא?
"הוּא בָּא," עוֹנָה אִמָּא. "אֲבָל הוּא אָמַר שֶׁקֹּדֶם כֹּל נַדְיָה תִּתְרַחֵץ וְאַחַר כָּךְ תֹּאכַל בֵּיצָה רַכָּה וְתִשְׁתֶּה חָלָב חַם."
"הוּא טוֹב לֵב?"
"הוּא טוֹב לֵב. תֹּאכְלִי, יַקִּירָתִי. תֶּכֶף נֵלֵךְ אֵלָיו."
"הוּא נִרְאֶה מַצְחִיק?"
"קְצָת. תִּלְבְּשִׁי אֶת הַסְּוֵדֶר הַחַם שֶׁלָּךְ."

------------------
הֵם חִכּוּ עַד שֶׁהַדֶּלֶת הִגִּיעָה אֲלֵיהֶם, וְאָז נִכְנְסוּ פְּנִימָה. זֶה בֶּאֱמֶת הָיָה בַּיִת מוּזָר. אָמְנָם הָיָה זֶה קַיִץ, אֲבָל בָּאֲרֻבָּה בָּעֲרוּ קוֹרוֹת עֵץ עֲנָקִיּוֹת, וְעַל הָאֵשׁ יָשַׁב אִישׁ נִכְבָּד וְהִתְחַמֵּם. מִדֵּי פַּעַם אָסַף הָאִישׁ בְּכַף יָדוֹ גֶּחָלִים לוֹחֲשׁוֹת וּבָלַע אוֹתָן בְּתֵאָבוֹן רַב. מִשֶּׁרָאָה אֶת הָאוֹרְחִים, נִגַּשׁ אֲלֵיהֶם, קַד קִדָּה וְאָמַר:
"הַאֻמְנָם אַתֶּם אָדוֹן חוּטְחוּטוֹן וְהָרוֹזֵן אָדוֹן דַּחְלִיל?"

------------------
וּבְעוֹד הוּא יוֹשֵׁב שָׁם וְלוֹחֵשׁ תְּפִלּוֹת בְּקֶצֶב שֶׁל רַכֶּבֶת דּוֹהֶרֶת, הוּא שׁוֹמֵעַ פִּתְאוֹם תַּ'קּוֹל שֶׁל דּוֹן שׁוּעָל הַזָּקֵן שֶׁצּוֹעֵק אֵלָיו:
"הָלוֹ, דּוֹן אַרְנָב, אֶת מִי'תָּה מְבַקֵּר שָׁם לְמַטָּה?" הוּא אוֹמֵר.

------------------
"הַחֵצִי הַשֵּׁנִי שֶׁלִּי נִמְצָא שָׁם," הוּא אָמַר וְהִצְבִּיעַ עַל הַגְּלוֹבּוּס הַשְּׁמֵימִי. "תְּנוּ לָרַגְלַיִם שֶׁלִּי לָצֵאת בְּבַקָּ -"
אֲבָל הַנָּסִיךְ סֵקוּנְדוּס אָמַר, "לֹא כָּל כָּךְ מַהֵר," וְהַנָּסִיךְ טֶרְטְיוּס שָׁאַל, "לָמָּה סָגְרוּ אוֹתְךָ?"
"סָגְרוּ אוֹתִי רַק בִּגְלַל מְעַט קֶסֶם טְרַקְלִינִי שֶׁלֹּא רָאִיתָ כָּל יָמֶיךָ עַל פְּנֵי הָאֲדָמָה," אָמַר פֶּלֶג גּוּפוֹ הָעֶלְיוֹן שֶׁל הַפְּרוֹפֵסוֹר לְקֶסֶם.

------------------
"מָה אַתֶּם חוֹשְׁבִים, שֶׁלֹּא רוֹאִים שֶׁאַתֶּם ג'וּקִים? ג'וּקִים לְבוּשִׁים זֶה מַמָּשׁ ג'וּקִים עֲרֻמִּים עִם בְּגָדִים."
אָמוֹרִי הִתְיַשְּׁבָה עַל הַסַּפְסָל. מַזָּל שֶׁכֻּלָּם כְּבָר הָלְכוּ הַבַּיְתָה, חָשְׁבָה. "מָה אַתֶּם רוֹצִים שֶׁאֶעֱשֶׂה?" שָׁאֲלָה בְּקוֹל נָמוּךְ.
"אֲנַחְנוּ רוֹצִים שֶׁתַּגִּידִי לָנוּ: 'וִילְבֵּרְס וְזִילְבֵּרְס הַנִּפְלָאִים וְהַנְּעִימִים, אָנָּא רִקְדוּ לְפָנַי.'"

------------------
יְחִי מַלְכֵּנוּ הַקָּטָן!" וְזֶה נִשְׁמַע כְּאִלּוּ הֵם שָׁרוּ אֶת הַמִּלִּים לִצְלִילֵי הַפַּעֲמוֹנִים.
בַּהַתְחָלָה הוּא קְצָת הִצְטַעֵר שֶׁהוּא לֹא לָבַשׁ אֶת בְּגָדָיו הַטּוֹבִים בְּיוֹתֵר, אֲבָל עַד מְהֵרָה הוּא שָׁכַח מִזֶּה. אִלּוּ הָיָה יַלְדָּה, כַּנִּרְאֶה זֶה הָיָה מַטְרִיד אוֹתוֹ כָּל הַזְּמַן.
כְּשֶׁהֵם הִמְשִׁיכוּ הָלְאָה, הַגֶּ'נְטְלְמֵנִים חֲמוּרֵי הַסֵּבֶר, שֶׁהָיוּ רֹאשׁ הַמֶּמְשָׁלָה וְשַׂר הָאוֹצָר, הִסְבִּירוּ לְלָיוֹנֵל אֶת הַדְּבָרִים שֶׁלֹּא הֵבִין.

------------------
וְכָךְ הָיָה. אִיסִיס וְנֶפְתִיס הִפְלִיגוּ בְּרַחֲבֵי הַנִּילוּס בְּסִירַת קָנִים קְטַנָּה וּבְסַבְלָנוּת רַבָּה אָסְפוּ שְׁלוֹשָׁה-עָשָׂר מֵחֲלָקָיו שֶׁל אוֹסִירִיס – אֶת הַחֵלֶק הָאַרְבָּעָה-עָשָׂר לֹא מָצְאוּ. אַךְ הֵן חִבְּרוּ כָּל כָּךְ יָפֶה אֶת שְׁאַר הַחֲלָקִים, עַד שֶׁאוֹסִירִיס הָיָה מֻשְׁלָם גַּם בִּלְעָדָיו.
כָּעֵת בֶּאֱמֶת נוֹתַר רַק לְחַבֵּר אֶת לַחַשׁ הַכְּשָׁפִים.

------------------
 מִפֵּרוֹתַי יִתָּלֶה עַל הָעֵץ שֶׁמִּמֶּנּוּ גָּנַב. זֶה הַכֹּל. מָה אַתָּה אוֹמֵר?"
הַמֶּלֶךְ אָמַר "כֵּן", כִּי מֵאָז אֲסוֹנוֹ לֹא הָיָה אִכְפַּת לוֹ מִדָּבָר, פְּרָט לִבְנִיָּה, וְנַפְשׁוֹ הַמַּלְכוּתִית הִשְׁתּוֹקְקָה לִרְאוֹת אֶת גִּבְעַת הַתִּלְתָּן הַיְּרֻקָּה נֶאֱכֶלֶת עַל יְדֵי זַחֲלֵי לְבֵנִים צְהֻבּוֹת בַּעֲלֵי גַּגּוֹת צִפְחָה. הוּא לֹא הִתְלַהֵב כְּלָל מֵהָרַעְיוֹן לִבְנוֹת שְׁתֵּי חוֹמוֹת מִלְּבֵנִים אֲדֻמּוֹת, אַךְ הוּא עָשָׂה זֹאת.

------------------
 יוֹם אֶחָד נִשְׁאֲרוּ הַבֻּבּוֹת לְגַמְרֵי לְבַדָּן.
אִמָּא הַקְּטַנָּה שֶׁלָּהֶן סִדְּרָה כָּל בֻּבָּה בִּמְקוֹמָהּ בַּחֶדֶר וְאָמְרָה לָהֶן לְהִתְנַהֵג יָפֶה וְלִשְׁמֹר עַל שֶׁקֶט עַד שֶׁהִיא תַּחְזֹר.
וְהֵן יָשְׁבוּ יָפֶה כֻּלָּן, לֹא זָזוּ מִמְּקוֹמָן וְלֹא הוֹצִיאוּ הֶגֶה, עַד שֶׁאִמָּא הַקְּטַנָּה יָצְאָה מֵהַחֶדֶר.
וְאָז הִפְנָה חַיַּל הַבְּדִיל אֶת רֹאשׁוֹ וְקָרַץ קְרִיצָה רַבַּת-מַשְׁמָעוּת לִסְמַרְטוּתִי.

------------------
עַל כֵּן הֵסִיר אֶת כֻּמְתָּתוֹ שֶׁל הַשּׁוֹטֵר וְאֶת מְעִילוֹ בַּעַל כַּפְתּוֹרֵי הַפְּלִיז וְהִלְבִּישׁ בָּהֶם אֶת הָעַלְמָה סְקְרַאפֶּל, וְאִלּוּ אֶת מִגְבַּעְתָּהּ הַמְּעֻטֶּרֶת בִּסְרָטִים וּבְנוֹצוֹת הֵנִיחַ נִמְרָצוֹת עַל רֹאשׁוֹ שֶׁל הַשּׁוֹטֵר.

------------------
"תּוֹדָה, יַנְשׁוּף לֹא-תּוּכְלִי-לִרְאוֹת-אוֹתוֹ-כְּשֶׁתַּבִּיטִי," אָמְרָה מוֹלִי, "וְתִמְסֹר לְמַלְכַּת הַיַּנְשׁוּפִים תּוֹדָה בִּשְׁמִי." הִיא פָּקְחָה עֵינַיִם וְרָאֲתָה אֶת מוֹלִי הַיְּרֻקָּה הַקְּטַנָּה וְהַזּוֹעֶמֶת חוֹמֶקֶת וְגוֹלֶשֶׁת מַטָּה עַל גֶּזַע הָעֵץ. מוֹלִי הָרְגִילָה מָשְׁכָה וְהִדְּקָה אֶת שׁוּלֵי הַכּוֹבַע הָאָדֹם עַל אָזְנֶיהָ, קָפְצָה מִבַּעַד לְחַלּוֹן הַחֶדֶר וְנָחֲתָה עַל הַמִּטָּה שֶׁלָּהּ.

------------------
אֲבָל סֵת פָּגַשׁ גַּם כַּמָּה אֲנָשִׁים שֶׁהָיוּ, כָּמוֹהוּ, מְמֻרְמָרִים מִלֵּדָה. תָּמִיד הָיָה נִדְמֶה לָהֶם שֶׁאֵצֶל הַשָּׁכֵן הַכֹּל מֻצְלָח יוֹתֵר: הַחִטָּה גְּבוֹהָה יוֹתֵר, הַפָּרָה שְׁמֵנָה יוֹתֵר, הַבַּיִת גָּדוֹל יוֹתֵר, הָאִשָּׁה חֲכָמָה יוֹתֵר, וְהַיְּלָדִים גַּם הֵם טוֹבִים וַחֲכָמִים וּלְעוֹלָם אֵינָם מְשֻׁעֲמָמִים.
