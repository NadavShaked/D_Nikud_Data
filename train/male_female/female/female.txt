וְכָךְ, כְּדֵי לְהָקֵל עַל בִּיל, נַעֲנֵיתִי, וּבְשִׁיתּוּף-פְּעוּלָּה חִיבַּרְנוּ מִכְתָּב בָּזוֹ הַלָּשׁוֹן:  גְּבֶרֶת אֶבֶן-עֵזֶר דּוֹרְסְט, אֲנַחְנוּ מַחְבִּיאִים אֶת הַבֵּן שֶׁלָּךְ אֶצְלֵנוּ, רָחוֹק מִפִּסְגָּה.
אִם תְּנַסִּי לְרַמּוֹת אוֹתָנוּ בְּדֶרֶךְ כָּלְשֶׁהִי, אוֹ אִם לֹא תְּמַלְּאִי אֶת דְּרִישָׁתֵנוּ לְפִי הָאָמוּר, לְעוֹלָם לֹא תִּרְאִי אֶת הַיֶּלֶד שֶׁלָּךְ.
שִׁמְעִי, אַתְּ תַּפְסִיקִי לִבְעוֹט בִּי אוֹ שֶׁאֲנִי קָם וּמְנַפֵּחַ אוֹתָךְ.

------------------
הִיא לוֹחֶצֶת עָלַי, "תֹּאכְלִי, תֹּאכְלִי, זֶה יַזְכִּיר לָךְ אֶת הַבַּיִת, שֶׁכְּבָר רָחוֹק," וַאֲנִי אוֹמֶרֶת לְעַצְמִי שֶׁאַף אֶחָד לֹא יָכוֹל בֶּאֱמֶת לִקְבּוֹעַ בִּשְׁבִיל מִישֶׁהוּ אַחֵר אֵיפֹה זֶה הַבַּיִת, וְשֶׁאֶת זֶה אֲנִי לָמַדְתִּי מִיֶּלֶד, קָטָן.
זֶה הַצַּד שֶׁאֲנִי נוֹתֵן לְךָ, וּמָה אַתְּ נוֹתֶנֶת לִי בִּתְמוּרָה.
מוּבָן שֶׁהִקְשַׁבְתִּי, וְלָכֵן אֲנִי שׁוֹאֵל אוֹתְךָ מָה אַתְּ נוֹתֶנֶת לִי בִּתְמוּרָה.

------------------
וְאִם תַּעֲשִׂי זֹאת, לְעוֹלָם לֹא תִּרְאִי אֶת דֶּיְיזִי שֶׁלָּךְ שׁוּב.
תַּתְחִילִי לֶאֱסוֹף אֶת הַצַּעֲצוּעִים שֶׁלָּךְ עַכְשָׁיו וּמִימוֹ יוּכַל לְהִישָּׁאֵר אֶצְלֵךְ.
רַק תַּאַסְפִי אֶת הַצַּעֲצוּעִים וּמִימוֹ יוּכַל לְהִישָּׁאֵר אֶצְלֵךְ.
אֲנִי אוֹהֵב אוֹתְךָ, אִמָּא.
אַף פַּעַם לֹא הֶרְאֵית לִי מָה שֶׁאַתְּ כּוֹתֶבֶת, אֲבָל אֲנִי סוֹמֵךְ עָלַיִךְ שֶׁלֹּא תְּוַותְּרִי.
תַּגִּידִי לַסּוֹדִי שֶׁלָּךְ.
תַּחְשְׁבִי עָלַי, אִם אַתְּ לֹא מוּכָנָה לַחֲשׁוֹב עַל עַצְמְךָ.

------------------
הָיוּ אֲפִילּוּ שְׁמוּעוֹת עַל כִּישּׁוּף, אִם כִּי אָנְדְּרֶה פְלִיבֵּיין, הִיסְטוֹרְיוֹנִית הֶחָצֵר שֶׁל לוּאִי הָאַרְבָּעָה-עָשָׂר, דּוֹחָה הַשְׁעָרוֹת מֵעֵין אֵלֶּה.
וַאֲפִילּוּ בִּזְמַן כְּתִיבָתָן שֶׁל שׁוּרוֹת אֵלֶּה, בְּיוֹשְׁבֵי אֶל שׁוּלְחַן הָעֵץ בְּבֵיתִי, בְּעוֹדִי שׁוֹמַעַת אֶת צְלִיל קוֹלָהּ שֶׁל בַּעֲלִי הַטּוֹב מְדַבֵּר אֶל יַלְדֵי בַּחֲדָרִים שֶׁבַּקּוֹמָה הָעֶלְיוֹנָה, תּוֹהָה אֲנִי שֶׁמָּא עוֹדֶנִּי בַּגַּן הַהוּא, שֶׁרוֹעָה עַל רִצְפַּת הַמְּעָרָה, שְׁבוּרָה וּמְפוֹרֶקֶת לְגוֹרְמִים.

------------------
אָה, נָאדִין, אַתְּ הֲרֵי לֹא מְצַפֶּה שֶׁאֶסְתַּפֵּק בַּטִּיפּ-טִיפָּה שֶׁאַתְּ מַצִּיעָה לִי?
כְּשֶׁהָיְתָה פּוֹסַעַת בִּרְחוֹב לֵס קָאז, אִי שֶׁל צְלָלִים בְּדִמְדּוּמִים שֶׁל יְמֵי הַחוֹרֶף, וּמְזַהָה מֵעַל הַדֶּלֶת אֶת דְּמוּת הָאִישָּׁה הַמְּחַיֶּיכֶת, הַמְּפוּסֶּלֶת בְּאֶבֶן, אֶת הַפָּנִים הָרַכּוֹת וְהַמּוּכָּרוֹת הַמְּעוּטָּרוֹת בִּסְרָטִים דַּקִּים, הָיְתָה חָשָׁה מְרוּכֶּכֶת, מְפוּיֶּסֶת, שְׁטוּפָה בְּאוֹשֶׁר רוֹגַע.

------------------
מִשְׁתַּהָה, וְאָז צוֹחֶקֶת בְּקוֹל גָּדוֹל וְטוֹפַחַת בְּיָדָהּ הַגְּדוֹלָה עַל הַלֶּחִי שֶׁלִּי.
הַדּוֹדָה אֵלָה מִתְגָּאֶה בְּנֶכְדָּתָהּ וְנוֹתֶנֶת לָהּ חִיבּוּק, שֶׁמִּמֶּנּוּ הִיא מְנַסָּה לְהִשְׁתַּחְרֵר שֶׁלֹּא בְּהַצְלָחָה.
וְהִיא מִתְעַנָּה עַל הַמִּיטָּה, מִתְהַפֶּכֶת וְגוֹנַחַת, כָּל גּוּפָהּ מִתְעַוֵּות.
עַל אַף שֶׁהִיא רַק בְּתִשְׁעִים וְשֶׁבַע הִיא מִתְפַּתָּה לְהָצִיץ מָה קוֹרֶה שָׁם.
הָרוֹפְאָה אָמְרָה, "עִם כָּל הַכָּבוֹד, אֲנִי מִתְקַשָּׁה לְהַאֲמִין".

------------------
הִיא שׁוֹתָה אֶת הַדָּם שֶׁלָּהֶם.
שֶׁאִם לֹא כֵּן, הָיְיתָה עוֹלָה בָּאֵשׁ בִּגְלַל חוֹמָה שֶׁל הַשֶּׁמֶשׁ.
נִדְמֶה לִי שֶׁאֲנִי רוֹאָה טוֹב יוֹתֵר וְשׁוֹמַעַת טוֹב יוֹתֵר אַחֲרֵי שֶׁבָּכִיתִי בְּמֶשֶׁךְ שָׁבוּעַ.
חוּץ מֵהַכְּנָפַיִם, אַתְּ דּוֹמָה מְאוֹד לְסוּס הַנַּדְנֵדָה הַמִּסְכֵּן, הַיָּקָר וְהָאָבוּד שֶׁלִּי.
אַתְּ לֹא רוֹאָה שֶׁאֲנַחְנוּ לֹא לְבַד?
וּשְׁתֵּיהֶן, תִּסְלַח לִי, מִלְּבֵנִים אֲדָמוֹת; לֹא מֵהַחֲמַר הַזּוֹל הַצַּהַב שֶׁאַתְּ בּוֹנֶה בּוֹ בְּדֶרֶךְ כְּלָל.

------------------
כַּאֲשֶׁר בְּשׁוּבָה הָיְיתָה כְּמִשְׁתָּאָה לְאַהֲבָתוֹ שֶׁעַזָּה הִיא מִן הָרָגִיל.
מָזִי, שֶׁסּוֹפֶרֶת מַטְבְּעוֹת, מִשְׁתַּהָה קְצָת בְּתַעֲנוּג עִם כָּל צִילְצוּל קָלוּשׁ שֶׁל מַטְבֵּעַ בְּמַטְבֵּעַ.
צְחוֹק עָצוּר בְּתוֹכָהּ וּמְטַלְטֵל אוֹתָהּ הֵנָּה וָהֵנָּה, עַד שֶׁהִיא מִתְקַשָּׁה לְהַמְשִׁיךְ.
הִיא מִתְכַּסָּה בַּשְּׂמִיכָה שֶׁלָּהּ.
אֵיךְ אַתְּ צוֹחֶקֶת בְּשֶׁקֶט וְאֵיךְ אַתְּ מִשְׁתַּנָּה עִם גְּדִילָתֵךְ.
הֵבֵאתִי אוֹתָם לַחֲבֶרְתִּי מֵחֶבְרָה אַחֶרֶת שֶׁמִּתְמַחָה בְּקֵרָמִיקָה.

------------------
אוּלַי הָיִית מְנַצַּחַת.
אוּלַי הָיִית מַשְׁאִירָה לָהּ כַּמָּה זִיכְרוֹנוֹת רָעִים.
טוֹב, הִיא אָמְרָה, "אֲנִי מַנִּיחָה שֶׁהָיִית צְרִיכָה לְהַעֲסִיק אֶת עַצְמֵךְ."
בֶּטַח גַּם אַתְּ עָבַדְתְּ אִיתָּם, אָמְרָה סְטֶיְיסִי לְאָנוּשְׁקָה, וְהִיא הִנְהֲנָה בְּהֶיסַּח דַּעַת.
סִלְחִי לִי, אָמְרָה וְהִתְאַמְּצָה לִפְשׁוֹט אֶת זְרוֹעָהּ, "חָשַׁבְתִּי שֶׁאַתְּ הִיא שֶׁהָרַגְתְּ אוֹתוֹ!"
הָיִיתִי כֹּה בּוֹטַחַת וְכֹה טְהוֹרָה, וְאַתְּ בָּאת כְּדֵי לֶאֱהוֹב אוֹתִי, וַאֲנִי אָהַבְתִּי אוֹתְךָ!

------------------
יַלְדָּתִי הַיְּקָרָה, אֶעֱשֶׂה כָּל דָּבָר בִּשְׁבִילְךָ, אֲבָל אֶת זֶה אֲנִי לֹא יָכוֹל לַעֲשׂוֹת.
תִּלְבְּשִׁי אֶת הַסּוּדָר הַחַם שֶׁלָּךְ.
זֶה מְסַפֵּק אֲבָל הַבֶּטֶן שֶׁלְּךָ מִתְהַפֶּכֶת וּמֵאֵיזוֹשֶׁהִי סִיבָּה אַתְּ רוֹצֶה לְהַמְשִׁיךְ לְהִסְתַּכֵּל.
אֲנִי יוֹדַעַת שֶׁאַתְּ בֶּטַח שׁוֹאֶלֶת אֶת עַצְמְךָ לָמָּה אֲנִי מְסַפֶּרֶת לָךְ אֶת הַסִּיפּוּר הַזֶּה, הָיִיתִי אוֹמֶרֶת.

------------------
אֵצֶל עַיִן רָעָה הִיא עוֹבֶרֶת, דְּמוּמָה, אֶת מְשׂוּכַת שִׁדְרַת הַפִּיגָם, וְכִמְעַט מִתְגַּלָּה בְּצִוְוחַת הָאֲוָוזִים שֶׁל בָּרוּךְ הַמְּעוּקָּל.
הִיא מִשְׁתַּהָה מְעַט, מְנַסָּה לִקְרוֹעַ אֶת מַעֲטֵה הָאַפְלוּלִית וּלְפַעְנֵחַ אֶת הַהִתְרַחֲשׁוּת, וְנֶחֱרֶדֶת מֵרַעַשׁ שֶׁהוֹלֵךְ וּמִתְגַּלְגֵּל לְעֶבְרָהּ.
הָיִיתִי בָּאָה לְבַקֵּר אוֹתוֹ מִדַּי עֶרֶב, מִתְחַזָּה לְלָקוֹחָה עִם עֲגָלָה בִּזְמַן שֶׁהוּא סִידֵּר בַּקְבּוּקֵי יַיִן עַל הַמַּדָּפִים, אוֹ מַעֲדַנֵּי חָלָב בַּמְּקָרֵר.

------------------
אַחֵר כך הִרְכִּיב אוֹתָם שׁוּב וְאָמַר לְזִקְנָה: סַפְּרִי לָנוּ כֵּיצַד קָנִית דַּיּוֹ, וַאֲנַחְנוּ נִכְתַּב עָלֶיךָ סֵפֶר וְנָתַן לָךְ דַּיּוֹ.
טָאצֶ'ר אָמַר לְךָ לַעֲנוֹת, אַחֶרֶת בַּחַיִּים לֹא הָיִיתָ פּוֹתַחַת אֶת הַדֶּלֶת, אָמַרְתִּי.
קָרְאָה הַיַּלְדָּה הַקְּטַנָּה, "קְחִי אוֹתִי אִיתְּךָ!
בְּפִיגּוּר מְסוּיָּם הִיא נוֹתֶנֶת לִצְוָוחָה לְהִימָּלֵט מִפִּיהָ, צְוָוחָה שֶׁהִיא אֵינָהּ מְזַהָה כְּשֶׁלָּהּ.
הִיא לֹא מְנַסָּה לַעֲצוֹר בַּעֲדָהּ, כִּי הִיא יוֹדַעַת שֶׁזֶּה יִהְיֶה לַשָּׁוְוא.

------------------
יוֹם אֶחָד אֶגַּשׁ לַמַּרְצָה וְאוֹמֵר לָהּ מָה דַּעְתִּי.
רָמַת הַשְּׁלִיטָה בַּשָּׂפָה אֵצֶל עוֹלָה אֲשֶׁר שֵׁירְתָה שֵׁירוּת סָדִיר גְּבוֹהָה מַשְׁמָעוּתִית מִמִּי שֶׁלֹּא שֵׁירְתָה.
כָּל רוֹעָה יוֹדַעַת לִסְפּוֹר עִיזִּים.
בַּקְּרִיסָה נֶהֶרְגָה כּוֹרָה אַחַת.
לֹא צָרִיךְ לְהַחֲזִיר אֶת הַכֶּסֶף הַיּוֹם, אָמְרָה הַמָּלְוָוה.
אַתְּ לֹא הַמְּמוּנָּה!
לֹא יִיתָּכֵן שֶׁמַּנְחָה לֹא תַּכִּיר אֶת הַסְּטוּדֶנְטִים שֶׁלָּהּ.

------------------
בִּטְרִיקָה קוֹלָנִית, מַרְאָה לָהּ, לַחוּצְפָּנִית, סָגְרָה אַחֲרֶיהָ אֶת הַדֶּלֶת.

------------------
הִיא כְּבָר עוֹלָה לַבֵּן שֶׁלָּהּ עִם זֶה עַל הָעֲצַבִּים.
הִיא רוֹצָה לִלְחוֹץ עָלָיו?
מֵרִיקִיטָה, אַתְּ בּוֹכָה?
שָׁוָוה הִיא אֶת כָּל הַזָּהָב בְּתֵבֵל, וִיקָרָה מִכָּל אַבְנֵי הָאוֹדֶם שֶׁבִּידֵי מְלָכִים.
אֵין הִיא שָׁוָוה מֵאָה שְׁחוּקָה שֶׁל כֶּסֶף.
אֵין אֵל אַחֵר, אֶלָּא רַק הָרְאִי שֶׁאַתְּ רוֹאָה, כִּי זֶהוּ רְאִי הַחָכְמָה.

------------------
כָּל פַּעַם הִיא שׁוֹכַחַת אֵיפֹה שָׂמָה אֶת הַמַּפְתְּחוֹת אוֹ מִתְקַשָּׁה לְהִיזָּכֵר בְּשֵׁם שֶׁל מִישֶׁהוּ שֶׁלָּמַד אִיתָּהּ בַּתִּיכוֹן.
הִיא מִשְׁתַּהָה, וְאָז מִתְמַתַּחַת, מְבַקֶּשֶׁת לַחֲזוֹר לְעַצְמָהּ.
אֲנִי לֹא יוֹדַעַת הִיא מִתְוַודָּה, מַשְׁפִּילָה מַבָּט.
יָצָאתִי מֵהַמַּיִם, גּוֹרֶרֶת בְּקוֹשִׁי אֶת רַגְלַי, מִתְקַשָּׁה בִּתְנוּעָה, כְּבֵדָה מֵהַקּוֹר.
הִיא מִתְכַּסָּה בַּסָּדִין וּמִסְתּוֹבֶבֶת בְּרוֹב טַאקְט לְעֵבֶר הַחַלּוֹן.

------------------
אָז לְבַסּוֹף הֵם שָׁאֲלוּ אֶת הַנְּסִיכָה הַקְּטַנָּה דְּנִיסָה לְאָן הִיא הוֹלֶכֶת כְּשֶׁהִיא רוֹצָה לְהַשִּׂיג תַּפּוּחִים.
אֲבָל כְּשֶׁהָפַכְתִּי אוֹתָהּ בַּחֲזָרָה, מִימוֹ עֲדַיִין הָיְיתָה שָׁם, חוֹלָה כְּמִקּוֹדֶם.
מִימוֹ נִרְאֲתָה חוֹלָה גַּם עַל הַצָּג שֶׁל בֵּקִי.
אַחַת בְּרִיאָה, הַשְּׁנִיָּיה חוֹלָה.
יַלְדָּה רָזָה שֶׁנָּפְלוּ לָהּ שְׁתֵּי שִׁינַּיִים קִדְמִיּוֹת הִבִּיטָה בְּנוֹ, תּוֹהָה מָה אֲנַחְנוּ עוֹשִׂים.

------------------
אַחַר כָּךְ נַדְיָה מַרְאָה לוֹ סֵפֶר גָּדוֹל עִם תְּמוּנוֹת וּמַסְבִּירָה: "זֶה סוּס, זֹאת קַנָרִי, זֶה רוֹבֶה."
הִיא לֹא הָיְתָה מוּדַעַת לְמַאֲמָץ כָּלְשֶׁהוּ, אֲבָל כְּאֵב חַד בְּמִפְרְקֵי יָדֶיהָ הֵעִיד עַל כָּךְ שֶׁהִיא מְנַסָּה לְשַׁחְרֵר אוֹתָן.
מִן הַשַּׁעַר "שֶׁלָּנוּ" יָצְאָה אִישָּׁה זְקֵנָה, נִשְׁעֶנֶת עַל מַקְלָה, לְעֵבֶר הַשַּׁעַר "שֶׁלָּהֶם", וְהִתְחִילָה לַחֲצוֹת אֶת "שֶׁטַח הַהֶפְקֵר" כְּשֶׁהִיא מַפְנָה מִזְּמַן לִזְמַן אֶת רֹאשָׁהּ וּמְנַפְנֶפֶת בְּיָדָהּ, וּמִתְקַדֶּמֶת הָלְאָה.

------------------
וּבִשְׂפָתֶיהָ הַפְּתוּחוֹת לְמֶחֱצָה, הַמְּפִיחוֹת חַיִּים, נִרְאֲתָה כִּמְצַפֶּה לִתְשׁוּבָה כָּלְשֶׁהִי, לְהַצְהָרַת אֱמוּנִים מִצִּדִּי, אַךְ דּוֹמֶה שֶׁלֹּא מָצְאָה אֶת שֶׁיִּיחֲלָה לִקְרוֹא בְּהַבָּעַת פָּנַיי, כִּי מַבָּטָהּ הַמְּחַפֵּשׂ וְהַמָּתוּחַ הִתְקַדֵּר, עַפְעַפֶּיהָ נִפְתְּחוּ וְנִסְגְּרוּ, וּפִתְאוֹם נֶעֶמְדָה לְיַד הַחַלּוֹן, הִפְנְתָה לִי אֶת גַּבָּהּ, הִצְמִידָה אֶת מִצְחָהּ בְּכָל כּוֹחָהּ אֶל הַתְּרִיס, וְגוּפָהּ הִזְדַּעְזֵעַ כָּל כָּךְ מִבֶּכִי דּוֹמֵם אַךְ עַז לְהַחֲרִיד, עַד
------------------
אֲבָל כְּשֶׁרָאִיתִי אוֹתְךָ יוֹשֶׁבֶת פֹּה לֹא יָכוֹלְתִּי שֶׁלֹּא לַחֲשׁוֹב עַל זֶה שֶׁרָצִיתִי שֶׁאִמָּא שֶׁלִּי תִּשְׁפּוֹךְ כְּבָר אֶת מֵסִיר הַלָּק.
אֲנִי לֹא אוֹמֶרֶת שֶׁאַתְּ חַיֶּיבֶת לְסַפֵּר לָהּ אֵיךְ זֶה הָיָה הַכְּאֵב הֲכִי נוֹרָא שֶׁהִרְגַּשְׁתָּ בְּחַיֶּיךָ וְאֵיךְ מֵאוֹתוֹ יוֹם ואילך אַתְּ תָּמִיד זוֹעֶמֶת עַל הַטֶּבַע, כִּי הַטֶּבַע הוּא שֶׁעִיצֵּב נָשִׁים בַּדֶּרֶךְ הַזֹּאת, עִם פֶּצַע תְּמִידִי פָּתוּחַ שֶׁבֶּאֱמֶת כָּל דֶּבִּיל יָכוֹל לְנַצֵּל רַק עַל יְדֵי זֶה שֶׁיְּמַלֵּא אוֹתוֹ.

------------------
צְרִיכָה הָיִית לִרְאוֹת בְּאֵיזוֹ תְּבוּנָה כִּלְכַּלְתִּי אֶת צַעֲדֵי – בְּאֵיזוֹ זְהִירוּת – בְּאֵיזוֹ רְאִייַּת-הַנּוֹלָד – בְּאֵיזוֹ צְבִיעוּת מְחוּשֶּׁבֶת נִיגַּשְׁתִּי לִמְלָאכָה!
וְאַתְּ לֹא רָאִית זֹאת?
הָיִית צְרִיכָה לִרְאוֹת אֶת הַפַּרְצוּף שֶׁלּוֹ, מִסְכֵּן.
הוּא נִרְאֶה מְחוּנָּךְ, אֲבָל אֲנִי לֹא חוֹשֶׁבֶת שֶׁהוּא כְּמוֹ שֶׁאַתְּ הָיִית רוֹצָה שֶׁהוּא יִהְיֶה.

------------------
חַדְלֵי לְךָ מִלַּחֲשׁוֹב עַל גֶּבֶר שֶׁיֵּשׁ לוֹ הַהֲגִינוּת שֶׁלֹּא לֶאֱהוֹב אוֹתְךָ עוֹד, וְאִם אַתְּ רוֹצָה לַעֲשׂוֹת עַמִּי טוֹבָה אַחֲרוֹנָה, שַׁלְּחִי לִי בִּמְהֵרָה חֲצִי לִיטֶר שֶׁל חוּמְצָה צִיאָן, שֶׁיִּיתֵּן לָךְ בְּחֵפֶץ לֵב, לְפִי הַמְלָצָתִי, מַזְכִּיר הָאָקָדֶמְיָה לְמַדָּעִים; הוּא כִּימַאי מוּכְשָׁר מְאוֹד.
פָּנָיו שֶׁל בַּעֲלָהּ הֶחֱוִוירוּ מְעַט, אַךְ לְבַסּוֹף אָמַר: "בְּסֵדֶר, אֶתֵּן לְךָ אֶת הַכֶּסֶף.
הָעִיקָּר שֶׁתַּשִּׂיגִי לְעַצְמְךָ שִׂמְלָה יָפָה לְנֶשֶׁף.

------------------
אֲבָל לִפְנֵי שֶׁהִגַּעְתִּי לְכָאן רָאִיתִי אֶת הָאִמָּא הַזְּקֵנָה שֶׁלִּי שֶׁקָּמָה כְּדֵי לִרְאוֹת אוֹתִי כְּשֶׁיָּצָאתִי, הִיא הֶעֱמִידָה פָּנִים שֶׁהִיא מְנַקָּה אֶת הַלּוּל, כְּאִילּוּ הַלֵּב שֶׁלָּהּ גִּילָּה לָהּ, וּבְחַיֵּי אֱלוֹהִים שֶׁאֲנִי אֶרְצַח אוֹתְךָ כְּמוֹ כֶּלֶב כְּדֵי שֶׁהַקְּשִׁישָׁה הַטּוֹבָה הַזֹּאת לֹא תִּבְכֶּה.
הִיא לוֹבֶשֶׁת רַק אֶת הַיְּקָרוֹת שֶׁבִּשְׂמָלוֹת וּמְצַפֶּה שֶׁגֶּבֶר צָעִיר יְחַיֵּיךְ וְיָבוֹא לְבַקָּרָה.

------------------
בְּעוֹד חוֹדֶשׁ אֲנִי מְתַחְתָּן, אִם אַתְּ אוֹהֶבֶת אוֹתִי כְּמוֹ שֶׁאַתְּ אוֹמֶרֶת תָּמִיד, זֶה צָרִיךְ לְשַׂמֵּחַ אוֹתְךָ, כֵּיווֹן שֶׁאֲנִי עוֹשֶׂה זֹאת לְמַעַן אוֹשְׁרִי.
אִישָּׁה אַחֶרֶת שֶׁיֹּאמַר לָהּ: "אֲנִי אוֹהֵב אוֹתְךָ!

------------------
הַזָּקֵן מַצִּיעַ לָךְ אֶת כָּל כַּסְפּוֹ, רְכוּשׁוֹ וּנְכָסָיו אִם תּוֹאִילִי לָחוֹן אֶת בְּנוֹ הַצָּעִיר.
וְכוּלָּם, בְּהוֹשִׁיטָם אֶת יְדֵיהֶם לְעֵבֶר מִי שֶׁהָיְתָה חַיֶּיבֶת לְהַצִּיל אֶת מִשְׁפַּחְתָּם מִתְּהוֹם הַנְּשִׁייָּה, נִשְׁמְעוּ כְּחוֹזְרִים עַל דִּבְרֵי הָאָב: "בִּתִּי, הָאֵין בָּךְ אוֹתָהּ נִמְרָצוֹת סְפָרַדִּית וּרְגִישׁוּת אֱמֶת?"
וְעַתָּה, מַרְקִיזָה, הַכִּי בְּלִי פַּחַד, כִּי אֵין בָּךְ רְבָב.
הוּא מִתְחַנֵּן לְפָנַיִךְ שֶׁתּוֹאִילִי לַעֲרוֹף אֶת רָאשֵׁיהֶם שֶׁל הָאֲצִילִים.

------------------
שָׁכַחְתָּ שֶׁיֵּשׁ לְךָ חָבֵר, תַּגִּידִי?
תַּגִּידִי מָה אַתְּ פְּסִיכִית", פָּנְתָה לִתְהִילָּה, "מָה עִם קוֹבִּי, יֵשׁ לְךָ חָבֵר, קוֹבִּי, אַל תִּפְלִי בַּפַּח שֶׁל הַיְּצוּר הַזֶּה.
אַתְּ גַּם לֹא רוֹצָה שֶׁאָבוֹא לֶאֱסוֹף אוֹתְךָ?
חַכִּי, אֲנִי אוֹמֵר, "בְּחַיֶּיךָ, חַכִּי רֶגַע.
אַתְּ בִּכְלָל לֹא נִרְאֵית כְּמוֹ בַּתְּמוּנָה שֶׁלְּךָ!
זֶה בִּשְׁבִילְךָ, שֶׁתִּשְׁמְעִי כְּשֶׁקּוֹרְאִים לָךְ שַׁי-לָהּ.
הַזָּקֵן שֶׁלִּי בְּאַשְׁדּוֹד עִם חִיתּוּלִים שֶׁתֵּדְעִי לְךָ, הִיא אָמְרָה.

------------------
בַּקַּיִץ, בַּחוֹפֶשׁ הַגָּדוֹל, נוּרִית הָיְתָה מִתְלַוָּוה אֵלָיו מִפַּעַם לְפַעַם.
אֲנִי שׁוֹמַעַת אוֹתָהּ מִזְדַּהָה, שׁוֹאֶלֶת מָה קָרָה.
מִפַּעַם לְפַעַם נִכְנְסָה לַחֲדַר-הַשֵּׁינָה וְהִמְשִׁיכָה בְּשִׁיטוּטָהּ הָאִילֵּם לְאוֹרֶךְ הַמִּיטָּה, מִשְׁתַּהָה רֶגַע בְּכָל אֶחָד מִקְּצוֹתֶיהָ לְהַבִּיט בְּבַעֲלָהּ.
אִישָּׁה אַחַת אָמְרָה, "יֵשׁ לִי דּוֹקְטוֹרָט, וּלְעִתִּים קְרוֹבוֹת אֲנִי מִתְפַּתָּה לַחֲנוֹת שָׁם."

------------------
וְעַכְשָׁיו, מָה דַּעְתֵּךְ שֶׁתָּשִׂימִי אֶת הַצַּלְעִיּוֹת עַל הָאֵשׁ?
אַתְּ גָּזַזְתְּ אֶת שַׂעֲרוֹתַיִךְ?
אַתְּ אוֹמֶרֶת שֶׁשְּׂעָרֵךְ אֵינֶנּוּ?
אֵינֵךְ אוֹהֶבֶת אוֹתִי בְּאוֹתָהּ מִידָּה, בֵּין כֹּה וָכֹה?
נַהֲגִי עִמִּי יָפֶה, כִּי לְמַעֲנֵךְ הָלַךְ וְאֵינֶנּוּ.
תְּנִי לִי אֶת שְׁעוֹנֵךְ.
הַקָּצִין הֶחֱוִויר, אַךְ אָזַר עוֹז לְהֵיחָפֵז וְלוֹמַר: "הַגֵּנֵרָל יַעֲנִיק לָךְ אֶת חַיַּיִךְ בְּמַתָּנָה אִם תִּינָּשְׁאִי לִי!"
אֵינֵךְ צְרִיכָה לְשַׁלֵּם עַל נִבְזוּתָם שֶׁל הַסְּפָרַדִּים.

------------------
הַסִּיפּוּר הַזֶּה הוּא בִּשְׁבִילְךָ – פּוּרְקָן, תֵּרַפְּיָה, אוֹטוֹתֵרַפְּיָה, מָה אַתְּ רוֹצָה מִמֶּנִּי?
צְאִי לְבָתֵּי הַזִּיקּוּק עַל חֶשְׁבּוֹנְךָ הַפְּרָטִי וְהָבִיאִי לִי דּוּגְמָה.
צְאִי לְאָן שֶׁמִּתְחַשֵּׁק לְךָ.
גַּם אִם אֲנַחְנוּ לֹא תָּמִיד מַסְכִּימוֹת עִם הַדֵּעוֹת שֶׁלְּךָ, מוֹסִיפָה וּמְחַיֶּיכֶת לְעֶבְרָהּ.
יוֹתֵר מֵהַמְּסָרִים עַצְמָם, חָשׁוּב לָנוּ שֶׁיִּרְאוּ אוֹתְךָ, שֶׁיָּבִינוּ שֶׁגַּם אַתְּ אִתָּנוּ.
אֲנִי מְבַקֶּשֶׁת מִכָּל אַחַת כָּאן, מִתְחַנֶּנֶת, תִּשְׁמְרִי עַל עַצְמְךָ.

------------------
הִיא לֹא טוֹעָה כִּמְעַט אַף פַּעַם בְּמִבְחָנִים בְּמָתֵמָטִיקָה.
הַיְּפֵהפִיָּה הַנִּרְדֶּמֶת הָיְיתָה טוֹוָה זָהָב מִקַּשׁ.
אֲנִי מִתְיַיפָּה אֶל מוּל הַמַּרְאֶה תּוֹךְ שֶׁאֲנִי מִתְאַפֶּרֶת.
שָׂרַת הַבִּיטָּחוֹן מְמַפָּה אֶת הָאִיּוּמִים עַל הַמְּדִינָה.
הָאִמָּא מִתְחָרָה עַל הָעוּגִיּוֹת.
הַבּוֹחֶנֶת חוֹנָה בַּחֲנָיַית נָכִים.
הִילָּה הָיְיתָה גֵּאָה בְּהֶשֵּׂגֶיהָ בַּשָּׁנָה הָאַחֲרוֹנָה.
הָאִמָּא מַרְשָׁה לַיֶּלֶד לִשְׁתּוֹת מִיץ כְּשֶׁהוּא חוֹלֶה.

------------------
וְאֶת יְכוֹלָה לְהוֹכִיחַ שֶׁזֶּה הָאַרְנָק שֶׁלָּךְ?
אַתְּ וְהָאִמָּא שֶׁלְּךָ וְאִם הָאֱלוֹהִים!
אִימָּא הָיְיתָה אוֹמֶרֶת לִי: "תִּתְבַּיְּישִׁי לְךָ, מְבוּגָּרִים לֹא בּוֹכִים.
אִמָּא, הֲרֵי אָמַרְתִּי לֵךְ!
אִמָּא, אֶפְשָׁר לָקַחַת אֶת הַצָּעִיף הָאָדוֹם שֶׁלָּךְ?

------------------
מוֹתֶק, אַתְּ תִּסְּעִי לְךָ בְּשֶׁקֶט, כּוֹתֶבֶת לָהּ אָנִיטָה, "אֲנַחְנוּ בְּסֵדֶר, אוֹ קֵיי?
אָז אַתְּ נוֹסַעַת, שֶׁבָּהּ אָנִיטָה וְאָמְרָה, "תִּזָּהֲרִי לְךָ".
וְטֵלֵפוֹנִים, שֶׁעָלוּ הָמוֹן כֶּסֶף אָז רַק לִפְעָמִים, תְּתָאֲרִי לְעַצְמְךָ.
אַתְּ תִּישְׁנִי, יֵשׁ לְךָ הַיּוֹם עוֹד נְסִיעָה אֲרוּכָּה".

------------------
אִילּוּ הָיִיתִי בֶּן שְׁבַע-עֶשְׂרֵה, אָמַר אֶלְיָקוּם בְּלִבּוֹ, הָיִיתִי יָכוֹל, כְּמוֹ כְּלוּם, לִקְפּוֹץ עָלֶיךָ, יָשָׁר עַל הָרִצְפָּה, בִּמְקוֹם שֶׁאַתְּ יוֹשֶׁבֶת, וְהָיִיתִי יָכוֹל לְחַבֵּק אוֹתְךָ בְּכוֹחַ כָּזֶה, שֶׁהָיִיתָ מִתְפָּרֶקֶת לַחֲתִיכוֹת.
– אַל תַּחְשְׁבִי שֶׁהָיִיתִי מִתְבַּיֵּישׁ בְּךָ מִפְּנֵי שֶׁאַתְּ מְבוּגֶּרֶת, אוֹ אֲפִילּוּ מִפְּנֵי שֶׁאַתְּ צוֹלַעַת.
תִּרְאִי, עַכְשָׁיו אֲנִי אָרוּץ רַק בִּשְׁבִילְךָ, בְּלִי אַף אֶחָד.

------------------
לְהוֹכִיחַ שֶׁהִיא מִשְׁתַּוָּוה אֵלַי בְּמָה שֶׁהִיא יוֹדַעַת, אִם לֹא לְמַעְלָה מִזֶּה.
אֲנִי גָּרָה בְּאֵזוֹר הַכְּפָר, וּמִתְקַשָּׁה לִמְצוֹא אֶת עַצְמִי בֶּעָרִים הַגְּדוֹלוֹת.
הִיא מִתְפַּתָּה לְהַגִּיד "תִּשְׁאַל אוֹתָם".
בְּשָׁלָב כָּלְשֶׁהוּ הַמִּתְמַחָה הַתּוֹרָנִית הִגִּיעָה וְשָׁאֲלָה הַאִם אֲנִי צָרִיךְ אֵיזוֹ עֶזְרָה.
בְּקִיצּוּר, כְּשֶׁאַתְּ מִתְפַּנָּה תַּגִּיעִי.

------------------
וְאָז הִיא אוֹמֶרֶת: "אֲנִי זוֹכֶרֶת אוֹתָךְ עוֹד שֶׁהָיִית כָּזוֹ קְטַנָּה."
מָה הִיא קְשׁוּרָה עַכְשָׁיו הֲשׁוּתָּפָהּ הַמְּטוּמְטֶמֶת הַזֹּאת, מָה הִיא שׁוֹלַחַת לִי אֶלֶף הוֹדָעוֹת, שָׁמַעְתִּי שֶׁאַתְּ בְּבוּאֵנוֹס וְשָׁמַעְתִּי שֶׁאַתְּ בְּזֶה וְעָשִׂית שָׁם טְרֶקִים וְאָכַלְתָּ צ'וֹרִיסוֹס דַּי מְטוּמְטֶמֶת תִּסְתְּמִי.
וּבְכָךְ, חֲבֵרָה צֶ'סְנוֹקוֹבָה, כֵּיוָֹון שֶׁהָיִית אִתִּי בַּחֲזִית לֹא חוֹדֶשׁ וְלֹא חוֹדְשַׁיִים, הֲרֵי לֹא תּוּכְלִי לְפַקְפֵּק.

------------------
חוּאָנִיטָה הַטּוֹבָה שֶׁלִּי, הֲרֵי לֹא רָצִיתָ שֶׁאֶהְיֶה שַׁיֶּיכֶת לָאִישׁ, וּבְכֵן!
אֲנִי מִתְעָרֵב שֶׁלֹּא אַתְּ פָּקַדְתְּ עַל כָּךְ.
אַתְּ יָדַעְתְּ, אָמַר לָהּ.
לְאַחַר שֶׁהָלְכוּ מַהֲלַךְ פַּרְסָה מִן הָעִיר, הִכָּה הַדַּיָּיג הַצָּעִיר עַל חָזֵהוּ וְאָמַר אֶל נִשְׁמָתוֹ, "מַדּוּעַ אָמַרְתְּ לִי לַהֲרוֹג אֶת הַסּוֹחֵר וְלָקַחַת אֶת זְהָבוֹ?
אָכֵן, תּוּכְלִי לְהִיכָּנֵס, אָמַר הַדַּיָּיג הַצָּעִיר, "כִּי וַדַּאי סָבַלְתְּ הַרְבֵּה בְּיָמִים בָּהֶם הָלַכְתְּ בָּעוֹלָם בְּלִי לֵב."

------------------
אֲנִי לֹא צָרִיךְ לְהִישָּׁאֵר כָּאן אִם אַתְּ לֹא רוֹצָה.
כְּשֶׁהֶחֱזַרְתִּי אֵלֶיהָ שׁוּב אֶת מַבָּטִי רָאִיתִי שֶׁהִיא בּוֹכָה.
מָה אַתְּ רוֹצָה מִמֶּנִּי, מָה?
רוֹצָה אֲנִי – וַאֲנִי בּוֹרַחַת לִסְטָארוֹדוֹב!
כָּל כָּךְ אֲנִי רוֹצָה לְהַגִּיד לָהּ אֶת זֶה אֲבָל בְּסוֹף רַק מְהַנְהֶנֶת, כְּדֵי לְצַמְצֵם כַּמָּה שֶׁיּוֹתֵר אֶת הַזְּמַן שֶׁל הַדִּיבּוּרִים וּלְהַרְוִויחַ עוֹד כַּמָּה דַּקּוֹת שֶׁל שֶׁקֶט.

------------------
אִילּוּ הָיִיתִי בֶּן שְׁבַע-עֶשְׂרֵה, אָמַר אֶלְיָקוּם בְּלִבּוֹ, הָיִיתִי יָכוֹל, כְּמוֹ כְּלוּם, לִקְפּוֹץ עָלַיִיךְ, יָשָׁר עַל הָרִצְפָּה, בַּמָּקוֹם שֶׁאַתְּ יוֹשֶׁבֶת, וְהָיִיתִי יָכוֹל לְחַבֵּק אוֹתָךְ בְּכוֹחַ כָּזֶה, שֶׁהָיִית מִתְפָּרֶקֶת לַחֲתִיכוֹת.
הָיִית נִבְהֶלֶת לִרְאוֹת כַּמָּה אֲנִי חָזָק.
אַתְּ, אַתְּ הָיִית הַיַּלְדָּה הַקְּטַנָּה שֶׁכָּל הַזְּמַן הִבִּיטָה אֵלַי, כָּל הַזְּמַן, כְּשֶׁיָּשַׁבְתִּי שָׁם וְקָרָאתִי, וְכוּלָּם נִרְדְּמוּ?

------------------
קָרְאָה, "הַשְׁבִּיעִי נא אֶת חוּאַנִיטוֹ שֶׁיְּמַלֵּא אַחַר פְּקוּדוֹתַיִךְ בְּנֶאֱמָנוּת."
חָוּאַנִיטָה, יַקִּירָתִי שֶׁלִּי, אָמְרָה בְּנִימָה שֶׁל עֲלִיצוּת בְּעוֹדָהּ כּוֹרֶכֶת אֶת זְרוֹעוֹתֶיהָ סְבִיב צַוָּוארָהּ וְנוֹשֶׁקֶת לְעַפְעַפֶּיהָ, "אִילּוּ רַק יָדַעְתְּ כַּמָּה מָתוֹק יִהְיֶה הַמָּוֶות מִיָּדַיִךְ."
עָלַיִךְ לְאֵזוֹר עוֹז, אָמַר לָהּ אָחִיהָ פִילִיפּ, "שֶׁאִם לֹא כֵּן, מִשְׁפַּחְתֵּנוּ תִּיכָּחֵד."

------------------
בֵּינְתַיִים, אַתְּ רוֹצָה לְהָבִיא מַשֶּׁהוּ לְבוֹסִית שֶׁלָּךְ?
מָה אִתְּךָ, סַבְתָּא, נָפַלְתָּ מֵהַיָּרֵחַ?
הָעוֹבְדִים הָיוּ צוֹחֲקִים: מָה אִתְּךָ, גְּבֶרֶת, מֵאֵיפֹה לָנוּ לַחְמָנִיּוֹת צָרְפָתִיּוֹת?
נִבְהֲלָה הַזְּקֵנָה פִּי כַּמָּה, אֲבָל לְמַרְבֵּה הַמַּזָּל נִמְצָא לוֹ אִישׁ טוֹב לֵב, תָּפַס אֶת יָדָהּ וְאָמַר: מָה אִתְּךָ, כָּאֵלּוּ נָפַלְתָּ מֵהַיָּרֵחַ!
הֲרֵי יָכְלוּ לְדֶרֶס אוֹתְךָ, וּמָשַׁךְ אוֹתָהּ לַצַּד הַשֵּׁנִי.

------------------
הִיא הָיְתָה מַשְׁקָה אֶת עֲלֵיהֶם, מַחֲלִיפָה אֶת הַחוֹל, כּוֹרַעַת עַל בִּרְכֶּיהָ כְּדֵי לְהֵיטִיב לַעֲדוֹר אֶת הָאֲדָמָה.
וּמָה הִיא רוֹצָה מִמֶּנּוּ, מְנַסָּה לְהִתְגָּרוֹת בּוֹ, אוּלַי?
אֲנִי טוֹרֶקֶת אֶת הַדֶּלֶת שֶׁל הָאוֹטוֹ הַשָּׂכוּר וְהַמְּצֻ'וקְמָק שֶׁלָּנוּ, מְנַסֶּה לְדַמְיֵין אֵיךְ תִּהְיֶה הַהַרְגָּשָׁה לִהְיוֹת הַבְּעָלִים שֶׁל חֵפֶץ כָּל כָּךְ גָּדוֹל וְיָקָר.

------------------
אִשְׁתּוֹ שׁוֹאֶלֶת מִתּוֹךְ שֵׁינָה, מְנַסָּה לְנַעֵר מֵעָלֶיהָ אֶת הַסִּיפּוּר שֶׁעוֹד לֹא מַמָּשׁ נוֹגֵעַ בָּהּ, בְּכַוָּונָה נְחוּשָׁה, שֶׁכָּרֶגַע נִפְעַר בָּהּ סֶדֶק רִאשׁוֹן, לְהַמְשִׁיךְ לִישׁוֹן עוֹד שָׁעָה וְאוּלַי שְׁעָתַיִים, שֶׁבְּרוֹב הַשַּׁבָּתוֹת הִיא מַצְלִיחָה.
גַּם הִיא מְנַסָּה לְהָזִיז אֶת הַבָּחוּר.

------------------
הִיא מִתְקַשָּׁה לִישׁוֹן, וְלֹא מְשַׁנֶּה כַּמָּה הִיא אוֹכֶלֶת, הִיא לֹא מַעֲלָה בְּמִשְׁקָל.
יוֹפִי, אָמַרְתִּי לְעַצְמִי, עַכְשָׁיו כָּל עַם יִשְׂרָאֵל יִרְאֶה אֶת הַיַּלְדָּה הַזֹּאת, מִתְעַנָּה עַל כִּיסֵּא הַגַּלְגַּלִּים, וּבֶטַח מִישֶׁהוּ יַגִּיד מִילָּה לַהוֹרִים שֶׁלָּהּ.
אִילּוּ הָיְתָה אֵיזוֹ יַלְדָּה יְפֵהפִיָּה, לְמָשָׁל, מִתְקַשָּׁה קְצָת בְּמַשֶּׁהוּ – אָז מָה?
אֵינִי יְכוֹלָה לִשְׁפּוֹט אוֹתָם, כִּי הָיִיתִי אַחַת מֵהֶם, אֲבָל אֲנִי מִזְדַּהֶה אִיתָּם בְּכָל לִבִּי.

------------------
#  בַּאֲרוּחַת הָעֶרֶב מוּלִי עָדִין הָיְתָה בְּמַצַּב רוּחַ רַע, אֲבָל אִמָּא שֶׁלָּהּ אָמְרָה, "אֲנִי מִצְטַעֶרֶת שֶׁנִּפְטַרְתִּי מֵהַכּוֹבַע הָאָדָם שֶׁלָּךְ, חֶמֶד.
וְגַם זֹאת רָצָה לוֹמַר: – אַל תַּחְשְׁבִי שֶׁשָּׁכַחְתִּי אוֹתְךָ.
אַתְּ יוֹדַעַת שֶׁאֲנִי מַכִּיר אוֹתְךָ?

------------------
הָיִיתָ מְסַלֶּקֶת אוֹתִי מִבֵּיתְךָ וְלֹא רוֹצָה לִשְׁמוֹעַ יוֹתֵר אַף מִילָּה, וְזוֹ הָיְיתָה תְּגוּבָה מוּבֶנֶת בְּהֶחְלֵט.
עַכְשָׁיו אֲנִי מֵבִין שֶׁהָיִיתִי צָרִיךְ לִשְׁמוֹר גַּם אֶת זוֹ שֶׁל הַבֵּן שֶׁלְּךָ, אֲבָל אֲנִי מְקַוֶּוה שֶׁתַּאֲמִינִי לָמָּה שֶׁאֲנִי אוֹמֵר.
עַכְשָׁיו אַתְּ מְבִינָה לָמָּה הָיִיתִי מוּכְרָח לְסַפֵּר לְךָ אֶת כָּל זֶה.
תִּתְבַּיְּשִׁי לְךָ, סְמַרְטוּתִי.
מָה קוֹרֶה, סְמַרְטוּתִי, כֻּלָּךְ דְּבִיקָה!
נִרְאֶה לִי שֶׁאַתְּ מֵרוּחָהּ כֻּלָּךְ בְּרַבָּה!

------------------
מָאדָאם אָטִינָה חָפְנָה אֶת פָּנַיי בְּיָדֶיהָ וְאָמְרָה: "יַלְדָּתִי, עִם הָעֵינַיִים הַיָּפוֹת שֶׁלָּךְ אֶת עֲתִידָה לִשְׂרוֹף אֶת הַלֵּב לְהַרְבֵּה גְּבָרִים."
כְּלוּם לֹא אָמַרְתִּי לָךְ שְׁמָהּ שֶׁאַתְּ טוֹעֶה לִרְאוֹת בּוֹ שִׁיגָּעוֹן אֵינוֹ אֶלָּא חַדּוּת-יֶתֶר שֶׁל הַחוּשִׁים?
אַתְּ רוֹשֶׁמֶת לְעַצְמֵךְ?
רְאִי כַּמָּה בְּרִיאָה הִיא דֶּרֶךְ סִיפּוּרִי, בְּאֵיזֶה קוֹר-רוּחַ אֲגוֹלֵל לָךְ אֶת הַסִּיפּוּר כּוּלּוֹ.
אַל לָךְ – לֹא תַּחֲזִי בַּדָּבָר הַזֶּה!

------------------
אַךְ לוּ הָיוּ לְךָ בָּנִים, עַתָּה אוֹ קוֹדֶם-לָכֵן, וּבְגִינָם הָיִית יְכוֹלָה לָדַעַת מָה עַזָּה הָאַהֲבָה אֲלֵיהֶם, כִּי אָז נִדְמֶה לִי שֶׁיָּכוֹלְתִּי לִבְטוֹחַ שֶׁתִּסְלְחִי לִי עַל חוּצְפָּתִי, וְלוּ גַּם מְעַט.
עָנָה הָאַבִּיר: מָה שֶׁאָכַלְתְּ הָיָה לַאֲמִיתּוֹ שֶׁל דָּבָר לִבּוֹ שֶׁל אָדוֹן גּוֹאִילְיֶילְמוֹ גְּוַארְדֶּאסְטָאנְיוֹ, שֶׁאַתְּ, אִשָּׁה בּוֹגְדָנִית, אוֹהֶבֶת כָּל-כָּךְ.
אַתְּ הָיִית רְגִילָה לַעֲבוֹד עַל הַגַּגּוֹת כְּמוֹ חֲתוּלָה, הוּא אָמַר לָהּ.

------------------
וּמִיכָאֵלָה הִנִּיחָה אֶת יָדֶיהָ עַל הַיַּלְדָּה הַמִּתְהַוָּוה וְחָשָׁה בְּאוֹשֶׁר.
הָיְיתָה מִתְרָאָה עִם אָבִיהָ בִּקְבִיעוּת בְּבֵית-הָעֵסֶק.
הַכַּלְבָּה שֶׁלִּי מִתְקַשָּׁה לָרֶדֶת אֶת הַמַּדְרֵגוֹת הַמְּעַטּוֹת.
אֲנִי מִתְקַשָּׁה לְחַבֵּר מִילִּים בּוֹדְדוֹת לִכְדֵי מִשְׁפָּט, אֲבָל בְּסוֹף אֲנִי מַצְלִיחָה.
הִיא מִתְקַשָּׁה לְהַאֲמִין שֶׁהוּא נוֹתַר אָדִישׁ מוּל זֶה.
בְּשֶׁל רִפְיוֹן הַהַחְלָמָה, הָיְתָה מִתְגָּאָה כִּמְעַט בְּחוֹסֶר-הָאוֹנִים שֶׁבָּהּ.

------------------
הִיא בָּאָה לְבֵית הַסֵּפֶר לְהַחֲלִיף זְמַנִּית אֶת הַמּוֹרָה שֶׁחָלְתָה; וּמֵאַחַר שֶׁלֹּא הָיְתָה מוּסְמֶכֶת לְהוֹרָאָה הֶעֱסִיקָה אֶת הַיְּלָדִים בַּסִּיפּוּרִים שֶׁקָּרְאָה בְּאוֹזְנֵיהֶם מִתּוֹךְ סֵפֶר קָטָן, עַב כֶּרֶס, עֲגַלְגַּל כְּמוֹתָהּ.
אַתְּ יוֹדַעַת שֶׁאֶת הַמּוֹרָה הַמַּחֲלִיפָה שֶׁקָּרְאָה לָנוּ סִיפּוּרִים?

------------------
חֲבוּרַת פּוֹחְחִים, הַנִּיצּוֹלִים מִטִּאְטוּאֵי מַטְאֲטֵא הַשֶּׁמֶשׁ הַלּוֹהֵב, יָשְׁבָה אֶל הַקִּיר בְּפִינַּת הַכִּיכָּר, כְּשֶׁהִיא מְנַסָּה אוֹתוֹ כָּל פַּעַם מֵחָדָשׁ בַּהֲטָלַת כַּפְתּוֹרִים וּמַטְבְּעוֹת, כְּאִילּוּ אֶפְשָׁר הָיָה לִקְרוֹא מִתּוֹךְ הוֹרוֹסְקוֹפּ עִיגּוּלֵי הַמַּתֶּכֶת הַלָּלוּ אֶת סוֹדוֹ הָאֲמִיתִּי שֶׁל הַקִּיר, שֶׁנְּקִיקִים וּפְרָצוֹת כִּיסּוּהוּ כִּכְתָב חַרְטוּמִּים.

------------------
חַדְלֵי מִן הַמִּשְׂחָק הַיַּלְדוּתִי, אָמַר הֶרְמָן וְלָקַח אֶת יָדָהּ, "אֲנִי שׁוֹאֵל אוֹתְךָ בַּפַּעַם הָאַחֲרוֹנָה: אַתְּ מוּכָנָה לוֹמַר לִי מֵהֶם שְׁלוֹשֶׁת הַקְּלָפִים שֶׁלָּךְ?
אוּלַי בְּחַדְרְךָ, כְּשֶׁאַתְּ יָשַׁנְתָּ: הוּא מְסוּגָּל לָזֶה.
מָה לָךְ, יַקִּירָתִי!
בכחך לַעֲשׂוֹת אֶת חַיַּי לִמְאוּשָּׁרִים, הִמְשִׁיךְ הֶרְמָן, "וְזֶה לֹא יַעֲלֶה לָךְ בִּכְלוּם: אֲנִי יוֹדֵעַ שֶׁאַתְּ יְכוֹלָה לְנַחֵשׁ שְׁלוֹשָׁה קְלָפִים בִּרְצִיפוּת.

------------------
רָאִינוּ אֶת הַמַּסּוֹקִים שֶׁחָגוּ מֵעַל הַכְּפָרִים בֶּהָרִים, וְאֵיךְ נֶאֱלַצְתֶּם לְהִתְחַבֵּא בֵּין הָעִזִּים, וְאֵיךְ נָגַעְתְּ בְּעַמּוּד הַגְּבוּל לְטוּרְקִיָּה שָׁלוֹשׁ פְּעָמִים כְּדֵי לְהִשְׁתַּכְנֵעַ שֶׁאַתְּ לֹא מְדַמְיֶינֶת.
רָאִינוּ אֶת הַשֶּׁלֶג הַכְּחַלְחַל וְאֶת הַמְּנוֹפִים וְאֶת גֶּדֶר הַתַּיִל, שֶׁבִּכְלָל לֹא הָיוּ מַמָּשִׁיִּים, וַאֲנַחְנוּ יוֹדְעִים שֶׁהָעֱלִי הֲכִי חָזָק הִיכָּה אוֹתָךְ, וָאטָאן, כִּי הָיִית תְּשׁוּשָׁה כָּל כָּךְ.

------------------
הַמּוֹרָה דִּיבְּרָה בְּמִילִּים מְסוּבָּכוֹת שֶׁקָּשֶׁה לְהָבִין בְּעִבְרִית.
לְיָאסוֹקִיצִ'י הָיָה נִדְמֶה שֶׁהִיא מְנַסָּה לַעֲצוֹר בַּעֲדוֹ בְּלֹא אוֹמֵר.
כְּמוֹ בְּכָל הַבְּקָרִים, גַּם בַּבּוֹקֶר הַזֶּה מַפְנֶה רָאִיסָה מִדֵּי פַּעַם אֶת רֹאשָׁהּ אֶל הַמּוֹשָׁב הָאֲחוֹרִי וּמְחַיֶּיכֶת אֶל גֶנְיָה חִיּוּךְ מְעוֹדֵד.
אֶת הַכֶּסֶף הִיא תַּחֲזִיר לְרַאִיסַה בְּקַלּוּת – הִיא מְזַהָה אֶת הַמַּטְבְּעוֹת לְפִי הַגּוֹדֶל.

------------------
סָבְתָא זוֹהֳרָהּ לֹא הֵבִינָה הַרְבֵּה מִמָּה שֶׁהַמּוֹרֶה אָמְרָה, אֲבָל יָשְׁבָה זָקוּף וְהַצַּמָּה שֶׁלָּהּ הָיְתָה מְתוּחָה יָפֶה יָפֶה מֵאֲחוֹרֵי הָרֹאשׁ.
בְּסוֹף, כְּשֶׁהַמּוֹרֶה סִייְּמָה לְדַבֵּר, הִיא אָמְרָה: "אִמָּא שֶׁלָּהּ חוֹלֶה, אֲבָל שֶׁהִיא תִּהְיֶה בְּרִיאָה אֲנִי יַגִּיד לָהּ שֶׁתִּתֵּן לָהּ עוֹנֶשׁ".
כָּכָה יָצָא לִי לִרְאוֹת אֶת סִימָה מְרִימָה אֶת הַחֲצָאִית, וּמַרְאָה לַבְּעָלִים שֶׁל הָאִמָּהוֹת, שֶׁאֲנִי לֹא רוֹצֶה לְהַגִּיד אֶת הַשֵּׁם שֶׁלָּהֶם עַכְשָׁיו.

------------------
תִּבְחֲרִי לְךָ אֵיזֶה תַּכְשִׁיט שֶׁתִּרְצִי, אָמְרָה.
הִיא נִרְאֵית נֶהְדֶּרֶת עָלֶיךָ," אָמַר בַּעֲלָהּ.
נְשִׁיקוֹת בִּשְׂפָתַיִים זֶה רַק בֵּין מִי שֶׁאוֹהֲבִים בֶּאֱמֶת, לֹא מַשֶּׁהוּ שֶׁעוֹשִׂים סְתָם עִם כָּל אֶחָד, גַּם יֵשׁ לְךָ חָבֵר שֶׁאַתְּ אוֹהֶבֶת, לֹא?
סַמְדִּי, נִדְהַמְתִּי לִשְׁמוֹעַ אֶת קוֹלָהּ שֶׁל תְּהִילָּה, "עַכְשָׁיו זוֹ אֲנִי שֶׁמְּבַקֶּשֶׁת מִמְּךָ, צְאִי מֵהַמַּיִם, אַתְּ לֹא רְצוּיָה כָּאן", אָמְרָה כְּשֶׁהִיא מַשְׁפִּילָה אֶת רֹאשָׁהּ.

------------------
 שֶׁהַמִּילִּים נֶעֶתְקוּ מִפִּי וְלֹא הֵעַזְתִּי לָגַעַת בָּהּ.
הֵכַנּוּ דַּף מְסָרִים, פְּשׁוּטִים, בְּרוּרִים, אֲנִי בֶּאֱמֶת חוֹשֶׁבֶת שֶׁאִם תְּעַיְּינִי בּוֹ תִּרְאִי בְּעַצְמֵךְ שֶׁיֵּשׁ בּוֹ הַרְבֵּה הִיגָּיוֹן, שֶׁאַתְּ מִזְדַּהֶה עִם לֹא מְעַט מֵהַתְּכָנִים.
אַחַר כָּךְ יִשְׁכְּחוּ וְיִצְעֲקוּ וְיִצְוְוחוּ זֶה אֶל זֶה וְזֶה עַל זֶה בַּצְּלִילִים הַגְּרוֹנִיִּים הָאֵלֶּה שֶׁהָאִישָּׁה הַמְּבוּגֶּרֶת שֶׁאִיתָּם מִתְקַשָּׁה כָּל כָּךְ לִלְמוֹד לַהֲגוֹתָם כַּהֲלָכָה.

------------------
לַחֲזוֹר לְעוֹלָם שֶׁבּוֹ הִיא יְכוֹלָה לִשְׁכַּב לָנוּחַ בְּלִי שֶׁעוֹד זוּג עֵינַיִים צוֹפָה בָּהּ כְּשֶׁהִיא מַרְשָׁה לְעַצְמָהּ לְהִינָּתֵק, לְהִיסָּחֵף.
סְפֶּנְסֶר תָּהָה אִם הִיא מְנַסָּה לְהִימָּלֵט מִ'תְּחוּם הַמּוֹשָׁב' שֶׁנִּקְבַּע בְּרוּסְיָה.
הִיא מְנַסָּה לְהִתְאוֹשֵׁשׁ, אוֹמֵר "אַבָּא קוֹרֵא לָהּ 'הַחִיפּוּשִׁית'."
הִיא מַרְאָה לְבֵּרְט אֵיךְ לִסְרוֹק אֶת זִיפֵי הַמַּתֶּכֶת בַּעֲדִינוּת לְאוֹרֶךְ רַגְלָיו הָאֲחוֹרִיּוֹת שֶׁל הַכֶּלֶב.

------------------
אוֹתוֹ מְשׁוֹרֵר מָצָא מָזוֹר לְיֵיאוּשׁוֹ בְּאַהֲבָתוֹ לְאִישָּׁה נְשׂוּאָה אֲשֶׁר מַעֲרִיכָה אֶת חוֹכְמָתוֹ וְנוֹתֶנֶת לוֹ הַשְׁרָאָה בְּהַדְלִיקָהּ אֶת לְבָבוֹ בְּלַפִּיד הָאַהֲבָה, וַאֲשֶׁר בָּחֲנָה וּבְיוֹפֶיהָ מַרְאֶה לוֹ אֶת הַיָּפִים וְהַנִּצְחִיִּים שֶׁבַּבִּיטּוּיִים.
הִיא מְגַלָּה כִּי נִשְׁמָתָהּ מְרַחֶפֶת סְבִיב הָאִישׁ שֶׁהִיא מַעֲרִיצָה בְּכָל נִימֵי אַהֲבָתָהּ הָאֲמִיתִּית הַטְּהוֹרָה וְהַיָּפָה.

------------------
הַתַּנּוּר בְּסֵדֶר גָּמוּר, זוֹ אַתְּ, גְּבֶרֶת מוּמְחִית, סָגַרְתְּ אֶת הַחִיפּוּי.
אִם לֹא הָיִית מִתְעוֹרֶרֶת וּמְעִירָה אוֹתָנוּ הָיִינוּ נֶחְנָקִים בְּוַודָּאוּת, אִיגוֹר הִצִּית סִיגַרְיָה וְהוֹשִׁיט לִי, "צָרִיךְ לְהַטְבִּיעַ אֶת הַתַּנּוּר הַזֶּה בַּנָּהָר."
וְתָשִׂימִי לֵב שֶׁלֹּא יַעְקְבוּ אַחֲרַיִיךְ כְּמוֹ שֶׁאַתְּ הָיִית עוֹקֶבֶת אַחֲרַיי, צָחַק אָבִי בְּקוֹל רַע.
וְאַתְּ יָדַעְתְּ כָּל הַזְּמַן שֶׁאֲנִי קוֹרֵא אֶת הַסְּפָרִים הָאֵלּוּ?

------------------
הִיא הָיְיתָה חוֹלָה רַק יוֹמַיִים.
אֲנִי אוֹמֶרֶת לַיַּלְדָּה, "אַתְּ רוֹצָה לַחֲזוֹר הַבַּיְתָה?"
אֲנִי רוֹצָה לִהְיוֹת הַצּוֹפָה הַשְּׁחוֹרָה.
יָעֵל מִסְתּוֹבֶבֶת, וּכְשֶׁהִיא רוֹאָה אֶת הַיַּלְדָּה, הַסּוֹמֶק נֶעֱלַם מִפָּנֶיהָ וְהִיא מִתְיַישֶּׁבֶת בְּרִפְיוֹן עַל הָאָרֶץ וּמַתְחִילָה לִתְלוֹשׁ עֲשָׂבִים וּנְצָרִים קְטַנִּים מִפֹּה וּמִשָּׁם.
הֵם יוּכְלוּ לִרְאוֹת אוֹתָהּ מִמֶּרְחָק רַב כְּשֶׁהִיא חוֹצָה אֶת הַשָּׂדוֹת אוֹ מִתְקַדֶּמֶת בְּדֶרֶךְ.

------------------
תָּאֲרִי לְעַצְמְךָ, חָשַׁבְתִּי שֶׁהוּא מְטוּנָּף בִּפְנִים אֲבָל הִתְבָּרֵר שֶׁלֹּא כָּל כך, רַק מְאוּבָּק.
תִּרְאִי אֵיזוֹ שָׂפָה עֲשִׁירָה יֵשׁ לְךָ, וָרֶה, יָשָׁר רוֹאִים שֶׁאַתְּ סְטוּדֶנְטִית לְבַלְשָׁנוּת.
נוּ, אָז תָּאִירִי אֶת עֵינֵינוּ, מָה הָאֱמֶת לְדַעְתְּךָ?
זַנְזוֹנֶת אַתְּ בְּעַצְמֵךְ!
אָז לָמָּה אַתְּ סוֹחֶבֶת עַל עַצְמְךָ אֶת גֶּרְבֶּנְשְׁצִ'יקוֹב וְלֹא אֶת מָמוֹנוֹב?
אֲנִי מְחַבֶּבֶת אוֹתְךָ, אַלֵּיף, וַאֲנִי רוֹצֶה שֶׁתַּצְלִיחִי.

------------------
לִפְנֵי הַמִּבְחָן הַסְּטוּדֶנְטִית מְמַצָּה כָּל דַּקָּה.
הַמְּשָׁרֶתֶת מִשְׁתַּחֲוָוה לְמַלְכָּה.
שִׁירָן קוֹנָה בְּזָארָה כָּל שָׁבוּעַ אַחֲרֵי שֶׁהִיא מְסַיֶּימֶת אֶת הָאִימּוּן שֶׁלָּהּ.
הָאִישָּׁה תּוֹלָה אֶת תִּקְווֹתֶיהָ בְּבַת זוּגָתָהּ.
------------------
וּבְאוֹמְרוֹ זֹאת קִיבֵּל אוֹתָהּ בִּשְׁפֵלוֹת-רוּחַ בְּתוֹךְ בֵּיתוֹ וְהוֹלִיכָה מִמֶּנּוּ אֶל גִּינָּתוֹ, וְכֵיווֹן שֶׁלֹּא הָיָה שָׁם אִישׁ לְאָרֵחַ עִמָּהּ לַחֶבְרָה, אָמַר לָהּ: "גְּבִירְתִּי, הֱיוֹת שֶׁאֵין כָּאן אִישׁ זוּלַת הָאִשָּׁה הַטּוֹבָה הָאָתְּ, רַעְיָיתוֹ שֶׁל הָאִיכָּר, תִּהְיֶה לָךְ הִיא לַחֶבְרָה בְּשָׁעָה שֶׁאלך לְהַשְׁגִּיחַ כִּי יַעַרְכוּ אֶת הַשּׁוּלְחָן.
אַתְּ פּוֹחֶדֶת שֶׁאֲנִי אוֹכֶל אוֹתְךָ?
אַתְּ יֵשׁ לְךָ דָּם חַם!

------------------
הָרוֹזֶנֶת שָׁתְקָה; הֶרְמָן הִמְשִׁיךְ: "בִּשְׁבִיל מִי אַתְּ שׁוֹמֶרֶת אֶת הַסּוֹד שֶׁלְּךָ?
גַּלִּי לִפְנֵי אֶת סוֹדְךָ!
חִשְׁבִי: אַתְּ זְקֵנָה; לֹא נוֹתַר לְךָ עוֹד זְמַן רַב לִחְיוֹת.
אֲנִי מוּכָן לְקַבֵּל עָלַי אֶת חֲטָאָךְ: רַק גַּלִּי לִי אֶת סוֹדְךָ.
חִשְׁבִי עַל כך שֶׁאוּשְּׁרוּ שֶׁל אָדָם מָסוּר בְּיָדַיִיךְ; שֶׁלֹּא רַק אֲנִי, כִּי אִם בְּנֵי, בְּנֵי-בְּנֵי וְנִינִי, יְבָרֲכוּ אֶת זִכְרְךָ וִיכַבְּדוּ אוֹתוֹ כְּזָכָר קָדוֹשׁ.

------------------
מָה שְׁמֵךְ, גְּבֶרֶת יְקָרָה וּמַקְסִימָה?
גֶּ'נְטֶלְמֵן אֶחָד הִגִּיעַ לִרְאוֹתֵךְ, הָעַלְמָה, אָמְרָה.
אַחֲרֵי שָׁנָה הִיא אָמְרָה לַמַּלְכָּה – "יַקִּירָתִי, אֶת חַיֶּיבֶת לַעֲשׂוֹת מַשֶּׁהוּ כְּדֵי לְהַסִּיחַ אֶת דַּעְתֵּךְ."
הִיא אָמְרָה – "הַשִּׁיבִי אֶת הַשְּׁטָחִים הַיְּרוּקִּים, הוֹד מַלְכוּתְךָ."
אַתְּ דַּיְּקָנִית כְּתָמִיד, בְּדִיּוּק כְּפִי שֶׁנִּכְתַּב בְּהַמְלָצוֹתֶיךָ.

------------------
אֵיךְ קוֹרְאִים לָךְ, אֵיפֹה אַתְּ גָּרָה, הַפָּרָמֵדִיק עֲדַיִין צוֹעֵק, דָּבֵק בַּמְּשִׂימָה שֶׁלּוֹ, אֲבָל הַנַּעֲרָה לֹא מְגַלָּה אֵיךְ קוֹרְאִים לָהּ וְגַם לֹא אֵיפֹה הִיא גָּרָה, "אֲנִי בִּכְלָל לֹא מִכָּאן," הִיא אוֹמֶרֶת.
אִם הַנַּעֲרָה נִבְהֲלָה מֵהַמִּילָּה מִשְׁטָרָה, הִיא לֹא מַרְאָה אֶת זֶה.
בַּהֲלִיכָה זְהִירָה הִיא מְנַסָּה לִשְׁמוֹר שֶׁהַקָּפֶה לֹא יִישָּׁפֵךְ בַּדֶּרֶךְ אֶל הַשּׁוּלְחָן, וּמְנִיחָה אֶת הַכּוֹס בְּמָקוֹם שֶׁבּוֹ הָעַכְבָּר לֹא יָכוֹל לְהַגִּיעַ אֵלֶיהָ.

------------------
יֵשׁ לָהּ בַּיִת בְּכָל עִיר בִּירָה, הִיא אוֹכֶלֶת מִצַּלָּחוֹת זָהָב, וְאִם הִיא רוֹצָה, הִיא יְכוֹלָה לִמְנוֹעַ אֶת רוּסְיָה מִלָּצֵאת לְמִלְחָמָה.
לֹא יָדַעְתִּי אִם הָאוּסְבֵּרְג הָיְיתָה רוֹצָה שֶׁשְּׁמָהּ יוּזְכַּר.
הִיא פּוֹנָה אֵלַיי פִּתְאוֹם.
שִׂיחַקְתִּי אַתְּ הַתּוֹמֶכֶת, הַמְּבִינָה, הַמְּכִילָה, עַד שֶׁעֶרֶב אֶחָד שָׁאַל: "אַתְּ רוֹצָה לְנַסּוֹת?"
לְהַסְתִּיר אוֹתוֹ, הִיא עוֹנָה לְעַצְמָהּ, "זֹאת הַדֶּרֶךְ הַיְּחִידָה, לְהַסְתִּיר אוֹתוֹ כָּאן בַּבַּיִת שֶׁלָּנוּ."

------------------
בִּמְקוֹם לִכְתּוֹב דְּרָמוֹת, מוּטָב שֶׁהָיִית בּוֹלַעַת מָרָק אוֹקְרוֹשְׁקָה קַר וְנִרְדֶּמֶת בַּמְּזָוֶוה.
וְאַחַר כָּךְ אָמַרְתִּי בְּבִיטָּחוֹן: אַתְּ מֵעוֹלָם לֹא הָיִית בְּמַדְּרָאס.
מָה עִם הַסֵּפֶר הַהוּא שֶׁאָמַרְתְּ שֶׁאַתְּ לֹא מְבִינָה, נִיסְּתָה לְהִתְעַשֵּׁת, אֲבָל הַחֶדֶר צָף לָהּ מוּל הָעֵינַיִים וְאָן אָמְרָה שֶׁאוּלַי זֶה לֹא בְּסֵדֶר שֶׁהִנִּיחָה לָהּ לִשְׁתּוֹת.
תּוֹדָה לָאֵל שֶׁחָזַרְתָּ בְּשָׁלוֹם, אַתְּ רוֹאֶה, טוֹב שֶׁשַּׂמְתִּי לְךָ אֶת הַשּׁוּם הַזֶּה בַּתִּיק.

------------------
בַּפַּעַם הַבָּאָה שֶׁאַתְּ נוֹסַעַת לִנְיוּ-יוֹרְק, אָמַר לִסְטֶיְיסִי, "קְחִי אוֹתִי אִיתְּךָ.
קְחִי אוֹתִי אִיתְּךָ, הוּא אָמַר.
תַּעֲשִׂי לְךָ מַשֶּׁהוּ.
הַנַּעֲרָה אָמְנָם לֹא הָיְתָה בַּשְׁלָנִית גְּדוֹלָה, אֲבָל נִיכָּר בָּהּ שֶׁהִיא בַּעֲלַת רָצוֹן, וּדְרִישׁוֹתֶיהָ הָיוּ צְנוּעוֹת כָּל-כך עַד שָׁמַרְתָּ אוּבַּן אָמְרָה לְבַסּוֹף: "טוֹב, אֶקַּח אוֹתְךָ.
אַתְּ אֵינְךָ שׁוֹכַחַת אוֹתָהּ!

------------------
אַפְּרוֹפּוֹ טַעַם חָדָשׁ שֶׁל שׁוֹקוֹלָד, אֲנִי מַעֲלָה לְעַצְמִי, מֵרוֹב תִּסְכּוּל עִם הַמַּחֲשָׁבָה הַזֹּאת, אֶת הַדּוֹפֶק, בְּאֵיזוֹ זְכוּת הָאֲנָשִׁים הָאֵלֶּה מַרְשִׁים לְעַצְמָם לְהַחְלִיט שֶׁזֶּה שֶׁאֲנִי יוֹשֶׁבֶת לְיָדָם אוֹמֵר שֶׁאֲנִי אִיתָּם.
בַּחֲצִי לֵב אֲנִי מוֹדָה גַּם שֶׁחַמִּים וּבָטוּחַ לִי בֵּינֵיהֶם, כְּבָר לֹא קְשׁוּרָה אֲבָל עֲטוּפָה מִכָּל הַכִּיוּוּנִים.

------------------
בְּסוֹף גַּם זֶה לֹא הוֹלֵךְ אָז הִיא מְנַסָּה בְּעוֹד כַּמָּה מְקוֹמוֹת אֲבָל בְּסוֹף הִיא מְוַותֶּרֶת וַאֲנִי אוֹמֶרֶת לָהּ שָׁלוֹם וְיוֹצֵאת.
בְּסוֹף הִיא כְּאִילּוּ שָׂמָה לֵב שֶׁאֲנִי שָׁם בִּכְלָל וּבְלִי לְהַגִּיד שָׁלוֹם הִיא פָּשׁוּט נִדְבֶּקֶת לִי לַבֶּטֶן, מְנַסָּה לְהִתְחַבֵּר.
אֲנִי נִשְׁאֶרֶת בַּבַּיִת וּמְנַסָּה לַחֲשׁוֹב מָה לַעֲשׂוֹת.
אִשְׁתִּי הָיְתָה נוֹשֶׁפֶת כָּל פַּעַם וּמְנַקָּה אֶת אַפָּהּ, כְּאִילּוּ יֵשׁ לָהּ נַזֶּלֶת.

------------------
אִם אַתְּ רוֹאָה פִּתְאוֹם בָּאִי בַּרְנָשׁ יָחֵף וּמוּזְנָח מִתְגַּנֵּב בֵּין הַשִּׂיחִים קָרוֹב לַיָּם, וּבוֹ בַּזְּמַן אַתְּ מַבְחִינָה בְּמֶרְחָק שֶׁל פָּחוֹת מִקִּילוֹמֶטֶר וָחֵצִי מֵהַחוֹף בְּמִפְרָשִׂית מְלֵאָה כּוּשִׁים מִתְרַחֶקֶת בִּמְהִירוּת, לֹא הָיִית חוֹשֶׁבֶת שֶׁהָאִישׁ הַזֶּה נָפַל מֵהַשָּׁמַיִם, נָכוֹן?
הָיִית צְרִיכָה לִרְאוֹת אוֹתָם קָמִים לִתְחִיָּיה וְנִרְתָּמִים שׁוּב לָעֲבוֹדָה שֶׁלָּהֶם!
אַתְּ מְעִזָּה לְהַכְחִישׁ שֶׁמִּשָּׁם בָּאת?

------------------
אִם רַק תִּגְרְמִי לָהּ לִשְׁאוֹל שְׁאֵלָה אַחַת עַל אוֹפְנַת הַחוֹרֶף בְּשַׁרְווּלֵי מְעִילִים, אֲנִי מַבְטִיחַ לָךְ בִּשְׁבִילָהּ סִיכּוּי שֶׁל אַחַת לְחָמֵשׁ, בִּמְקוֹם אַחַת לְעֶשֶׂר.
יֵשׁ לִי מַשֶּׁהוּ לְסַפֵּר לָךְ, עַכְבָּרִית לְבָנָה, אָמְרָה.
אִם לֹא תִּתְנַהֲגִי כְּמוֹ שֶׁצָּרִיךְ, אוֹמֵר אֲנִי, "אֲנִי לוֹקֵחַ אוֹתָךְ יָשָׁר הַבַּיְתָה."
אַתְּ יוֹדַעַת, סָמֶנְתָה, אוֹמֵר בִּיל, "אֲנִי עָמַדְתִּי לִימִינֵךְ בְּלִי לְהָנִיד עַפְעַף בִּרְעִידוֹת אֲדָמָה, בְּאֵשׁ וּבְמַיִם."

------------------
כָּכָה אַתְּ מִתְחַזָּה לְחוֹלָה כְּדֵי לֹא לָלֶכֶת לְבֵית הַסֵּפֶר...
לְבַסּוֹף הִיא מִתְרַצָּה וּמְעַוָּוה אֶת פָּנֶיהָ בְּחִיּוּךְ.
כָּל-כָּךְ מְעַט מִילִּים נֶהֱגוּ עַל שְׂפָתֶיהָ שֶׁלִּפְעָמִים הָיְיתָה מִשְׁתַּהָה זְמַן רַב עַד שֶׁיָּכְלָה לַעֲנוֹת עַל שְׁאֵלָה פְּשׁוּטָה כְּמוֹ: "בַּת כַּמָּה אַתְּ?"
אֲבָל עִם יְלָדֶיהָ הַקְּטַנִּים הָיְיתָה מִשְׁתַּטָּה, בִּצְלִילִים, מַמְצִיאָה מִילִּים וְרוֹקֶדֶת עִימָּם בְּשָׂפָה שֶׁל הֲבָרוֹת.

------------------
הָאֶצְבָּעוֹת עַכְשָׁיו שֶׁאֲנִי רוֹאָה אֶת זֶה פִּתְאוֹם אֲנִי מְבִינָה שֶׁאֵין לָהֶן מִפְרָקִים.
הִיא רוֹצָה לָצֵאת מֵהַתִּיק וּלְהִסְתַּכֵּל מֵהַחַלּוֹן אָז הַפַּעַם אֲנִי פָּשׁוּט נוֹתֶנֶת לָהּ וְזֶהוּ.
כָּל מִינֵי שְׁטוּיוֹת שֶׁבֶּטַח יֵשׁ לָהּ בָּרֹאשׁ וּבַבּוֹקֶר הִיא עֲיֵיפָה מִדַּי וְלֹא רוֹצָה לָלֶכֶת אִיתִּי לָאוּנִיבֶרְסִיטָה.
אֲנִי תּוֹלָה אֶת הַמְּעִיל וְאוֹמֶרֶת שָׁלוֹם.
כָּל מָה שֶׁאַתְּ רוֹצָה לַעֲשׂוֹת זֶה לְחַפֵּשׂ פֹּה עוֹד סְפָרִים.

------------------
עַל כֵּן, בּוֹאִי וּקְחִינָה, וְכָל אוֹצְרוֹת הָעוֹלָם יִהְיוּ שֶׁלָּךְ.
בּוֹאִי עִמִּי וְאַרְאֶה אוֹתָן לָךְ.
אַל תַּטְרִידִי אֶת עַצְמֵךְ, אֶלָּא בּוֹאִי אִתִּי אֶל עִיר אַחֶרֶת.
הַנִּיחִי לִי עַתָּה לְסַפֵּר לָךְ עַל כְּאֵב הָעוֹלָם, אוּלַי תִּשְׁמְעִי לִי.
בִּרְחִי עַל נַפְשֵׁךְ, קָרְאָה הַנְּשָׁמָה.
עָלָיִךְ לִרְקוֹד אִתִּי, נַעֲרָה נֶחְמָדָה, מִלְמְלָה, וְחִייְּכָה אֵלֶיהָ בְּדַבְּרָהּ.
וְאִם יַנְשׁוּף יְדַבֵּר אֵלַיִךְ, אַל תָּשִׁיבִי לוֹ תְּשׁוּבָה.

------------------
זֹאת הַמִּסְעָדָה שֶׁלִּי, אֲנִי הַבּוֹסִית, אָז אֲנִי שׁוֹאֶלֶת אוֹתוֹ שְׁאֵלוֹת, מְנַסָּה לְפַעְנֵחַ אוֹתוֹ.
אֲנִי שׁוֹמַעַת אוֹתָהּ מְזַהָה אוֹתִי.

------------------
הִיא נִזְכֶּרֶת בְּשֵׁם שֶׁל סֵפֶר: "אֲנִי שׂוֹנֵאת אֶת הַצַּוָּואר שֶׁלִּי", מְנַסָּה לְהִיזָּכֵר אִם קָרְאָה אוֹתוֹ בִּכְלָל.
אֲבָל הִיא מְזַהָה אֶת הַחוּלְצָה, הַכְּתֵפַיִים, הַהֲלִיכָה.
כְּשֶׁהִיא מִתְיַישֶּׁבֶת בְּעוֹמֶק הַבָּמָה הִיא מְזַהָה אֶת כַּרְמֶל.
הִנֵּה, הִיא מְזַהָה פִּתְאוֹם אֶת כַּרְמֶל, אֲבָל כַּרְמֶל לֹא מִסְתַּכֶּלֶת עָלֶיהָ.
הִיא מַרְגִּישָׁה אֶת הַגּוּף שֶׁלָּהּ, מְרִיחָה אוֹתָהּ, וְרַק אָז מַפְנָה אֶת הַמַּבָּט.

------------------
אֲנַחְנוּ מָצָאנוּ אֶת הַבָּחוּר הַבּוֹקֶר וְאֶת הַשְּׁנַיִים בְּלִי הָעֵדִים, אֲבָל אַתְּ מַמָּשׁ רָאִית אֶת זֶה קוֹרֶה.
אַתְּ רָאִית אֶת גְּלָאס הַצָּעִיר לִפְנֵי שֶׁהָיִיתִי שֵׁרִיף, וְאֶת הַבַּת שֶׁל סִילְבָה.
לֹא הָיְתָה סִיבָּה לְבַקֵּר בִּמְדִינַת ווֹשִׁינְגְּטוֹן בְּאוֹתָם יָמִים, אֶלָּא אִם הָיִית חוֹטֶבֶת עֵצִים חוֹבֶבֶת, אוֹ שֶׁהִתְחַשֵּׁק לְךָ לִרְאוֹת אֶת נְהַר הֲקוֹלוֹמְבִּיָּה מְכוּסֶּה בְּמֵשָׁט שֶׁל בּוּלֵי עֵץ.

------------------
אֲנִי אַחֲזִיר לָךְ הַכֹּל בַּסּוֹף שֶׁל הַחוֹדֶשׁ כְּשֶׁאֲנִי אֲקַבֵּל אֶת הַמַּשְׂכּוֹרֶת, אַתְּ יוֹדַעַת שֶׁאֲנִי אַחֲזִיר.
אוּלַי הוּא יַסְבִּיר לָךְ עַד כַּמָּה אַתְּ חֲסֵרָה לִי.
מַדּוּעַ אֵינֵךְ אוֹכֶלֶת?
רוֹמַנְטִיקָה חַסְרַת שַׁחַר, כְּשֶׁתֵּשְׁבִי בַּכֶּלֶא, לִפְנֵי שֶׁיּוֹצִיאוּ אוֹתָךְ לַהוֹרֵג, לֹא תַּחְשְׁבִי אֶת כָּל הַדְּבָרִים הַיָּפִים הָאֵלֶּה, מוֹתֶק.
אַתְּ רוֹצֶה בֶּטַח שֶׁאַבְרִיחַ לָךְ אֶחָד כָּזֶה בַּמְּכוֹנִית שֶׁל הַקָּצִין שֶׁלִּי, מָה?

------------------
אֲבָל אֲנִי אוֹמֵר לְךָ, מוֹתֶק, אִי אֶפְשָׁר, אִי אֶפְשָׁר לִשְׂרוֹד שָׁם, עָדִיף לְהִצְטָרֵף לְמֵאוֹשָׁרִים שֶׁנָּפַל בְּחֶלְקָם הַמַּזָּל לֹא לַחֲשׁוֹב, כְּדֵי לִשְׂרוֹד צָרִיךְ לִלְמוֹד לְשַׂחֵק טֵנִיס בְּלִי כַּדּוּר וּבְלִי מַחְבֵּט.
אֲבָל אַחֲרֵי חֲשִׁיבָה קְצָרָה, הוֹסִיפָה: "בְּעֶצֶם, עָדִיף שֶׁתֵּלְכִי לְבַדְּךָ.
אֵיךְ אֲנִי יָכוֹל לְהֵירָדֵם בִּידִיעָה שֶׁאַתְּ שׁוֹכֶבֶת פֹּה לְיָדִי כְּאִילּוּ כְּלוּם, וְיֵשׁ לְךָ הַיֵּקִי עַל הַכָּתֵף שֶׁאַתְּ לֹא יְכוֹלָה לְהַסְבִּיר?

------------------
עַכְשָׁיו אִשְׁתּוֹ מְבַלָּה אֶת זְמַנָּהּ בְּסִלְסוּל שַׂעֲרוֹתֶיהָ, בִּצְבִיעַת שְׂפָתֶיהָ וּבְבִישּׂוּם גּוּפָהּ.
אִמָּא שֶׁלִּי לֹא מַרְשָׁה לִי לְשַׂחֵק עִם כּוּשׁוֹנִים, הִיא אָמְרָה לִי.
בְּעַד מֵאָה פְרַנְק לְשָׁנָה, הָיְתָה מְבַשֶּׁלֶת לָהּ וּמְנַקֶּה, תּוֹפֶרֶת, מְכַבֶּסֶת, מְגַהֶצֶת, יָדְעָה לְמַתֵּג סוּס, לְפַטֵּם אֶת הָעוֹפוֹת, לְחַבֵּץ חֶמְאָה, וְשָׁמְרָה אֱמוּנִים לִגְבִרְתָּהּ – שֶׁלֹּא הָיְתָה בְּרִיָּיה נְעִימָה דַּוְוקָא.

------------------
הַכַּבָּאִית מְכַבָּה אֶת הַשְּׂרֵיפָה.
הַחֲתוּלָה מְכַסָּה אֶת צְרָכֶיהָ.
תָּמָרָה מַמְרָה אֶת פִּי הוֹרֶיהָ.
הַגְּבֶרֶת הִתְנַהֲגָה בְּאוֹפֶן לֹא יָאֶה.
הַטַּבָּחִית מְנַפָּה אֶת הַסּוּכָּר.
הַכַּלְבָּה מִתְפַּנָּה שָׁלוֹשׁ פְּעָמִים בְּיוֹם.
הָרַקְדָנִית מִתְנַסָּה בְּסִגְנוֹנוֹת רִיקּוּד אֲחֵרִים.
הָרוֹפְאָה מַפְרָה אֶת הַבֵּיצִית בְּמַבְחֵנָה.
הַגְּבֶרֶת עוֹטָה אֶת מְעִילָהּ.
הִיא עוֹשָׂה שְׁטוּיוֹת בְּמַעְבָּדָה.
נוֹגַהּ מִתְפַּנָּה בְּשֵׁירוּתֵי הַנָּשִׁים לְעִיתִּים תְּכוּפוֹת.

------------------
אֲנִי מְגַלֶּה שֶׁהוּא אַשַּׁף בְּמִסְפָּרִים וְנוֹתֶנֶת לוֹ לְפַקֵּחַ עַל כָּל הַקְּנִיּוֹת.
זֹאת לֹא הַפַּעַם הָרִאשׁוֹנָה שֶׁאֲנִי עוֹצֶמֶת עֵינַיִים וּמְנַסֶּה לְהַעֲלוֹת בָּאוֹב אֶת הַדָּבָר שֶׁלְּדַעְתִּי אֲנִי לֹא מְסוּגֶּלֶת לִחְיוֹת בִּלְעָדָיו, אֲבָל לִפְנֵי שֶׁהוּא הִגִּיעַ נִשְׁבַּעְתִּי שֶׁזֹּאת תִּהְיֶה הַפַּעַם הָאַחֲרוֹנָה.
אֲנִי מִתְיַישֶּׁבֶת עַל רִצְפַּת חֲדַר הַקֵּירוּר, הַלִּימוֹן הָאִיּוּם בְּכַף יָדִי, וּמְנַסֶּה לִמְצוֹא הִיגָּיוֹן בַּקֶּסֶם.

------------------
ג'וֹרְגְ'יָה פְּסֶלְמֶנְצֶר הָיְתָה מִתְחַזָּה שֶׁטָּעֲנָה כִּי מוֹצָאָהּ בַּמִּזְרָח הָרָחוֹק.
רוֹרִי עָקְבָה אַחֲרֶיהָ בְּהַעֲרָצָה מֵעַל לְכוֹס הַבּוֹרְבּוֹן שֶׁלָּהּ, מִתְקַשָּׁה לְהַאֲמִין שֶׁהִיא יוֹשֶׁבֶת עִם וְזוֹבֵי אַחֲרֵי פְּעָמִים רַבּוֹת כָּל כָּךְ שֶׁרָאֲתָה אוֹתָהּ מִסְתַּלֶּקֶת לְדַרְכָּהּ בְּמוֹנִיּוֹת, וְהִרְגִּישָׁה אָז שֶׁהַדָּבָר הֲכִי חָשׁוּב בָּעוֹלָם הוּא לְהֵיעָלֵם אִיתָּהּ.
סְטֶיְיסִי אָמְרָה לוֹ אֶת שָׁם הַסּוֹכְנוּת שֶׁלָּהּ, מִתְקַשָּׁה לְהָכִיל אֶת שִׂמְחָתָהּ.

------------------
כַּעֲבוֹר חוֹדָשִׁים סְפוּרִים הוֹפִיעָה בַּמְּעָרָה, לְבַדָּהּ, תִּינוֹקֶת בּוֹכָה, אוֹחֶזֶת בְּיָדֶיהָ תִּלְיוֹן קָטָן בְּצֶבַע כֶּסֶף, חָלוּד בְּשׁוּלָיו, שֶׁחִיסֵּל אֶת מָה שֶׁנּוֹתַר מֵהָאֱנוֹשׁוּת.
הָאֱמֶת, אֲנִי לֹא יוֹדַעַת אִם הִתְעַצְבַּנְתִּי בִּגְלַל מָה שֶׁסִּיפְּרָה לִי אוֹ כִּי אֶתְמוֹל הָיָה יוֹם רִאשׁוֹן וַאֲנִי לֹא רוֹצָה שֶׁיַּפְרִיעוּ לִי בְּסוֹפֵי שָׁבוּעַ.
הִיא נִרְאֵית מְחוּנֶּכֶת, אֲבָל אֲנִי לֹא חוֹשֶׁבֶת שֶׁהִיא כְּמוֹ שֶׁאַתְּ הָיִית רוֹצָה שֶׁהִיא תִּהְיֶה.

------------------
וְאָז פִּתְאוֹם שָׁאֲלָה אֶת אִמָּהּ אִם הִיא הָיְיתָה רוֹצָה לָדַעַת מִי הִלְשִׁין עָלֶיהָ.
לְעוּמַּת זֹאת שָׁאֲלָה אֶת אִמָּהּ אִם הִיא רוֹצָה קִינּוּחַ, אֲבָל הָאֵם הֶעֱדִיפָה כּוֹס מַיִם.
בְּטוּחָה שֶׁאַתְּ לֹא רוֹצָה מָרָק?
אַחֲרֵי הַכּוֹל זוֹ הָיְיתָה אִישָּׁה שֶׁיָּדְעָה מָה הִיא רוֹצָה.
הִיא אֲפִילּוּ הָיְיתָה כִּמְעַט גֵּאָה בְּעַצְמָהּ.
הוּא תָּמִיד חָשַׁב מֵרֹאשׁ עַל פִּתְרוֹן, בְּדֶרֶךְ כְּלָל עַל יְדֵי כָּךְ שֶׁנִּיסָּה לַחֲשׁוֹב מָה יְהוּדִית הָיְיתָה רוֹצָה בְּאוֹתוֹ רֶגַע.

------------------
אָנָה מַתְחִילָה לִבְכּוֹת וְאוֹמֶרֶת שֶׁהִיא צְרִיכָה לִפְגּוֹשׁ אוֹתִי, הִיא רוֹצָה לְסַפֵּר לִי מָה קָרָה.
אֲנִי לֹא רוֹאָה טֵלֵוִויזְיָה וְלֹא שׁוֹמַעַת רַדְיוֹ, כְּדֵי לֹא לִשְׁמוֹעַ חֲדָשׁוֹת שֶׁטֶּרֶם קָרָאתִי.
הַתְּמוּנָה הָיְתָה מְעוּרְפֶּלֶת, אֲבָל הַבַּחוּרָה הַזּוֹ דּוֹמָה לָךְ כָּל כָּךְ עַד שֶׁהֶחְלַטְתִּי לִבְחוֹן אֶת הַתְּמוּנָה בִּזְכוּכִית מַגְדֶּלֶת.

------------------
כְּשֶׁשָּׁאַלְתִּי אֶת הַמְּנַקָּה בֶּן כַּמָּה אֲנִי, הִיא נִיחֲשָׁה שֶׁאֲנִי צָעִיר מִגִּילִי.
אִם הִרְצֵית פַּעַם, אַתְּ מַרְצָה.
לִהְיוֹת עוֹלָה מֵאֶרֶץ רְחוֹקָה פֵּירוּשׁוֹ לִהְיוֹת בּוֹדְדָה.
הָיִיתִי רוֹעָה, הָיִיתִי זַמֶּרֶת, הָיִיתִי כָּל מִינֵי דְּבָרִים.
שָׁאַלְנוּ אֶת הַכּוֹרָה מֵאֵיפֹה הִיא בָּאָה.
לֹא חָשַׁבְתִּי שֶׁאֶזְדַּקֵּק לְטוֹבוֹת מִמָּלְוָוה בְּרִיבִּית מְחוּרְבֶּנֶת.
אוֹרֶן כַּעַס עַל הַמְמוּנָּה הַחַשְׁדָנִית.
אֲנִי לֹא חוֹשֶׁבֶת שֶׁאֶהְיֶה מַנְחָה.

------------------
אַתְּ לֹא שׁוֹבֶרֶת אֶת הַמְּפָרֶקֶת שֶׁל מִישֶׁהִי שֶׁאֵין לָךְ רְגָשׁוֹת כְּלַפֶּיהָ.
הִיא אָמְרָה, "אַתְּ יְכוֹלָה לְהַרְאוֹת לִי מֵאֵיפֹה יוֹצֵא הַפִּיפִּי שֶׁלָּךְ?
נִתְרַחֵק מֵהַכּוֹסוֹת, הַכַּפִּיּוֹת, הַתַּצְלוּמִים שֶׁלָּךְ כְּלוֹחֶמֶת גֵּרִילָה צְעִירָה עַל הַמְּקָרֵר.

------------------
תּוֹדָה לָךְ, מְתוּקָה שֶׁלִּי, לָחֲשָׁה לָהּ בְּפוֹלָנִית, חִיבְּקָה אוֹתָהּ וְנִיסְּתָה לְהַסְדִּיר מְעַט אֶת הִתְנַשְּׁפוּיוֹתֶיהָ וְאֶת דּוֹפֶק הַלֵּב, אֲבָל אִירִינָה, שֶׁהִתְעוֹרְרָה לְגַמְרֵי, בִּיקְּשָׁה בִּתְנוּעוֹת יָדַיִים וּבִילָלוֹת לְהֵיחָלֵץ מֵחִיבּוּקָהּ וּלְהִתְקָרֵב אֶל הַתַּנּוּר.
אִיזָבֵּלָה פָּלְטָה צָעֲקָה, וְהִפְנְתָה אֶת רֹאשָׁהּ לְאָחוֹר, נֶאֱחֶזֶת בְּרוֹבֶּרְטוֹ: "מִלֶּכְתִּי", קָרָא זֶה, "אִישׁ לֹא יַעֲקוֹר אוֹתְךָ מִזְּרוֹעוֹת אֵלֶּה!

------------------
לְדַעְתִּי אַתֶּם מַגְזִימִים קְצָת בִּדְרִישׁוֹתֵיכֶם, וּבַזֶּה אֲנִי מַגִּישָׁה לָכֶם הַצָּעָה נֶגְדִּית, שֶׁאֲנִי נוֹטֶה לְהַאֲמִין שֶׁתְּקַבְּלוּ אוֹתָהּ.
אֲנִי קוֹנָה שְׂעָרוֹת, אָמְרָה הַגְּבֶרֶת.
הִיא שָׁאֲלָה אוֹתָהּ אִם הִיא רוֹצָה שֶׁתַּגִּיד לוֹ מָה קָרְאָה שֶׁכָּתוּב בַּתִּיק וְאִמָּהּ הֵשִׁיבָה שׁוּב – "לֹא".
אָז אָזְרָה אוֹמֶץ וְשָׁאֲלָה אוֹתָהּ לָמָּה הִיא לֹא רוֹצָה לְסַפֵּר לָהּ כְּלוּם.
יוֹם אֶחָד אָמְרָה לְבַעֲלָהּ שֶׁהָיְיתָה רוֹצָה לִנְסוֹעַ לְבוּקָרֵשְׂט.

------------------
מוּזָר הָיָה לָהּ לִרְאוֹת בְּמִין אוֹרַחַת פּוֹרַחַת כָּזוֹ מִתְחָרָה לַמֶּיְיגּ'וֹר הַנֶּהְדָּר שֶׁלָּהּ.
אֲנִי מִתְאַוָּוה לְמֵלַנְכּוֹלְיָה וּסְבוּרָה שֶׁבְּסוֹף טִיּוּל טִבְעִי לַחֲשׁוֹב עַל הָעֲצָמוֹת הַשְּׂרוּעוֹת מִתַּחַת לָעֵשֶׂב...
יַאֲשִׁימוּ אוֹתָךְ שֶׁאַתְּ קוֹמוּנִיסְטִית, כּוֹפֶרֶת, אָתֵיאִיסְטִית, לְאוּמָּנִית, מִתְחַזָּה לְמַאֲמִינָה, בּוֹגֶדֶת... לֹא חָשׁוּב.
אַתְּ בְּטוּחָה שֶׁאַתְּ מִתְנַקָּה כְּמוֹ שֶׁצָּרִיךְ לִפְנֵי שֶׁאַתְּ חוֹזֶרֶת הַבַּיְתָה?

------------------
הַיַּלְדָּה שֶׁמְּנַקָּה אֶת הַלּוּחַ פִסְפְסָה חֵלֶק גָּדוֹל.
הַיַּלְדָּה רוֹצָה סוּכָּרִיָּיה.
הַפְּסִיכוֹלוֹגִית מְנַסָּה לַעֲזוֹר לַמְּטוּפָּל.
עוֹרֶכֶת הַדִּין מְבַזָּה אֶת בֵּית הַמִּשְׁפָּט.
רֹאשַׁת הַמֶּמְשָׁלָה מְגַנָּה אֶת הִתְנַהֲגוּת הַשָּׂרִים כְּשֶׁהִיא כּוֹעֶסֶת.
הָאָחוֹת הַגְּדוֹלָה מִתְגָּרָה בְּאָחִיהָ הַקָּטָן.
הַמְּתַרְגֶּלֶת לֹא מִזְדַּהָה עִם סֵבֶל הַתַּלְמִידִים.
כְּשֶׁאַלּוֹנָה כּוֹעֶסֶת הִיא חוֹצָה כָּל גְּבוּל.
הָאֲשֵׁמָה מִתְוַודָּה עַל הַפֶּשַׁע.

------------------
הִיא בִּיצְּעָה לִכְאוֹרָה מְעִילָה וְהוֹנָאַת מִיסִּים שֶׁבְּגִינָהּ, בְּעֵת כְּתִיבַת שׁוּרוֹת אֵלֶּה, הִיא מְרַצָּה עוֹנֶשׁ מַאֲסָר.
הַמּוֹרָה הָיְתָה דּוֹקְטוֹרַנְטִית יְלִידַת סָמַרְקַנְד בְּשֵׁם גּוֹלְנוֹרָה.
נָכוֹן, "שָׁם" הִיא מִילָּה דֵּי שְׁגוּרָה בְּקוּרְסֵי שָׂפָה שָׁנָה א', וְהַמּוֹרָה אֲמוּרָה, כַּנִּרְאֶה, לָדַעַת אִם הִיא בְּזָכָר אוֹ בִּנְקֵבָה.

------------------
וְאָז אִמָּא מְכַבָּה לָנוּ אֶת הַטֵּלֵוִויזְיָה, וַאֲנַחְנוּ מוֹחִים, לַמְרוֹת שֶׁגַּם לָנוּ קְצָת נִמְאַס.
אִמָּא מְנַסָּה לְהַלְהִיב אוֹתִי.
הִיא מַרְאָה אֶת אֵיךְ שֶׁבְּלֶיְיק מְצַיֵּיר אֶת הַשְּׁנִיָּה שֶׁל הָאֲכִילָה מֵעֵץ הַדַּעַת.
מָה שֶׁהִיא כֵּן עוֹשָׂה זֶה מִתְקָרֶבֶת לִי לַבֶּטֶן וּמְנַסָּה לְהַרְכִּיב אֶת עַצְמָהּ עַל הַיָּרֵךְ שֶׁלִּי.
כְּשֶׁהִיא לֹא מַצְלִיחָה הִיא מְנַסָּה יָשָׁר עַל הַכָּתֵף.

------------------
הַמַּרְצֵעַ יָצָא מֵהַשֶּׁק, הִיא חוֹשֶׁבֶת מִיָּד, וּבִכְלָל לֹא בְּטוּחָה מָה זֶה מַרְצֵעַ, וַאֲבִיבָה עוֹנָה בְּעִנְיָינִיּוּת, "הֵכַנּוּ דַּף מְסָרִים, פְּשׁוּטִים, בְּרוּרִים, אֲנִי בֶּאֱמֶת חוֹשֶׁבֶת שֶׁאִם תְּעַיְּינִי בּוֹ תִּרְאִי בְּעַצְמֵךְ שֶׁיֵּשׁ בּוֹ הַרְבֵּה הִיגָּיוֹן, שֶׁאַתְּ מִזְדַּהֶה עִם לֹא מְעַט מֵהַתְּכָנִים.
וְהִיא מַרְגִּישָׁה סְחַרְחוֹרֶת, וְרוֹצֶה לִצְעוֹק אוֹ שֶׁאוּלַי הִיא בֶּאֱמֶת צוֹעֶקֶת, כַּרְמֶלִּי, תִּיזָּהֲרִי, תִּשְׁמְרִי עַל עַצְמֵךְ, בּוֹאִי הַבַּיְתָה.

------------------
נִבְהֲלָה הַזְּקֵנָה פִּי כַּמָּה, אֲבָל לְמַרְבֵּה הַמַּזָּל נִמְצָא אִישׁ טוֹב לֵב שֶׁתָּפַס אֶת יָדָהּ וְאָמַר: מָה אִיתָּךְ, כְּאִילּוּ נָפַלְתָּ מֵהַיָּרֵחַ!
אַחַר כָּךְ הִרְכִּיב אוֹתָם שׁוּב וְאָמַר לַזְּקֵנָה: סַפְּרִי לָנוּ כֵּיצַד קָנִית דְּיוֹ, וַאֲנַחְנוּ נִכְתּוֹב עָלַיִיךְ סֵפֶר וְנִיתֵּן לָךְ דְּיוֹ.
טָאצֶ'ר אָמַר לָךְ לַעֲנוֹת, אַחֶרֶת בַּחַיִּים לֹא הָיִית פּוֹתַחַת אֶת הַדֶּלֶת, אָמַרְתִּי.

------------------
אַתְּ גָּנַבְתְּ אֶת הַנַּעֲלַיִים?
וְאַתְּ, פָּנָה אֶל הַסּוֹחֶרֶת, "הָיִית יוֹעַצְתּוֹ הָעִסְקִית מִדֵּי פַּעַם."
זוֹכֶרֶת מָה אָמַרְתְּ לִי פַּעַם שֶׁעָבְרָה?
מָה אָמַרְתְּ, גְּבִרְתִּי הַצְּעִירָה?
כַּמָּה טִיפְּשָׁה אַתְּ, אֱלִישֶׁבַע, עַל שֶׁלֹּא אָמַרְתְּ לָהּ לָבוֹא אִיתָּךְ לָאִיצְטַדְיוֹן.
אִילּוּ הָיִית שָׁם, בַּכִּיתָּה הַהִיא, אַתְּ צְרִיכָה לִהְיוֹת עַתָּה בַּת שְׁתֵּים-עֶשְׂרֵה אוֹ שְׁלוֹשׁ-עֶשְׂרֵה.

------------------
כְּשֶׁחָזְרָה הַבַּיְתָה, רָצָה לְחַלּוֹן – הַקָּצִין עָמַד כְּמִקוֹדֶם, נוֹעֵץ בָּהּ אֶת עֵינָיו: הִיא הִתְרַחֲקָה, מִתְעַנָּה בְּסַקְרָנוּת וְנִרְגֶּשֶׁת מִתְּחוּשָׁה שֶׁהָיְתָה לָהּ חֲדָשָׁה לְגַמְרֵי.
הִיא לֹא הֶאֱמִינָה לְמַרְאֵה עֵינֶיהָ, מִתְקַשָּׁה לְהָבִין אֵיךְ יְכוֹלָה הָיְתָה לְהַחֲטִיא.
אַתְּ מִתְגָּרֶה בִּי, שִׁיגֵּר אֶרְנַנְדֶּס.
כְּשֵׁם שֶׁהַבַּחוּרָה הִתְאַוְּותָה לַמַּיִם, כָּךְ מִתְאַוָּוה הִיא לְשִׂיחָה.

------------------
לֹא סִיפַּרְתִּי לָךְ כְּדֵי שֶׁלֹּא תִּדְאֲגִי.
לָמָּה מִינּוּ אוֹתָךְ לִהְיוֹת הָאַחְרָאִית?
הֵבַנְתִּי אֶת הַשְּׁאֵלָה שֶׁמּוּסְוֵוית בַּשְּׁאֵלָה הַזֹּאת: מָה הָיִית צְרִיכָה לַעֲשׂוֹת כְּדֵי שֶׁיִּמְנוּ אוֹתָךְ לְאַחְרָאִית?
תּוּכְלִי לָתֵת לִי שִׁשָּׁה פֵּנִי, אִמָּא, אָמְרָה אֵפִי, "כִּי הֵבֵאתִי לָךְ מִין חָדָשׁ."
אֶתֵּן לָךְ חֲצִי כֶּתֶר תְּמוּרָתוֹ, בֶּרְטָה נַעֲרָתִי, אָמַר בִּמְהִירוּת רַבָּה, וְאָז אָסַף אוֹתוֹ בִּזְהִירוּת בְּמִטְפַּחְתּוֹ.

------------------
בְּקוּמִי הִרְגַּשְׁתִּי חוֹלָה וְחַלָּשָׁה עַד מְאוֹד – הוֹ, עַד כְּלוֹת כָּל הַכּוֹחוֹת!
כְּלוּם לֹא אָמַרְתִּי לָךְ שְׁמָהּ שֶׁאַתְּ טוֹעָה לִרְאוֹת בּוֹ שִׁיגָּעוֹן אֵינוֹ אֶלָּא חַדּוֹת-יֶתֶר שֶׁל הַחוּשִׁים?
כִּי רְאִיתִיהָ בּוֹהָה לָרִיק בְּמֶשֶׁךְ שָׁעוֹת אֲרוּכּוֹת, וְכָל-כּוּלָּהּ דְּרוּכָה כְּאִילּוּ הָיְיתָה מַאֲזִינָה לְאֵיזֶה צְלִיל דִּמְיוֹנִי.

------------------
הִיא מוֹנָה שְׁלוֹשׁ מֵאוֹת אַרְבָּעִים וְתֵשַׁע נְקִישׁוֹת מָחוֹג וְנִשְׁמַעַת יְרִיָּיה.
לוֹ יָכוֹלְתִּי לִקְרוֹעַ אֶת הַמּוֹסְרוֹת מֵעַל מַרְפְּקִי כִּי אָז הָיִיתִי תּוֹפֶסֶת בַּמְּטוּטֶּלֶת וּמְנַסָּה לַעֲצוֹר אוֹתָהּ.
אַתְּ מְדַמָּה שֶׁאֲנִי מְשׁוּגָּע.
כְּלוּם אֵינִי שׁוֹמַעַת פְּעִימָה כְּבֵדָה וַאֲיוּמָּה, וּמְזַהָה בָּהּ אֶת הַלְמוּת לִבָּהּ?
פַּעַם אַחַת כְּבָר תָּפַסְתִּי אוֹתָהּ מְנַסָּה לִפְתּוֹחַ אֶת הָאָרוֹן שֶׁלָּךְ, אָז צָרַחְתִּי עָלֶיהָ וְהִיא בָּרְחָה.

------------------
אֲנִי מוֹרֶה לַלָּשׁוֹן, גְּרוּשָׁה, בַּת אַרְבָּעִים, בְּלִי יְלָדִים וְעַכְשָׁיו גַּם בְּלִי רַחֵם".
בְּסוֹף כָּל מִשְׁמֶרֶת עֶרֶב, בַּחֲצוֹת, כְּשֶׁאֲנִי נִכְנֶסֶת לְשֵׁירוּתֵי הָעוֹבְדִים וּמַתִּיזָה מַיִם קָרִים עַל פָּנַי, אֲנִי מְגַלֶּה שֶׁהַמַּסְקֵרָה שֶׁמָּרַחְתִּי בַּבּוֹקֶר, מֶחֱווֹה גַּאַוְותָנִית קְטַנָּה, נָטְשָׁה אֶת הָרִיסִים וְנִקְוְותָה בְּסִפְלֵי הָעוֹר הָעֲמוּקִּים שֶׁמִּתַּחַת לְעֵינֵי.

------------------
הָאִישָּׁה שֶׁלּוֹ עוֹנָה לוֹ שֶׁבְּוַודָּאוֹת הִיא לֹא יוֹדַעַת, אֲבָל מָה שֶׁבָּטוּחַ זֶה שֶׁאֲנַחְנוּ כְּבָר כָּאן וַאֲנִי אוֹמֶרֶת לָהֶם שֶׁכָּאן, כָּאן זֶה שְׁוֵויְיצָרְיָה.
הַסִּיפּוּר הַזֶּה הוּא בִּשְׁבִילֵךְ – פּוּרְקָן, תֵּרַפְּיָה, אוֹטוֹתֵרַפְּיָה, מָה אַתְּ רוֹצָה מִמֶּנִּי?
אַךְ צְרִיכָה הָיִית לִרְאוֹת אוֹתִי.

------------------
צָעַק, חֲתִיכַת חָרָא קְטַנָּה, תִּרְאִי אֵיךְ גָּדַלְתְּ!
שַׁחַר הַדָּמִים יִזְרַח עוֹד מְעַט, וְאַתְּ הֵן יָדַעְתְּ, אַבִּירָה, שֶׁלּוֹרְד אַרְתּוּרוֹ הָאַכְזָר אֵינוֹ יוֹדֵעַ רַחֲמֵי אֱנוֹשׁ וְאֵינוֹ יָרֵא אֶת הַמָּוֶות.
אַתְּ דַּאֲגַת לַכֶּלֶב וְהוּא דָּאַג לְךָ.
הָיִית צְרִיכָה לְשַׂחֵק אִיתּוֹ, לְהַאֲכִיל אוֹתוֹ, לָקַחַת אוֹתוֹ לַמְּלוּנָה.
הִיא פָּנְתָה לַקּוּפָּאִית, "אַתְּ רָאִית אֶת הַיַּלְדּוֹנֶת הַזֹּאת, נָכוֹן?"
חֵרַבְתְּ לָנוּ אֶת כָּל סוֹף הַשָּׁבוּעַ, זוֹנָה שֶׁכָּמוֹךְ.

------------------
זוֹ אִישָּׁה קְטַנָּה, צְהוּבָּה כִּזְעַפְרָן, וּבְמִשְׁחַת זְעַפְרָן הִיא גַּם מוּשְׁחָת אֶת הָרְצָפוֹת, אֶת שׁוּלְחָנוֹת הָאוֹרֶן, הַשְּׁרַפְרַפִּים וְהַכְּלוֹנְסָאוֹת שֶׁהִיא מְנַקָּה בַּחֲדָרֵיהֶם שֶׁל אֲנָשִׁים עֲנִיִּים.

------------------
עַכְשָׁיו בְּיַחַד הִיא רוֹצָה שֶׁנִּבְדּוֹק שֶׁהַדֶּלֶת נְעוּלָה, כְּמוֹ שֶׁלִּימַּדְתִּי אֶת עַצְמִי וְאוֹתָהּ.
הָאִישׁ שֶׁלּוֹ אוֹמֵר שֶׁצְּפַרְדֵּעַ, וְהָאִישָּׁה שֶׁלָּהּ עוֹנָה שֶׁלֹּא יָכוֹל לִהְיוֹת וְשֶׁמָּה הוּא מֵבִין מֵהַחַיִּים שֶׁלּוֹ.
כְּמוֹ שֶׁעֶרֶב קַיִץ אֶחָד, אַחֲרֵי שֶׁבָּאַמְבַּטְיָה, לֹא שָׁטַפְתִּי לָהּ טוֹב-טוֹב אֶת כַּפּוֹת הָרַגְלַיִים, הַיַּלְדָּה שֶׁלִּי אָמְרָה לִי שֶׁהִיא רוֹצָה הַבַּיְתָה.

------------------
הוֹ, אֲנִי רוֹצָה הַבַּיְתָה, אָמְרָה אֵפִי.
הָארִי, אָמְרָה אֵפִי, "אֲנִי תּוֹהָה מָתַי הוּא יֹאכַל אוֹתָנוּ!"
וְהֵם רָאוּ אֶת אִמָּם בּוֹכָה בַּבַּיִת עַל יְלָדֶיהָ שֶׁנֶּעֶלְמוּ בְּאוֹר הַיּוֹם הָאָיוֹם וּמֵטִיל הָאֵימָה.
וְאָז לִיאַת חָבְשָׁה אֶת כִּתְרָהּ וְיָשְׁבָה עַל כֵּס הַמְּלוּכָה שֶׁלָּהּ וְאָמְרָה שֶׁהִיא רוֹצָה לְחוֹקֵק כַּמָּה חוּקִּים.
הִיא הָפְכָה לְמִי שֶׁאֵינָהּ טוֹעָה לְעוֹלָם – זֹאת אוֹמֶרֶת, כָּל מָה שֶׁאָמְרָה הָיָה נָכוֹן, וְהִיא לֹא הָיְיתָה מְסוּגֶּלֶת לִטְעוֹת.

------------------
אִירִיס אוֹ עֲדִינָה – הִיא מְנַסָּה לְגָרֵד מֵהָאַרְכִיּוֹן הַמְּשׁוּבָּשׁ שֶׁלָּהּ אֶת הַשֵּׁם שֶׁל הַמַּנְהִיגָה הָרוּחָנִית שֶׁל נְשׁוֹת הַגַּרְעִין הַתּוֹרָנִי שֶׁהִגִּיעָה אֲלֵיהֶם מֵהַשּׁוֹמְרוֹן וְהָפְכָה לִמְנַהֶלֶת הַמַּתְנָ"ס.
אֲנַחְנוּ מְאוֹד מַעֲרִיכוֹת אוֹתְךָ, הִיא מַדְגִּישָׁה אֶת הַמִּילִּים הַחִיּוּבִיּוֹת, כְּמוֹ מוֹרָה בְּיוֹם הוֹרִים שֶׁצְּרִיכָה לְהַגִּיד מַשֶּׁהוּ טוֹב לִפְנֵי דִּיוּוּחַ עַל צִיּוֹנִי מַחֲצִית גְּרוּעִים.
הִיא מְנַסָּה לְהַרְוִויחַ זְמַן.

------------------
תְּחִילָּה חִיפְּשָׂה אוֹתוֹ בֵּין הַשִּׂיחִים, עַל שְׂפַת הַמַּיִם וְעַל הַגַּגּוֹת, וְלֹא שֶׁעַתָּה לִגְבִרְתָּהּ, שֶׁקָּרְאָה לָהּ: "הִיזָּהֲרִי שָׁם, יָצָאתָ מִדַּעְתְּךָ!
הִיא בָּכְתָה כָּל-כך עַד שֶׁגְּבִרְתָּהּ אָמְרָה לָהּ: "אָז תְּנִי שֶׁיַּעֲשׂוּ לְךָ מִמֶּנּוּ פּוּחְלָץ!
מֶה עָשִׂיתִי לְךָ כְּדֵי שֶׁתֹּאהֲבֵינִי כָּל כך?
אֲנִי אוֹמֵר לְךָ פַּעַם נוֹסֶפֶת, גְּבִרְתִּי, רוֹצָה אֲנִי שֶׁאָהֲבָה תִּהְיֶה מְתוּנָה.

------------------
אִילּוּ הָיִיתִי עוֹשֶׂה כָּךְ, רוֹב הַסִּיכּוּיִים שֶׁכְּשֶׁהָיִיתִי מַגִּיעַ לַנְּקוּדָּה שֶׁתֵּיאַרְתִּי זֶה עַתָּה הָיִית מְצַוָּוה עָלַיי לִשְׁתּוֹק בְּזַעַם וְלֹא מַאֲמִינָה לִי.
הָיִית מְסַלֶּקֶת אוֹתִי מִבֵּיתְךָ וְלֹא רוֹצָה לִשְׁמוֹעַ יוֹתֵר אַף מִילָּה, וְזוֹ הָיְיתָה תְּגוּבָה מוּבֶנֶת בְּהֶחְלֵט.
בַּמִּקְרֶה הַטּוֹב הָיִית דּוֹרֶשֶׁת מִמֶּנִּי הֶסְבֵּרִים שֶׁאֲנִי לֹא יָכוֹל לְסַפֵּק.
אַתְּ הָיִית בַּמְּזָוֶה עִם כָּל הָאֲחֵרוֹת!
יָפֶה אָמַרְתְּ, פְּיּוּטָה עַלְמָתִי.

------------------
לָכֵן אֲנִי אוֹמֶרֶת לְךָ שֶׁאִם תִּבְטְלִי אֶת הַצָּעַת הַמֶּחְקָר שֶׁלָּךְ עַכְשָׁיו, הַסִּיכּוּי שֶׁהַוַּועֲדָה הַזּוֹ תַּעֲנִיק לְךָ בֶּעָתִיד מִלְגָּה נוֹסֶפֶת הוּא קָטָן מְאוֹד.
יֵשׁ מוֹצָא אַחֵר: אַתְּ יְכוֹלָה לִזְכּוֹת בְּכַסְפְּךָ בַּחֲזָרָה'.
מָה לְךָ, יַקִּירָה?
מָה קָרָה לְךָ, יַקִּירָתִי, אַתְּ נִמְצֵאת בְּאֵיזֶה טְרַאנְס אוֹ מָה?
בְּבַקָּשָׁה מִמְּךָ, חֲבִיבָה!
וּלְזֶה שֶׁשָּׁלַח אוֹתְךָ תַּגִּידִי שֶׁהוּא צָרִיךְ לְהִתְבַּיֵּישׁ.

------------------
אֵין הָאוֹפָה מְעִידָה עַל עִיסָּתָהּ.
בִּתְאוּנָה בְּמִכְרֵה נֶהֶרְגָה כּוֹרָה אַחַת.
הִיא מַלְוָוה בְּרִיבִּית, תִּתְרַחֲקִי מִמֶּנָּה.
לֹא רָצִינוּ לְהַרְגִּיז אֶת הַמְּמוּנָּה, אָז לֹא סִיפַּרְנוּ לָהּ מָה עָשִׂינוּ.
נִפְגְּשׁוּ עִם הַמַּנְחָה וְאָמְרוּ לָהּ אֶת דַּעְתָּם.
לִהְיוֹת מְנַקָּה שֶׁעוֹבֶדֶת בִּשְׂכַר מִינִימוּם זֶה לֹא קַל.
לֹא קִיבַּלְתִּי תֶּקֶן בְּתוֹר מַרְצָה, אָז הָלַכְתִּי לַעֲבוֹד כְּרוֹפֵאת מִשְׁפָּחָה.

------------------
תִּישָּׁאֲרִי עִם הַכַּדּוּר הַיָּשָׁן שֶׁלְּךָ.
קַל לְךָ לְהַגִּיד, אָמַרְתִּי לָהּ.
צָרַחְתִּי וְקַמְתִּי צוֹרֵחַ וְהַפְּלֵתִי אֶת הַנֵּר: "אַל תַּעֲשִׂי אֶת זֶה, כַּלְבָּה, בַּחַיִּים שֶׁלְּךָ אַל תַּעֲשִׂי אֶת זֶה!
תַּרְשִׁי לִי לְהַזְמִין אוֹתְךָ לְמַשְׁקֶה, אָמַרְתִּי לְאִישָּׁה.
תַּקְפִּידִי לִשְׁמוֹר עַל צִיּוּד הַתְּפִירָה בַּחֶדֶר שֶׁלְּךָ, תַּקְפִּידִי לִנְעוֹל אֶת דֶּלֶת הַחֶדֶר שֶׁלָּךְ.
תּוֹרִידִי אֶת הָעֵינַיִים שֶׁלְּךָ מֵהַיֶּלֶד שֶׁלִּי!
אָמַרְתִּי לָךְ, לְכִי הַבַּיְתָה.

------------------
בְּצוּרָתָהּ הַזָּוִויתִית, בְּקִפְאוֹנָהּ, בְּרַגְלֵי-הַכְּלוֹנָס שֶׁלָּהּ, הִיא מַעֲלָה עַל הַדַּעַת, אֲפִילּוּ מִקָּרוֹב, דּוּבְשָׁן דְּמוּי-סוּס בִּפְרוּטָה.
הִיא מִתְיַישֶּׁבֶת עַל הָרִצְפָּה, מְנַקָּה אֶת הָעַרְדָּלַיִים, וְחוֹשֶׁבֶת שֶׁטּוֹב הָיָה לִתְחוֹב אֶת הָרֹאשׁ לְעַרְדָּל גָּדוֹל וְעָמוֹק וּלְנַמְנֵם קְצָת.
וְרַקָּה שׁוֹטֶפֶת אֶת הַמַּדְרֵגוֹת וּמְנַקָּה אֶת הַחֲדָרִים, אַחֵר-כָּךְ מֵסִיקָה תַּנּוּר נוֹסָף וְרָצָה לַחֲנוּת.
הִיא מְנַסָּה לִנְשׁוֹם בִּמְקוֹמוֹ.

------------------
אֲנִי אוֹמֶרֶת לְךָ לֹא וּמוֹדָה לְךָ, אָמְרָה.
אֲנִי מְקַוָּה שֶׁלֹּא הִקְדַּמְתִּי, אָמְרָה.
טוֹב, הָעַלְמָה פִיצְרוֹי רוֹבִּינְסוֹן, אָמְרָה, "אֲנִי מְרוּצָּה לַחֲלוּטִין מֵהִתְקַדְּמוּתָהּ שֶׁל דֶּיְיזִי מִיּוֹם שֶׁהִיא נְתוּנָה לְהַשְׁגָּחָתֵךְ."
אַחֲרֵי שֶׁסִּיפְּרוּ אֶת שֶׁהִתְרַחֵשׁ, אָמְרָה – "חֲבֵרִים יְקָרִים, אֲנִי מְקַוָּה שֶׁתִּסְלְחוּ לִי – חֻקֵּי הַכְנָסַת הָאוֹרְחִים אָמְנָם נִקְשִׁים – אֲבָל בַּעֲסָקִים כְּמוֹ בַּעֲסָקִים, אַחֲרֵי הַכֹּל."

------------------
כָּל אַחַת בַּחֲצִי שְׂמִיכָה שֶׁלָּהּ קְצָת מְכַסָּה אוֹתִי, וְלִי יֵשׁ פִּתְאוֹם שְׂמִיכָה שְׁלֵמָה, בְּדִיּוּק כְּשֶׁהָאִוְורוּר מַתְחִיל לְהַקְפִּיא לִי אֶת כַּפּוֹת הָרַגְלַיִים.
מִיְּמִינָהּ וּקְצָת מֵאֲחוֹרֶיהָ – כְּאִילּוּ מְנַסָּה לְהִסְתַּתֵּר, יָעֵל, אִמָּא שֶׁל גּוֹנִי.
וְאֶת כִּישְׁרוֹן הַכְּתִיבָה שֶׁלָּךְ, כַּמּוּבָן, יָעֵל מְמַהֶרֶת לְהוֹסִיף, מְנַסָּה לִלְכּוֹד אֶת מַבָּטָהּ.
וּמוֹדָה בְּלִבָּהּ לְלִיבִּי שֶׁדּוֹחֶפֶת אֶת הַפָּנִים שֶׁלָּהּ בִּתְבִיעָה לְלִיטּוּפִים.

------------------
הִיא מוֹדָה גַּם לְצִ'ילָנִי הָרִאשׁוֹן, בִּזְמַן שֶׁהֵם מְלַטְּפִים זֶה אֶת זֶה בִּידִיעָה שֶׁהַלַּיְלָה לֹא יַעֲשׂוּ אַהֲבָה וְיַעֲבִירוּ אֶת הַשָּׁעוֹת בְּשִׂיחָה, צְמוּדִים זֶה לָזֶה, צְמוּדִים עַד סַכָּנָה, סוֹלִידָארִיִּים.
יַלְדָּה, אֲנִי מְדַבֶּרֶת אֵלַיִיךְ, אִישָּׁה מְבוּגֶּרֶת וּשְׁמֵנָה דִּידְּתָה לְכִיוּוּנִי  בְּכָל הַכּוֹחַ, תּוֹמֶכֶת בְּיָד אַחַת אֶת הֶחָזֶה וְאִילּוּ אֶת הַשְּׁנִיָּיה הִיא הוֹשִׁיטָה אֵלַיי, כְּאִילּוּ מְנַסָּה לַחֲטוֹף לִי אֶת הַסִּיגַרְיָה מֵהַפֶּה.

------------------
וּבְכֵן, אֲנִי מִצְטַעֵר שֶׁהִכְעַסְתִּי אוֹתָךְ, אָמַר לָיוֹנֵל, "בּוֹאִי, קְחִי נְשִׁיקָה וְנִהְיֶה חֲבֵרִים."
אַתְּ מַלְכָּה קְטַנָּה וְשׁוֹבָבָה, אָמְרָה, "וְאַף אֶחָד לֹא אוֹהֵב אוֹתָךְ."
גְּבֶרֶת יְקָרָה, הַמֶּלֶךְ שֶׁל אַרְצֵנוּ, סָב-סָב-סָב-סָב-סָב-סָב-סָבֵךְ, מֵת, וְעַתָּה עָלַיִךְ לָבוֹא וְלִהְיוֹת מַלְכָּה.
אֶלֶף לִירוֹת שְׂטֶרְלִינְג לִשְׁלוֹשָׁה-עָשָׂר שְׂעוֹרִים – מְחִיר בְּמֵיחֵד בִּשְׁבִילֵךְ, עַלְמָתִי הַיְּקָרָה.
אֶצְטָרֵךְ לִתְלוֹת אוֹתָךְ, אַתְּ יוֹדַעַת, אָמַר הַמֶּלֶךְ בְּסַפְקָנוּת.

------------------
אַתְּ כָּל הַשְּׁרִירִים אֲנִי אֲפָרֵק לָךְ, כְּמוֹ שֶׁאַתְּ אוֹהֶבֶת.
יֵשׁ לָךְ שָׁם אֶת הַחֶדֶר הַנּוֹסָף הַזֶּה, שֶׁאַתְּ לֹא מִשְׁתַּמֶּשֶׁת בּוֹ, נָכוֹן?
וְאָז הִיא אוֹמֶרֶת: "אֲנִי זוֹכֶרֶת אוֹתָךְ עוֹד שֶׁהָיִיתָ כָּזֹאת קְטַנָּה."
אֲנִי עוֹשֶׂה לָהּ פַּרְצוּף שֶׁל "שָׂמַחְתִּי לֹא לְהַכִּיר אוֹתָךְ", מִסְתּוֹבֵב מִמֶּנָּה וּמוֹסִיף עוֹד חֵלֶק יָרוֹק שֶׁל לֵגוֹ עַל הַכַּף.
גַּם אַתְּ לִפְעָמִים כּוֹתֶבֶת בַּמַּחְבָּרוֹת הַצְּהוּבּוֹת שֶׁלָּךְ.

------------------
לַמְרוֹת הַחוֹם אֲנִי נִשְׁכֶּבֶת עַל הַמִּיטָּה וּמִתְכַּסָּה בִּשְׂמִיכַת הַפִּיקֶה.
הִיא תָּמִיד צִיפְּתָה מֵהַחַיִּים שֶׁיִּהְיוּ קְצָת יוֹתֵר יָפִים מִמָּה שֶׁהָיוּ בֶּאֱמֶת וּבַסּוֹף תָּמִיד הָיְיתָה מִתְבַּדָּה.
גְּבִרְתִּי, אִם אַתְּ לֹא יוֹדַעַת לְאָן אַתְּ רוֹצָה לִנְסוֹעַ, לָמָּה נִכְנַסְתְּ לַמּוֹנִית שֶׁלִּי?
הָאַחֶרֶת הִתְרַגְּזָה וְאָמְרָה: "מָה פִּתְאוֹם אַתְּ נוֹתֶנֶת לִי שָׁלוֹשׁ לִירוֹת, אֲנִי רוֹצָה לִירָה אַחַת."

------------------
הִנֵּה הִיא עוֹבֶרֶת עַל פֶּתַח הַדִּירָה, מִשְׁתַּהָה רֶגַע, וְחוֹזֶרֶת, וְעֵינֶיהָ הַגְּדוֹלוֹת לוֹטְפוֹת אֶת דֶּלֶת הַבַּרְזֶל.
הִיא מִתְקַשָּׁה לִבְלוֹעַ וּצְרִיכָה לִשְׁתּוֹת רוֹק מְלָאכוּתִי.
אַתְּ חוֹשֶׁבֶת שֶׁהִיא רַק מִתְחַזָּה לְעִיוֶּורֶת?
אֲנִי נִכְנֶסֶת לַחֲדַר הַשֵּׁינָה וּמִתְכַּסָּה בִּשְׂמִיכָה.

------------------
אַחֵר כָּךְ בִּיקַּשְׁתִּי מִמֶּנּוּ שֶׁיְּסַפֵּר לִי עַל הַמִּשְׁפָּחָה שֶׁלּוֹ וְהוּא סִיפֵּר לִי עַל הַתִּיכוֹן שֶׁהַהוֹרִים שֶׁלּוֹ נִיהֲלוּ בִּבְרוֹנְקְס, אַבָּא שֶׁלּוֹ הָיָה הַמְּנַהֵל וְאִמָּא שֶׁלּוֹ הָיְיתָה מוֹרָה לְאַנְגְּלִית.
הִיא חָתְכָה וּבַחֲשָׁה וְטִיגְּנָה בַּמִּטְבָּח, כָּל יוֹמַיִים, לְעֵרֶךְ, בְּעוֹד אמא מוֹרָה מָה וְאֵיךְ לַעֲשׂוֹת, מַשְׁגִּיחָה שֶׁחָס וְחָלִילָה לֹא תָּשִׂים לָהּ חָרִיף, שֶׁלֹּא תּוֹסִיף אֶת הַתַּבְלִינִים שֶׁלָּהּ שֶׁמַּסְרִיחִים אֶת הַבַּיִת.

------------------
וְהַצֵּל עָנָה, "גְּשִׁי אֵלַיי, וַאֲדַבֵּר אִתָּךְ; כִּי דְּבָרִים נִפְלָאִים רָאִיתִי.
אָמַר לָהּ הַצֵּל, "כַּאֲשֶׁר עָזַבְתִּי אוֹתָךְ, שַׂמְתִּי אֶת פָּנַיי מִזְרָחָה וְיָצָאתִי לְמַסָּע."
הָאִינָךְ יוֹדַעַת כִּי קֵיסָר אָנוֹכִי בָּעִיר הַזֹּאת?
הַאִם נְבִיאָה הִנֵּךְ שֶׁאֵין לְהַזִּיק לָהּ, אוֹ בִּתּוֹ שֶׁל נָבִיא שֶׁלֹּא אוּכַל לִפְגּוֹעַ בָּהּ?
אָנָּא, עִזְבִי אֶת עִירִי הַלַּיְלָה, כִּי כָּל עוֹד הִנֵּךְ בָּעִיר, לֹא אוּכַל לִהְיוֹת אָדוֹן בָּהּ.
תְּנִי לִי אֶת מַחֲצִית אוֹצָרֵךְ, וְאֵלֵךְ לִי.

------------------
אַתְּ מְבִינָה בְּקוֹלְנוֹעַ, אָהַבְתְּ אַתְּ "אַתָּה יֶלֶד גָּדוֹל עַכְשָׁיו", לֹא כָּכָה?
לֹא, אַתְּ לֹא מְבִינָה, אַתְּ לֹא מְבִינָה, בָּאתִי לְדַבֵּר עַל הַסֶּרֶט, אַתְּ אָהַבְתְּ אוֹתוֹ, לֹא כָּכָה?
תַּגִּידִי לִי, אַלְבֶּרְטָה, אַתְּ נָפַלְתְּ עַל הָרֹאשׁ?
תַּגִּידִי לִי אַתְּ, אֲנִי מַפְצִיר, לֹא רָאִית מַסְפִּיק עֲוָולוֹת בִּימֵי חַיַּיִךְ?
תַּגִּידִי לִי בְּבַקָּשָׁה, רָאִית תֵּיאַטְרוֹן בְּלִי נִיצָּבִים?

------------------
אֲנִי הָלַכְתִּי וְהִרְגַּשְׁתִּי אוֹתָהּ מִסְתַּכֶּלֶת לִי עַל הָעוֹרֶף וְחָשַׁבְתִּי שֶׁהִיא רוֹצָה לַהֲרוֹג אוֹתִי.
קְלָאוּדְיָה לֹא עוֹנָה כְּשֶׁאֲנִי מְנַסֶּה לְהַשִּׂיג אוֹתָהּ בַּטֵּלֵפוֹן.
מִישֶׁהִי רוֹאָה אוֹתָהּ.
אֲבָל מִי שֶׁחוֹשֶׁבֶת שֶׁהַמָּקוֹם הַזֶּה הוּא כְּמוֹ הַפִּיאַסְטָה אוֹ הַחַוָּוה הִיא טוֹעָה וּבְגָדוֹל.
אֲנִי לֹא מַאֲמִינָה לְמָה שֶׁאֲנִי רוֹאָה.
מָה אַתְּ רוֹצָה שֶׁאֶעֱשֶׂה, מוֹתֶק?
אַתְּ רוֹצָה בֶּטַח שֶׁאַבְרִיחַ לָךְ אֶחָד כָּזֶה בַּמְּכוֹנִית שֶׁל הַקָּצִין שֶׁלִּי, מָה?

------------------
הָיִית מַאֲמִינָה?
הָיִית אֲמוּרָה לְהִישָּׂרֵף בַּזְּרִיחָה, וּפָחַדְתְּ שֶׁהוּא יַעֲשֶׂה אֶת זֶה.
כְּשֶׁעָמַדְתִּי לָצֵאת, הַיֶּלֶד נִיגַּשׁ אֵלַי וְאָמַר: "הוֹ, עֵין-נָחָשׁ, אַתְּ אָמַרְתְּ שֶׁאֲנִי יָכוֹל לְשַׂחֵק בַּצּוֹפֶה הַשָּׁחוֹר כְּשֶׁלֹּא תִּהְיִי כָּאן."
חוּאָנִיטָה, יַקִּירָתִי שֶׁלִּי, אָמְרָה בְּנִימָה שֶׁל עֲלִיצוּת בְּעוֹדָהּ כּוֹרֶכֶת אֶת זְרוֹעוֹתֶיהָ סְבִיב צַוָּוארָהּ וְנוֹשֶׁקֶת לְעַפְעַפֶּיהָ, "אִילּוּ רַק יָדַעְתְּ כַּמָּה מָתוֹק יִהְיֶה הַמָּוֶות מִיָּדַיִיךְ."

------------------
וּלְפֶתַע נֶעֶשְׂתָה זוֹעֶמֶת וְתוֹקְפָנִית וְהִמְשִׁיכָה בֵּטוֹן גָּבוֹהַּ יוֹתֵר: "אַתְּ צְרִיכָה לְהִתְבַּיֵּישׁ לְךָ, אַנְטוֹנְיָה!
שִׂימִי לֵב לַתִּיק שֶׁלְּךָ, גְּבֶרֶת, אָמְרָה הַזְּקֵנָה.
סִיפַּרְתִּי לְךָ אֶת זֶה כְּבָר, אַתְּ לֹא זוֹכֶרֶת?
מָתַי כְּבָר תַּפְסִיקִי עִם הַשְּׁטוּיוֹת שֶׁלְּךָ!
תַּגִּידִי לוֹ רַק שֶׁיֵּירָגַע, כִּי בִּגְלַל הַבֵּיצִים שֶׁלּוֹ נִהְיֶה לִי חֵשֶׁק לְזַיֵּין אוֹתְךָ, אַרְגֶּנְטִינָאִית.
לָמָּה הִיא קָרְאָה לְךָ קְטִינָה?

------------------
מַדּוּעַ אַתְּ מַרְשָׁה לַטִּיפְּשׁוּת הַזֹּאת לְהִיכָּנֵס לְרֹאשׁ?
בַּמִּשְׁפָּחָה שֶׁלָּהּ אוּלַי חוֹשְׁבִים שֶׁהִיא מְבַלָּה אֶת הַלַּיְלָה אֵצֶל הַדּוֹדָה גֵ'יְין אוֹ אֵצֶל אֶחָד הַשְּׁכֵנִים.
אֲנִי לֹא מְנַסָּה, אָמְרָה הִיא, "לְזַלְזֵל בַּצַּד הַמּוּסָרִי הַמְּהוּלָּל שֶׁל אַהֲבַת הוֹרִים לִילָדִים."
כָּל אֵימַת שֶׁהָיְתָה מַגִּיעָה לְסוֹפוֹ שֶׁל דַּף, הָיְתָה מְגַלָּה דַּף נוֹסָף בַּתִּיק.
הִיא לְעוֹלָם לֹא הָיְתָה מוֹדָה בְּכָךְ.

------------------
מָרִיָּה נָטְלָה קְלָף וְהִנִּיחַתּוּ, מְכַסָּה אוֹתוֹ בַּעֲרֵימָה שֶׁל שְׁטָרוֹת.
לְמַעֲשֶׂה, זֶה מְשַׁעֲמֵם אוֹתָהּ, אֲבָל כְּדֵי לֹא לְצַעֵר אֶת אָבִיהָ הִיא לוֹחֶשֶׁת בְּהַכְנָעָה: "אֲנִי מְאוֹד-מְאוֹד מוֹדָה לְךָ, אַבָּא יָקָר."
הִיא שׁוֹכֶבֶת זֶה חוֹדֶשׁ בְּמִטָּתָהּ, מַרְזָה וְנֶחֱלֶשֶׁת מִיּוֹם לְיוֹם, לֹא מִתְעַנְיֶינֶת בַּדָּבָר, מִשְׁתַּעֲמֶמֶת וְדוֹעֶכֶת לְאִטָּהּ.
שׁוֹטֶרֶת עוֹבֶרֶת בַּקָּהָל וּמְנַסָּה לְדַבֵּר אֶל לִבּוֹ: "רַבּוֹתַי, אֲנִי מְבַקֶּשֶׁת מִכֶּם לְהִתְפַּזֵּר."

------------------
הוּא גּוֹעֵר בָּהּ בְּקוֹל סָדוּק; נוֹשֵׁף בְּיֵיאוּשׁ לְתוֹךְ בְּשָׂרָהּ, עוֹצֵם מוּבָס אֶת עֵינָיו, נִמְאַסְתָּ כְּבָר – " "חוֹב, ג'וֹנָאם," הִיא אוֹמֶרֶת וּמְרַפָּה מִן הַחִיבּוּק, יָדָהּ מַחֲלִיקָה סַלְחָנִית בְּשַׁעֲרוֹ, "יִהְיֶה בְּסֵדֶר בְּעֶזְרַת הַשֵּׁם, אֱלוֹהִים גָּדוֹל – " וּכְשֶׁשְּׂפָתֶיהָ נָעוֹת, מְמַלְמְלוֹת חֶרֶשׁ, מִתְיַיפְּחוֹת אֶל הַתִּקְרָה, הוּא מְלַוֶּוה אוֹתָהּ מִתְרַחֶקֶת עִם הַשַּׂקִּיּוֹת לַמִּטְבָּח.

------------------
בְּכָל זֹאת לֹא זָכִיתִי לַעֲסוֹק בַּמִּקְצוֹעַ הַמְּפוּקְפָּק בְּיוֹתֵר בָּעוֹלָם – אֲנִי מְנַסָּה לְשַׁכְנֵעַ אֶת עַצְמִי, וּבְקוֹשִׁי מַצְלִיחָה לִשְׁמוֹר עַל קוֹר רוּחִי מֵרוֹב שִׂמְחָה.
אִם תַּגִּיד לָהֶם שֶׁהִיא מְנַקָּה מְכוֹנִיּוֹת, יִשְׁאֲלוּ מָה הִיא עוֹשָׂה בַּמּוֹשָׁב הָאֲחוֹרִי.
הִיא מְנַקָּה לִי אֶת הַבַּיִת, הַמְּפַקֵּד.
וְהָרוּחַ נוֹשֶׁבֶת, וְהִיא מַעֲלָה אוֹתִי עַל הַסּוּס שֶׁלָּהּ, וַאֲנַחְנוּ בּוֹרְחִים בַּלַּיְלָה.

------------------
הָיִיתִי יֶלֶד תָּמִים וּבַדְּמָעוֹת אָמַרְתִּי לוֹ שֶׁאִמָּא שֶׁלִּי לֹא מַרְשָׁה.
הִיא פָּרְצָה בִּצְחוֹק וְאָמְרָה שֶׁהוּא לֹא פֹּה וְשֶׁהִיא מַרְשָׁה וְאָז פָּשׁוּט תָּלְשָׁה אוֹתוֹ מֵעָלַיי וְחָבְשָׁה אוֹתוֹ עַל רֹאשָׁהּ.
אֲנִי יוֹדַעַת זֹאת הֵיטֵב כִּי אֲנִי הָיִיתִי הַמּוֹרָה הַפְּרָטִית שֶׁלּוֹ.
הָיָה לִי קָשֶׁה לְהַאֲמִין שֶׁזּוֹ הַסִּיבָּה הָאֲמִיתִּית וְהָיָה לִי בָּרוּר שֶׁיֵּשׁ סוֹד אַחֵר שֶׁאִמִּי מְנַסֶּה לְהַסְתִּיר מִמֶּנִּי וְהַסּוֹד הַזֶּה הֵצִיק לִי.

------------------
אֲנִי לֹא מַכִּירָה אוֹתוֹ, לֹא יוֹדַעַת מִיהוּ אוֹ מִי הוּא הָיָה וְלָכֵן מְנַסֶּה לְהַשְׁגִּיחַ עָלָיו בְּכָל הִזְדַּמְּנוּת.
אֲנִי שׁוֹאֶלֶת אֵיךְ הוּא הֵכִין אֶת הַמּוּס וְהוּא מְסָרֵב לְגַלּוֹת, וְכָךְ אֲנִי מְגַלֶּה שֶׁהוּא שֵׁף מוּסְמָךְ.
אֲנִי מַתְחִילָה לְחַבֵּב אוֹתוֹ וּמְגַלֶּה שֶׁנָּעִים לִי בְּחֶבְרָתוֹ.
אֲנִי מַרְשֶׁה לוֹ לְהַתְקִין תּוֹכְנָהּ בַּמַּחְשֵׁב הַיָּשָׁן שֶׁלִּי, וְהִיא הוֹפֶכֶת אֶת הַנְהָלַת הַחֶשְׁבּוֹנוֹת לְמַשֶּׁהוּ שֶׁאֲנִי מַמָּשׁ נֶהֱנֵית לַעֲשׂוֹת.

------------------
אֵין לִי חוּשׁ טַעַם, אָז אֲנִי לֹא יְכוֹלָה לִהְיוֹת אוֹפָה.
שָׁאַלְתִּי אֶת הַכּוֹרָה אֵיפֹה הִיא גָּרָה, אֲבָל לֹא קִיבַּלְתִּי תְּשׁוּבָה.
הַמַּלְוָוה בִּיקְּשָׁה עֵרָבוֹן.
הִיא הַמְּמוּנָּה, תִּשְׁאֲלִי אוֹתָהּ.
מַנְחָה טוֹבָה יוֹדַעַת לְהַקְשִׁיב.
הַמְּנַקָּה בִּיקְּשָׁה לֹא לְלַכְלֵךְ.
לֹא כָּל שִׁיעוּר הַמַּרְצָה מְאַחֶרֶת, אֲבָל זֶה קוֹרֶה לְעִתִּים קְרוֹבוֹת מִדַּי.
אֲנִי עוֹלָה חֲדָשָׁה מֵרוּסְיָה.
לָשֶׁבֶת עִם הַכְּבָשִׂים, כְּמוֹ רוֹעָה יְתוֹמָה בַּכְּפָר.

------------------
נַעֲרָתִי הַיְּקָרָה, אָמַר טְרֶווֹר בְּחִיּוּךְ, "הַקַּבְּצָן הַזָּקֵן הַהוּא, כְּפִי שֶׁאַתְּ מְכַנָּה אוֹתוֹ, הוּא אַחַד הָאֲנָשִׁים הָעֲשִׁירִים בְּיוֹתֵר בְּאֵירוֹפָּה."
אֲפִילּוּ הַקְּמָטִים הַקְּטַנִּים שֶׁמַּתְחִילִים לְהִיוָּוצֵר אֶצְלָהּ — הִיא לֹא מְנַסָּה לְהַסְתִּיר אוֹתָם, כֵּן?
רַק אַחֲרֵי שֶׁהֵם עוֹזְבִים אוֹתִי וְהוֹלְכִים לָשֶׁבֶת עִם הַהוֹרִים שֶׁלִּי עַל קָפֶה וּפוֹלִיטִיקָה, אֲנִי חוֹזֶרֶת בְּשֶׁקֶט לַכַּף שֶׁל הַטְּרַקְטוֹר שֶׁלִּי, מְנַסֶּה לְיַיצֵּב עָלָיו חֵלֶק אָדוֹם שֶׁל לֵגוֹ.

------------------
הִסְתַּלְּקִי, מִלְמֵל, "וְאַל תַּנִּיחִי לִי עוֹד לִרְאוֹת אֶת פָּנַיִךְ."
עֲשִׂי בָּהּ כְּחֶפְצְךָ.
הָרְאִי לִי אֶת הָאֵל,' קָרָאתִי, 'וְאִם לֹא, אֶהֶרְגֵךְ.'
אֶמְכּוֹר לָךְ אֶת נִשְׁמָתִי, עָנָה, "אָנָּא, קְנִי אוֹתָהּ מִמֶּנִּי, כִּי עָיַיפְתִּי מִמֶּנָּה."
וְהִיא דִּיבְּרָה אֶל הַדַּיֶּיגֶת הַצְּעִירָה וְאָמְרָה, "סִיפַּרְתִּי לָךְ עַל שִׂמְחוֹת הָעוֹלָם, וְהִפְנֵית אֵלַיי אוֹזֶן אֲטוּמָה."
וְכַמָּה אַתְּ מְקַבֶּלֶת עַל הַתְּמוּנָה שֶׁלָּךְ, אֵלֶן?

------------------
לֹא כְּדַאי שֶׁתִּהְיִי מְנַקָּה.
הַמַּרְצָה עֲשִׁירָה.
הַיַּלְדָּה הַזֶּה, עוֹלֶה מֵאַרְהָ"ב, יוֹדַעַת מָה הִיא עוֹשָׂה.
אָסַפְתִּי אוֹתָם כְּמוֹ הָרוֹעָה הַחֲבִיבָה שֶׁאֲנִי.
------------------
וְהִיא צָחֲקָה וְאָמְרָה: "אִמָּא מַרְשָׁה לְךָ?"
קָשֶׁה שֶׁיָּלְדָה מַבִּיטָה בְּךָ וּמְצַפָּה לַשָּׁוְוא לִתְשׁוּבָה מִשְּׁנֵי הוֹרִים מְבוּלְבָּלִים.
כֵּן, אֲנִי מוֹדָה שֶׁאֲנִי חָשָׁה הַעֲרָצָה מְסוּיֶּמֶת, אֲבָל לָמָּה מֵעוֹלָם לֹא רְאִיתֶם בִּי חַיֶּילֶת בַּגְּדוּדִים שֶׁלָּכֶם?
אַחַת הַשּׁוֹדְדוֹת מְנַסָּה לָקַחַת מִיָּדָהּ שֶׁל הַשּׁוֹמֶרֶת אֶת הָאֶקְדָּח וְלֹא מַבְחִינָה שֶׁהִיא עֲדַיִין בַּחַיִּים.
אֲבָל בִּמְקוֹמְךָ לֹא הָיִיתִי מַשְׁאִירָה אֶת זֶה כָּכָה, הָיִיתִי מַרְאֶה לָהֶם.

------------------
וְאֶפְשָׁר שֶׁהָיְיתָה מַאֲרִיכָה יָמִים בְּחֵיק אוֹרְחוֹתָיו הַשְּׁלֵווֹת וְלֹא מִתְפַּתָּה לְהַעֲמִיד אֶת כִּישְׁרוֹנָהּ בְּמִבְחָן, אִילּוּלֵא הִתְרַחֵשׁ מְאוֹרָע יוֹצֵא מִגֶּדֶר הָרָגִיל שֶׁטָּרַף אֶת מַסְלוּל חַיֶּיהָ.
הִיא מִשְׁתַּחֲוָוה, וְהָעֵינַיִים שֶׁלָּהּ עֲדַיִין עָלַיי וַאֲנִי הָעֵינַיִים שֶׁלִּי עֲדַיִין עָלֶיהָ.
אֲנִי מִתְכַּסֶּה בִּשְׂמִיכָה הַדַּקָּה שֶׁאֶחָד הַדַּיָּילִים סִיפֵּק לִי וּמִסְתּוֹבֶבֶת בַּכִּיסֵּא הַמְּצוּמָּק.

------------------
אִם אֲנִי רוֹצָה לְיַישֵּׁר אֶת הָרַגְלַיִים, אֲנִי לֹא יְכוֹלָה כִּי אֲנִי יוֹתֵר אֲרוּכָּה מֵהַמִּזְרָן.
הָאִישָּׁה שֶׁלּוֹ אוֹמֶרֶת שֶׁבְּלִי קָפֶה הִיא לֹא יְכוֹלָה לְהִתְעוֹרֵר, שֶׁהִנֵּה הִיא כְּבָר קְצָת רוֹאָה לְמַטָּה הַרְבֵּה יָרוֹק עִם גַּגּוֹת אֲדוּמִּים, וְשֶׁאוּלַי מִישֶׁהוּ מֵאִיתָּנוּ יוֹדֵעַ אֵיפֹה אֲנַחְנוּ בִּכְלָל.

------------------
תְּחִילָּה, לְמַרְבֵּה הָאַכְזָבָה, הִיא מְגַלָּה שֶׁגַּגּוֹתֵיהֶן שֶׁל כָּל הַבְּקָתוֹת שְׁקוּעִים פְּנִימָה וַעֲשָׂבִים גְּבוֹהִים מְבַצְבְּצִים מִתּוֹךְ תִּלֵּי רְעָפִים שֶׁנָּפְלוּ, תְּרִיסִים תְּלוּיִים עַל צִירֵיהֶם חַסְרֵי-נֶחָמָה, וְהַמָּקוֹם כּוּלּוֹ אֵינוֹ מְיוּשָּׁב.
כְּשֶׁהִיא כּוֹרַעַת וּמְנַסָּה לֶאֱסוֹף אֶת שִׁבְרֵי הַזְּכוּכִית, רְסִיס חַד נִנְעַץ עָמוֹק בְּכָרִית אֲגוּדָלָהּ; הִיא צוֹעֶקֶת צְעָקָה מָרָה, אֲמִיתִּית.
אַתְּ מְגַלֶּה שֶׁאֵין מָה לְהַצִּיל.

------------------
אָז בּוֹאִי נִבְדּוֹק רֶגַע מָה זֶה אוֹמֵר: אַתְּ פּוֹנָה אֶל הַיַּלְדָּה, אוֹ אֶל הָאִמָּא וְאֶל הָאַבָּא, אִם יֵשׁ אַבָּא בַּתְּמוּנָה, וּמַזְמִינָה אוֹתָם אֵלֵינוּ.
מִישֶׁהִי רוֹצָה לִשְׁתּוֹת עוֹד מַשֶּׁהוּ, תֶּה?
לְכִי אֲלֵיהֶם, מָה אַתְּ רוֹצָה מִמֶּנִּי?
הִיא לֹא מְעַנְיֶינֶת אוֹתִי – הַיַּלְדָּה שֶׁלָּהֶם – בְּדֶרֶךְ כְּלָל הִיא יוֹתֵר קְטַנָּה מִמֶּנִּי, בּוֹהָה בִּי עִם פַּרְצוּף הַנַּזֶּלֶת שֶׁלָּהּ.
כִּי הַיַּלְדָּה שֶׁלָּהֶם בּוֹכָה?

------------------
אַתְּ כַּנִּרְאֶה טָעִית, חֲבִיבָה, אָמְרָה, "הַפֶּתֶק הַזֶּה לֹא מְיוֹעָד אֵלַי".
הֲרֵי זֶה כְּאִילּוּ הָיִית אוֹמֶרֶת לִי פִּתְאוֹם: 'אַבָּא, תּוֹרִיד לִי אֶת הַשֶּׁמֶשׁ מֵהַשָּׁמַיִם.'
אִילּוּ הַיֶּלֶד הָיָה נֶאֱנָח הָיָה יוֹתֵר פָּשׁוּט, הָיִית מַשְׁכִּיבָה אוֹתוֹ בַּמְּכוֹנִית וְלוֹקַחַת אוֹתוֹ לְבֵית חוֹלִים אוֹ מַזְמִינָה אֶת כּוֹחוֹת הַהַצָּלָה.

------------------
כַּאֲשֶׁר רָאוּ אוֹתָהּ הַסּוֹחֲרִים בָּאָה לִקְרָאתָם, הֵחֵלּוּ לִלְחוֹשׁ אִישׁ לְרֵעֵהוּ; אֶחָד מֵהֶם צָעַד קָדִימָה לְפָגְשָׁהּ, קָרָא לָהּ בִּשְׁמָהּ וְאָמַר לָהּ, "מָה יֵשׁ לָךְ לִמְכּוֹר?"
מָה תִּיתְּנִי לִי אִם אוֹמֵר לָךְ?
אִם יָבוֹא לִקְרָאתֵךְ כֶּלֶב שָׁחוֹר, הַכִּי אוֹתוֹ בְּמַטֵּה מֵעֵץ הָעֲרָבָה, וְהוּא יֵלֵךְ לוֹ.
אַל נָא תְּשַׁלְּחִי אוֹתִי מֵעַל פָּנַיִךְ עַכְשָׁיו, כִּי מָה רָעָה עָשִׂיתִי לְךָ?
אַכְזָר הוּא הָעוֹלָם; אָנָּא, תְּנִי לִי אֶת לִבֵּךְ לָקַחַת אִתִּי.

------------------
אֱלוֹהִים יִשְׁמוֹר עָלֶיךָ, אַל תֵּלְכִי עִם זָרִים לִמְקוֹמוֹת מְסוּכָּנִים וְתִתְקַשְּׁרִי.
הוּא שִׁחְרֵר לְפֶתַע אֶת יָדוֹ מֵהַפִּצְעוֹן, תָּפַשׂ אֶת הַזְּקֵנָה בְּשַׂעֲרוֹת רֹאשָׁהּ וְנָבַח לְעֶבְרָהּ: "אִם כך, לֹא תַּחְשְׁבִי עָלַי רָעוֹת אִם אַשְׁדּוֹד אוֹתְךָ, הָא?
אִם לֹא תֹּאמְרִי זֶה סוֹפֵךְ!
זֶה אֶצְלֵךְ, מְתוּקָה?
הוּא הִמְתִּין רֶגַע קָט, אַחֵר כך נִישֵּׁק אוֹתָהּ בְּמִצְחָה וְאָמַר, "אֲנִי אוֹהֵב אוֹתְךָ.
תִּתְעַלְּמִי מִשְּׁטוּיוֹת, אָמְרָה לְעַצְמָהּ, זֶה סְתָם הוֹרֵס לָךְ אֶת הַיּוֹם.

------------------
בְּכוֹחַ אֲנִי מְנַסָּה לְהַחֲזִיק אֶת עַצְמִי לֹא לְצַחְקֵק בְּנִימוּס מֵהַבְּדִיחָה הַזֹּאת, הַמַּעֲלִיבָה, אֲבָל בְּכָל זֹאת מְצִיצָה בַּתְּמוּנָה וּמְהַנְהֶנֶת בְּמִין חֲצִי חִיּוּךְ שֶׁל הַסְכָּמָה.
בַּלֵּב שֶׁלִּי אֲנִי מְנַסָּה לַחֲשׁוֹב אֵיךְ אֲנִי יְכוֹלָה לִגְרוֹם לָהּ לְהַפְסִיק לְדַבֵּר.

------------------
הֵבַנְתִּי אֶת הַשְּׁאֵלָה שֶׁמּוּסְוֵוית בַּשְּׁאֵלָה הַזֹּאת: מָה הָיִית צְרִיכָה לַעֲשׂוֹת כְּדֵי שֶׁיְּמַנּוּ אוֹתְךָ לְאַחְרָאִית?
הֵבַנְתִּי אֶת הַשְּׁאֵלָה שֶׁלֹּא שׁוֹאֲלִים אַף פַּעַם: מָה הָיִית צְרִיכָה לַעֲשׂוֹת כְּדֵי לִשְׂרוֹד?
בַּדַּקָּה הָאַחֲרוֹנָה רָץ לָיוֹנֵל בְּמַעֲלֵה הַמַּדְרֵגוֹת, נָשַׁק לְאוֹמֶנֶת וְאָמַר, "תּוֹדָה שֶׁרָחַצְתְּ אוֹתִי."
אֲבָל הָאוֹמֶנֶת אָמְרָה, אַל תִּרְתְּמִי אֶת הָעֲגָלָה לִפְנֵי הַסּוּסִים, יַקִּירָתִי, עוֹד לֹא רָאִית לְמָה הַדְּרָקוֹן מְסוּגָּל.

------------------
הַפָּרָה פּוֹעָה בַּלַּיְלָה.
הַמּוֹרָה מְנַהֶלֶת בְּכִיתָּה שִׂיחָה פּוֹרֶה.
חַוָּוה מְפַתֶּה אֶת אָדָם.
הַמְּאַלֶּפֶת מְצַוָּוה עַל הַכֶּלֶב לָשֶׁבֶת.
הַתַּלְמִידָה מְצַפָּה בְּכִילְיוֹן עֵינַיִים לַחוֹפֶשׁ הַגָּדוֹל.
הַבַּחוּרָה מְשַׁנָּה אֶת דַּעְתָּהּ בְּאוֹפֶן יוֹם יוֹמִי.
אֲחוֹתִי מְרַמָּה אוֹתִי לְעִיתִּים קְרוֹבוֹת בְּפוֹקֵר.
הַבַּחוּרָה מָשְׁלָה אֶת הַבָּחוּר.
כָּל אַחַת מִשְׁתַּנָּה לְאוֹרֶךְ הַחַיִּים שֶׁלָּהּ.
הִילָּה שׁוֹתָה מִכּוֹס הַיַּיִן שֶׁלְּפָנֶיהָ.

------------------
הִיא מְנַסָּה לְחַלֵּק אֶת רְצוּעַת הַמֶּלֶל לַחֲתִיכוֹת, לְזַהוֹת מִלִּים, לְנַחֵשׁ אוֹ לְהִיזָּכֵר בְּמַשְׁמָעוּתָן.
אֲבָל כְּשֶׁהִיא פּוֹקַחַת אֶת הָעֵינַיִים, הִיא מְגַלָּה שֶׁהָאָהִיל הַתָּלוּי מֵעַל רֹאשָׁהּ טָס הַצִּידָּה בִּמְהִירוּת עֲצוּמָה.
הִיא מִסְתַּכֶּלֶת אִם יֵשׁ בַּסְּבִיבָה מִישֶׁהוּ מֵרוּסְיָה שֶׁיְּתַרְגֵּם, הִיא מְזַהָה אֶת הָרוּסִים לְפִי הַפָּנִים.
רַק הַמְּנַקָּה הַצַּחְקָנִית אוֹמֶרֶת בְּכָל פַּעַם "רָאִיסָה יוֹק!

------------------
אַתְּ קָנִית שַׁרְשֶׁרֶת יַהֲלוֹמִים?
כַּאֲשֶׁר הֶחֱזִירָה מָטִילְדָּה אֶת הַמַּחְרוֹזֶת לַחֲבֶרְתָּהּ אָמְרָה לָהּ זָ'אן בִּקְרִירוּת: "הָיִית צְרִיכָה לְהַחֲזִיר לִי אֶת הַשַּׁרְשֶׁרֶת יוֹתֵר מַהֵר."
אִם הָיִית פּוֹרֶצֶת פְּנִימָה, אָמַר בְּעֶצֶב, "זֶה בֶּאֱמֶת הָיָה יָכוֹל לִהְיוֹת מִבְצָע רָאוּי לִכְנוּפְיָה."
הָיִית מֵתָה!
בְּ-סֵ-דֶ-ר, הוּא שׁוּב קוֹטֵעַ אוֹתָהּ, הַפַּעַם בְּגַסּוּת, "כְּבָר אָמַרְתְּ אֶת זֶה – "
כָּתַבְתְּ שֶׁאַתְּ צְרִיכָה עֶזְרָה.
מָה אִתָּךְ, סַבְתָּא, נָפַלְתְּ מֵהַיָּרֵחַ?

------------------
אַתְּ, מָה אִתְּךָ, מְנוּוֶּלֶת?
שָׁנִים פִנְטַזְתִּי אֵיךְ אֲנִי מוֹצֵא אוֹתָהּ, נִיגַּשׁ אֵלֶיהָ, מַרְחִיק אוֹתָהּ לְרֶגַע מֵהַפָּרָנוֹאִיד אוֹ הַהִיפֶּרְאַקְטִיבִי אוֹ הַסְּכִיזוֹאִיד הַתּוֹרָן וְאוֹמֵר לָהּ: "הִנֵּה הָעֵט שֶׁלְּךָ".
הָאָדוֹן הַצָּעִיר יְקַבֵּל אֶת פָּנֶיךָ בַּחֲדַר הָעֲבוֹדָה כְּשֶׁתְּסַיְּימִי לְצַלֵּם אֶת הַגַּן, אָמְרָה לָהּ הָאִישָּׁה, כְּפִי שֶׁסִּיכַּמְנוּ מֵרֹאשׁ.
תּוֹדָה לָאֵל שֶׁחָזַרְתָּ בְּשָׁלוֹם, אַתְּ רוֹאֶה, טוֹב שֶׁשַּׂמְתִּי לְךָ אֶת הַשּׁוּם הַזֶּה בַּתִּיק.

------------------
אֲנִי מַנִּיחָה שֶׁהִיא הָיְתָה רוֹצָה שֶׁהַיּוֹמָן יִיגָּמֵר בַּתְּמוּנָה הַזֹּאת שֶׁגָּזַרְתִּי מֵהָעִיתּוֹן וַאֲנִי עוֹמֶדֶת לְהַדְבִּיק פֹּה.
אַחֵר כָּךְ חָשַׁבְתִּי שֶׁאַחֲרֵי שֶׁשָּׁתְתָה מֵעֵבֶר לַמִּידָּה, בְּמַצָּבָהּ הַגּוּפָנִי הַנּוֹכְחִי, מָה שֶׁהִיא רוֹצָה לְסַפֵּר לִי עָלוּל לְטַלְטֵל אוֹתָהּ, אוּלַי יוֹתֵר מִדַּי.
לָקַחְתִּי אֶת הַבַּקְבּוּק וְשָׁאַלְתִּי אוֹתָהּ אִם הִיא רוֹצָה שֶׁאֶמְזוֹג לָהּ עוֹד קְצָת, אֲבָל הִיא לֹא עָנְתָה לִי וְאוּלַי לֹא שָׁמְעָה.

------------------
אֲבָל מֵעַל הַכּוֹבַע אֲנִי מְרִיחָה רֵיחַ שֶׁל דָּגִים, לֹא מְקוּלְקָלִים, אֶלָּא טְרִיִּים לְמַדַּי, עִם רֵיחַ לְוַואי כָּלְשֶׁהוּ שֶׁאֲנִי מִתְקַשָּׁה לְהַגְדִּיר.
עַכְשָׁיו, מִשֶּׁנִּרְגְּעָה, נִרְאֲתָה מְכוּנֶּסֶת בְּתוֹךְ עַצְמָהּ וְקַשּׁוּבָה, כְּמִי שֶׁמִּתְחַקָּה בִּזְהִירוּת אַחֵר עִקְבוֹתָיו שֶׁל חֲלוֹם.
וְהִיא, מִתְרַצָּה, נִרְדֶּמֶת.
מוֹרִין לִיקְּקָה מֵהַשָּׂפָם טִיפּוֹת בְּצֶבַע אַרְגָּמָן וְחִייְּכָה כְּמוֹ יַלְדָּה שֶׁמִּתְגַּלָּה מְכוֶּּוצֶת בְּמַחֲבוֹאָה.

------------------
וּכְמוֹ בַּחֲלוֹם שָׁמְעָה, אוֹ אוּלַי רַק נִדְמֶה לָהּ, שֶׁאָנִיטָה, לִפְנֵי שֶׁסָּגְרָה אַחֲרֶיהָ, בִּזְהִירוּת, בַּעֲדִינוּת,  אֶת דֶּלֶת הַחֶדֶר, לָחֲשָׁה וְהוֹסִיפָה, "תִּשְׁמְרִי עַל עַצְמְךָ, יִקְרֶה".
זֶה זָהָב אֲמִיתִּי, שֶׁתֵּדְעִי לָךְ, אָמְרָה קְרוֹבַת הַמִּשְׁפָּחָה הָרְחוֹקָה.
וּבְעוֹדֵנוּ צוֹעֲדִים וַאֲנִי מִתְנַדְנֶדֶת עַל מַגְּפֵי הֶעָקֵב הַלֹּא–נוֹחִים שֶׁלִּי, נֶעֱצַר לְרֶגַע וִינְסֵנְט וְשָׁאַל: "אָז מָה שְׁלוֹמְךָ?
הוּא אוֹמֵר, "תִּרְאִי, אֲנִי יוֹדֵעַ שֶׁהִבְהַלְתִּי אוֹתְךָ.

------------------
